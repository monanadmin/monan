35def xtime2numtime(xtime):
36 # First parse the xtime character array into a string
37 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
38
39 dt = []
40 for stritem in xtimestr:
41 itemarray = stritem.strip().replace('_', '-').replace(':', '-').split('-') # Get an array of strings that are Y,M,D,h,m,s
42 results = [int(i) for i in itemarray]
43 if (results[0] < 1900): # datetime has a bug where years less than 1900 are invalid on some systems
44 results[0] += 1900
45 dt.append( datetime.datetime(*results) ) # * notation passes in the array as arguments
46
47 numtime = netCDF4.date2num(dt, units='seconds since '+str(dt[0])) # use the netCDF4 module's function for converting a datetime to a time number
48 return numtime
