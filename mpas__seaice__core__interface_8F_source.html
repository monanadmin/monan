131 subroutine setup_packages_dynamics(configPool, packagePool, ierr)!{{{
132
133 type (mpas_pool_type), intent(in) :: configPool
134 type (mpas_pool_type), intent(in) :: packagePool
135 integer, intent(out) :: ierr
136
137 logical, pointer :: &
138 config_use_velocity_solver
139
140 character(len=strKIND), pointer :: &
141 config_stress_divergence_scheme, &
142 config_variational_basis
143
144 logical, pointer :: &
145 pkgWeakActive, &
146 pkgVariationalActive, &
147 pkgWachspressActive, &
148 pkgPieceWiseLinearActive
149
150 !pkgWeak
151 !pkgVariational
152 !pkgWachspress
153 !pkgPieceWiseLinear
154
155 call mpas_pool_get_config(configpool, "config_use_velocity_solver", config_use_velocity_solver)
156 call mpas_pool_get_config(configpool, "config_stress_divergence_scheme", config_stress_divergence_scheme)
157 call mpas_pool_get_config(configpool, "config_variational_basis", config_variational_basis)
158
159 call mpas_pool_get_package(packagepool, "pkgWeakActive", pkgweakactive)
160 call mpas_pool_get_package(packagepool, "pkgVariationalActive", pkgvariationalactive)
161 call mpas_pool_get_package(packagepool, "pkgWachspressActive", pkgwachspressactive)
162 call mpas_pool_get_package(packagepool, "pkgPieceWiseLinearActive", pkgpiecewiselinearactive)
163
164 ! is the velocity solver on
165 if (config_use_velocity_solver) then
166
167 ! stress divergence scheme
168 if (trim(config_stress_divergence_scheme) == "weak") then
169
170 ! weak stress divergence scheme
171 pkgweakactive = .true.
172
173 else if (trim(config_stress_divergence_scheme) == "variational") then
174
175 ! variational stress divergence scheme
176 pkgvariationalactive = .true.
177
178 ! variational basis function scheme
179 if (trim(config_variational_basis) == "wachspress") then
180
181 ! wachspress basis functions
182 pkgwachspressactive = .true.
183
184 else if (trim(config_variational_basis) == "pwl") then
185
186 ! piecewise linear basis functions
187 pkgpiecewiselinearactive = .true.
188
189 else
190
191 ! unknown value of config_variational_basis
192 call mpas_log_write("Unknown variational scheme: "//trim(config_variational_basis))
193 ierr = 1
194
195 endif ! config_variational_basis
196
197 else
198
199 ! unknown value of config_stress_divergence_scheme
200 call mpas_log_write("Unknown stress divergence scheme: "//trim(config_stress_divergence_scheme))
201 ierr = 1
202
203 endif ! config_stress_divergence_scheme
204
205 endif ! config_use_velocity_solver
206
207 end subroutine setup_packages_dynamics!}}}

601
602 use mpas_derived_types
603 use mpas_decomp
604
605 implicit none
606
607 type (mpas_decomp_list), pointer :: decomplist
608
609 integer :: ierr
610 procedure(mpas_decomp_function), pointer :: decompfunc
611
612 ierr = 0
613
614 nullify(decomplist)
615
616 call mpas_decomp_create_decomp_list(decomplist)
617
618 decompfunc => mpas_uniform_decomp
619
620 if ( associated(decomplist) ) then
621 call mpas_log_write(' It is associated (decompList that is)')
622 end if
623
624 call mpas_decomp_register_method(decomplist, 'uniform', decompfunc, ierr)
625

627 ierr = 0
628 end if
