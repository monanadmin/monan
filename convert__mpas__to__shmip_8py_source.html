1#!/usr/bin/env python
2'''
3Convert MPAS hydro output to the SHMIP format regquired by SHMIP

15parser.add_option("-s", "--scenario", dest="scenario", type='string', help="name of SHMIP scenario this run corresponds to, e.g., A1", metavar="SCENARIO")
16parser.add_option("-t", "--title", dest="title", type='string', help="string to use for title for this test. Defulat: 'hoffman_mpas_SCENARIO'", metavar="TITLE")
17parser.add_option("-v", "--version", dest="hash", type='string', help="version of MPAS used", metavar="HASH")
18options, args = parser.parse_args()
19if not options.filename:
20 options.filename = 'output.nc'
21 print 'No file specified. Attempting to use output.nc'
22if not options.icfilename:
23 options.icfilename = 'landice_grid.nc'
24 print 'No file specified. Attempting to use '+options.icfilename
25if not options.scenario:
26 print 'ERROR: Scenario has not been specified. Please include this with the -s option.'
27 sys.exit()
28if not options.title:
29 options.title = 'hoffman_mpas_'+options.scenario

31if not options.hash:

33 print 'Not version specified. Using: '+options.hash

61# ============================================
62outfile.createDimension('time', None) # None=unlimited

64outfile.createDimension('n_nodes_ch', 2) # how many nodes to make a channel edge. Fixed(?) at 2
65outfile.createDimension('index1', len(infile.dimensions['nCells']))

70print nNSEdges, len(infile.dimensions['nEdges'])

173setattr(thevar, 'units', 'm^2')

177setattr(thevar, 'long_name', 'channel discharge')
