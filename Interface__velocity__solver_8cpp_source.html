536void velocity_solver_compute_2d_grid(int const* _verticesMask_F, int const* _cellsMask_F, int const* _dirichletCellsMask_F, int const* _floatingEdgesMask_F) {
537 int numProcs, me;
538
539 verticesMask_F = _verticesMask_F;
540 cellsMask_F = _cellsMask_F;
541 verticesMask_F = _verticesMask_F;
542 dirichletCellsMask_F = _dirichletCellsMask_F;
543 floatingEdgesMask_F = _floatingEdgesMask_F;
544
545 MPI_Comm_size(comm, &numProcs);
546 MPI_Comm_rank(comm, &me);
547 std::vector<int> partialOffset(numProcs + 1), globalOffsetTriangles(
548 numProcs + 1), globalOffsetVertices(numProcs + 1), globalOffsetEdge(
549 numProcs + 1);
550
551 triangleToFVertex.clear();
552 triangleToFVertex.reserve(nVertices_F);
553 std::vector<int> fVertexToTriangle(nVertices_F, NotAnId);
554 bool changed = false;
555 for (int i(0); i < nVerticesSolve_F; i++) {
556 if ((verticesMask_F[i] & dynamic_ice_bit_value) && !isGhostTriangle(i)) {
557 fVertexToTriangle[i] = triangleToFVertex.size();
558 triangleToFVertex.push_back(i);
559 }
560 changed = changed || (verticesMask_F[i] != mask[i]);
561 }

1057//This function computes the normal velocity on the edges midpoints of MPAS 2d cells.
1058//The function rely on the assumption that the locations of the edges midpoints are inside the union of the triangles that share the dual of that edge.
1059//The normal velocity is computed assuming to have linear/bilinear finite elements on the prisms.
1060//The normal velocity will be second order accurate for other finite elements (e.g. linear or quadratic on Tetrahedra).
1061
1062void get_prism_velocity_on_FEdges(double * uNormal,
1063 const std::vector<double>& velocityOnCells,
1064 const std::vector<int>& edgeToFEdge) {
1065
1066 //using layout of velocityOnCells
1067 int columnShift = 1;
1068 int layerShift = (nLayers + 1);
1069
1070 UInt nPoints3D = nCells_F * (nLayers + 1);
1071
1072 // Loop over all edges of the triangulation - MPAS will decide which edges it should use.
1073 for (int i = 0; i < nEdges; i++) {
1074
1075 //identifying vertices on the edge
1076 ID lId0 = verticesOnEdge[2 * i];
1077 ID lId1 = verticesOnEdge[2 * i + 1];
1078 int iCell0 = vertexToFCell[lId0];
1079 int iCell1 = vertexToFCell[lId1];
1080
1081 //computing normal to the cell edge (dual of triangular edge)
1082 double nx = xCell_F[iCell1] - xCell_F[iCell0];
1083 double ny = yCell_F[iCell1] - yCell_F[iCell0];
1084 double n = sqrt(nx * nx + ny * ny);
1085 nx /= n;
1086 ny /= n;
1087
1088 //identifying triangles that shares the edge
1089 ID iEdge = edgeToFEdge[i];
1090 ID fVertex0 = verticesOnEdge_F[2 * iEdge] - 1;
1091 ID fVertex1 = verticesOnEdge_F[2 * iEdge + 1] - 1;
1092 double t0[2*3], t1[2*3]; //t0[0] contains the x-coords of vertices of triangle 0 and t0[1] its y-coords.
1093 for (int j = 0; j < 3; j++) {
1094 int iCell = cellsOnVertex_F[3 * fVertex0 + j] - 1;
1095 t0[0 + 2 * j] = xCell_F[iCell];
1096 t0[1 + 2 * j] = yCell_F[iCell];
1097 iCell = cellsOnVertex_F[3 * fVertex1 + j] - 1;
1098 t1[0 + 2 * j] = xCell_F[iCell];
1099 t1[1 + 2 * j] = yCell_F[iCell];
1100 }
1101
1102 //getting triangle circumcenters (vertices of MPAS cells).
1103 double bcoords[3];
1104
1105 ID iCells[3]; //iCells[k] is the array of cells indexes of triangle k on iEdge
1106
1107 //computing midpoint of fortran edge
1108 double e_mid[2];
1109 e_mid[0] = 0.5*(xVertex_F[fVertex0] + xVertex_F[fVertex1]);
1110 e_mid[1] = 0.5*(yVertex_F[fVertex0] + yVertex_F[fVertex1]);
1111
1112 if((verticesMask_F[fVertex0] & dynamic_ice_bit_value) && belongToTria(e_mid, t0, bcoords)) {
1113 // triangle1 is in the mesh AND midpoint is in triangle1
1114 for (int j = 0; j < 3; j++)
1115 iCells[j] = cellsOnVertex_F[3 * fVertex0 + j] - 1;
1116 }
1117 else if((verticesMask_F[fVertex1] & dynamic_ice_bit_value) && belongToTria(e_mid, t1, bcoords)) {
1118 //triangle2 is in the mesh AND midpoint is in triangle2
1119 for (int j = 0; j < 3; j++)
1120 iCells[j] = cellsOnVertex_F[3 * fVertex1 + j] - 1;
1121 }
1122 else if(i<numBoundaryEdges) {
1123 //edge is on boundary and wasn't found by previous two cases
1124 //For boundary edges one of the two triangles sharing the edge won't be part of the velocity solver's mesh and the dynamic_ice_bit_value will be 0.

1403double signedTriangleAreaOnSphere(const double* x, const double* y,
1404 const double *z) {
1405 double u[3] = { x[1] - x[0], y[1] - y[0], z[1] - z[0] };
1406 double v[3] = { x[2] - x[0], y[2] - y[0], z[2] - z[0] };
1407
1408 double crossProduct[3] = { u[1] * v[2] - u[2] * v[1], u[2] * v[0]
1409 - u[0] * v[2], u[0] * v[1] - u[1] * v[0] };
1410 double area = 0.5
1411 * std::sqrt(
1412 crossProduct[0] * crossProduct[0] + crossProduct[1] * crossProduct[1]
1413 + crossProduct[2] * crossProduct[2]);
1414 return
1415 (crossProduct[0] * x[0] + crossProduct[1] * y[0] + crossProduct[2] * z[0]
1416 > 0) ? area : -area;
