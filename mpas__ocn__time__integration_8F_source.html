80 subroutine ocn_timestep(domain, dt, timeStamp)!{{{
81 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
82 ! Advance model state forward in time by the specified time step
83 !
84 ! Input: domain - current model state in time level 1 (e.g., time_levs(1)state%h(:,:))
85 ! plus mesh meta-data
86 ! Output: domain - upon exit, time level 2 (e.g., time_levs(2)%state%h(:,:)) contains
87 ! model state advanced forward in time by dt seconds
88 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
89
90 implicit none
91
92 type (domain_type), intent(inout) :: domain
93 real (kind=rkind), intent(in) :: dt
94 character(len=*), intent(in) :: timestamp
95
96 type (dm_info) :: dminfo
97 type (block_type), pointer :: block
98
99 type (mpas_pool_type), pointer :: diagnosticspool, statepool, meshpool
100
101 character (len=StrKIND), pointer :: xtime
102 real (kind=rkind), pointer :: dayssincestartofsim
103 character (len=StrKIND), pointer :: simulationstarttime
104 type (mpas_time_type) :: xtime_timetype, simulationstarttime_timetype
105
106
107 if (rk4on) then
108 call ocn_time_integrator_rk4(domain, dt)
109 elseif (spliton) then
110 call ocn_time_integrator_split(domain, dt)
111 endif
112
113 block => domain % blocklist
114 do while (associated(block))
115 call mpas_pool_get_subpool(block % structs, 'state', statepool)
116 call mpas_pool_get_subpool(block % structs, 'diagnostics', diagnosticspool)
117 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
118
119 call mpas_pool_get_array(diagnosticspool, 'xtime', xtime)
120
121 xtime = timestamp
122
123 ! compute time since start of simulation, in days
124 call mpas_pool_get_array(diagnosticspool, 'simulationStartTime', simulationstarttime)
125 call mpas_pool_get_array(diagnosticspool, 'daysSinceStartOfSim',dayssincestartofsim)
126 call mpas_set_time(xtime_timetype, datetimestring=xtime)
127 call mpas_set_time(simulationstarttime_timetype, datetimestring=simulationstarttime)
128 call mpas_get_timeinterval(xtime_timetype - simulationstarttime_timetype,dt=dayssincestartofsim)
129
130 !$omp single
131 dayssincestartofsim = dayssincestartofsim*days_per_second
132 !$omp end single
133
134 block => block % next
135 end do
136
137 end subroutine ocn_timestep!}}}
138
139 subroutine ocn_timestep_init(err)!{{{
140
141 integer, intent(out) :: err
142
143 character (len=StrKIND), pointer :: config_time_integrator
144
145 err = 0
146
147 call mpas_pool_get_config(ocnconfigs, 'config_time_integrator', config_time_integrator)
148
149 rk4on = .false.
150 spliton = .false.
151
152 if (trim(config_time_integrator) == 'RK4') then
153 rk4on = .true.
154 elseif (trim(config_time_integrator) == 'split_explicit' &
155 .or.trim(config_time_integrator) == 'unsplit_explicit') then
156 spliton = .true.
157 else
158 err = 1
159 call mpas_log_write('Incorrect choice for config_time_integrator:' // trim(config_time_integrator) // &
160 ' choices are: RK4, split_explicit, unsplit_explicit', mpas_log_crit)
161 endif
162
163
164 end subroutine ocn_timestep_init!}}}
