62 function li_core_init(domain, startTimeStamp) result(err)
63
64 use mpas_derived_types
65 use mpas_pool_routines
66 use mpas_stream_manager
67 use li_velocity
68 use li_velocity_external
69 use li_thermal
70 use li_setup
71 use li_constants
72 use li_subglacial_hydro
73!!! use mpas_tracer_advection
74!!! use li_global_diagnostics
75
76 implicit none
77
78 !-----------------------------------------------------------------
79 !
80 ! input variables
81 !
82 !-----------------------------------------------------------------
83
84 !-----------------------------------------------------------------
85 !
86 ! input/output variables
87 !
88 !-----------------------------------------------------------------
89 type (domain_type), intent(inout) :: domain
90
91 !-----------------------------------------------------------------
92 !
93 ! output variables
94 !
95 !-----------------------------------------------------------------
96 character(len=*), intent(out) :: starttimestamp
97
98 !-----------------------------------------------------------------
99 !
100 ! local variables
101 !
102 !-----------------------------------------------------------------
103 type (block_type), pointer :: block
104 type (mpas_time_type) :: starttime
105 integer :: i, err, err_tmp, globalerr
106 logical, pointer :: config_do_restart
107 real (kind=rkind), pointer :: deltat ! variable in each block
108 real (kind=rkind) :: dtseconds ! local variable
109 type (mpas_pool_type), pointer :: meshpool
110 type (mpas_timeinterval_type) :: timestepinterval
111 character (len=StrKIND), pointer :: xtime, simulationstarttime
112 real (kind=rkind), pointer :: dayssincestart
113 integer, dimension(:), pointer :: cellprocid
114 type (mpas_time_type) :: xtime_timetype, simulationstarttime_timetype
115
116
117 err = 0
118 err_tmp = 0
119 globalerr = 0
120
121 call mpas_pool_get_config(domain % configs, 'config_do_restart', config_do_restart)
122
123 !
124 ! Initialize config option settings as needed
125 !
126 call li_setup_config_options( domain, err_tmp )
127 err = ior(err, err_tmp)
128
129 if (config_do_restart) then
130 call mpas_stream_mgr_read(domain % streamManager, streamid='restart', ierr=err_tmp)
131 err = ior(err, err_tmp)
132 else
133 call mpas_stream_mgr_read(domain % streamManager, streamid='input', ierr=err_tmp)
134 err = ior(err, err_tmp)
135 end if
136 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='restart', ierr=err_tmp)
137 err = ior(err, err_tmp)
138 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='input', ierr=err_tmp)
139 err = ior(err, err_tmp)
140 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_output, ierr=err_tmp)
141 err = ior(err, err_tmp)
142
143 !
144 ! Read the remaining input streams
145 !
146 call mpas_timer_start('io_read', .false.)
147 call mpas_stream_mgr_read(domain % streamManager, ierr=err_tmp)
148 err = ior(err, err_tmp)
149 call mpas_timer_stop('io_read')
150 call mpas_timer_start('reset_io_alarms', .false.)
151 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_input, ierr=err_tmp)
152 err = ior(err, err_tmp)
153 call mpas_timer_stop('reset_io_alarms')
154
155 ! ===
156 ! Initialize some time stuff on each block
157 ! ===
158 ! Set startTimeStamp based on the start time of the simulation clock
159 starttime = mpas_get_clock_time(domain % clock, mpas_start_time, err_tmp)
160 call mpas_get_time(starttime, datetimestring=starttimestamp) ! Get the start time as a time stamp
161 err = ior(err, err_tmp)
162
163 timestepinterval = mpas_get_clock_timestep(domain % clock, ierr=err_tmp) ! get timestep interval object
164 err = ior(err,err_tmp)
165 call mpas_get_timeinterval(timestepinterval, starttimein=starttime, dt=dtseconds, ierr=err_tmp) ! Get config dt in seconds
166 err = ior(err,err_tmp)
167
168 block => domain % blocklist
169 do while (associated(block))
170 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
171
172 ! Assign initial time stamp
173 call mpas_pool_get_array(meshpool, 'xtime', xtime)
174 xtime = starttimestamp
175
176 ! Set simulationStartTime on a cold start only - it should have been read from restart otherwise
177 call mpas_pool_get_array(meshpool, 'simulationStartTime', simulationstarttime)
178 if (trim(simulationstarttime)=="no_date_available") then
179 if (config_do_restart) then
180 call mpas_log_write("simulationStartTime is undefined but this is a restart, " // &
181 "so setting its value to the current start time which may be incorrect.", mpas_log_warn)
182 endif
183 simulationstarttime = starttimestamp
184 elseif ((trim(simulationstarttime)/="no_date_available") .and. (.not. config_do_restart)) then
185 call mpas_log_write("This is a cold start but simulationStartTime is already defined, " // &
186 "so its value may be incorrect.", mpas_log_warn)
187 endif
188
189 ! compute time since start of simulation, in days
190 call mpas_pool_get_array(meshpool, 'daysSinceStart',dayssincestart)
191 call mpas_set_time(xtime_timetype, datetimestring=xtime)
192 call mpas_set_time(simulationstarttime_timetype, datetimestring=simulationstarttime)
193 call mpas_get_timeinterval(xtime_timetype - simulationstarttime_timetype,dt=dayssincestart)
194 dayssincestart = dayssincestart / seconds_per_day
195
196 ! Initialize dt in seconds
197 call mpas_pool_get_array(meshpool, 'deltat', deltat)
198 deltat = dtseconds
199
200 block => block % next
201 end do
202
203 ! ===
204 ! === Initialize modules ===
205 ! ===
206
207 call li_velocity_init(domain, err_tmp)
208 err = ior(err, err_tmp)
209
210!!! call mpas_tracer_advection_init(err_tmp) ! Calling signature may be incorrect here.
211!!! err = ior(err,err_tmp)
212
213 ! ===
214 ! === Initialize blocks ===
215 ! ===
216 block => domain % blocklist
217 do while (associated(block))
218 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
219 call mpas_pool_get_array(meshpool, 'cellProcID', cellprocid)
220 cellprocid(:) = domain % dminfo % my_proc_id
221
222 call landice_init_block(block, domain % dminfo, err_tmp)
223 err = ior(err, err_tmp)
224
225 block => block % next
226 end do
227
228 ! initialize thermal solver
229 ! Note: This subroutine includes a loop over blocks
230 call li_thermal_init(domain, err_tmp)
231 err = ior(err, err_tmp)
232
233 ! Initialize subglacial hydrology solver
234 call li_sgh_init(domain, err_tmp)
235 err = ior(err, err_tmp)
236
237 ! initialize analysis driver
238 call li_analysis_init(domain, err_tmp)
239 err = ior(err, err_tmp)
240
241 ! halo update for reconstruction coefficients
242 ! Note: Results on multiple processors may be incorrect without this update
243
244 call mpas_timer_start("halo updates")
245 call mpas_dmpar_field_halo_exch(domain, 'coeffs_reconstruct')
246 call mpas_timer_stop("halo updates")
247
248 ! ===
249 ! Perform albany mesh write if requested - Note: model terminates after this.
250 ! I think this has to come here so we 1. have the vertical coordinate initialized first,
251 ! which occurs in landice_init_block and 2. have li_velocity_init.
252 ! ===
253 call li_velocity_external_write_albany_mesh(domain)
254
255 ! check for errors and exit
256
257 call mpas_dmpar_max_int(domain % dminfo, err, globalerr) ! Find out if any blocks got an error
258 if (globalerr > 0) then
259 call mpas_log_write("An error has occurred in li_core_init. Aborting...", mpas_log_crit)
260 endif
261
262 !--------------------------------------------------------------------
263 end function li_core_init

278
279 function li_core_run(domain) result(err)
280
281 use mpas_derived_types
282 use mpas_pool_routines
283 use mpas_kind_types
284 use mpas_stream_manager
285 use mpas_timer
286 use mpas_io_streams, only: mpas_stream_latest_before
287 use mpas_io_units
288 use li_diagnostic_vars
289 use li_setup
290 use li_statistics
291 use li_time_integration
292
293 implicit none
294
295 !-----------------------------------------------------------------
296 !
297 ! input variables
298 !
299 !-----------------------------------------------------------------
300
301 !-----------------------------------------------------------------
302 !
303 ! input/output variables
304 !
305 !-----------------------------------------------------------------
306 type (domain_type), intent(inout) :: domain
307
308 !-----------------------------------------------------------------
309 !
310 ! output variables
311 !
312 !-----------------------------------------------------------------
313
314 !-----------------------------------------------------------------
315 !
316 ! local variables
317 !
318 !-----------------------------------------------------------------
319 integer, pointer :: timestepnumber
320 type (block_type), pointer :: block
321 type (mpas_pool_type), pointer :: geometrypool, meshpool
322 integer, pointer :: config_stats_interval
323 character(len=StrKIND), pointer :: config_restart_timestamp_name
324 character(len=StrKIND), pointer :: config_velocity_solver
325 ! Variables needed for printing timestamps
326 type (mpas_time_type) :: currtime
327 character(len=StrKIND) :: timestamp
328 logical :: isringing
329 integer :: restarttimestampunit
330
331 integer :: err, err_tmp, err_tmp2, globalerr
332
333
334 err = 0
335 err_tmp = 0
336 globalerr = 0
337
338 ! Get Pool stuff that will be needed
339 call mpas_pool_get_config(liconfigs, 'config_restart_timestamp_name', config_restart_timestamp_name)
340 call mpas_pool_get_config(liconfigs, 'config_stats_interval', config_stats_interval)
341 call mpas_pool_get_config(liconfigs, 'config_velocity_solver', config_velocity_solver)
342
343 call mpas_timer_start("land ice core run")
344
345 ! initialize time step number. initial time is 0
346 block => domain % blocklist
347 do while(associated(block))
348 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
349 call mpas_pool_get_array(meshpool, 'timestepNumber', timestepnumber)
350 timestepnumber = 0
351 block => block % next
352 end do
353
354 ! ===
355 ! Solve initial state before beginning time stepping
356 ! ===
357 err_tmp = li_core_initial_solve(domain)
358 err = ior(err,err_tmp) ! li_core_initial_solve would abort if there was an error, but being safe.
359
360 ! During integration, time level 1 stores the model state at the beginning of the
361 ! time step, and time level 2 stores the state advanced dt in time by timestep(...)
362 ! ===
363 ! === Time step loop
364 ! ===
365 do while (.not. mpas_is_clock_stop_time(domain % clock))
366
367 ! Update time step number
368 block => domain % blocklist
369 do while(associated(block))
370 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
371 call mpas_pool_get_array(meshpool, 'timestepNumber', timestepnumber)
372 timestepnumber = timestepnumber + 1
373 block => block % next
374 end do
375
376 call mpas_log_write('Starting timestep number $i', intargs=(/timestepnumber/), flushnow=.true.)
377
378 ! ===
379 ! === Perform Timestep
380 ! ===
381 call mpas_timer_start("time integration")
382
383 call li_timestep(domain, err_tmp)
384 err = ior(err,err_tmp)
385
386 ! Write statistics at designated interval
387 if (config_stats_interval > 0) then
388 if (mod(timestepnumber, config_stats_interval) == 0) then
389 call mpas_timer_start("compute_statistics")
390 call li_compute_statistics(domain, timestepnumber)
391 call mpas_timer_stop("compute_statistics")
392 end if
393 end if
394
395 call mpas_timer_stop("time integration")
396
397 ! ===
398 ! === Read time-varying inputs, if present (i.e., forcing)
399 ! ===
400 ! This should happen at the end of the time step so that if we write out
401 ! the forcing it is at the correct time level.
402 ! For an explicit time-stepping method, we want the forcing to be at the
403 ! *old* time when it is applied during time integration. Reading it here
404 ! will allow that.
405 ! Finally, set whence to latest_before so we have piecewise-constant forcing.
406 ! Could add, e.g., linear interpolation later.
407 call mpas_stream_mgr_read(domain % streamManager, whence=mpas_stream_latest_before, ierr=err_tmp)
408 err = ior(err, err_tmp)
409 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_input, ierr=err_tmp)
410 err = ior(err, err_tmp)
411
412 ! call analysis driver compute, restart, write subroutines
413 ! (note: alarms and timers are handled by the analysis member code)
414 call li_analysis_compute(domain, err_tmp)
415 err = ior(err, err_tmp)
416 call li_analysis_restart(domain, err_tmp)
417 err = ior(err, err_tmp)
418 call li_analysis_write(domain, err_tmp)
419 err = ior(err, err_tmp)
420
421 ! ===
422 ! === Write Output and/or Restart, if needed
423 ! ===
424
425 call mpas_timer_start("write streams")
426 call mpas_stream_mgr_write(domain % streamManager, ierr=err_tmp)
427 if (err_tmp > 0) then
428 call mpas_log_write('Error writing output or restart streams.', mpas_log_err)
429 endif
430 err = ior(err, err_tmp)
431 ! Update the restart_timestamp file with the new time, if needed.
432 if ( domain % dminfo % my_proc_id == 0 ) then ! Only have 1 task write this file.
433 ! Do this only after writing has completed successfully.
434 ! (Otherwise we would be setting the restart_timestamp to an invalid time level.)
435 ! (This ignores the situation where restart wrote successfully but one or more other output streams did not.)
436 if (err_tmp == 0) then
437 isringing = mpas_stream_mgr_ringing_alarms(domain % streamManager, streamid='restart', &
438 direction=mpas_stream_output, ierr=err_tmp2)
439 if (err_tmp2 > 0) then
440 call mpas_log_write('Error checking restart stream alarm.', mpas_log_err)
441 endif
442 err = ior(err, err_tmp2)
443
444 if (isringing) then
445 ! Need updated timestamp for writing restart timestamp
446 currtime = mpas_get_clock_time(domain % clock, mpas_now, err_tmp2)
447 if (err_tmp2 > 0) then
448 call mpas_log_write('Error in mpas_get_clock_time when writing restart timestamp.', mpas_log_err)
449 endif
450 err = ior(err, err_tmp2)
451 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=err_tmp2)
452 if (err_tmp2 > 0) then
453 call mpas_log_write('Error in mpas_get_time when writing restart timestamp.', mpas_log_err)
454 endif
455 err = ior(err, err_tmp2)
456
457 ! write the timestamp to file
458 call mpas_new_unit(restarttimestampunit)
459 open(restarttimestampunit, file=config_restart_timestamp_name, form='formatted', status='replace', iostat=err_tmp2)
460 if (err_tmp2 > 0) then
461 call mpas_log_write('Error opening file to write restart timestamp:' // &
462 trim(config_restart_timestamp_name), mpas_log_err)
463 endif
464 err = ior(err, err_tmp2)
465 write(restarttimestampunit, *, iostat=err_tmp2) timestamp
466 if (err_tmp2 > 0) then
467 call mpas_log_write('Error writing restart timestamp to file ' // &
468 trim(config_restart_timestamp_name), mpas_log_err)
469 endif
470 err = ior(err, err_tmp2)
471 close(restarttimestampunit)
472 call mpas_release_unit(restarttimestampunit)
473 end if ! isRinging
474 end if ! err_tmp==0
475 end if ! master task
476 ! Now that we are done here, reset output alarms (can be called whether or not it was ringing)
477 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_output, ierr=err_tmp)
478 if (err_tmp > 0) then
479 call mpas_log_write('Error resetting output-direction stream alarms.', mpas_log_err)
480 endif
481 err = ior(err, err_tmp)
482 call mpas_timer_stop("write streams")
483
484 ! === (end of output section) ===
485
486
487 ! Move time level 1 fields (current values) into time level 2 (old values) for next time step
488 ! (for those fields with multiple time levels)
489 block => domain % blocklist
490 do while(associated(block))
491 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
492 call mpas_pool_shift_time_levels(geometrypool)
493 block => block % next
494 end do
495
496 ! Reset the alarm for checking for force setting of the adaptive timestep interval
497 if (mpas_is_alarm_ringing(domain % clock, 'adaptiveTimestepForceInterval', ierr=err_tmp)) then
498 err = ior(err, err_tmp)
499 call mpas_reset_clock_alarm(domain % clock, 'adaptiveTimestepForceInterval', ierr=err_tmp)
500 err = ior(err, err_tmp)
501 endif
502 err = ior(err, err_tmp)
503
504 ! === error check and exit
505 call mpas_dmpar_max_int(domain % dminfo, err, globalerr) ! Find out if any blocks got an error
506 if (globalerr > 0) then
507 call mpas_log_write("An error has occurred in li_core_run. Aborting...", mpas_log_crit)
508 endif
509
510 end do
511 call mpas_timer_stop("land ice core run")
512
513 !--------------------------------------------------------------------
514 end function li_core_run

604! function li_core_initial_solve
605!
614!
615!-----------------------------------------------------------------------
616 function li_core_initial_solve(domain) result(err)
617
618 use mpas_derived_types
619 use mpas_pool_routines
620 use mpas_kind_types
621 use mpas_stream_manager
622 use mpas_timer
623 use li_diagnostic_vars
624 use li_setup
625 use li_statistics
626 use li_advection
627 use mpas_io_streams, only: mpas_stream_latest_before
628
629 implicit none
630
631 !-----------------------------------------------------------------
632 !
633 ! input variables
634 !
635 !-----------------------------------------------------------------
636
637 !-----------------------------------------------------------------
638 !
639 ! input/output variables
640 !
641 !-----------------------------------------------------------------
642 type (domain_type), intent(inout) :: domain
643
644 !-----------------------------------------------------------------
645 !
646 ! output variables
647 !
648 !-----------------------------------------------------------------
649
650 !-----------------------------------------------------------------
651 !
652 ! local variables
653 !
654 !-----------------------------------------------------------------
655 type (block_type), pointer :: block
656 type (mpas_pool_type), pointer :: geometrypool, meshpool, velocitypool
657 logical, pointer :: config_do_restart, config_write_output_on_startup, config_write_stats_on_startup
658 character(len=StrKIND), pointer :: config_velocity_solver
659 ! Variables needed for printing timestamps
660 type (mpas_time_type) :: currtime
661 character(len=StrKIND) :: timestamp
662
663 integer :: err, err_tmp, globalerr
664 logical :: solvevelo
665
666 integer, dimension(:), pointer :: vertexmask
667
668
669 err = 0
670 err_tmp = 0
671 globalerr = 0
672
673 ! Get Pool stuff that will be needed
674 call mpas_pool_get_config(liconfigs, 'config_do_restart', config_do_restart)
675 call mpas_pool_get_config(liconfigs, 'config_write_output_on_startup', config_write_output_on_startup)
676 call mpas_pool_get_config(liconfigs, 'config_write_stats_on_startup', config_write_stats_on_startup)
677 call mpas_pool_get_config(liconfigs, 'config_velocity_solver', config_velocity_solver)
678
679 currtime = mpas_get_clock_time(domain % clock, mpas_now, err_tmp)
680 err = ior(err, err_tmp)
681 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=err_tmp)
682 err = ior(err, err_tmp)
683 call mpas_log_write('Initial timestep ' // trim(timestamp))
684
685
686 ! ===
687 ! === Calculate Initial state
688 ! ===
689 call mpas_timer_start("initial state calculation")
690
691 ! On a restart, we already have the exact velocity field we need,
692 ! so don't do the expensive calculation again.
693 if (config_do_restart) then
694 solvevelo = .false.
695 else
696 ! Otherwise, we need to calculate velocity for the initial state
697 ! (Note: even if the velocity is supplied, we should still calculate it
698 ! to ensure it is consistent with the current geometry/B.C. If the
699 ! velocity solver is iterative, the supplied field will be used as an
700 ! initial guess, so the solution should be quick.
701 solvevelo = .true.
702 endif
703
704 ! Update mask and geometry
705 block => domain % blocklist
706 do while (associated(block))
707 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
708 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
709 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
710
711 call li_calculate_mask(meshpool, velocitypool, geometrypool, err_tmp)
712 call li_update_geometry(geometrypool)
713
714 block => block % next
715 end do
716
717 ! Initial velocity solve
718 call li_velocity_solve(domain, solvevelo=solvevelo, err=err_tmp)
719 err = ior(err, err_tmp)
720
721 ! Calculate diagnostic vars
722 call li_calculate_diagnostic_vars(domain, err=err_tmp)
723 err = ior(err, err_tmp)
724
725 call mpas_timer_stop("initial state calculation")
726
727 if (config_write_stats_on_startup) then
728 call mpas_timer_start("compute_statistics")
729 call li_compute_statistics(domain, 0) ! itimestep = 0
730 call mpas_timer_stop("compute_statistics")
731 endif
732
733 ! compute analysis members on startup if option activated (checked inside routine)
734 ! If we are doing a restart, any required output from this time level should already be written, so don't recalculate/rewrite
735 ! Note: for debugging inexact restarts, it can be useful to enable this write on a restart
736 if (.not. config_do_restart) then
737 call mpas_timer_start("analysis member startup calculations")
738 call li_analysis_compute_startup(domain, err_tmp)
739 err = ior(err, err_tmp)
740 call mpas_timer_stop("analysis member startup calculations")

744 ! === Write Initial Output
745 ! ===

749 ! Note: for debugging inexact restarts, it can be useful to enable this write on a restart

752 call mpas_stream_mgr_write(domain % streamManager, 'output', forcewritenow=.true., ierr=err_tmp)

764 block => block % next

774 call mpas_pool_get_array(geometrypool, 'vertexMask', vertexmask, timelevel=2) ! Get the old vertexMask

811 subroutine landice_init_block(block, dminfo, err)

814 use mpas_pool_routines

816 use mpas_vector_reconstruction
817 use li_setup

853 character (len=StrKIND), pointer :: config_velocity_solver
854 logical, pointer :: config_do_velocity_reconstruction_for_external_dycore
855 logical, pointer :: config_adaptive_timestep_include_DCFL
856 logical, pointer :: config_SGH
857 integer :: err_tmp
858
859 err = 0
860 err_tmp = 0
861
862 ! Get pool stuff
863 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
864 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
865 call mpas_pool_get_config(liconfigs, 'config_velocity_solver', config_velocity_solver)
866 call mpas_pool_get_config(liconfigs, 'config_do_velocity_reconstruction_for_external_dycore', &
867 config_do_velocity_reconstruction_for_external_dycore)

869 call mpas_pool_get_config(liconfigs, 'config_SGH', config_sgh)

871 ! Copy data from first time level into all other time levels
872 call mpas_pool_initialize_time_levels(geometrypool)
873
874 ! Initialize vertexMask on time level 2 to 0, so diagnostic_solve_before_velocity in li_diagnostic_vars
875 ! says that the vertexMask has changed (needed by external dycore)
876 call mpas_pool_get_array(geometrypool, 'vertexMask', vertexmask, timelevel = 2)
877 vertexmask = 0
878
879 ! ===
880 ! === Call init routines ===
881 ! ===
882 call li_setup_vertical_grid(meshpool, geometrypool, err_tmp)
883 err = ior(err, err_tmp)
884
885 call li_setup_sign_and_index_fields(meshpool)
886
887! This was needed to init FCT once.
888!!! ! Init for FCT tracer advection
889!!! mesh % maxLevelCell % array = mesh % nVertLevels ! Needed for FCT tracer advection
890!!! mesh % maxLevelEdgeTop % array = mesh % nVertLevels ! Needed for FCT tracer advection
891!!! mesh % maxLevelEdgeBot % array = mesh % nVertLevels ! Needed for FCT tracer advection
892!!! call ocn_initialize_advection_rk(mesh, err)
893!!! call mpas_ocn_tracer_advection_coefficients(mesh, err_tmp)
894!!! err = ior(err, err_tmp)
895
896 ! Init for reconstruction of velocity
897 ! Note: mpas_init_reconstruct fails with the MISMIP3d periodic mesh
898 ! because x_period and y_period are not set properly.
899 ! That should be fixed for that test, but a workaround is to
900 ! disable these two calls for those runs.
901 call mpas_rbf_interp_initialize(meshpool)
902 call mpas_init_reconstruct(meshpool)
903
904 ! Initialize velocity solver
905 !WHL - This is now after the call to mpas_init_reconstruct, so that the reconstruction coefficients are available.
906 call mpas_timer_start("initialize velocity")
907 call li_velocity_block_init(block, err_tmp)
908 err = ior(err, err_tmp)
909 call mpas_timer_stop("initialize velocity")
910
911 ! Higher-order velo solvers needs vertex-to-cell interp routine initialized
912 if ( (trim(config_velocity_solver) == 'L1L2') .or. &
913 (trim(config_velocity_solver) == 'FO') .or. &
914 (trim(config_velocity_solver) == 'Stokes') ) then
915 call li_setup_wachspress_vertex_to_cell_weights(meshpool)
916 endif
917
918 ! Mask init identifies initial ice extent
919 call li_calculate_mask_init(geometrypool, err=err_tmp)
920 err = ior(err, err_tmp)
921
922 ! Initialize thicknessOld variable used to calculate dHdt
923 call mpas_pool_get_array(geometrypool, 'thickness', thickness)
924 call mpas_pool_get_array(geometrypool, 'thicknessOld', thicknessold)
925 thicknessold = thickness
926
927 ! === error check
928 if (err > 0) then
929 call mpas_log_write("An error has occurred in landice_init_block.", mpas_log_err)
930 endif
931
932 !--------------------------------------------------------------------
933 end subroutine landice_init_block

987 config_output_interval, config_restart_interval ! MPAS standard configs
988 character (len=StrKIND), pointer :: config_dt ! MPAS LI-specific config option
989 character (len=StrKIND), pointer :: config_adaptive_timestep_force_interval ! MPAS LI-specific config option

1061 alarmtimeinterval=adaptdtforceinterval, ierr=err_tmp)
1062 ierr = ior(ierr,err_tmp)
1063 ! Reset the alarm for checking for force setting of the adaptive timestep interval
1064 if (mpas_is_alarm_ringing(core_clock, 'adaptiveTimestepForceInterval', ierr=err_tmp)) then
1065 ierr = ior(ierr, err_tmp)
1066 call mpas_reset_clock_alarm(core_clock, 'adaptiveTimestepForceInterval', ierr=err_tmp)
1067 ierr = ior(ierr, err_tmp)
1068 endif
1069 ierr = ior(ierr, err_tmp)
1070
1071 ! === error check
1072 if (ierr > 0) then
1073 call mpas_log_write("An error has occurred in li_simulation_clock_init.", mpas_log_err)
1074 endif

1077 end subroutine li_simulation_clock_init
