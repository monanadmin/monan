56 subroutine mpas_atm_update_bdy_tend(clock, streamManager, block, firstCall, ierr)
57
58 use mpas_constants, only : rvord
59 use mpas_stream_manager, only : mpas_stream_mgr_read
60 use mpas_log, only : mpas_log_write
61 use mpas_derived_types, only : mpas_stream_mgr_noerr, mpas_log_err
62 use mpas_timekeeping, only : mpas_get_time
63
64 implicit none
65
66 type (mpas_clock_type), intent(in) :: clock
67 type (mpas_streammanager_type), intent(inout) :: streammanager
68 type (block_type), intent(inout) :: block
69 logical, intent(in) :: firstcall
70 integer, intent(out) :: ierr
71
72 character(len=StrKIND) :: lbc_intv_start_string
73 character(len=StrKIND) :: lbc_intv_end_string
74
75 type (mpas_pool_type), pointer :: mesh
76 type (mpas_pool_type), pointer :: state
77 type (mpas_pool_type), pointer :: lbc
78 real (kind=rkind) :: dt
79
80 integer, pointer :: ncells
81 integer, pointer :: nedges
82 integer, pointer :: index_qv
83
84 real (kind=rkind), dimension(:,:), pointer :: u
85 real (kind=rkind), dimension(:,:), pointer :: ru
86 real (kind=rkind), dimension(:,:), pointer :: rho_edge
87 real (kind=rkind), dimension(:,:), pointer :: w
88 real (kind=rkind), dimension(:,:), pointer :: theta
89 real (kind=rkind), dimension(:,:), pointer :: rtheta_m
90 real (kind=rkind), dimension(:,:), pointer :: rho_zz
91 real (kind=rkind), dimension(:,:), pointer :: rho
92 real (kind=rkind), dimension(:,:,:), pointer :: scalars
93 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_u
94 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_ru
95 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_rho_edge
96 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_w
97 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_theta
98 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_rtheta_m
99 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_rho_zz
100 real (kind=rkind), dimension(:,:), pointer :: lbc_tend_rho
101 real (kind=rkind), dimension(:,:,:), pointer :: lbc_tend_scalars
102
103 integer, dimension(:,:), pointer :: cellsonedge
104 real (kind=rkind), dimension(:,:), pointer :: zz
105
106 integer :: dd_intv, s_intv, sn_intv, sd_intv
107 type (mpas_time_type) :: currtime
108 type (mpas_timeinterval_type) :: lbc_interval
109 character(len=StrKIND) :: read_time
110 integer :: iedge
111 integer :: cell1, cell2
112
113
114 ierr = 0
115
116 call mpas_pool_get_subpool(block % structs, 'mesh', mesh)
117 call mpas_pool_get_subpool(block % structs, 'state', state)
118 call mpas_pool_get_subpool(block % structs, 'lbc', lbc)
119
120 if (firstcall) then
121 call mpas_stream_mgr_read(streammanager, streamid='lbc_in', timelevel=2, whence=mpas_stream_latest_before, &
122 actualwhen=read_time, ierr=ierr)
123 if (ierr /= mpas_stream_mgr_noerr) then
124 call mpas_log_write('Could not read from ''lbc_in'' stream on or before the current date '// &
125 'to update lateral boundary tendencies', messagetype=mpas_log_err)
126 ierr = 1
127 end if
128 else
129 call mpas_pool_shift_time_levels(lbc)
130 call mpas_stream_mgr_read(streammanager, streamid='lbc_in', timelevel=2, whence=mpas_stream_earliest_strictly_after, &
131 actualwhen=read_time, ierr=ierr)
132 if (ierr /= mpas_stream_mgr_noerr) then
133 call mpas_log_write('Could not read from ''lbc_in'' stream after the current date '// &
134 'to update lateral boundary tendencies', messagetype=mpas_log_err)
135 ierr = 1
136 end if
137 end if
138 if (ierr /= 0) then
139 return
140 end if
141
142 call mpas_set_time(currtime, datetimestring=trim(read_time))
143
144 !
145 ! Compute any derived fields from those that were read from the lbc_in stream
146 !
147 call mpas_pool_get_array(lbc, 'lbc_u', u, 2)
148 call mpas_pool_get_array(lbc, 'lbc_ru', ru, 2)
149 call mpas_pool_get_array(lbc, 'lbc_rho_edge', rho_edge, 2)
150 call mpas_pool_get_array(lbc, 'lbc_theta', theta, 2)
151 call mpas_pool_get_array(lbc, 'lbc_rtheta_m', rtheta_m, 2)
152 call mpas_pool_get_array(lbc, 'lbc_rho_zz', rho_zz, 2)
153 call mpas_pool_get_array(lbc, 'lbc_rho', rho, 2)
154 call mpas_pool_get_array(lbc, 'lbc_scalars', scalars, 2)
155
156 call mpas_pool_get_array(mesh, 'cellsOnEdge', cellsonedge)
157 call mpas_pool_get_dimension(mesh, 'nCells', ncells)
158 call mpas_pool_get_dimension(mesh, 'nEdges', nedges)
159 call mpas_pool_get_dimension(lbc, 'index_qv', index_qv)
160 call mpas_pool_get_array(mesh, 'zz', zz)
161
162 ! Compute lbc_rho_zz
163 zz(:,ncells+1) = 1.0_rkind ! Avoid potential division by zero in the following line
164 rho_zz(:,:) = rho(:,:) / zz(:,:)
165
166 ! Average lbc_rho_zz to edges
167 do iedge=1,nedges
168 cell1 = cellsonedge(1,iedge)
169 cell2 = cellsonedge(2,iedge)
170 if (cell1 > 0 .and. cell2 > 0) then
171 rho_edge(:,iedge) = 0.5_rkind * (rho_zz(:,cell1) + rho_zz(:,cell2))
172 end if
173 end do
174
175 ru(:,:) = u(:,:) * rho_edge(:,:)
176 rtheta_m(:,:) = theta(:,:) * rho_zz(:,:) * (1.0_rkind + rvord * scalars(index_qv,:,:))
177
178 if (.not. firstcall) then
179 lbc_interval = currtime - lbc_intv_end
180 call mpas_get_timeinterval(interval=lbc_interval, dd=dd_intv, s=s_intv, s_n=sn_intv, s_d=sd_intv, ierr=ierr)
181 dt = 86400.0_rkind * real(dd_intv, kind=rkind) + real(s_intv, kind=rkind) &
182 + (real(sn_intv, kind=rkind) / real(sd_intv, kind=rkind))
183
184 call mpas_pool_get_array(lbc, 'lbc_u', u, 2)
185 call mpas_pool_get_array(lbc, 'lbc_ru', ru, 2)
186 call mpas_pool_get_array(lbc, 'lbc_rho_edge', rho_edge, 2)
187 call mpas_pool_get_array(lbc, 'lbc_w', w, 2)
188 call mpas_pool_get_array(lbc, 'lbc_theta', theta, 2)
189 call mpas_pool_get_array(lbc, 'lbc_rtheta_m', rtheta_m, 2)
190 call mpas_pool_get_array(lbc, 'lbc_rho_zz', rho_zz, 2)
191 call mpas_pool_get_array(lbc, 'lbc_rho', rho, 2)
192 call mpas_pool_get_array(lbc, 'lbc_scalars', scalars, 2)
193
194 call mpas_pool_get_array(lbc, 'lbc_u', lbc_tend_u, 1)
195 call mpas_pool_get_array(lbc, 'lbc_ru', lbc_tend_ru, 1)
196 call mpas_pool_get_array(lbc, 'lbc_rho_edge', lbc_tend_rho_edge, 1)
197 call mpas_pool_get_array(lbc, 'lbc_w', lbc_tend_w, 1)
198 call mpas_pool_get_array(lbc, 'lbc_theta', lbc_tend_theta, 1)
199 call mpas_pool_get_array(lbc, 'lbc_rtheta_m', lbc_tend_rtheta_m, 1)
200 call mpas_pool_get_array(lbc, 'lbc_rho_zz', lbc_tend_rho_zz, 1)
201 call mpas_pool_get_array(lbc, 'lbc_rho', lbc_tend_rho, 1)
202 call mpas_pool_get_array(lbc, 'lbc_scalars', lbc_tend_scalars, 1)
203
204
205 dt = 1.0_rkind / dt
206 lbc_tend_u(:,:) = (u(:,:) - lbc_tend_u(:,:)) * dt
207 lbc_tend_ru(:,:) = (ru(:,:) - lbc_tend_ru(:,:)) * dt
208 lbc_tend_rho_edge(:,:) = (rho_edge(:,:) - lbc_tend_rho_edge(:,:)) * dt
209 lbc_tend_w(:,:) = (w(:,:) - lbc_tend_w(:,:)) * dt
210 lbc_tend_theta(:,:) = (theta(:,:) - lbc_tend_theta(:,:)) * dt
211 lbc_tend_rtheta_m(:,:) = (rtheta_m(:,:) - lbc_tend_rtheta_m(:,:)) * dt
212 lbc_tend_rho_zz(:,:) = (rho_zz(:,:) - lbc_tend_rho_zz(:,:)) * dt
213 lbc_tend_rho(:,:) = (rho(:,:) - lbc_tend_rho(:,:)) * dt
214 lbc_tend_scalars(:,:,:) = (scalars(:,:,:) - lbc_tend_scalars(:,:,:)) * dt
215
216 !
217 ! Logging the lbc start and end times appears to be backwards, but
218 ! until the end of this function, LBC_intv_end == the last interval
219 ! time and currTime == the next interval time.
220 !
221 call mpas_get_time(lbc_intv_end, datetimestring=lbc_intv_start_string)
222 call mpas_get_time(currtime, datetimestring=lbc_intv_end_string)
223 call mpas_log_write('----------------------------------------------------------------------')
224 call mpas_log_write('Updated lateral boundary conditions. LBCs are now valid')
225 call mpas_log_write('from '//trim(lbc_intv_start_string)//' to '//trim(lbc_intv_end_string))
226 call mpas_log_write('----------------------------------------------------------------------')
227
228 end if
229
230 lbc_intv_end = currtime
231
232 end subroutine mpas_atm_update_bdy_tend

268 function mpas_atm_get_bdy_tend(clock, block, vertDim, horizDim, field, delta_t) result(return_tend)
269
270 implicit none
271
272 type (mpas_clock_type), intent(in) :: clock
273 type (block_type), intent(inout) :: block
274 integer, intent(in) :: vertdim, horizdim
275 character(len=*), intent(in) :: field
276 real (kind=rkind), intent(in) :: delta_t
277
278 real (kind=rkind), dimension(vertDim,horizDim+1) :: return_tend
279
280 type (mpas_pool_type), pointer :: lbc
281 integer, pointer :: idx
282 real (kind=rkind), dimension(:,:), pointer :: tend
283 real (kind=rkind), dimension(:,:,:), pointer :: tend_scalars
284 integer :: ierr
285
286
287 call mpas_pool_get_subpool(block % structs, 'lbc', lbc)
288
289 nullify(tend)
290 call mpas_pool_get_array(lbc, 'lbc_'//trim(field), tend, 1)
291
292 if (associated(tend)) then
293 return_tend(:,:) = tend(:,:)
294 else
295 call mpas_pool_get_array(lbc, 'lbc_scalars', tend_scalars, 1)
296 call mpas_pool_get_dimension(lbc, 'index_'//trim(field), idx)
297
298 return_tend(:,:) = tend_scalars(idx,:,:)
299 end if
300
301 end function mpas_atm_get_bdy_tend

336 !-----------------------------------------------------------------------
337 function mpas_atm_get_bdy_state(clock, block, vertDim, horizDim, field, delta_t) result(return_state)
338
339 use mpas_pool_routines, only : mpas_pool_get_error_level, mpas_pool_set_error_level
340 use mpas_derived_types, only : mpas_pool_silent
341
342 implicit none
343
344 type (mpas_clock_type), intent(in) :: clock
345 type (block_type), intent(inout) :: block
346 integer, intent(in) :: vertdim, horizdim
347 character(len=*), intent(in) :: field
348 real (kind=rkind), intent(in) :: delta_t
349
350 real (kind=rkind), dimension(vertDim,horizDim+1) :: return_state
351
352 type (mpas_pool_type), pointer :: lbc
353 integer, pointer :: idx
354 real (kind=rkind), dimension(:,:), pointer :: tend
355 real (kind=rkind), dimension(:,:), pointer :: state
356 real (kind=rkind), dimension(:,:,:), pointer :: tend_scalars
357 real (kind=rkind), dimension(:,:,:), pointer :: state_scalars
358 type (mpas_time_type) :: currtime
359 type (mpas_timeinterval_type) :: lbc_interval
360 integer :: dd_intv, s_intv, sn_intv, sd_intv
361 real (kind=rkind) :: dt
362 integer :: err_level
363 integer :: ierr
364
365
366 currtime = mpas_get_clock_time(clock, mpas_now, ierr)
367
368 lbc_interval = lbc_intv_end - currtime
369
370 call mpas_get_timeinterval(interval=lbc_interval, dd=dd_intv, s=s_intv, s_n=sn_intv, s_d=sd_intv, ierr=ierr)
371 dt = 86400.0_rkind * real(dd_intv, kind=rkind) + real(s_intv, kind=rkind) &
372 + (real(sn_intv, kind=rkind) / real(sd_intv, kind=rkind))
373
374 dt = dt - delta_t
375
376 call mpas_pool_get_subpool(block % structs, 'lbc', lbc)
377
378 !
379 ! The first two calls to mpas_pool_get_array, below, may cause harmless warning
380 ! messages, which we can silence by setting the pool error level
381 !
382 err_level = mpas_pool_get_error_level()
383 call mpas_pool_set_error_level(mpas_pool_silent)
384
385 nullify(tend)
386 nullify(state)
387
388 call mpas_pool_get_array(lbc, 'lbc_'//trim(field), tend, 1)
389 call mpas_pool_get_array(lbc, 'lbc_'//trim(field), state, 2)
390
391 ! Reset the pool error level to its previous value
392 call mpas_pool_set_error_level(err_level)
393
394 !
395 ! If we have both a tendency and state for this boundary field from the tend and state
396 ! pools, then the requested field was not a scalar constituent; otherwise, we need to
397 ! query the field as a scalar constituent
398 !
399 if (associated(tend) .and. associated(state)) then
400 return_state(:,:) = state(:,:) - dt * tend(:,:)
401 else
402 call mpas_pool_get_array(lbc, 'lbc_scalars', tend_scalars, 1)
403 call mpas_pool_get_array(lbc, 'lbc_scalars', state_scalars, 2)
404 call mpas_pool_get_dimension(lbc, 'index_'//trim(field), idx)
405
406 return_state(:,:) = state_scalars(idx,:,:) - dt * tend_scalars(idx,:,:)
407 end if
408
409 end function mpas_atm_get_bdy_state
410

413 !
414 ! routine mpas_atm_setup_bdy_masks
415 !
424 !
425 !-----------------------------------------------------------------------
426 subroutine mpas_atm_setup_bdy_masks(mesh, configs)
427
428 implicit none
429
430 type (mpas_pool_type), intent(inout) :: mesh
431 type (mpas_pool_type), intent(in) :: configs
432
433 integer :: icell, i, j, ii, jj
434 real (kind=rkind) :: d, dmin
435
436 integer, pointer :: ncells
437 integer, dimension(:), pointer :: bdymaskcell, bdymaskedge, bdymaskvertex
438 integer, dimension(:), pointer :: nearestrelaxationcell
439 integer, dimension(:), pointer :: nedgesoncell
440 integer, dimension(:,:), pointer :: cellsoncell
441 real (kind=rkind), dimension(:), pointer :: speczonemaskcell, speczonemaskedge, speczonemaskvertex
442 real (kind=rkind), dimension(:), pointer :: xcell, ycell, zcell
443
444 call mpas_pool_get_dimension(mesh, 'nCells', ncells)
445
446 call mpas_pool_get_array(mesh, 'bdyMaskCell', bdymaskcell)
447 call mpas_pool_get_array(mesh, 'bdyMaskEdge', bdymaskedge)
448 call mpas_pool_get_array(mesh, 'bdyMaskVertex', bdymaskvertex)
449 call mpas_pool_get_array(mesh, 'specZoneMaskCell', speczonemaskcell)
450 call mpas_pool_get_array(mesh, 'specZoneMaskEdge', speczonemaskedge)
451 call mpas_pool_get_array(mesh, 'specZoneMaskVertex', speczonemaskvertex)
452 call mpas_pool_get_array(mesh, 'nearestRelaxationCell', nearestrelaxationcell)
453 call mpas_pool_get_array(mesh, 'nEdgesOnCell', nedgesoncell)
454 call mpas_pool_get_array(mesh, 'cellsOnCell', cellsoncell)
455 call mpas_pool_get_array(mesh, 'xCell', xcell)
456 call mpas_pool_get_array(mesh, 'yCell', ycell)
457 call mpas_pool_get_array(mesh, 'zCell', zcell)
458
459 !
460 ! Setup mask identifying cells/edges/vertices in the specified zone
461 ! NB: The specZoneMask{Cell,Edge,Vertex} fields receive a default value of 0.0 from the Registry,
462 ! so no need to initialize them here
463 !
464 where (bdymaskcell(:) > nrelaxzone) speczonemaskcell(:) = 1.0_rkind
465 where (bdymaskedge(:) > nrelaxzone) speczonemaskedge(:) = 1.0_rkind
466 where (bdymaskvertex(:) > nrelaxzone) speczonemaskvertex(:) = 1.0_rkind
467
468
469 nearestrelaxationcell(:) = ncells+1
470
471 !
472 ! For nearest relaxation cell to inner specified zone, just search
473 ! all cellsOnCell with bdyMaskCell == nRelaxZone
474 !
475 do icell=1,ncells
476 if (bdymaskcell(icell) == (nrelaxzone+1)) then
477 dmin = 1.0e36
478 do j=1,nedgesoncell(icell)
479 i = cellsoncell(j,icell)
480 if (bdymaskcell(i) == nrelaxzone) then
481 d = (xcell(i) - xcell(icell))**2 + (ycell(i) - ycell(icell))**2 + (zcell(i) - zcell(icell))**2
482 if (d < dmin) then
483 dmin = d
484 nearestrelaxationcell(icell) = i
485 end if
486 end if
487 end do
488 end if
489 end do
490
491 !
492 ! For nearest relaxation cell to outer specified zone, search
493 ! all cellsOnCell of cellsOnCell
494 !
495 do icell=1,ncells
496 if (bdymaskcell(icell) == (nrelaxzone+2)) then
497 dmin = 1.0e36
498 do j=1,nedgesoncell(icell)
499 i = cellsoncell(j,icell)
500 if (bdymaskcell(i) == (nrelaxzone+1)) then
501
502 do jj=1,nedgesoncell(i)
503 ii = cellsoncell(jj,i)
504 if (bdymaskcell(ii) == nrelaxzone) then
505
506 d = (xcell(ii) - xcell(icell))**2 + (ycell(ii) - ycell(icell))**2 + (zcell(ii) - zcell(icell))**2
507 if (d < dmin) then
508 dmin = d
509 nearestrelaxationcell(icell) = ii
510 end if
511
512 end if
513 end do
514
515 end if
516 end do
517 end if
518 end do
519
520 end subroutine mpas_atm_setup_bdy_masks

541 !-----------------------------------------------------------------------
542 subroutine mpas_atm_bdy_checks(dminfo, blockList, streamManager, ierr)
543
544 use mpas_log, only : mpas_log_write
545 use mpas_kind_types, only : strkind
546 use mpas_derived_types, only : dm_info, block_type, mpas_pool_type, mpas_log_err, mpas_stream_property_record_intv, &
547 mpas_stream_mgr_noerr, mpas_stream_input
548 use mpas_stream_manager, only : mpas_stream_mgr_get_property
549 use mpas_pool_routines, only : mpas_pool_get_config, mpas_pool_get_dimension, mpas_pool_get_subpool, mpas_pool_get_array
550 use mpas_dmpar, only : mpas_dmpar_max_int
551
552 implicit none
553
554 type (dm_info), pointer :: dminfo
555 type (block_type), pointer :: blocklist
556 type (mpas_streammanager_type), pointer :: streammanager
557 integer, intent(out) :: ierr
558
559 character(len=StrKIND) :: input_interval
560 logical, pointer :: config_apply_lbcs => null()
561 integer, pointer :: ncellssolve => null()
562 type (mpas_pool_type), pointer :: meshpool => null()
563 type (block_type), pointer :: block => null()
564 integer, dimension(:), pointer :: bdymaskcell => null()
565 integer :: maxvar2d_local, maxvar2d_global
566
567
568 call mpas_pool_get_config(blocklist % configs, 'config_apply_lbcs', config_apply_lbcs)
569
570 call mpas_log_write('')
571 call mpas_log_write('Checking consistency of limited-area settings...')
572 call mpas_log_write(' - config_apply_lbcs = $l', logicargs=(/config_apply_lbcs/))
573
574 !
575 ! Check whether any elements of bdyMaskCell have non-zero values
576 !
577 maxvar2d_local = -huge(maxvar2d_local)
578 block => blocklist
579 do while (associated(block))
580 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
581 call mpas_pool_get_dimension(meshpool, 'nCellsSolve', ncellssolve)
582 call mpas_pool_get_array(meshpool, 'bdyMaskCell', bdymaskcell)
583
584 maxvar2d_local = max(maxvar2d_local, maxval(bdymaskcell(1:ncellssolve)))
585
586 block => block % next
587 end do
588
589 call mpas_dmpar_max_int(dminfo, maxvar2d_local, maxvar2d_global)
590 call mpas_log_write(' - Maximum value in bdyMaskCell = $i', intargs=(/maxvar2d_global/))
591
592 !
593 ! If there are boundary cells, config_apply_lbcs must be set to true
594 !
595 if (.not. config_apply_lbcs .and. maxvar2d_global > 0) then
596 call mpas_log_write('Boundary cells found in the bdyMaskCell field, but config_apply_lbcs = false.', &
597 messagetype=mpas_log_err)
598 call mpas_log_write('Please ensure that config_apply_lbcs = true for limited-area simulations.', &
599 messagetype=mpas_log_err)
600 ierr = 1
601 return
602 end if
603
604 !
605 ! If there are no boundary cells, config_apply_lbcs must be set to false
606 !
607 if (config_apply_lbcs .and. maxvar2d_global == 0) then
608 call mpas_log_write('config_apply_lbcs = true, but no boundary cells found in the bdyMaskCell field.', &
609 messagetype=mpas_log_err)
