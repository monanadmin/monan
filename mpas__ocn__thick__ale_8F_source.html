71!-----------------------------------------------------------------------
72 subroutine ocn_ale_thickness(meshPool, verticalMeshPool, SSH, ALE_thickness, err, newHighFreqThickness)!{{{
73
74 !-----------------------------------------------------------------
75 !
76 ! input variables
77 !
78 !-----------------------------------------------------------------
79
80 type (mpas_pool_type), intent(in) :: &
81 meshpool
82
83 type (mpas_pool_type), intent(in) :: &
84 verticalmeshpool
85
86 real (kind=rkind), dimension(:), intent(in) :: &
87 ssh
88
89 real (kind=rkind), dimension(:,:), intent(in), optional :: &
90 newhighfreqthickness
91
92 !-----------------------------------------------------------------
93 !
94 ! output variables
95 !
96 !-----------------------------------------------------------------
97
98 real (kind=rkind), dimension(:,:), intent(out) :: &
99 ale_thickness
100
101 integer, intent(out) :: err
102
103 !-----------------------------------------------------------------
104 !
105 ! local variables
106 !
107 !-----------------------------------------------------------------
108
109 integer :: icell, k, i, kmax
110 integer, pointer :: ncells, nvertlevels
111 integer, dimension(:), pointer :: maxlevelcell
112
113 real (kind=rkind) :: thicknesssum, remainder, newthickness, thicknesswithremainder
114 real (kind=rkind), dimension(:), pointer :: vertcoordmovementweights
115 real (kind=rkind), dimension(:), allocatable :: &
116 ssh_ale_thickness, &
117 prelim_ale_thickness, &
118 min_ale_thickness_down, &
119 min_ale_thickness_up
120 real (kind=rkind), dimension(:,:), pointer :: &
121 restingthickness
122
123 logical, pointer :: thicknessfilteractive
124 logical, pointer :: config_use_min_max_thickness
125 real (kind=rkind), pointer :: config_max_thickness_factor
126 real (kind=rkind), pointer :: config_min_thickness
127
128 err = 0
129
130 call mpas_pool_get_package(ocnpackages, 'thicknessFilterActive', thicknessfilteractive)
131
132 call mpas_pool_get_config(ocnconfigs, 'config_use_min_max_thickness', config_use_min_max_thickness)
133 call mpas_pool_get_config(ocnconfigs, 'config_max_thickness_factor', config_max_thickness_factor)
134 call mpas_pool_get_config(ocnconfigs, 'config_min_thickness', config_min_thickness)
135
136 call mpas_pool_get_array(meshpool, 'maxLevelCell', maxlevelcell)
137 call mpas_pool_get_array(meshpool, 'vertCoordMovementWeights', vertcoordmovementweights)
138
139 call mpas_pool_get_array(verticalmeshpool, 'restingThickness', restingthickness)
140
141 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
142 call mpas_pool_get_dimension(meshpool, 'nVertLevels', nvertlevels)
143
144 allocate(ssh_ale_thickness(nvertlevels), prelim_ale_thickness(nvertlevels), &
145 min_ale_thickness_down(nvertlevels), min_ale_thickness_up(nvertlevels))
146
147 !
148 ! ALE thickness alteration due to SSH (z-star)
149 !
150 !$omp do schedule(runtime) private(kMax, thicknessSum, k)
151 do icell = 1, ncells
152 kmax = maxlevelcell(icell)
153
154 thicknesssum = 1e-14_rkind
155 do k = 1, kmax
156 ssh_ale_thickness(k) = ssh(icell) * vertcoordmovementweights(k) * restingthickness(k, icell)
157 thicknesssum = thicknesssum + vertcoordmovementweights(k) * restingthickness(k, icell)
158 end do
159
160 ! Note that restingThickness is nonzero, and remaining terms are perturbations about zero.
161 do k = 1, kmax
162 ssh_ale_thickness(k) = ssh_ale_thickness(k) / thicknesssum
163 ale_thickness(k, icell) = restingthickness(k, icell) + ssh_ale_thickness(k)
164 end do
165 enddo
166 !$omp end do
167
168 if (thicknessfilteractive) then
169 !$omp do schedule(runtime) private(kMax)
170 do icell = 1, ncells
171 kmax = maxlevelcell(icell)
172
173 ale_thickness(1:kmax, icell) = &
174 ale_thickness(1:kmax, icell) &
175 + newhighfreqthickness(1:kmax,icell)
176 enddo
177 !$omp end do
178 end if
179
180 !
181 ! ALE thickness alteration due to minimum and maximum thickness
182 !
183 if (config_use_min_max_thickness) then
184
185 !$omp do schedule(runtime) private(kMax, remainder, k, newThickness)
186 do icell = 1, ncells
187 kmax = maxlevelcell(icell)
188
189 ! go down the column:
190 prelim_ale_thickness(1:kmax) = ale_thickness(1:kmax, icell)
191 remainder = 0.0_rkind
192 do k = 1, kmax
193 newthickness = max( min(prelim_ale_thickness(k) + remainder, &
194 config_max_thickness_factor * restingthickness(k,icell) ), &
195 config_min_thickness)
196 min_ale_thickness_down(k) = newthickness - prelim_ale_thickness(k)
197 remainder = remainder - min_ale_thickness_down(k)
198 end do
199
200 ! go back up the column:
201 min_ale_thickness_up(kmax) = 0.0_rkind
202 prelim_ale_thickness(1:kmax) = prelim_ale_thickness(1:kmax) + min_ale_thickness_down(1:kmax)
203 do k = kmax-1, 1, -1
204 newthickness = max( min(prelim_ale_thickness(k) + remainder, &
205 config_max_thickness_factor * restingthickness(k,icell) ), &
206 config_min_thickness)
207 min_ale_thickness_up(k) = newthickness - prelim_ale_thickness(k)
208 remainder = remainder - min_ale_thickness_up(k)
209 end do
210 min_ale_thickness_up(1) = min_ale_thickness_up(1) + remainder
211
212 ale_thickness(1:kmax, icell) = ale_thickness(1:kmax, icell) + min_ale_thickness_down(1:kmax) &
213 + min_ale_thickness_up(1:kmax)
214
215 enddo
216 !$omp end do
217
218 endif ! config_use_min_max_thickness
219
220 deallocate(ssh_ale_thickness, prelim_ale_thickness, min_ale_thickness_down, min_ale_thickness_up)
221
222 end subroutine ocn_ale_thickness!}}}
