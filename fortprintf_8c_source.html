26int fortprintf(FILE * fd, char * str, ...)/*{{{*/
27{
28 int i, nl, sp, sp_inquotes, inquotes, q;
29 int lastchar;
30 int errorcode;
31 va_list ap;
32
33#ifdef FORTPRINTF_DEBUG
34 printf("call to fortprintf\n");
35 printf("Format string is: %s\n", str);
36 printf("\n");
37#endif
38
39 /* Assume no errors */
40 errorcode = 0;
41
42 /* Add formatted string to the buffer of fortran code to be written */
43 va_start(ap, str);
44 i = vsnprintf(fbuffer+nbuf, 1024-nbuf, str, ap);
45#ifdef FORTPRINTF_DEBUG
46 printf("Full buffer is:\n");
47 printf("%s\n", fbuffer);
48#endif
49 va_end(ap);
50
51 /* Set the next free position in the fortran buffer */
52 nbuf = nbuf + i;
53
54 inquotes = 0;
55 q = -1;
56
57 do {
58
59 nl = sp = -1;
60
61 /* Scan through the max line length - 1 (since we may have to add an & character) or the end of the buffer, whichever comes first */
62 for (i=0; i<MAX_LINE_LEN-1 && i<nbuf; i++) {
63 lastchar = (i == nbuf-1) ? 1 : 0;
64 if (fbuffer[i] == '\'') {
65 if ( fbuffer[i+1] != '\'' || lastchar ) {
66 inquotes = (inquotes + 1) % 2;
67 q = inquotes ? i : -1;
68 } else if ( !lastchar && fbuffer[i+1] == '\'' ) {
69 i++;
70 }
71 }
72 if (fbuffer[i] == '\n') nl = i; /* The last occurrence of a newline */
73 if (fbuffer[i] == ' ' && !lastchar && fbuffer[i+1] != '&') { /* The last occurrence of a space */
74 sp = i;
75 sp_inquotes = inquotes;
76 }
77 }
78
79#ifdef FORTPRINTF_DEBUG
80 printf("1: i = %d, nl = %d, sp = %d, fbuffer[i] = %c, inquotes = %d\n", i, nl, sp, fbuffer[i], inquotes);
81#endif
82
83 /* If we haven't reached the column limit, don't consider breaking the line yet */
84 if (nbuf <= MAX_LINE_LEN) sp = -1;
85
86#ifdef FORTPRINTF_DEBUG
87 printf("2: i = %d, nl = %d, sp = %d, fbuffer[i] = %c\n", i, nl, sp, fbuffer[i]);
88#endif
89
90 /* If the charater at column MAX_LINE_LEN happens to be a newline or the only space, though, we mark it */
91 if (i == MAX_LINE_LEN-1) {
92 if (fbuffer[i] == '\n') nl = i;
93 if (fbuffer[i] == ' ' && sp < 0) sp = i-1; /* Insert & before the space so it will fit within column limit */
94 }
95
96#ifdef FORTPRINTF_DEBUG
97 printf("3: i = %d, nl = %d, sp = %d, fbuffer[i] = %c\n", i, nl, sp, fbuffer[i]);
98#endif
99
100 /* If we have a newline */
101 if (nl >= 0) {
102 snprintf(printbuf, nl+2, "%s", fbuffer);
103 fprintf(fd, "%s", printbuf);
104 nl++;
105
106 /* Shift unprinted contents of fortran buffer to the beginning */
107 for (i=0; nl<nbuf; i++, nl++)
108 fbuffer[i] = fbuffer[nl];
109 nbuf = i;
110 }
111 /* Else if we found a place to break the line */
112 else if (sp >= 0) {
113 snprintf(printbuf, sp+2, "%s", fbuffer);
114 i = sp+1;
115 if (sp_inquotes && (sp > q)) printbuf[i++] = '\'';
116 printbuf[i++] = '&';
117 printbuf[i++] = '\n';
118 printbuf[i++] = '\0';
119 fprintf(fd, "%s", printbuf);
120 sp++;
121 i = 0;
122 if (sp_inquotes && (sp > q)) {
123 inquotes = (inquotes + 1) % 2;
124 fbuffer[i++] = '/';
125 fbuffer[i++] = '/';
126 fbuffer[i++] = '\'';
127 }
128
129 /* Shift unprinted contents of fortran buffer to the beginning */
130 for ( ; sp<nbuf; i++, sp++)
131 fbuffer[i] = fbuffer[sp];
132 nbuf = i;
133 }
134 else if (nbuf > MAX_LINE_LEN) {
135 fprintf(fd, "!!! Error: Output could not be formatted by fortprintf() !!!\n");
136 nbuf = 0;
137 errorcode = 1;
138 }
139
140 } while (nl >= 0 || sp >= 0);
141
142 return errorcode;
143}/*}}}*/
