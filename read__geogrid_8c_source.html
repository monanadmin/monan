46int read_geogrid(
47 char * fname, /* The name of the file to read from */
48 float * rarray, /* The array to be filled */
49 int nx, /* x-dimension of the array */
50 int ny, /* y-dimension of the array */
51 int nz, /* z-dimension of the array */
52 int isigned, /* 0=unsigned data, 1=signed data */
53 int endian, /* 0=big endian, 1=little endian */
54 float scalefactor, /* value to multiply array elements by before truncation to integers */
55 int wordsize, /* number of bytes to use for each array element */
56 int * status)
57{
58 int i, ival, cnt, narray;
59 int A2, B2;
60 int A3, B3, C3;
61 int A4, B4, C4, D4;
62 unsigned char * c;
63 FILE * bfile;
64
65 *status = 0;
66
67 narray = (nx) * (ny) * (nz);
68
69 /* Attempt to open file for reading */
70 if (!(bfile = fopen(fname,"rb")))
71 {
72 *status = 1;
73 return 1;
74 }
75
76 /* Allocate memory to hold bytes from file and read data */
77 c = (unsigned char *)malloc(sizeof(unsigned char)* wordsize * narray);
78 cnt = fread((void *)c, sizeof(unsigned char), narray * wordsize, bfile);
79
80 fclose(bfile);
81
82 if (cnt == 0)
83 {
84 *status = 1;
85 return 1;
86 }
87
88 /*
89 Set up byte offsets for each wordsize depending on byte order.
90 A, B, C, D give the offsets of the LSB through MSB (i.e., for
91 word ABCD, A=MSB, D=LSB) in the array from the beginning of a word
92 */
93 if (endian == GEOG_BIG_ENDIAN) {
94 A2 = 0; B2 = 1;
95 A3 = 0; B3 = 1; C3 = 2;
96 A4 = 0; B4 = 1; C4 = 2; D4 = 3;
97 }
98 else {
99 B2 = 0; A2 = 1;
100 C3 = 0; B3 = 1; A3 = 2;
101 D4 = 0; C4 = 1; B4 = 2; A4 = 3;
102 }
103
104 /* Convert words from native byte order */
105 switch(wordsize) {
106 case 1:
107 for(i=0; i<narray; i++)
108 {
109 ival = (int)(c[i]);
110 if ((isigned) && (ival > (1 << 7))) ival -= (1 << 8);
111 rarray[i] = (float)ival;
112 }
113 break;
114
115 case 2:
116 for(i=0; i<narray; i++)
117 {
118 ival = (int)((c[2*i+A2]<<8) | (c[2*i+B2]));
119 if ((isigned) && (ival > (1 << 15))) ival -= (1 << 16);
120 rarray[i] = (float)ival;
121 }
122 break;
123
124 case 3:
125 for(i=0; i<narray; i++)
126 {
127 ival = (int)((c[3*i+A3]<<16) | (c[3*i+B3]<<8) | c[3*i+C3]);
128 if ((isigned) * (ival > (1 << 23))) ival -= (1 << 24);
129 rarray[i] = (float)ival;
130 }
131 break;
132
133 case 4:
134 for(i=0; i<narray; i++)
135 {
136 ival = (int)((c[4*i+A4]<<24) | (c[4*i+B4]<<16) | (c[4*i+C4]<<8) | c[4*i+D4]);
137 if ((isigned) && (ival > (1 << 31))) ival = -(~ival + 1);
138 rarray[i] = (float)ival;
139 }
140 break;
141 }
142
143 free(c);
144
145 /* Scale real-valued array by scalefactor */
146 if (scalefactor != 1.0)
147 {
148 for (i=0; i<narray; i++)
149 rarray[i] = rarray[i] * (scalefactor);
150 }
151
152 return 0;
153}
