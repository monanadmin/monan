37def xtime2numtime(xtime):
38 """Define a function to convert xtime character array to numeric time values using datetime objects"""
39 # First parse the xtime character array into a string
40 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
41
42 dt = []
43 for stritem in xtimestr:
44 itemarray = stritem.strip().replace('_', '-').replace(':', '-').split('-') # Get an array of strings that are Y,M,D,h,m,s
45 results = [int(i) for i in itemarray]
46 if (results[0] < 1900): # datetime has a bug where years less than 1900 are invalid on some systems
47 results[0] += 1900
48 dt.append( datetime.datetime(*results) ) # * notation passes in the array as arguments
49
50 numtime = netCDF4.date2num(dt, units='seconds since '+str(dt[0])) # use the netCDF4 module's function for converting a datetime to a time number
51 return numtime

53def xtimeGetYear(xtime):
54 """Get an array of years from an xtime array, ignoring any partial year information"""
55 # First parse the xtime character array into a string
56 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
57 years = np.zeros( (len(xtimestr),) )
58 for i in range(len(xtimestr)):
59 years[i] = ( int(xtimestr[i].split('-')[0]) ) # Get the year part and make it an integer
60 return years
