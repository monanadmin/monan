35! subroutine camradinit calls the main subroutines needed to initialize the long- and short-wave
36! CAM radiation codes, and read input data from auxillary files.
37!
38! subroutines called in mpas_atmphys_camrad_init:
39! -----------------------------------------------
40! radini :initialization of radiation constants.
41! esinti :initialization of saturation vapor pressures.
42! oznini :initialization of climatological monthly-mean ozone profiles.
43! aerosol_init:initialization of aerosol optical properties.

45! add-ons and modifications to sourcecode:
46! ----------------------------------------
47! * added initialization of variable mxaerl which is the number of layers below 900 hPa in
48! which background aerosols are present. mxaerl is computed using the pressure-base array.
49! -> added diag in the argument list of subroutines camradinit and aerosol_init.
50! -> in subroutine aerosol_init, added initialization of variable mxaerl.
51! Laura D. Fowler (birch.ucar.edu) / 2013-07-01.
52! * moved the arrays pin and ozmixm from the mesh structure to the atm_input structure in
53! subroutine oznini.
54! Laura D. Fowler (birch.ucar.edu) / 2013-07-08.
55! * Replaced the variable g (that originally pointed to gravity) with gravity, for simplicity.
56! Laura D. Fowler (laura@ucar.edu) / 2014-03-21.

154!=================================================================================================================
155 subroutine radini(dminfo,gravx,cpairx,epsilox,stebolx,pstdx)
156!=================================================================================================================
157!
158! Purpose:
159! Initialize various constants for radiation scheme; note that
160! the radiation scheme uses cgs units.
161!
162! Method:
163! <Describe the algorithm(s) used in the routine.>
164! <Also include any applicable external references.>
165!
166! Author: W. Collins (H2O parameterization) and J. Kiehl
167!
168!-----------------------------------------------------------------------
169! use shr_kind_mod, only: r8 => shr_kind_r8
170! use ppgrid, only: pver, pverp
171! use comozp, only: cplos, cplol
172! use pmgrid, only: masterproc, plev, plevp
173! use radae, only: radaeini
174! use physconst, only: mwdry, mwco2
175#if ( defined SPMD )
176! use mpishorthand
177#endif
178 implicit none
179
180!------------------------------Arguments--------------------------------
181!
182! Input arguments
183!
184 type(dm_info),intent(in):: dminfo
185
186 real, intent(in) :: gravx ! Acceleration of gravity (MKS)
187 real, intent(in) :: cpairx ! Specific heat of dry air (MKS)
188 real, intent(in) :: epsilox ! Ratio of mol. wght of H2O to dry air
189 real, intent(in) :: stebolx ! Stefan-Boltzmann's constant (MKS)
190 real(r8), intent(in) :: pstdx ! Standard pressure (Pascals)
191!
192!---------------------------Local variables-----------------------------
193!
194 integer k ! Loop variable
195
196 real(r8) v0 ! Volume of a gas at stp (m**3/kmol)
197 real(r8) p0 ! Standard pressure (pascals)
198 real(r8) amd ! Effective molecular weight of dry air (kg/kmol)
199 real(r8) goz ! Acceleration of gravity (m/s**2)
200!
201!-----------------------------------------------------------------------
202!
203! Set general radiation consts; convert to cgs units where appropriate:
204!
205 gravit = 100.*gravx
206 rga = 1./gravit
207 gravmks = gravx
208 cpair = 1.e4*cpairx
209 epsilo = epsilox
210 sslp = 1.013250e6
211 stebol = 1.e3*stebolx
212 rgsslp = 0.5/(gravit*sslp)
213 dpfo3 = 2.5e-3
214 dpfco2 = 5.0e-3
215 dayspy = 365.
216 pie = 4.*atan(1.)
217!
218! Initialize ozone data.
219!
220 v0 = 22.4136 ! Volume of a gas at stp (m**3/kmol)
221 p0 = 0.1*sslp ! Standard pressure (pascals)
222 amd = 28.9644 ! Molecular weight of dry air (kg/kmol)
223 goz = gravx ! Acceleration of gravity (m/s**2)
224!
225! Constants for ozone path integrals (multiplication by 100 for unit
226! conversion to cgs from mks):
227!
228 cplos = v0/(amd*goz) *100.0
229 cplol = v0/(amd*goz*p0)*0.5*100.0
230!
231! Derived constants
232! If the top model level is above ~90 km (0.1 Pa), set the top level to compute
233! longwave cooling to about 80 km (1 Pa)
234! WRF: assume top level > 0.1 mb
235! if (hypm(1) .lt. 0.1) then
236! do k = 1, pver
237! if (hypm(k) .lt. 1.) ntoplw = k
238! end do
239! else
240 ntoplw = 1
241! end if
242! if (masterproc) then
243! call mpas_log_write('RADINI: ntoplw = $i pressure: $r', intArgs=(/ntoplw/), realArgs=(/hypm(ntoplw)/))
244! endif
245
246 call radaeini(dminfo,pstdx,mwdry,mwco2)
247
248 end subroutine radini

250!=================================================================================================================
251 subroutine radaeini(dminfo,pstdx,mwdryx,mwco2x)
252!=================================================================================================================
253
254!input arguments:
255 type(dm_info),intent(in):: dminfo
256
257 real(r8), intent(in) :: pstdx ! Standard pressure (dynes/cm^2)
258 real(r8), intent(in) :: mwdryx ! Molecular weight of dry air
259 real(r8), intent(in) :: mwco2x ! Molecular weight of carbon dioxide
260
261!local variables:
262
263!variables for loading absorptivity/emissivity:
264 integer:: ncid_ae ! NetCDF file id for abs/ems file
265 integer:: pdimid ! pressure dimension id
266 integer:: psize ! pressure dimension size
267 integer:: tpdimid ! path temperature dimension id
268 integer:: tpsize ! path temperature size
269 integer:: tedimid ! emission temperature dimension id
270 integer:: tesize ! emission temperature size
271
272 integer:: udimid ! u (H2O path) dimension id
273 integer:: usize ! u (H2O path) dimension size
274
275 integer:: rhdimid ! relative humidity dimension id
276 integer:: rhsize ! relative humidity dimension size
277
278 integer:: ah2onwid ! var. id for non-wndw abs.
279 integer:: eh2onwid ! var. id for non-wndw ems.
280 integer:: ah2owid ! var. id for wndw abs. (adjacent layers)
281 integer:: cn_ah2owid ! var. id for continuum trans. for wndw abs.
282 integer:: cn_eh2owid ! var. id for continuum trans. for wndw ems.
283 integer:: ln_ah2owid ! var. id for line trans. for wndw abs.
284 integer:: ln_eh2owid ! var. id for line trans. for wndw ems.
285
286!character*(NF_MAX_NAME) tmpname! dummy variable for var/dim names
287 character(len=StrKIND):: locfn ! local filename
288 integer:: tmptype ! dummy variable for variable type
289 integer:: ndims ! number of dimensions
290!integer dims(NF_MAX_VAR_DIMS) ! vector of dimension ids
291 integer:: natt ! number of attributes
292
293!Variables for setting up H2O table:
294 integer:: t ! path temperature
295 integer:: tmin ! mininum path temperature
296 integer:: tmax ! maximum path temperature
297 integer:: itype ! type of sat. pressure (=0 -> H2O only)
298 real(r8):: tdbl
299
300 integer:: i,istat,cam_abs_unit
301 logical:: opened
302 character(len=StrKIND):: errmess
303
304 integer:: i_te,i_rh
305
306!-----------------------------------------------------------------------------------------------------------------
307
308!... constants to set:
309 p0 = pstdx
310 amd = mwdryx
311 amco2 = mwco2x
312
313!... coefficients for h2o emissivity and absorptivity for overlap of H2O and trace gases:
314 c16 = coefj(3,1)/coefj(2,1)
315 c17 = coefk(3,1)/coefk(2,1)
316 c26 = coefj(3,2)/coefj(2,2)
317 c27 = coefk(3,2)/coefk(2,2)
318 c28 = .5
319 c29 = .002053
320 c30 = .1
321 c31 = 3.0e-5
322
323!... initialize further longwave constants referring to far wing correction for overlap of H2O
324! and trace gases; R&D refers to:
325! Ramanathan,V., and P.Downey, 1986:A Nonisothermal Emissivity and Absorptivity Formulation
326! for Water Vapor Journal of Geophysical Research, vol. 91., D8, pp 8649-8666.
327
328 fwcoef = .1 ! See eq(33) R&D
329 fwc1 = .30 ! See eq(33) R&D
330 fwc2 = 4.5 ! See eq(33) and eq(34) in R&D
331 fc1 = 2.6 ! See eq(34) R&D
332
333 istat = -999
334 if(dminfo % my_proc_id == io_node) then
335 do i = 10,99
336 inquire(i,opened = opened,iostat=istat)
337 if(.not. opened) then
338 cam_abs_unit = i
339 exit
340 endif
341 enddo
342 if(istat /= 0) &
343 call physics_error_fatal('module_ra_cam: radaeinit: Cannot find unused '//&
344 'fortran unit to read in lookup table.')
345 endif
346
347!distribute unit to other processors:
348 call mpas_dmpar_bcast_int(dminfo,cam_abs_unit)
349
350!open init file:
351 if(dminfo % my_proc_id == io_node) then
352
353 open(cam_abs_unit,file='CAM_ABS_DATA.DBL',form='UNFORMATTED',status='OLD',iostat=istat)
354 if(istat /= 0) then
355 write(errmess,'(A,I4)') 'module_ra_cam: error reading CAM_ABS_DATA on unit', &
356 cam_abs_unit
357 call physics_error_fatal(errmess)
358 endif
359
360 read(cam_abs_unit,iostat=istat) ah2onw
361 read(cam_abs_unit,iostat=istat) eh2onw
362 read(cam_abs_unit,iostat=istat) ah2ow
363 read(cam_abs_unit,iostat=istat) cn_ah2ow
364 read(cam_abs_unit,iostat=istat) cn_eh2ow
365 read(cam_abs_unit,iostat=istat) ln_ah2ow
366 read(cam_abs_unit,iostat=istat) ln_eh2ow
367
368 endif
369
370 dm_bcast_macro(ah2onw)
371 dm_bcast_macro(eh2onw)
372 dm_bcast_macro(ah2ow)
373 dm_bcast_macro(cn_ah2ow)
374 dm_bcast_macro(cn_eh2ow)
375 dm_bcast_macro(ln_ah2ow)
376 dm_bcast_macro(ln_eh2ow)
377
378 if(dminfo % my_proc_id == io_node) close(cam_abs_unit)
379
380! Set up table of H2O saturation vapor pressures for use in calculation effective path RH.
381! Need separate table from table in wv_saturation because:
382! (1. Path temperatures can fall below minimum of that table; and
383! (2. Abs/Emissivity tables are derived with RH for water only.
384
385 tmin = nint(min_tp_h2o)
386 tmax = nint(max_tp_h2o)+1
387 itype = 0
388 do t = tmin, tmax
389! call gffgch(dble(t),estblh2o(t-tmin),itype)
390 tdbl = t
391 call gffgch(tdbl,estblh2o(t-tmin),itype)
392 enddo
393
394 end subroutine radaeini

396!=================================================================================================================
397 subroutine aerosol_init(dminfo,mesh,diag,diag_physics,state,time_lev)
398!=================================================================================================================
399
400!This subroutine assumes a uniform aerosol distribution in both time and space. It should be
401!modified if aerosol data are available from WRF-CHEM or other sources.
402
403!input arguments:!
404 type(dm_info),intent(in) :: dminfo
405 type(mpas_pool_type),intent(in):: mesh
406 type(mpas_pool_type),intent(in):: diag
407 type(mpas_pool_type),intent(in):: diag_physics
408
409 integer,intent(in):: time_lev
410
411!inout arguments:
412 type(mpas_pool_type),intent(inout):: state
413
414!local pointers:
415 integer, pointer:: naerlevels,ncells,ncellssolve,nvertlevels
416 real(kind=rkind),dimension(:),pointer:: m_psp,m_psn
417 real(kind=rkind),dimension(:,:),pointer:: m_hybi
418 real(kind=rkind),dimension(:,:),pointer:: pressure_b
419 real(kind=rkind),dimension(:,:,:),pointer:: aerosolcn,aerosolcp
420
421 integer,pointer:: idxsul_l,idxsslt_l,idxdustfirst_l,idxocpho_l,idxcarbonfirst_l, &
422 idxbcpho_l,idxocphi_l,idxbcphi_l,idxbg_l,idxvolc_l
423
424!local variables:
425 integer:: max_mxaerl
426 integer,dimension(:),allocatable:: mxaerl_local
427 integer:: icell,k,kk
428
429 real(kind=rkind):: psurf
430
431 real(kind=rkind),dimension(29) :: hybi
432 data hybi/0, 0.0065700002014637, 0.0138600002974272, 0.023089999333024 , &
433 0.0346900001168251, 0.0491999983787537, 0.0672300010919571, &
434 0.0894500017166138, 0.116539999842644 , 0.149159997701645 , &
435 0.187830001115799 , 0.232859998941422 , 0.284209996461868 , &
436 0.341369986534119 , 0.403340011835098 , 0.468600004911423 , &
437 0.535290002822876 , 0.601350009441376 , 0.66482001543045 , &
438 0.724009990692139 , 0.777729988098145 , 0.825269997119904 , &
439 0.866419970989227 , 0.901350021362305 , 0.930540025234222 , &
440 0.954590022563934 , 0.974179983139038 , 0.990000009536743 , 1/

442!-----------------------------------------------------------------------------------------------------------------
443
444!initialization:
445 call mpas_pool_get_dimension(mesh,'nCells',ncells)
446 call mpas_pool_get_dimension(mesh,'nCellsSolve',ncellssolve)
447 call mpas_pool_get_dimension(mesh,'nAerLevels',naerlevels)
448 call mpas_pool_get_dimension(mesh,'nVertLevels',nvertlevels)
449
450 call mpas_pool_get_array(diag_physics,'m_hybi',m_hybi)
451 call mpas_pool_get_array(diag_physics,'m_ps',m_psp)
452 call mpas_pool_get_array(diag_physics,'m_ps',m_psn)
453
454 call mpas_pool_get_array(diag,'pressure_base',pressure_b)
455
456 call mpas_pool_get_array(state,'aerosols',aerosolcp,1)
457 call mpas_pool_get_array(state,'aerosols',aerosolcn,2)
458
459!initialization of aerosol levels:
460 do k = 1, naerlevels
461 do icell = 1, ncells
462 m_hybi(k,icell) = hybi(k)
463 enddo
464 enddo
465
466 psurf = 1.e05
467 do icell = 1, ncells
468 m_psp(icell) = psurf
469! m_psn(iCell) = psurf ! m_psp and m_psn both point to the same array for m_ps, so we only
470 ! need to initialize one of them
471 enddo
472
473!mxaerl = max number of levels (from bottom) for background aerosol. Limit background aerosol
474!height below 900 mb:
475 if(.not.allocated(mxaerl_local)) allocate(mxaerl_local(1:ncellssolve))
476 mxaerl_local(1:ncellssolve) = 0
477 do k = 1, nvertlevels
478 do icell = 1, ncellssolve
479 if(pressure_b(k,icell) .ge. 9.e04) mxaerl_local(icell) = mxaerl_local(icell) + 1
480 enddo
481 enddo
482
483 max_mxaerl = maxval(mxaerl_local(1:ncellssolve))
484 call mpas_dmpar_max_int(dminfo,max_mxaerl,mxaerl)
485 call mpas_log_write('--- AEROSOLS:mxaerl on node = $i', intargs=(/max_mxaerl/))
486 call mpas_log_write(' AEROSOLS:Background aerosol limited to bottom $i model interfaces', intargs=(/mxaerl/))
487 if(allocated(mxaerl_local)) deallocate(mxaerl_local)
488
489!initialize indices for water species:
490 ozncyc = .true.
491 indirect = .true.
492 ixcldliq = 2
493 ixcldice = 3
494
495!initialize indices for aerosol species:
496 call mpas_pool_get_dimension(state,'index_sul' ,idxsul_l )
497 call mpas_pool_get_dimension(state,'index_sslt' ,idxsslt_l )
498 call mpas_pool_get_dimension(state,'index_dust1',idxdustfirst_l )
499 call mpas_pool_get_dimension(state,'index_ocpho',idxocpho_l )
500 call mpas_pool_get_dimension(state,'index_ocpho',idxcarbonfirst_l)
501 call mpas_pool_get_dimension(state,'index_bcpho',idxbcpho_l )
502 call mpas_pool_get_dimension(state,'index_ocphi',idxocphi_l )
503 call mpas_pool_get_dimension(state,'index_bcphi',idxbcphi_l )
504 call mpas_pool_get_dimension(state,'index_bg' ,idxbg_l )
505 call mpas_pool_get_dimension(state,'index_volc' ,idxvolc_l )
506
507 idxsul = idxsul_l
508 idxsslt = idxsslt_l
509 idxdustfirst = idxdustfirst_l
510 idxocpho = idxocpho_l
511 idxcarbonfirst = idxcarbonfirst_l
512 idxbcpho = idxbcpho_l
513 idxocphi = idxocphi_l
514 idxbcphi = idxbcphi_l
515 idxbg = idxbg_l
516 idxvolc = idxvolc_l
517
518 call mpas_log_write(' idxSUL = $i', intargs=(/idxsul/))
519 call mpas_log_write(' idxSSLT = $i', intargs=(/idxsslt/))
520 call mpas_log_write(' idxDUSTfirst = $i', intargs=(/idxdustfirst/))
521 call mpas_log_write(' idxOCPHO = $i', intargs=(/idxocpho/))
522 call mpas_log_write(' idxCARBONfirst = $i', intargs=(/idxcarbonfirst/))
523 call mpas_log_write(' idxBCPHO = $i', intargs=(/idxbcpho/))
524 call mpas_log_write(' idxOCPHI = $i', intargs=(/idxocphi/))
525 call mpas_log_write(' idxBCPHI = $i', intargs=(/idxbcphi/))
526 call mpas_log_write(' idxBG = $i', intargs=(/idxbg/))
527 call mpas_log_write(' idxVOLC = $i', intargs=(/idxvolc/))
528
529 do icell = 1, ncells
530 do k = 1, naerlevels
531!aerosolc arrays are upward cumulative (kg/m2) at each level. Here we assume uniform vertical
532!distribution (aerosolc linear with hybi)
533 aerosolcp(idxsul,k,icell)=1.e-7*(1.-hybi(k))
534 aerosolcn(idxsul,k,icell)=1.e-7*(1.-hybi(k))
535 aerosolcp(idxsslt,k,icell)=1.e-22*(1.-hybi(k))
536 aerosolcn(idxsslt,k,icell)=1.e-22*(1.-hybi(k))
537 aerosolcp(idxdustfirst,k,icell)=1.e-7*(1.-hybi(k))
538 aerosolcn(idxdustfirst,k,icell)=1.e-7*(1.-hybi(k))
539 aerosolcp(idxdustfirst+1,k,icell)=1.e-7*(1.-hybi(k))

560!=================================================================================================================
561 subroutine aer_optics_initialize(dminfo)
562!=================================================================================================================
563
564!input arguments:
565 type(dm_info):: dminfo
566
567!local variables:
568 integer:: nrh_opac ! number of relative humidity values for OPAC data
569 integer:: nbnd ! number of spectral bands, should be identical to nspint
570 integer:: krh_opac ! rh index for OPAC rh grid
571 integer:: krh ! another rh index
572 integer:: ksz ! dust size bin index
573 integer:: kbnd ! band index
574
575 real(r8), parameter :: wgt_sscm = 6.0 / 7.0
576 real(r8):: rh ! local relative humidity variable
577
578 integer, parameter :: irh=8
579 real(r8):: rh_opac(irh) ! OPAC relative humidity grid
580 real(r8):: ksul_opac(irh,nspint) ! sulfate extinction
581 real(r8):: wsul_opac(irh,nspint) ! single scattering albedo
582 real(r8):: gsul_opac(irh,nspint) ! asymmetry parameter
583 real(r8):: ksslt_opac(irh,nspint) ! sea-salt
584 real(r8):: wsslt_opac(irh,nspint)
585 real(r8):: gsslt_opac(irh,nspint)
586 real(r8):: kssam_opac(irh,nspint) ! sea-salt accumulation mode
587 real(r8):: wssam_opac(irh,nspint)
588 real(r8):: gssam_opac(irh,nspint)
589 real(r8):: ksscm_opac(irh,nspint) ! sea-salt coarse mode
590 real(r8):: wsscm_opac(irh,nspint)
591 real(r8):: gsscm_opac(irh,nspint)
592 real(r8):: kcphil_opac(irh,nspint) ! hydrophilic organic carbon
593 real(r8):: wcphil_opac(irh,nspint)
594 real(r8):: gcphil_opac(irh,nspint)
595 real(r8):: dummy(nspint)
596
597 integer:: i,istat,cam_aer_unit
598 logical:: opened
599 character(len=StrKIND):: errmess
600
601!-----------------------------------------------------------------------------------------------------------------
602
603!call mpas_log_write('--- enter subroutine aer_optics_initialize:')
604
605!READ AEROSOL OPTICS DATA:
606 istat = -999
607 if(dminfo % my_proc_id == io_node) then
608 do i = 10,99
609 inquire(i,opened = opened,iostat=istat)
610 if(.not. opened) then
611 cam_aer_unit = i
612 exit
613 endif
614 enddo
615 if(istat /= 0) &
616 call physics_error_fatal('module_ra_cam: aer_optics_initialize: Cannot find unused '//&
617 'fortran unit to read in lookup table.')
618 endif
619
620!distribute unit to other processors:
621 call mpas_dmpar_bcast_int(dminfo,cam_aer_unit)
622
623!open init file:
624 if(dminfo % my_proc_id == io_node) then
625
626 open(cam_aer_unit,file='CAM_AEROPT_DATA.DBL',form='UNFORMATTED',status='OLD',iostat=istat)
627 if(istat /= 0) then
628 write(errmess,'(A,I4)') 'module_ra_cam: error reading CAM_AEROPT_DATA on unit', &
629 cam_aer_unit
630 call physics_error_fatal(errmess)
631 endif
632
633 read(cam_aer_unit,iostat=istat) dummy
634 read(cam_aer_unit,iostat=istat) rh_opac
635 read(cam_aer_unit,iostat=istat) ksul_opac
636 read(cam_aer_unit,iostat=istat) wsul_opac
637 read(cam_aer_unit,iostat=istat) gsul_opac
638 read(cam_aer_unit,iostat=istat) kssam_opac
639 read(cam_aer_unit,iostat=istat) wssam_opac
640 read(cam_aer_unit,iostat=istat) gssam_opac
641 read(cam_aer_unit,iostat=istat) ksscm_opac
642 read(cam_aer_unit,iostat=istat) wsscm_opac
643 read(cam_aer_unit,iostat=istat) gsscm_opac

647 read(cam_aer_unit,iostat=istat) kcb
648 read(cam_aer_unit,iostat=istat) wcb
649 read(cam_aer_unit,iostat=istat) gcb
650 read(cam_aer_unit,iostat=istat) kdst

655 read(cam_aer_unit,iostat=istat) gbg

657 read(cam_aer_unit,iostat=istat) wvolc
658 read(cam_aer_unit,iostat=istat) gvolc

660 endif

662 dm_bcast_macro(rh_opac)
663 dm_bcast_macro(ksul_opac)
664 dm_bcast_macro(wsul_opac)
665 dm_bcast_macro(gsul_opac)
666 dm_bcast_macro(kssam_opac)
667 dm_bcast_macro(wssam_opac)
668 dm_bcast_macro(gssam_opac)

670 dm_bcast_macro(wsscm_opac)
671 dm_bcast_macro(gsscm_opac)
672 dm_bcast_macro(kcphil_opac)

674 dm_bcast_macro(gcphil_opac)

680 dm_bcast_macro(gdst)

685 dm_bcast_macro(wvolc)
686 dm_bcast_macro(gvolc)

688 if(dminfo % my_proc_id == io_node) close(cam_aer_unit)
689
690! map OPAC aerosol species onto CAM aerosol species
691! CAM name OPAC name
692! sul or SO4 = suso sulfate soluble
693! sslt or SSLT = 1/7 ssam + 6/7 sscm sea-salt accumulation/coagulation mode
694! cphil or CPHI = waso water soluble (carbon)
695! cphob or CPHO = waso @ rh = 0
696! cb or BCPHI/BCPHO = soot
697
698 ksslt_opac(:,:) = (1.0 - wgt_sscm) * kssam_opac(:,:) + wgt_sscm * ksscm_opac(:,:)
699
700 wsslt_opac(:,:) = ( (1.0 - wgt_sscm) * kssam_opac(:,:) * wssam_opac(:,:) &
701 + wgt_sscm * ksscm_opac(:,:) * wsscm_opac(:,:) ) &
702 / ksslt_opac(:,:)
703
704 gsslt_opac(:,:) = ( (1.0 - wgt_sscm) * kssam_opac(:,:) * wssam_opac(:,:) * gssam_opac(:,:) &
705 + wgt_sscm * ksscm_opac(:,:) * wsscm_opac(:,:) * gsscm_opac(:,:) ) &
706 / ( ksslt_opac(:,:) * wsslt_opac(:,:) )
707
708 do i = 1, nspint
709 kcphob(i) = kcphil_opac(1,i)
710 wcphob(i) = wcphil_opac(1,i)
711 gcphob(i) = gcphil_opac(1,i)
712 end do
713
714!interpolate optical properties of hygrospopic aerosol species onto a uniform
715!relative humidity grid:
716
717 nbnd = nspint
718
719 do krh = 1, nrh
720 rh = 1.0_r8 / nrh * (krh - 1)
721 do kbnd = 1, nbnd
722 ksul(krh,kbnd) = exp_interpol(rh_opac, &
723 ksul_opac(:,kbnd) / ksul_opac(1,kbnd),rh) * ksul_opac(1,kbnd)
724 wsul(krh,kbnd) = lin_interpol(rh_opac, &
725 wsul_opac(:,kbnd) / wsul_opac(1,kbnd),rh) * wsul_opac(1,kbnd)
726 gsul(krh,kbnd) = lin_interpol(rh_opac, &
727 gsul_opac(:,kbnd) / gsul_opac(1,kbnd),rh) * gsul_opac(1,kbnd)
728 ksslt(krh,kbnd) = exp_interpol(rh_opac, &
729 ksslt_opac(:,kbnd) / ksslt_opac(1,kbnd),rh) * ksslt_opac(1,kbnd)
730 wsslt(krh,kbnd) = lin_interpol(rh_opac, &
731 wsslt_opac(:,kbnd) / wsslt_opac(1,kbnd),rh) * wsslt_opac(1,kbnd)
732 gsslt(krh,kbnd) = lin_interpol(rh_opac, &
733 gsslt_opac(:,kbnd) / gsslt_opac(1,kbnd),rh) * gsslt_opac(1,kbnd)
734 kcphil(krh,kbnd) = exp_interpol(rh_opac, &
735 kcphil_opac(:,kbnd) / kcphil_opac(1,kbnd),rh) * kcphil_opac(1,kbnd)
736 wcphil(krh,kbnd) = lin_interpol(rh_opac, &
737 wcphil_opac(:,kbnd) / wcphil_opac(1,kbnd),rh) * wcphil_opac(1,kbnd)
738 gcphil(krh,kbnd) = lin_interpol(rh_opac, &
739 gcphil_opac(:,kbnd) / gcphil_opac(1,kbnd),rh) * gcphil_opac(1,kbnd)
740 enddo
741 enddo
742
743! call mpas_log_write(' end subroutine aer_optics_initialize:')
744
745 end subroutine aer_optics_initialize

873! ozmixm(1,k,iCell),ozmixin(1,k,i2,1)/))
874! enddo
