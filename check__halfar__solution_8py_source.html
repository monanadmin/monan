34def halfar(t,x,y, A, n, rho):
35 # A # s^{-1} Pa^{-3}
36 # n # Glen flow law exponent
37 # rho # ice density kg m^{-3}
38
39 # These constants should come from setup_dome_initial_conditions.py.
40 # For now they are being hardcoded.
41 R0 = 60000.0 * np.sqrt(0.125) # initial dome radius
42 H0 = 2000.0 * np.sqrt(0.125) # initial dome thickness at center
43 g = 9.80616 # gravity m/s/s -- value used by mpas_constants
44 alpha = 1.0/9.0
45 beta = 1.0/18.0
46 secpera = 3600.0*24.0*365.0 # value for gregorian_noleap calendar
47 Gamma = 2.0 / (n+2.0) * A * (rho * g)**n
48
49 x0 = 0.0; y0 = 0.0 # The IC file puts the center of the dome and the domain at (0,0)
50
51 t0 = (beta/Gamma) * (7.0/4.0)**3 * (R0**4/H0**7) # NOTE: These constants assume n=3 - they need to be generalized to allow other n's
52 t=t+t0
53 t=t/t0
54
55 H=np.zeros(len(x))
56 for i in range(len(x)):
57 r = np.sqrt( (x[i] - x0)**2 + (y[i] - y0)**2)
58 r=r/R0
59 inside = max(0.0, 1.0 - (r / t**beta)**((n+1.0) / n))
60
61 H[i] = H0 * inside**(n / (2.0*n+1.0)) / t**alpha
62 #print "max r=", x[H>0.0].max(), y[H>0.0].max()
63 return H

65# Define a function to convert xtime character array to numeric time values using datetime objects
66def xtime2numtime(xtime):
67 # First parse the xtime character array into a string
68 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
69
70 dt = []
71 for stritem in xtimestr:
72 itemarray = stritem.strip().replace('_', '-').replace(':', '-').split('-') # Get an array of strings that are Y,M,D,h,m,s
73 results = [int(i) for i in itemarray]
74 if (results[0] < 1900): # datetime has a bug where years less than 1900 are invalid on some systems
75 results[0] += 1900
76 dt.append( datetime.datetime(*results) ) # * notation passes in the array as arguments
77
78 numtime = netCDF4.date2num(dt, units='seconds since '+str(dt[0])) # use the netCDF4 module's function for converting a datetime to a time number
79 return numtime
