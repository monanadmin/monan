44#define EZXML_BUFSIZE 1024 // size of internal memory buffers
45#define EZXML_NAMEM 0x80 // name is malloced

49typedef struct ezxml *ezxml_t;
50struct ezxml {
51 char *name; // tag name
52 char **attr; // tag attributes { name, value, name, value, ... NULL }
53 char *txt; // tag character content, empty string if none
54 size_t off; // tag offset from start of parent tag character content
55 ezxml_t next; // next tag with same name in this section at this depth
56 ezxml_t sibling; // next tag with different name in same section and depth
57 ezxml_t ordered; // next tag, same section and depth, in original order
58 ezxml_t child; // head of sub tag list, NULL if none
59 ezxml_t parent; // parent tag, NULL if current tag is root tag
60 short flags; // additional information
61};

63// Given a string of xml data and its length, parses it and creates an ezxml
64// structure. For efficiency, modifies the data by adding null terminators
65// and decoding ampersand sequences. If you don't want this, copy the data and
66// pass in the copy. Returns NULL on failure.
67ezxml_t ezxml_parse_str(char *s, size_t len);

69// A wrapper for ezxml_parse_str() that accepts a file descriptor. First
70// attempts to mem map the file. Failing that, reads the file into memory.
71// Returns NULL on failure.

74// a wrapper for ezxml_parse_fd() that accepts a file name
75ezxml_t ezxml_parse_file(const char *file);

77// Wrapper for ezxml_parse_str() that accepts a file stream. Reads the entire
78// stream into memory and then parses it. For xml files, use ezxml_parse_file()
79// or ezxml_parse_fd()
80ezxml_t ezxml_parse_fp(FILE *fp);

103// Traverses the ezxml sturcture to retrieve a specific subtag. Takes a
104// variable length list of tag names and indexes. The argument list must be
105// terminated by either an index of -1 or an empty string tag name. Example:
106// title = ezxml_get(library, "shelf", 0, "book", 2, "title", -1);
107// This retrieves the title of the 3rd book on the 1st shelf of library.
108// Returns NULL if not found.
