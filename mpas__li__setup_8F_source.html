157 subroutine li_setup_vertical_grid(meshPool, geometryPool, err)
158
159 !-----------------------------------------------------------------
160 !
161 ! input variables
162 !
163 !-----------------------------------------------------------------
164
165 !-----------------------------------------------------------------
166 !
167 ! input/output variables
168 !
169 !-----------------------------------------------------------------
170 type (mpas_pool_type), intent(inout) :: meshpool
171 type (mpas_pool_type), intent(inout) :: geometrypool
172
173 !-----------------------------------------------------------------
174 !
175 ! output variables
176 !
177 !-----------------------------------------------------------------
178
179 integer, intent(out) :: err
180
181 !-----------------------------------------------------------------
182 !
183 ! local variables
184 !
185 !-----------------------------------------------------------------
186 ! Pool pointers
187 integer, pointer :: nvertlevels ! Dimensions
188 real (kind=rkind), dimension(:), pointer :: layerthicknessfractions, layercentersigma, layerinterfacesigma
189 real (kind=rkind), dimension(:), pointer :: thickness
190 logical, pointer :: config_do_restart
191 ! Truly locals
192 integer :: k
193 real (kind=rkind) :: fractiontotal
194
195 ! Get pool stuff
196 call mpas_pool_get_config(liconfigs, 'config_do_restart', config_do_restart)
197 call mpas_pool_get_dimension(meshpool, 'nVertLevels', nvertlevels)
198 ! layerThicknessFractions is provided by input
199 call mpas_pool_get_array(meshpool, 'layerThicknessFractions', layerthicknessfractions)
200 call mpas_pool_get_array(meshpool, 'layerCenterSigma', layercentersigma)
201 call mpas_pool_get_array(meshpool, 'layerInterfaceSigma', layerinterfacesigma)
202 call mpas_pool_get_array(geometrypool, 'thickness', thickness)
203
204 ! Check that layerThicknessFractions are valid
205 ! TODO - switch to having the user input the sigma levels instead???
206 if (.not. config_do_restart) then ! This would be applied an additional time on each restart,
207 ! messing up the vertical levels (very, very slightly)
208 fractiontotal = sum(layerthicknessfractions)
209 if (fractiontotal /= 1.0_rkind) then
210 if (abs(fractiontotal - 1.0_rkind) > 0.001_rkind) then
211 call mpas_log_write('The sum of layerThicknessFractions is different from 1.0 by more than 0.001.', mpas_log_err)
212 err = 1
213 end if
214 call mpas_log_write('Adjusting upper layerThicknessFrac by small amount because sum of ' // &
215 'layerThicknessFractions is slightly different from 1.0.')
216 layerthicknessfractions(1) = layerthicknessfractions(1) - (fractiontotal - 1.0_rkind)
217 endif
218 endif
219
220 ! layerCenterSigma is the fractional vertical position (0-1) of each layer center,
221 ! with 0.0 at the ice surface and 1.0 at the ice bed
222 ! layerInterfaceSigma is the fractional vertical position (0-1) of each layer interface,
223 ! with 0.0 at the ice surface and 1.0 at the ice bed.
224 ! Interface 1 is the surface, interface 2 is between layers 1 and 2, etc., and interface nVertLevels+1 is the bed.
225 layercentersigma(1) = 0.5_rkind * layerthicknessfractions(1)
226 layerinterfacesigma(1) = 0.0_rkind
227 do k = 2, nvertlevels
228 layercentersigma(k) = layercentersigma(k-1) + 0.5_rkind * layerthicknessfractions(k-1) &
229 + 0.5_rkind * layerthicknessfractions(k)
230 layerinterfacesigma(k) = layerinterfacesigma(k-1) + layerthicknessfractions(k-1)
231 end do
232 layerinterfacesigma(nvertlevels+1) = 1.0_rkind
233
234 !--------------------------------------------------------------------
235 end subroutine li_setup_vertical_grid

472! subroutine li_cells_to_vertices_1dfield_using_kiteAreas
473!
483!-----------------------------------------------------------------------
484 subroutine li_cells_to_vertices_1dfield_using_kiteareas(meshPool, fieldCells, fieldVertices)
485 !-----------------------------------------------------------------
486 ! input variables
487 !-----------------------------------------------------------------
488 type (mpas_pool_type), intent(in) :: &
489 meshpool
490 real (kind=rkind), dimension(:), intent(in) :: &
491 fieldcells
492
493 !-----------------------------------------------------------------
494 ! input/output variables
495 !-----------------------------------------------------------------
496
497 !-----------------------------------------------------------------
498 ! output variables
499 !-----------------------------------------------------------------
500 real (kind=rkind), dimension(:), intent(out) :: &
501 fieldvertices
502
503 !-----------------------------------------------------------------
504 ! local variables
505 !-----------------------------------------------------------------
506 real (kind=rkind), dimension(:,:), pointer :: kiteareasonvertex
507 integer, dimension(:,:), pointer :: cellsonvertex
508 integer, pointer :: nvertices, vertexdegree
509 integer :: icell, icell2, ivertex, cellindex
510 real (kind=rkind) :: fvertexaccum, baryweight, weightaccum
511
512 ! Get needed items from mesh pool
513 call mpas_pool_get_dimension(meshpool, 'nVertices', nvertices)
514 call mpas_pool_get_dimension(meshpool, 'vertexDegree', vertexdegree)
515
516 call mpas_pool_get_array(meshpool, 'kiteAreasOnVertex', kiteareasonvertex)
517 call mpas_pool_get_array(meshpool, 'cellsOnVertex', cellsonvertex)
518
519 ! Calculate h on vertices using barycentric interpolation
520 do ivertex = 1, nvertices ! Loop over vertices
521 fvertexaccum = 0.0_rkind
522 weightaccum = 0.0_rkind
523 ! Loop over cells on this vertex
524 do icell = 1, vertexdegree
525 cellindex = cellsonvertex(icell, ivertex)
526 baryweight = 0.0_rkind
527 do icell2 = 1, vertexdegree
528 if (icell2 /= icell) baryweight = baryweight + 0.5 * kiteareasonvertex(icell2, ivertex)
529 enddo
530 fvertexaccum = fvertexaccum + baryweight * fieldcells(cellindex) ! add the contribution from this cell's kite
531 weightaccum = weightaccum + kiteareasonvertex(icell, ivertex) ! This doesn't match areaTriangle for obtuse triangles!!!
532 enddo
533 fieldvertices(ivertex) = fvertexaccum / weightaccum ! I assume this should never be 0...
534 enddo
535
536 end subroutine li_cells_to_vertices_1dfield_using_kiteareas

614! routine li_compute_gradient_2d
615!
623!
624!-----------------------------------------------------------------------
625
626 subroutine li_compute_gradient_2d(&
627 meshPool, &
628 field, &
629 xGrad, yGrad, &
630 err)
631
632 !-----------------------------------------------------------------
633 ! input variables
634 !-----------------------------------------------------------------
635 type (mpas_pool_type), intent(in) :: &
636 meshpool
637
638 real(kind=rkind), dimension(:), intent(in) :: &
639 field
640
641 !-----------------------------------------------------------------
642 ! input/output variables
643 !-----------------------------------------------------------------
644
645 !-----------------------------------------------------------------
646 ! output variables
647 !-----------------------------------------------------------------
648 real(kind=rkind), dimension(:), intent(out) :: &
649 xgrad, &
650 ygrad
651
652 integer, intent(out) :: err
653
654 !-----------------------------------------------------------------
655 ! local variables
656 !-----------------------------------------------------------------
657 integer, pointer :: &
658 ncells, & !< number of cells
659 maxedges
660
661 integer, dimension(:), pointer :: &
662 nedgesoncell
663
664 integer, dimension(:,:), pointer :: &
665 edgesoncell, & !< index for each edge of a given cell
666 cellsoncell, & !< index for each cell neighbor of a given cell
667 cellsonedge
668
669 real(kind=rkind), dimension(:), pointer :: &
670 dcedge
671
672 real(kind=rkind), dimension(:,:,:), pointer :: &
673 coeffs_reconstruct
675
676 integer :: icell, iedge, iedgeoncell, icellneighbor
677
678 real(kind=rkind) :: &
679 signgradient ! = 1 or -1, depending on which direction is taken as positive at a given edge
680
681 real(kind=rkind), dimension(:), allocatable :: &
682 normalgrad ! normal components of the gradient, defined on cell edges. For one cell.
683
684 err = 0
685
686 ! get needed variables
687 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
688 call mpas_pool_get_dimension(meshpool, 'maxEdges', maxedges)
689 call mpas_pool_get_array(meshpool, 'nEdgesOnCell', nedgesoncell)
690 call mpas_pool_get_array(meshpool, 'edgesOnCell', edgesoncell)
691 call mpas_pool_get_array(meshpool, 'cellsOnCell', cellsoncell)
692 call mpas_pool_get_array(meshpool, 'cellsOnEdge', cellsonedge)
693 call mpas_pool_get_array(meshpool, 'dcEdge', dcedge)
694 call mpas_pool_get_array(meshpool, 'coeffs_reconstruct', coeffs_reconstruct)
695
696 ! find dimensions and allocate arrays
697 allocate(normalgrad(maxedges))
698
699 ! initialize the gradient
700 xgrad(:) = 0.0_rkind
701 ygrad(:) = 0.0_rkind
702
703 ! loop over cells
704 do icell = 1, ncells
705
706 ! initialize normal gradient components on edges of this cell
707 normalgrad(:) = 0.0_rkind
708
709 ! loop over edges of this cell
710 do iedgeoncell = 1, nedgesoncell(icell)
711
712 icellneighbor = cellsoncell(iedgeoncell, icell)
713
714 ! compute the normal component of the gradient on this edge
715 if (icellneighbor >= 1 .and. icellneighbor <= ncells) then ! there is a cell neighbor on this edge
716 iedge = edgesoncell(iedgeoncell,icell)
717 if (icell == cellsonedge(1,iedge)) then
718 signgradient = 1.0_rkind
719 else
720 signgradient = -1.0_rkind
721 end if
722 normalgrad(iedgeoncell) = signgradient * (field(icellneighbor) - field(icell)) / dcedge(iedge)
723 else ! there is no cell neighbor on this edge
724 ! set gradient component = 0
725 normalgrad(iedgeoncell) = 0.0_rkind
726 endif
727
728 ! add the contribution of this normal component to the reconstructed
729 ! gradient at the cell center (in global x/y/z coordinates)
730 xgrad(icell) = xgrad(icell) + coeffs_reconstruct(1,iedgeoncell,icell) * normalgrad(iedgeoncell)
731 ygrad(icell) = ygrad(icell) + coeffs_reconstruct(2,iedgeoncell,icell) * normalgrad(iedgeoncell)
732 enddo ! iEdgeOnCell
733
734 enddo ! iCell
735
736 ! cleanup
737 deallocate(normalgrad)
738
739 end subroutine li_compute_gradient_2d
