29 subroutine init_atm_case_sfc(domain, dminfo, stream_manager, mesh, fg, state, dims, configs)
30!==================================================================================================
31
32 use mpas_stream_manager
33
34 implicit none
35
36!input arguments:
37 type (domain_type), intent(inout) :: domain
38 type (dm_info), intent(in) :: dminfo
39 type (mpas_streammanager_type), intent(inout) :: stream_manager
40 type (mpas_pool_type), intent(inout) :: mesh
41 type (mpas_pool_type), intent(inout) :: fg
42 type (mpas_pool_type), intent(inout) :: state
43 type (mpas_pool_type), intent(in) :: dims
44 type (mpas_pool_type), intent(in) :: configs
45
46!local variables:
47 type (mpas_time_type) :: curr_time, stop_time
48 character(len=StrKIND) :: timestring
49
50 character(len=StrKIND), pointer :: config_sfc_prefix
51 character(len=StrKIND), pointer :: xtime
52 integer :: ierr
53
54
55!==================================================================================================
56
57
58 call mpas_pool_get_config(configs, 'config_sfc_prefix', config_sfc_prefix)
59
60 call mpas_pool_get_array(state, 'xtime', xtime)
61
62!loop over all times:
63 curr_time = mpas_get_clock_time(domain % clock, mpas_now)
64 stop_time = mpas_get_clock_time(domain % clock, mpas_stop_time)
65
66 do while (curr_time <= stop_time)
67 call mpas_get_time(curr_time, datetimestring=timestring)
68 xtime = timestring
69
70! call mpas_log_write('Processing '//trim(config_sfc_prefix)//':'//timeString(1:13))
71
72 !read the sea-surface temperature and sea-ice data from the surface file, and interpolate the
73 !data to the MPAS grid:
74 call interp_sfc_to_mpas(timestring(1:13), mesh, fg, dims, dminfo, config_sfc_prefix)
75
76 !write the interpolated SST/SKINTEMP field as a new time slice in the MPAS output file:
77 call mpas_stream_mgr_write(stream_manager, streamid='surface', ierr=ierr)
78 call mpas_stream_mgr_reset_alarms(stream_manager, streamid='surface', direction=mpas_stream_output, ierr=ierr)
79
80 call mpas_advance_clock(domain % clock)
81 curr_time = mpas_get_clock_time(domain % clock, mpas_now)
82
83 end do
84
85 !
86 ! Ensure that no output alarms are still ringing for the 'surface' stream after
87 ! we exit the time loop above; the main run routine may write out all other
88 ! output streams with ringing alarms.
89 !
90 call mpas_stream_mgr_reset_alarms(stream_manager, streamid='surface', direction=mpas_stream_output, ierr=ierr)
91
92 end subroutine init_atm_case_sfc

94!==================================================================================================
95 subroutine interp_sfc_to_mpas(timeString, mesh, fg, dims, dminfo, config_sfc_prefix)
96!==================================================================================================
97
98 use mpas_dmpar
99
100 implicit none
101
102!input arguments:
103 character(len=*), intent(in) :: timestring
104 type (mpas_pool_type), intent(in) :: mesh
105 type (mpas_pool_type), intent(in) :: dims
106 type (dm_info), intent(in) :: dminfo
107 character(len=*), intent(in) :: config_sfc_prefix
108
109!inout arguments:
110 type (mpas_pool_type), intent(inout) :: fg
111
112
113!local variables:
114 type(met_data) :: field !real*4 meteorological data.
115
116 integer :: istatus
117 integer :: masked
118 integer, dimension(5) :: interp_list
119 integer, dimension(:), pointer :: mask_array
120 logical :: have_landmask
121
122 real(kind=rkind) :: fillval, maskval, msgval
123 real(kind=rkind), dimension(:,:), allocatable :: maskslab
124
125 integer, dimension(:), pointer :: landmask
126 integer :: global_max_landmask
127 real(kind=rkind), dimension(:), pointer :: destfield1d
128 real(kind=rkind), dimension(:), pointer :: sst, xice
129
130 integer, pointer :: ncells
131
132!==================================================================================================
133
134 call mpas_pool_get_array(mesh, 'landmask', mask_array)
135 call mpas_pool_get_array(mesh, 'landmask', landmask)
136 call mpas_pool_get_array(fg, 'sst', sst)
137 call mpas_pool_get_array(fg, 'xice', xice)
138
139 call mpas_pool_get_dimension(dims, 'nCells', ncells)
140
141!
142! Try to determine whether we have used a 'grid.nc' or a 'static.nc' file as input.
143! If we are working from a 'grid.nc', we expect that the global maximum of the landmask
144! will be 0.
145!
146 call mpas_dmpar_max_int(dminfo, maxval(landmask(1:ncells)), global_max_landmask)
147 if (global_max_landmask == 0) then
148 call mpas_log_write('*******************************************************************************', messagetype=mpas_log_err)
149 call mpas_log_write('The global maximum of the ''landmask'' field is zero, which suggests that this', messagetype=mpas_log_err)
150 call mpas_log_write('field was not in the input file. A ''landmask'' field is needed to properly', messagetype=mpas_log_err)
151 call mpas_log_write('interpolate surface fields.', messagetype=mpas_log_err)
152 call mpas_log_write('Please rerun after specifying a static or initial conditions file as input in', messagetype=mpas_log_err)
153 call mpas_log_write('the ''streams.init_atmosphere'' file.', messagetype=mpas_log_err)
154 call mpas_log_write('*******************************************************************************', messagetype=mpas_log_crit)
155 end if
156
157!open intermediate file:
158 call read_met_init(trim(config_sfc_prefix),.false.,timestring,istatus)
159 if(istatus /= 0) then
160 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
161 call mpas_log_write('Error opening surface file '//trim(config_sfc_prefix)//':'//timestring(1:13), messagetype=mpas_log_err)
162 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
163 else
164 call mpas_log_write('Processing file '//trim(config_sfc_prefix)//':'//timestring(1:13))
165 end if
166
167!scan through all fields in the file, looking for the LANDSEA field:
168 have_landmask = .false.
169 call read_next_met_field(field,istatus)
170 do while (istatus == 0)
171 if(trim(field % field) == 'LANDSEA') then
172 have_landmask = .true.
173 if(.not.allocated(maskslab)) allocate(maskslab(-2:field % nx+3, field % ny))
174 maskslab(1:field % nx, 1:field % ny) = field % slab(1:field % nx, 1:field % ny)
175 maskslab(0, 1:field % ny) = field % slab(field % nx, 1:field % ny)
176 maskslab(-1, 1:field % ny) = field % slab(field % nx-1, 1:field % ny)
177 maskslab(-2, 1:field % ny) = field % slab(field % nx-2, 1:field % ny)
178 maskslab(field % nx+1, 1:field % ny) = field % slab(1, 1:field % ny)
179 maskslab(field % nx+2, 1:field % ny) = field % slab(2, 1:field % ny)
180 maskslab(field % nx+3, 1:field % ny) = field % slab(3, 1:field % ny)
181! call mpas_log_write('minval, maxval of LANDSEA = $r $r', realArgs=(/minval(maskslab), maxval(maskslab)/))
182 end if
183 deallocate(field % slab)
184 call read_next_met_field(field,istatus)
185 end do
186 call read_met_close()
187
188!read sea-surface temperatures and seaice data. open intermediate file:
189 call read_met_init(trim(config_sfc_prefix),.false.,timestring(1:13),istatus)
190 if(istatus /= 0) then
191 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
192 call mpas_log_write('Error opening surface file '//trim(config_sfc_prefix)//':'//timestring(1:13), messagetype=mpas_log_err)
193 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
194 end if
195
196 if(.not. have_landmask) then
197 call mpas_log_write('********************************************************************************')
198 call mpas_log_write('Landsea mask not available from the surface file ')
199 call mpas_log_write('********************************************************************************')
200 end if
201
202!scan through all fields in the file, looking for the SST,SKINTEMP, or SEAICE field:
203 call read_next_met_field(field,istatus)
204 do while (istatus == 0)
205
206 !sea-surface data:
207 if((trim(field % field) == 'SKINTEMP') .or. (trim(field % field) == 'SST')) then
208! call mpas_log_write('... Processing SST:')
209 sst(1:ncells) = 0.0_rkind
210 destfield1d => sst
211
212 !interpolation to the MPAS grid:
213 interp_list(1) = four_point
214 interp_list(2) = search
215 interp_list(3) = 0
216 msgval = -1.0e30_r4kind !missing value
217 masked = -1
218 maskval = -1.0_rkind
219 fillval = 0.0_rkind
220 if(have_landmask) then
221 call interp_to_mpas(mesh,ncells,field,destfield1d,interp_list,msgval,masked,maskval,fillval, &
222 mask_array,maskslab)
223 else
224 call interp_to_mpas(mesh,ncells,field,destfield1d,interp_list,msgval,masked,maskval,fillval, &
225 mask_array)
226 end if
227
228 !field%slab was allocated in the subroutine read_next_met_field
229 deallocate(field%slab)
230
231 !sea-ice data:
232 else if(trim(field % field) == 'SEAICE') then
233! call mpas_log_write('... Processing SEAICE:')
234 xice(1:ncells) = 0.0_rkind
235 destfield1d => xice
236
237 !interpolation to the MPAS grid:
238 interp_list(1) = four_point
239 interp_list(2) = w_average4
240 interp_list(3) = search
241 interp_list(4) = 0
242 msgval = -1.0e30_r4kind !missing value
243 masked = 1
244 maskval = 1.0_rkind
245 fillval = 0.0_rkind
246 if(have_landmask) then
247 call interp_to_mpas(mesh,ncells,field,destfield1d,interp_list,msgval,masked,maskval,fillval, &
248 mask_array,maskslab)
249 else
250 call interp_to_mpas(mesh,ncells,field,destfield1d,interp_list,msgval,masked,maskval,fillval, &
251 mask_array)
252 end if
253
254 !field%slab was allocated in the subroutine read_next_met_field
255 deallocate(field%slab)
256
257 else
258 deallocate(field%slab)
259
260 end if
261
262 call read_next_met_field(field,istatus)
263 end do
264
265!close intermediate file:
266 call read_met_close()
267 if(allocated(maskslab)) deallocate(maskslab)
268
269!freeze really cold oceans:
270 where (sst < 271.0_rkind .and. landmask == 0) xice = 1.0_rkind
271
272!limit XICE to values between 0 and 1. Although the input meteorological field is between 0. and 1.
273!interpolation to the MPAS grid can yield values of XiCE less than 0. and greater than 1.:
274 where (xice < 0._rkind) xice = 0._rkind
275 where (xice > 1._rkind) xice = 1._rkind
276
277 end subroutine interp_sfc_to_mpas

279!==================================================================================================
280 subroutine interp_to_mpas(mesh,nCells,field,destField1d,interp_list,msgval,masked,maskval,fillval, &
281 mask_array,maskslab)
282!==================================================================================================
283
284!input arguments:
285 type (mpas_pool_type), intent(in) :: mesh
286 integer, intent(in) :: nCells
287 type (met_data), intent(in) :: field !real*4 meteorological data.
288
289 integer, intent(in) :: masked
290 integer, dimension(5), intent(in) :: interp_list
291 integer, dimension(:), intent(in), pointer :: mask_array
292
293 real(kind=rkind), intent(in) :: fillval, maskval, msgval
294 real(kind=rkind), intent(in), dimension(*), optional :: maskslab
295
296!inout arguments:
297 real(kind=rkind), intent(inout), dimension(:), pointer :: destfield1d
298
299!local variables:
300 type(proj_info) :: proj
301 integer :: i, ninterppoints
302 real(kind=rkind) :: lat,lon,x,y
303 real(kind=rkind), dimension(:,:), allocatable :: rslab
304
305 real(kind=rkind), dimension(:), pointer :: latpoints, lonpoints
306 real(kind=rkind), dimension(:), pointer :: latcell, loncell
307
308!--------------------------------------------------------------------------------------------------
309
310 call mpas_pool_get_array(mesh, 'latCell', latcell)
311 call mpas_pool_get_array(mesh, 'lonCell', loncell)
312
313 call map_init(proj)
314 if(field % iproj == proj_latlon) then
315 call map_set(proj_latlon, proj, &
316 latinc = real(field % deltalat,rkind), &
317 loninc = real(field % deltalon,rkind), &
318 knowni = 1.0_rkind, &
319 knownj = 1.0_rkind, &
320 lat1 = real(field % startlat,rkind), &
321 lon1 = real(field % startlon,rkind))
322! call mpas_log_write('--- The projection is PROJ_LATLON.')
323 else if(field % iproj == proj_gauss) then
324 call map_set(proj_gauss, proj, &
325 nlat = nint(field % deltalat), &
326 loninc = 360.0_rkind / real(field % nx,rkind), &
327 lat1 = real(field % startlat,rkind), &
328 lon1 = real(field % startlon,rkind))
329! call mpas_log_write('--- The projection is PROJ_GAUSS.')
330 else if(field % iproj == proj_ps) then
331 call map_set(proj_ps, proj, &
332 dx = real(field % dx,rkind), &
333 truelat1 = real(field % truelat1,rkind), &
334 stdlon = real(field % xlonc,rkind), &
335 knowni = real(field % nx / 2.0,rkind), &
336 knownj = real(field % ny / 2.0,rkind), &
337 lat1 = real(field % startlat,rkind), &
338 lon1 = real(field % startlon,rkind))
339! call mpas_log_write('--- The projection is PROJ_PS.')
340 end if
341
342 ninterppoints = ncells
343 latpoints => latcell
344 lonpoints => loncell
345
346 allocate(rslab(-2:field % nx+3, field % ny))
347 rslab(1:field % nx, 1:field % ny) = field % slab(1:field % nx, 1:field % ny)
348 rslab( 0, 1:field % ny) = field % slab(field % nx , 1:field % ny)
349 rslab(-1, 1:field % ny) = field % slab(field % nx-1, 1:field % ny)
350 rslab(-2, 1:field % ny) = field % slab(field % nx-2, 1:field % ny)
351 rslab(field % nx+1, 1:field % ny) = field % slab(1, 1:field % ny)
352 rslab(field % nx+2, 1:field % ny) = field % slab(2, 1:field % ny)
353 rslab(field % nx+3, 1:field % ny) = field % slab(3, 1:field % ny)
354
355 do i = 1,ninterppoints
356 if(mask_array(i) /= masked) then
357 lat = latpoints(i) * deg_per_rad
358 lon = lonpoints(i) * deg_per_rad
359 call latlon_to_ij(proj, lat, lon, x, y)
360 if(y <= 0.5) then
361 y = 1.0
362 else if(y >= real(field%ny)+0.5) then
363 y = real(field % ny)
364 end if
365 if(x < 0.5) then
366 lon = lon + 360.0
367 call latlon_to_ij(proj, lat, lon, x, y)
368 else if (x >= real(field%nx)+0.5) then
369 lon = lon - 360.0
370 call latlon_to_ij(proj, lat, lon, x, y)
371 end if
372 if(present(maskslab)) then
373 destfield1d(i) = interp_sequence(x,y,1,rslab,-2,field%nx+3,1,field%ny,1,1, &
374 msgval,interp_list,1,maskval=maskval,mask_array=maskslab)
375 else
376 destfield1d(i) = interp_sequence(x,y,1,rslab,-2,field%nx+3,1,field%ny,1,1, &
377 msgval,interp_list,1,maskval=maskval)
378 end if
379 else
380 destfield1d(i) = fillval
381 end if
382 end do
383 deallocate(rslab)
384
385 end subroutine interp_to_mpas
