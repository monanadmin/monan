2# This script sets up a "Confined Shelf Experiment".
3# see http://homepages.vub.ac.be/~phuybrec/eismint/shelf-descr.pdf

15options, args = parser.parse_args()
16if not options.filename:
17 options.filename = 'landice_grid.nc'
18 print 'No file specified. Attempting to use landice_grid.nc'

21# Open the file, get needed dimensions
22try:

24 nCells = len(gridfile.dimensions['nCells'])
25 nVertLevels = len(gridfile.dimensions['nVertLevels'])
26 maxEdges = len(gridfile.dimensions['maxEdges'])
27 if nVertLevels != 5:
28 print 'nVerLevels in the supplied file was ', nVertLevels, '. 5 levels are typically used with this test case.'
29 # Get variables
30 xCell = gridfile.variables['xCell'][:]
31 yCell = gridfile.variables['yCell'][:]
32 xEdge = gridfile.variables['xEdge'][:]
33 yEdge = gridfile.variables['yEdge'][:]
34 xVertex = gridfile.variables['xVertex'][:]
35 yVertex = gridfile.variables['yVertex'][:]
36 cellsOnCell= gridfile.variables['cellsOnCell'][:]
37except:

40
41
42# put the domain origin in the center of the center cell in the x-direction and in the 2nd row on the y-direction
43# Only do this if it appears this has not already been done:
44if xVertex[:].min() == 0.0:
45 print 'Shifting domain origin to center of shelf front, because it appears that this has not yet been done.'
46 unique_xs=np.array(sorted(list(set(xCell[:]))))
47 targetx = (unique_xs.max() - unique_xs.min()) / 2.0 + unique_xs.min() # center of domain range
48 best_x=unique_xs[ np.absolute((unique_xs - targetx)) == np.min(np.absolute(unique_xs - (targetx))) ][0]
49 print 'Found a best x value to use of:' + str(best_x)
50
51 unique_ys=np.array(sorted(list(set(yCell[:]))))
52# print unique_ys
53 best_y = unique_ys[5] # get 6th value
54 print 'Found a best y value to use of:' + str(best_y)
55
56 xShift = -1.0 * best_x
57 yShift = -1.0 * best_y
58 xCell[:] = xCell[:] + xShift
59 yCell[:] = yCell[:] + yShift
60 xEdge[:] = xEdge[:] + xShift
61 yEdge[:] = yEdge[:] + yShift
62 xVertex[:] = xVertex[:] + xShift
63 yVertex[:] = yVertex[:] + yShift

65 gridfile.variables['xCell'][:] = xCell[:]
66 gridfile.variables['yCell'][:] = yCell[:]
67 gridfile.variables['xEdge'][:] = xEdge[:]
68 gridfile.variables['yEdge'][:] = yEdge[:]
69 gridfile.variables['xVertex'][:] = xVertex[:]
70 gridfile.variables['yVertex'][:] = yVertex[:]

74# Make a square ice mass
75# Define square dimensions - all in meters
76L = 200000.0
77
78shelfMask = np.logical_and(
79 np.logical_and(xCell[:]>=-L/2.0, xCell[:]<=L/2.0),
80 np.logical_and(yCell[:]>=0.0, yCell[:]<=L) )
81# now grow it by one cell
82shelfMaskWithGround = np.zeros( (nCells,), dtype=np.int16)
83for c in range(nCells):
84 if shelfMask[c] == 1:
85 for n in range(maxEdges):
86 neighbor = cellsOnCell[c,n] - 1 # fortran to python indexing
87 if neighbor >= 0:
88 shelfMaskWithGround[neighbor] = 1
89# but remove the south side extension
90shelfMaskWithGround[ np.nonzero(yCell[:]<0.0)[0] ] = 0
91
92thickness = gridfile.variables['thickness'][:]
93thickness[:] = 0.0 # initialize to 0.0
94thickness[0, np.nonzero(shelfMaskWithGround==1)[0] ] = 500.0
95gridfile.variables['thickness'][:] = thickness[:]
96gridfile.sync()
97del thickness

99# flat bed at -2000 m everywhere but grounded around the edges
100bedTopography = gridfile.variables['bedTopography'][0,:]
101bedTopography[np.nonzero(shelfMask==1)[0]] = -2000.0
102bedTopography[np.nonzero(shelfMask==0)[0]] = -440.0
103gridfile.variables['bedTopography'][0,:] = bedTopography[:]
104gridfile.sync()
105del bedTopography
106
107# Dirichlet velocity mask
108kinbcmask = gridfile.variables['dirichletVelocityMask'][:]
109kinbcmask[:] = 0
110#kinbcmask[:, np.nonzero(shelfMask==0)[0], :] = 1
111kinbcmask[:, np.nonzero(np.logical_and(shelfMask==0, shelfMaskWithGround==1))[0], :] = 1
112#kinbcmask[:, np.nonzero(yCell[:]<0.0)[0] ] = 0
113# Need to extend this mask south by one cell so that the extended FEM mask will still have the 0 velo on the edges...
114theSides = Counter(xCell[ np.nonzero(kinbcmask[0,:])[0] ]).most_common(4) # need the 4 most common x positions.
115for side in theSides:
116 thesideindices = np.nonzero( np.logical_and( xCell[:] == side[0] , yCell[:] <= 0.0 ) )[0]
117 kinbcmask[:, thesideindices] = 1
118gridfile.variables['dirichletVelocityMask'][:] = kinbcmask[:]
119gridfile.sync()
120del kinbcmask
121
122# Dirichlet velocity values
123gridfile.variables['uReconstructX'][:] = 0.0
124gridfile.variables['uReconstructY'][:] = 0.0
125
126# beta is 0 everywhere (strictly speaking it should not be necessary to set this)
127gridfile.variables['beta'][:] = 0.0
128
129# Setup layerThicknessFractions
130gridfile.variables['layerThicknessFractions'][:] = 1.0 / nVertLevels
131
132# boundary conditions
133SMB = gridfile.variables['sfcMassBal'][:]
134SMB[:] = 0.0 # m/yr
135# Convert from units of m/yr to kg/m2/s using an assumed ice density
136SMB[:] = SMB[:] *910.0/(3600.0*24.0*365.0)
137gridfile.variables['sfcMassBal'][:] = SMB[:]
138gridfile.sync()
139del SMB
140
