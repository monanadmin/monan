38 subroutine init_thompson_clouddroplets_formpas(mesh,sfc_input,diag_physics)
39!=================================================================================================================
40
41!input variables:
42 type(mpas_pool_type),intent(in):: mesh
43 type(mpas_pool_type),intent(in):: sfc_input
44
45!inout variables:
46 type(mpas_pool_type),intent(inout):: diag_physics
47
48!local variables and pointers:
49 integer,pointer:: ncellssolve
50 integer,dimension(:),pointer:: landmask
51
52 real(kind=rkind),dimension(:),pointer:: nt_c,mu_c
53
54 integer:: icell
55
56!-----------------------------------------------------------------------------------------------------------------
57!call mpas_log_write('')
58!call mpas_log_write('--- enter subroutine init_thompson_clouddroplets_forMPAS:')
59
60 call mpas_pool_get_dimension(mesh,'nCellsSolve',ncellssolve)
61
62 call mpas_pool_get_array(sfc_input,'landmask',landmask)
63 call mpas_pool_get_array(diag_physics,'nt_c',nt_c)
64 call mpas_pool_get_array(diag_physics,'mu_c',mu_c)
65
66!... initialize the prescribed number of cloud droplets, and mu_c (parameter in the exponential of the generalized
67!gamma distribution) as a function of the land-cean mask. as set in the thompson cloud microphysics scheme, nt_c
68!is set to 100 per cc (100.E6 m^-3) for maritime cases and 300 per cc (300.E6 m^-3) for continental cases.
69 do icell = 1, ncellssolve
70 if(landmask(icell) .eq. 1) then
71 nt_c(icell) = 300.e6
72 elseif(landmask(icell) .eq. 0) then
73 nt_c(icell) = 100.e6
74 endif
75 mu_c(icell) = min(15., (1000.e6/nt_c(icell) + 2.))
76 enddo
77
78!call mpas_log_write('--- end subroutine init_thompson_clouddroplets_forMPAS.')
79
80 end subroutine init_thompson_clouddroplets_formpas
