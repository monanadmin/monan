148 integer function mpas_stream_inclusion_count(fieldName, direction, includeInactive)
149
150 use mpas_kind_types, only : strkind
151 use mpas_derived_types, only : mpas_stream_input, mpas_stream_output, mpas_stream_input_output, &
152 mpas_stream_property_record_intv
153 use mpas_stream_manager, only : mpas_stream_mgr_begin_iteration, mpas_stream_mgr_get_next_stream, &
154 mpas_stream_mgr_ringing_alarms, mpas_stream_mgr_get_next_field, &

157 implicit none

160 integer, intent(in), optional :: direction
161 logical, intent(in), optional :: includeinactive

164 character (len=StrKIND) :: fieldnameitr
165 character (len=StrKIND) :: recordintervalin

168 logical :: streamactive
169 logical :: fieldactive
170 integer :: ierr
171

173 logical :: local_includeinactive

175
176 if (present(direction)) then
177 local_direction = direction
178 else
179 local_direction = mpas_stream_input_output
180 end if
181
182 if (present(includeinactive)) then
183 local_includeinactive = includeinactive
184 else
185 local_includeinactive = .false.
186 end if
187
188
189 mpas_stream_inclusion_count = 0
190
191 call mpas_stream_mgr_begin_iteration(streammanager)
192 stream_loop: do while (mpas_stream_mgr_get_next_stream(streammanager, streamid = streamnameitr, &
193 directionproperty = streamdirection, activeproperty = streamactive))
194
195 call mpas_stream_mgr_get_property(streammanager, trim(streamnameitr), mpas_stream_property_record_intv, &
196 recordintervalin, direction=mpas_stream_input, ierr=ierr)
197
198 call mpas_stream_mgr_get_property(streammanager, trim(streamnameitr), mpas_stream_property_record_intv, &
199 recordintervalout, direction=mpas_stream_output, ierr=ierr)
200
201 ! Determine whether this stream is "active" for the purposes of consideration here
202 if (.not. local_includeinactive) then
203 if (streamactive) then
204 streamactive = ((local_direction == mpas_stream_input .and. trim(recordintervalin) /= 'none') &
205 .or. (local_direction == mpas_stream_output .and. trim(recordintervalout) /= 'none') &
206 .or. (local_direction == mpas_stream_input_output .and. ((trim(recordintervalin) /= 'none') &
207 .or. (trim(recordintervalout) /= 'none'))))
208 end if
209 else
210 streamactive = .true.
211 end if
212
213 if (streamactive .and. ((local_direction == mpas_stream_input .and. streamdirection == mpas_stream_input) &
214 .or. (local_direction == mpas_stream_output .and. streamdirection == mpas_stream_output) &
215 .or. (local_direction == mpas_stream_input_output) &
216 .or. (streamdirection == mpas_stream_input_output))) then
217
218 call mpas_stream_mgr_begin_iteration(streammanager, streamid=streamnameitr)
219 do while (mpas_stream_mgr_get_next_field(streammanager, streamnameitr, fieldnameitr, isactive=fieldactive))
220
221 if (fieldactive .and. (fieldnameitr == fieldname)) then
222 mpas_stream_inclusion_count = mpas_stream_inclusion_count + 1
223 cycle stream_loop
224 end if
225
226 end do
227
228 end if
229
230 end do stream_loop
231
232 end function mpas_stream_inclusion_count
