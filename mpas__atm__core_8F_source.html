21 function atm_core_init(domain, startTimeStamp) result(ierr)
22
23 use mpas_timekeeping
24 use mpas_kind_types
25 use mpas_stream_manager
26 use mpas_atm_dimensions, only : mpas_atm_set_dims
27 use mpas_atm_diagnostics_manager, only : mpas_atm_diag_setup
28 use mpas_atm_threading, only : mpas_atm_threading_init
29
30 implicit none
31
32 type (domain_type), intent(inout) :: domain
33 character(len=*), intent(out) :: starttimestamp
34 integer :: ierr
35
36 real (kind=rkind), pointer :: dt
37 type (block_type), pointer :: block
38
39 integer :: i
40 logical, pointer :: config_do_restart
41
42 type (mpas_pool_type), pointer :: state
43 type (mpas_pool_type), pointer :: mesh
44 type (mpas_pool_type), pointer :: diag
45 type (field2dreal), pointer :: u_field, pv_edge_field, ru_field, rw_field
46 character (len=StrKIND), pointer :: xtime
47 character (len=StrKIND), pointer :: initial_time1, initial_time2
48 type (mpas_time_type) :: starttime
49
50 integer, pointer :: nvertlevels, maxedges, maxedges2, num_scalars
51
52 ierr = 0
53
54
55 !
56 ! Setup threading
57 !
58 call mpas_atm_threading_init(domain % blocklist, ierr)
59 if ( ierr /= 0 ) then
60 call mpas_log_write('Threading setup failed for core '//trim(domain % core % coreName), messagetype=mpas_log_crit)
61 end if
62
63
64 !
65 ! Set up inner dimensions used by arrays in optimized dynamics routines
66 !
67 call mpas_pool_get_subpool(domain % blocklist % structs, 'state', state)
68 call mpas_pool_get_dimension(state, 'nVertLevels', nvertlevels)
69 call mpas_pool_get_dimension(state, 'maxEdges', maxedges)
70 call mpas_pool_get_dimension(state, 'maxEdges2', maxedges2)
71 call mpas_pool_get_dimension(state, 'num_scalars', num_scalars)
72 call mpas_atm_set_dims(nvertlevels, maxedges, maxedges2, num_scalars)
73
74 !
75 ! Set "local" clock to point to the clock contained in the domain type
76 !
77 clock => domain % clock
78 mpas_log_info => domain % logInfo
79
80
81 call mpas_pool_get_config(domain % blocklist % configs, 'config_do_restart', config_do_restart)
82 call mpas_pool_get_config(domain % blocklist % configs, 'config_dt', dt)
83
84
85 !
86 ! If this is a restart run, read the restart stream, else read the input
87 ! stream.
88 ! Regardless of which stream we read for initial conditions, reset the
89 ! input alarms for both input and restart before reading any remaining
90 ! input streams.
91 !
92 if (config_do_restart) then
93 call mpas_stream_mgr_read(domain % streamManager, streamid='restart', ierr=ierr)
94 else
95 call mpas_stream_mgr_read(domain % streamManager, streamid='input', ierr=ierr)
96 end if
97 if (ierr /= mpas_stream_mgr_noerr) then
98 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
99 call mpas_log_write('Error reading initial conditions', messagetype=mpas_log_err)
100 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
101 end if
102 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='input', direction=mpas_stream_input, ierr=ierr)
103 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='restart', direction=mpas_stream_input, ierr=ierr)
104
105 !
106 ! Read all other inputs
107 ! For now we don't do this here to match results with previous code; to match requires
108 ! that we read in SST and seaice fields after the call to atm_mpas_init_block()
109 !
110! call MPAS_stream_mgr_read(domain % streamManager, ierr=ierr)
111! call MPAS_stream_mgr_reset_alarms(domain % streamManager, direction=MPAS_STREAM_INPUT, ierr=ierr)
112
113 if (.not. config_do_restart) then
114 block => domain % blocklist
115 do while (associated(block))
116 call mpas_pool_get_subpool(block % structs, 'state', state)
117 call mpas_pool_initialize_time_levels(state)
118 block => block % next
119 end do
120 end if
121
122 !
123 ! Read a new data stream for Incremental Analysis Update (IAU), if config_IAU_option /= 'off' : HA (June-15-2016)
124 ! FIXME: should I check xtime for the IAU fields? Maybe not.
125 ! Note: Because the 'iau' stream has the 'iau' package attached to it, the MPAS_stream_mgr_read( )
126 ! call here will actually try to read the stream only if IAU is being used in the run.
127 !
128 call mpas_stream_mgr_read(domain % streamManager, streamid='iau', whence=mpas_stream_nearest, ierr=ierr)
129 if (ierr /= mpas_stream_mgr_noerr) then
130 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
131 call mpas_log_write('Error reading IAU files', messagetype=mpas_log_err)
132 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
133 end if
134 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='iau', ierr=ierr)
135
136 !
137 ! Set startTimeStamp based on the start time of the simulation clock
138 !
139 starttime = mpas_get_clock_time(clock, mpas_start_time, ierr)
140 call mpas_get_time(starttime, datetimestring=starttimestamp)
141
142
143 call mpas_pool_get_subpool(domain % blocklist % structs, 'state', state)
144 call mpas_pool_get_field(state, 'u', u_field, 1)
145 call mpas_dmpar_exch_halo_field(u_field)
146
147
148 !
149 ! Perform basic compatibility checks among the fields that were read and the run-time options that were selected
150 !
151 call mpas_atm_run_compatibility(domain % dminfo, domain % blocklist, domain % streamManager, ierr)
152 if (ierr /= 0) then
153 call mpas_log_write('Please correct issues with the model input fields and/or namelist.', messagetype=mpas_log_err)
154 return
155 end if
156
157
158 block => domain % blocklist
159 do while (associated(block))
160 call mpas_pool_get_subpool(block % structs, 'mesh', mesh)
161 call mpas_pool_get_subpool(block % structs, 'state', state)
162
163 call atm_mpas_init_block(domain % dminfo, domain % streamManager, block, mesh, dt)
164
165 call mpas_pool_get_array(state, 'xtime', xtime, 1)
166 xtime = starttimestamp
167
168 ! Initialize initial_time in second time level. We need to do this because initial state
169 ! is read into time level 1, and if we write output from the set of state arrays that
170 ! represent the original time level 2, the initial_time field will be invalid.
171 call mpas_pool_get_array(state, 'initial_time', initial_time1, 1)
172 call mpas_pool_get_array(state, 'initial_time', initial_time2, 2)
173 initial_time2 = initial_time1
174
175 block => block % next
176 end do
177
178 call mpas_pool_get_subpool(domain % blocklist % structs, 'diag', diag)
179 call mpas_pool_get_field(diag, 'pv_edge', pv_edge_field)
180 call mpas_dmpar_exch_halo_field(pv_edge_field)
181
182 call mpas_pool_get_field(diag, 'ru', ru_field)
183 call mpas_dmpar_exch_halo_field(ru_field)
184
185 call mpas_pool_get_field(diag, 'rw', rw_field)
186 call mpas_dmpar_exch_halo_field(rw_field)
187
188 call mpas_atm_diag_setup(domain % streamManager, domain % blocklist % configs, &
189 domain % blocklist % structs, domain % clock, domain % dminfo)
190
191 end function atm_core_init

273
274 type (dm_info), intent(in) :: dminfo
275 type (MPAS_streamManager_type), intent(inout) :: stream_manager
276 type (block_type), intent(inout) :: block
277 type (mpas_pool_type), intent(inout) :: mesh !MGD does this need to be a pointer?

280 type (mpas_pool_type), pointer :: state
281 type (mpas_pool_type), pointer :: diag
282 type (mpas_pool_type), pointer :: tend
283 type (mpas_pool_type), pointer :: sfc_input
284 type (mpas_pool_type), pointer :: diag_physics

289 real (kind=rkind), dimension(:,:), pointer :: u, ureconstructx, ureconstructy, ureconstructz, ureconstructzonal, ureconstructmeridional

298 character(len=StrKIND), pointer :: mminlu

302 integer, dimension(:), pointer :: cellSolveThreadStart, cellSolveThreadEnd

304 integer, dimension(:), pointer :: edgeSolveThreadStart, edgeSolveThreadEnd

311 call atm_compute_signs(mesh)
312
313 call mpas_pool_get_subpool(block % structs, 'diag', diag)
314 call mpas_pool_get_subpool(block % structs, 'state', state)

321 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
322 !!!! Compute inverses to avoid divides later
323
324 call mpas_pool_get_array(mesh, 'areaCell', areacell)
325 call mpas_pool_get_array(mesh, 'invAreaCell', invareacell)
326
327 call mpas_pool_get_array(mesh, 'dvEdge', dvedge)
328 call mpas_pool_get_array(mesh, 'dcEdge', dcedge)
329 call mpas_pool_get_array(mesh, 'invDvEdge', invdvedge)
330 call mpas_pool_get_array(mesh, 'invDcEdge', invdcedge)
331
332 call mpas_pool_get_array(mesh, 'areaTriangle', areatriangle)
333 call mpas_pool_get_array(mesh, 'invAreaTriangle', invareatriangle)

335 call mpas_pool_get_dimension(mesh, 'nCells', ncells)

337 call mpas_pool_get_dimension(mesh, 'nVertices', nvertices)

340 invareacell(icell) = 1.0_rkind / areacell(icell)
341 end do
342
343 do iedge=1,nedges

346
347 do iedge=1,nedges
348 invdcedge(iedge) = 1.0_rkind / dcedge(iedge)
349 end do
350
351 do ivertex=1,nvertices
352 invareatriangle(ivertex) = 1.0_rkind / areatriangle(ivertex)

355 !!!!! End compute inverses

357

359

361
362 call mpas_pool_get_dimension(state, 'nVertices', nvertices)
363 call mpas_pool_get_dimension(state, 'nVertLevels', nvertlevels)

387!$OMP PARALLEL DO
388 do thread=1,nthreads

392 vertexthreadstart(thread), vertexthreadend(thread), &

400 cellthreadstart(thread), cellthreadend(thread), &
401 vertexthreadstart(thread), vertexthreadend(thread), &
402 edgethreadstart(thread), edgethreadend(thread))
403 end do
404!$OMP END PARALLEL DO
405
406 deallocate(ke_vertex)
407 deallocate(ke_edge)
408
409 call mpas_rbf_interp_initialize(mesh)

420 ureconstructy, &
421 ureconstructz, &

470 realargs=(/minval(meshscalingdel2(1:nedgessolve)), maxval(meshscalingdel2(1:nedgessolve))/))
471 call mpas_log_write('min/max of meshScalingDel4 = $r $r', &
472 realargs=(/minval(meshscalingdel4(1:nedgessolve)), maxval(meshscalingdel4(1:nedgessolve))/))
473
474 end subroutine atm_mpas_init_block

477 function atm_core_run(domain) result(ierr)
478
479 use mpas_timekeeping
480 use mpas_kind_types
481 use mpas_stream_manager
482 use mpas_derived_types, only : mpas_stream_latest_before, mpas_stream_input, mpas_stream_input_output
483 use mpas_timer
484 use mpas_atm_boundaries, only : mpas_atm_update_bdy_tend
485 use mpas_atm_diagnostics_manager, only : mpas_atm_diag_update, mpas_atm_diag_compute, mpas_atm_diag_reset
486
487 implicit none
488
489 type (domain_type), intent(inout) :: domain
490 integer :: ierr
491
492 real (kind=rkind), pointer :: dt
493 logical, pointer :: config_do_restart
494 logical, pointer :: config_apply_lbcs
495 type (block_type), pointer :: block_ptr
496
497 type (mpas_time_type) :: currtime
498 character(len=StrKIND) :: timestamp
499 character (len=StrKIND), pointer :: config_restart_timestamp_name
500 integer :: itimestep
501
502 integer :: stream_dir
503 character(len=StrKIND) :: input_stream, read_time
504
505 type (mpas_pool_type), pointer :: state, diag, mesh, diag_physics, tend, tend_physics
506
507 ! For high-frequency diagnostics output
508 character (len=StrKIND) :: tempfilename
509
510 ! For timing information
511 real (kind=r8kind) :: integ_start_time, integ_stop_time
512 real (kind=r8kind) :: diag_start_time, diag_stop_time
513 real (kind=r8kind) :: input_start_time, input_stop_time
514 real (kind=r8kind) :: output_start_time, output_stop_time
515
516 ierr = 0
517
518 clock => domain % clock
519 mpas_log_info => domain % logInfo
520
521 ! Eventually, dt should be domain specific
522 call mpas_pool_get_config(domain % blocklist % configs, 'config_dt', dt)
523 call mpas_pool_get_config(domain % blocklist % configs, 'config_do_restart', config_do_restart)
524 call mpas_pool_get_config(domain % blocklist % configs, 'config_restart_timestamp_name', config_restart_timestamp_name)
525
526 ! Avoid writing a restart file at the initial time
527 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='restart', direction=mpas_stream_output, ierr=ierr)
528
529 ! Also, for restart runs, avoid writing the initial history or diagnostics fields to avoid overwriting those from the preceding run
530 if (config_do_restart) then
531 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='output', direction=mpas_stream_output, ierr=ierr)
532 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='diagnostics', direction=mpas_stream_output, ierr=ierr)
533 end if
534
535 call mpas_dmpar_get_time(diag_start_time)
536
537 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)) then
538 block_ptr => domain % blocklist
539 do while (associated(block_ptr))
540 call mpas_pool_get_subpool(block_ptr % structs, 'state', state)
541 call mpas_pool_get_subpool(block_ptr % structs, 'diag', diag)
542 call mpas_pool_get_subpool(block_ptr % structs, 'diag_physics', diag_physics)
543 call mpas_pool_get_subpool(block_ptr % structs, 'mesh', mesh)
544 call atm_compute_output_diagnostics(state, 1, diag, mesh)
545
546 block_ptr => block_ptr % next
547 end do
548 end if
549 call mpas_atm_diag_reset()
550 call mpas_atm_diag_update()
551 call mpas_atm_diag_compute()
552
553 call mpas_dmpar_get_time(diag_stop_time)
554
555 call mpas_dmpar_get_time(output_start_time)
556 call mpas_stream_mgr_write(domain % streamManager, ierr=ierr)
557 call mpas_dmpar_get_time(output_stop_time)
558 if (ierr /= mpas_stream_mgr_noerr .and. &
559 ierr /= mpas_stream_mgr_err_clobber_file .and. &
560 ierr /= mpas_stream_mgr_err_clobber_rec) then
561 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
562 call mpas_log_write('Error writing one or more output streams', messagetype=mpas_log_err)
563 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
564 end if
565 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)) then
566 call mpas_log_write('Timing for diagnostic computation: $r s', realargs=(/real(diag_stop_time - diag_start_time, kind=rkind)/))
567 call mpas_log_write('Timing for stream output: $r s', realargs=(/real(output_stop_time - output_start_time, kind=rkind)/))
568 end if
569
570 call mpas_atm_diag_reset()
571
572 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)
573
574 block_ptr => domain % blocklist
575 call mpas_pool_get_subpool(block_ptr % structs, 'mesh', mesh)
576 call mpas_pool_get_subpool(block_ptr % structs, 'state', state)
577 call mpas_pool_get_subpool(block_ptr % structs, 'diag', diag)
578 call mpas_pool_get_subpool(block_ptr % structs, 'diag_physics', diag_physics)
579
580 call mpas_pool_get_config(domain % blocklist % configs, 'config_apply_lbcs', config_apply_lbcs)
581
582 !
583 ! Read initial boundary state
584 !
585 if (config_apply_lbcs .and. &
586 mpas_stream_mgr_ringing_alarms(domain % streamManager, streamid='lbc_in', direction=mpas_stream_input, ierr=ierr)) then
587 block_ptr => domain % blocklist
588 do while (associated(block_ptr))
589 call mpas_atm_update_bdy_tend(clock, domain % streamManager, block_ptr, .true., ierr)
590 if (ierr /= 0) then
591 currtime = mpas_get_clock_time(clock, mpas_now, ierr)
592 call mpas_get_time(curr_time=currtime, datetimestring=timestamp)
593 call mpas_log_write('Failed to process LBC data on or before '//trim(timestamp), messagetype=mpas_log_err)
594 return
595 end if
596
597 block_ptr => block_ptr % next
598 end do
599 end if
600
601 ! During integration, time level 1 stores the model state at the beginning of the
602 ! time step, and time level 2 stores the state advanced dt in time by timestep(...)
603 itimestep = 1
604 do while (.not. mpas_is_clock_stop_time(clock))
605
606 currtime = mpas_get_clock_time(clock, mpas_now, ierr)
607 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=ierr)
608
609 call mpas_log_write('')
610 call mpas_log_write('Begin timestep '//trim(timestamp))
611
612 !
613 ! Read future boundary state and compute boundary tendencies
614 !
615 if (config_apply_lbcs .and. &
616 mpas_stream_mgr_ringing_alarms(domain % streamManager, streamid='lbc_in', direction=mpas_stream_input, ierr=ierr)) then
617 block_ptr => domain % blocklist
618 do while (associated(block_ptr))
619 call mpas_atm_update_bdy_tend(clock, domain % streamManager, block_ptr, .false., ierr)
620 if (ierr /= 0) then
621 call mpas_log_write('Failed to process LBC data at next time after '//trim(timestamp), messagetype=mpas_log_err)
622 return
623 end if
624
625 block_ptr => block_ptr % next
626 end do
627 end if
628
629 ! Regardless of whether boundary tendencies were updated, above, we do not want to read the 'lbc_in' stream
630 ! as a general input stream, below.
631 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='lbc_in', direction=mpas_stream_input, ierr=ierr)
632
633
634 !
635 ! Read external field updates
636 !
637 call mpas_stream_mgr_begin_iteration(domain % streamManager, ierr=ierr)
638 do while (mpas_stream_mgr_get_next_stream(domain % streamManager, streamid=input_stream, directionproperty=stream_dir))
639 if (stream_dir == mpas_stream_input .or. stream_dir == mpas_stream_input_output) then
640 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, streamid=input_stream, &
641 direction=mpas_stream_input, ierr=ierr)) then
642 call mpas_dmpar_get_time(input_start_time)
643 call mpas_stream_mgr_read(domain % streamManager, streamid=input_stream, whence=mpas_stream_latest_before, &
644 actualwhen=read_time, ierr=ierr)
645 call mpas_dmpar_get_time(input_stop_time)
646 if (ierr /= mpas_stream_mgr_noerr) then
647 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
648 call mpas_log_write('Error reading input stream '//trim(input_stream), messagetype=mpas_log_err)
649 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
650 end if
651
652 call mpas_log_write('----------------------------------------------------------------------')
653 call mpas_log_write(' Read '''//trim(input_stream)//''' input stream valid at '//trim(read_time))
654 call mpas_log_write(' Timing for stream input: $r s', realargs=(/real(input_stop_time - input_start_time, kind=rkind)/))
655 call mpas_log_write('----------------------------------------------------------------------')
656
657 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid=input_stream, direction=mpas_stream_input, ierr=ierr)
658 end if
659 end if
660 end do
661
662 call mpas_timer_start("time integration")
663 call mpas_dmpar_get_time(integ_start_time)
664 call atm_do_timestep(domain, dt, itimestep)
665 call mpas_dmpar_get_time(integ_stop_time)
666 call mpas_timer_stop("time integration")
667 call mpas_log_write(' Timing for integration step: $r s', realargs=(/real(integ_stop_time - integ_start_time, kind=rkind)/))
668
669 ! Move time level 2 fields back into time level 1 for next time step
670 call mpas_pool_get_subpool(domain % blocklist % structs, 'state', state)
671 call mpas_pool_shift_time_levels(state)
672
673 ! Advance clock before writing output
674 itimestep = itimestep + 1
675 call mpas_advance_clock(clock)
676 currtime = mpas_get_clock_time(clock, mpas_now, ierr)
677
678
679 call mpas_dmpar_get_time(diag_start_time)
680
681 !
682 ! Write any output streams that have alarms ringing, after computing diagnostics fields
683 !
684 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=ierr)
685 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)) then
686 block_ptr => domain % blocklist
687 do while (associated(block_ptr))
688 call mpas_pool_get_subpool(block_ptr % structs, 'state', state)
689 call mpas_pool_get_subpool(block_ptr % structs, 'diag', diag)
690 call mpas_pool_get_subpool(block_ptr % structs, 'diag_physics', diag_physics)
691 call mpas_pool_get_subpool(block_ptr % structs, 'mesh', mesh)
692 call mpas_pool_get_subpool(block_ptr % structs, 'tend', tend)
693 call mpas_pool_get_subpool(block_ptr % structs, 'tend_physics', tend_physics)
694 call atm_compute_output_diagnostics(state, 1, diag, mesh)
695
696 block_ptr => block_ptr % next
697 end do
698 end if
699
700 call mpas_atm_diag_update()
701 call mpas_atm_diag_compute()
702 call mpas_dmpar_get_time(diag_stop_time)
703
704 call mpas_dmpar_get_time(output_start_time)
705 call mpas_stream_mgr_write(domain % streamManager, ierr=ierr)
706 call mpas_dmpar_get_time(output_stop_time)
707 if (ierr /= mpas_stream_mgr_noerr .and. &
708 ierr /= mpas_stream_mgr_err_clobber_file .and. &
709 ierr /= mpas_stream_mgr_err_clobber_rec) then
710 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
711 call mpas_log_write('Error writing one or more output streams', messagetype=mpas_log_err)
712 call mpas_log_write('********************************************************************************', messagetype=mpas_log_crit)
713 end if
714 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)) then
715 call mpas_log_write('Timing for diagnostic computation: $r s', realargs=(/real(diag_stop_time - diag_start_time, kind=rkind)/))
716 call mpas_log_write('Timing for stream output: $r s', realargs=(/real(output_stop_time - output_start_time, kind=rkind)/))
717 end if
718
719 ! reset any diagnostics here
720
721 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, streamid='diagnostics', direction=mpas_stream_output, ierr=ierr)) then
722 block_ptr => domain % blocklist
723 do while (associated(block_ptr))
724
725 call mpas_pool_get_subpool(block_ptr % structs, 'diag', diag)
726 call mpas_pool_get_subpool(block_ptr % structs, 'diag_physics', diag_physics)
727 call atm_reset_diagnostics(diag, diag_physics)
728
729 block_ptr => block_ptr % next
730 end do
731 end if
732
733
734 ! Only after we've successfully written the restart file should we we
735 ! write the restart_timestamp file
736 if (mpas_stream_mgr_ringing_alarms(domain % streamManager, streamid='restart', direction=mpas_stream_output, ierr=ierr)) then
737 if (domain % dminfo % my_proc_id == 0) then
738 open(22,file=trim(config_restart_timestamp_name),form='formatted',status='replace')
739 write(22,*) trim(timestamp)
740 close(22)
741 end if
742 end if
743
744 call mpas_atm_diag_reset()
745
746 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)
747
748 end do
749
750 end function atm_core_run

753 subroutine atm_compute_output_diagnostics(state, time_lev, diag, mesh)
754 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
755 ! Compute diagnostic fields for a domain to be written to history files
756 !
757 ! Input: state - contains model prognostic fields
758 ! mesh - contains grid metadata
759 !
760 ! Output: state - upon returning, diagnostic fields will have be computed
761 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
762
763 use mpas_constants
764
765 implicit none
766
767 type (mpas_pool_type), intent(inout) :: state
768 integer, intent(in) :: time_lev ! which time level to use from state
769 type (mpas_pool_type), intent(inout) :: diag
770 type (mpas_pool_type), intent(in) :: mesh
771
772 integer :: iCell, k
773 integer, pointer :: nCells, nVertLevels, index_qv
774 real (kind=rkind), dimension(:,:), pointer :: theta, rho, theta_m, rho_zz, zz
775 real (kind=rkind), dimension(:,:), pointer :: pressure_base, pressure_p, pressure
776 real (kind=rkind), dimension(:,:,:), pointer :: scalars
777
778 call mpas_pool_get_dimension(mesh, 'nCells', ncells)
779 call mpas_pool_get_dimension(mesh, 'nVertLevels', nvertlevels)
780 call mpas_pool_get_dimension(state, 'index_qv', index_qv)
781
782 call mpas_pool_get_array(state, 'theta_m', theta_m, time_lev)
783 call mpas_pool_get_array(state, 'rho_zz', rho_zz, time_lev)
784 call mpas_pool_get_array(state, 'scalars', scalars, time_lev)
785
786 call mpas_pool_get_array(diag, 'theta', theta)
787 call mpas_pool_get_array(diag, 'rho', rho)
788 call mpas_pool_get_array(diag, 'pressure_p', pressure_p)
789 call mpas_pool_get_array(diag, 'pressure_base', pressure_base)
790 call mpas_pool_get_array(diag, 'pressure', pressure)
791
792 call mpas_pool_get_array(mesh, 'zz', zz)
793
794 do icell=1,ncells
795 do k=1,nvertlevels
796 theta(k,icell) = theta_m(k,icell) / (1._rkind + rvord * scalars(index_qv,k,icell))
797 rho(k,icell) = rho_zz(k,icell) * zz(k,icell)
798 pressure(k,icell) = pressure_base(k,icell) + pressure_p(k,icell)
799 end do
800 end do
801
802 end subroutine atm_compute_output_diagnostics

927 subroutine atm_compute_mesh_scaling(mesh, configs)
928
929 implicit none
930
931 type (mpas_pool_type), intent(inout) :: mesh
932 type (mpas_pool_type), intent(in) :: configs
933
934 integer :: iCell,iEdge, cell1, cell2
935 integer, pointer :: nEdges, nCells
936 integer, dimension(:,:), pointer :: cellsOnEdge
937 real (kind=rkind), dimension(:), pointer :: meshdensity, meshscalingdel2, meshscalingdel4
938 real (kind=rkind), dimension(:), pointer :: meshscalingregionalcell, meshscalingregionaledge
939 logical, pointer :: config_h_ScaleWithMesh
940
941 call mpas_pool_get_array(mesh, 'meshDensity', meshdensity)
942 call mpas_pool_get_array(mesh, 'meshScalingDel2', meshscalingdel2)
943 call mpas_pool_get_array(mesh, 'meshScalingDel4', meshscalingdel4)
944 call mpas_pool_get_array(mesh, 'meshScalingRegionalCell', meshscalingregionalcell)
945 call mpas_pool_get_array(mesh, 'meshScalingRegionalEdge', meshscalingregionaledge)
946 call mpas_pool_get_array(mesh, 'cellsOnEdge', cellsonedge)
947
948 call mpas_pool_get_dimension(mesh, 'nEdges', nedges)
949 call mpas_pool_get_dimension(mesh, 'nCells', ncells)
950
951 call mpas_pool_get_config(configs, 'config_h_ScaleWithMesh', config_h_scalewithmesh)
952
953 !
954 ! Compute the scaling factors to be used in the del2 and del4 dissipation
955 !
956 meshscalingdel2(:) = 1.0
957 meshscalingdel4(:) = 1.0
958 if (config_h_scalewithmesh) then
959 do iedge=1,nedges
960 cell1 = cellsonedge(1,iedge)
961 cell2 = cellsonedge(2,iedge)
962 meshscalingdel2(iedge) = 1.0 / ( (meshdensity(cell1) + meshdensity(cell2) )/2.0)**0.25
963 meshscalingdel4(iedge) = 1.0 / ( (meshdensity(cell1) + meshdensity(cell2) )/2.0)**0.75
964 end do
965 end if
966
967 !
968 ! Compute the scaling factors to be used in relaxation zone of regional configuration
969 !
970 meshscalingregionalcell(:) = 1.0
971 meshscalingregionaledge(:) = 1.0
972 if (config_h_scalewithmesh) then
973 do iedge=1,nedges
974 cell1 = cellsonedge(1,iedge)
975 cell2 = cellsonedge(2,iedge)
976 meshscalingregionaledge(iedge) = 1.0 / ( (meshdensity(cell1) + meshdensity(cell2) )/2.0)**0.25
977 end do
978
979 do icell=1,ncells
980 meshscalingregionalcell(icell) = 1.0 / (meshdensity(icell))**0.25
981 end do
982 end if
983
984 end subroutine atm_compute_mesh_scaling

987 subroutine atm_compute_signs(mesh)
988
989 implicit none
990
991 type (mpas_pool_type), intent(inout) :: mesh
992
993 integer :: i, j, iCell, iVtx
994 integer, pointer :: nCells, nVertices, nEdges, vertexDegree
995 integer, dimension(:), pointer :: nEdgesOnCell
996 integer, dimension(:,:), pointer :: edgesOnVertex, verticesOnEdge, cellsOnEdge, edgesOnCell
997 integer, dimension(:,:), pointer :: verticesOnCell, cellsOnVertex, kiteForCell
998 real (kind=rkind), dimension(:,:), pointer :: edgesonvertex_sign, edgesoncell_sign
999 real (kind=rkind), dimension(:,:,:), pointer :: zb, zb3, zb_cell, zb3_cell
1000
1001
1002 call mpas_pool_get_array(mesh, 'edgesOnVertex', edgesonvertex)
1003 call mpas_pool_get_array(mesh, 'verticesOnEdge', verticesonedge)
1004 call mpas_pool_get_array(mesh, 'nEdgesOnCell', nedgesoncell)
1005 call mpas_pool_get_array(mesh, 'edgesOnCell', edgesoncell)
1006 call mpas_pool_get_array(mesh, 'cellsOnEdge', cellsonedge)
1007 call mpas_pool_get_array(mesh, 'verticesOnCell', verticesoncell)
1008 call mpas_pool_get_array(mesh, 'cellsOnVertex', cellsonvertex)
1009 call mpas_pool_get_array(mesh, 'edgesOnVertex_sign', edgesonvertex_sign)
1010 call mpas_pool_get_array(mesh, 'edgesOnCell_sign', edgesoncell_sign)
1011 call mpas_pool_get_array(mesh, 'kiteForCell', kiteforcell)
1012 call mpas_pool_get_array(mesh, 'zb', zb)
1013 call mpas_pool_get_array(mesh, 'zb3', zb3)
1014 call mpas_pool_get_array(mesh, 'zb_cell', zb_cell)
1015 call mpas_pool_get_array(mesh, 'zb3_cell', zb3_cell)
1016
1017 call mpas_pool_get_dimension(mesh, 'nCells', ncells)
1018 call mpas_pool_get_dimension(mesh, 'nVertices', nvertices)
1019 call mpas_pool_get_dimension(mesh, 'nEdges', nedges)
1020 call mpas_pool_get_dimension(mesh, 'vertexDegree', vertexdegree)
1021
1022
1023 do ivtx=1,nvertices
1024 do i=1,vertexdegree
1025 if (edgesonvertex(i,ivtx) <= nedges) then
1026 if (ivtx == verticesonedge(2,edgesonvertex(i,ivtx))) then
1027 edgesonvertex_sign(i,ivtx) = 1.0
1028 else
1029 edgesonvertex_sign(i,ivtx) = -1.0
1030 end if
1031 else
1032 edgesonvertex_sign(i,ivtx) = 0.0
1033 end if
1034 end do
1035 end do
1036
1037 do icell=1,ncells
1038 do i=1,nedgesoncell(icell)
1039 if (edgesoncell(i,icell) <= nedges) then
1040 if (icell == cellsonedge(1,edgesoncell(i,icell))) then
1041 edgesoncell_sign(i,icell) = 1.0
1042 zb_cell(:,i,icell) = zb(:,1,edgesoncell(i,icell))
1043 zb3_cell(:,i,icell) = zb3(:,1,edgesoncell(i,icell))
1044 else
1045 edgesoncell_sign(i,icell) = -1.0
1046 zb_cell(:,i,icell) = zb(:,2,edgesoncell(i,icell))
1047 zb3_cell(:,i,icell) = zb3(:,2,edgesoncell(i,icell))
1048 end if
1049 else
1050 edgesoncell_sign(i,icell) = 0.0
1051 end if
1052 end do
1053 end do
1054
1055 do icell=1,ncells
1056 do i=1,nedgesoncell(icell)
1057 ivtx = verticesoncell(i,icell)
1058 if (ivtx <= nvertices) then
1059 do j=1,vertexdegree
1060 if (icell == cellsonvertex(j,ivtx)) then
1061 kiteforcell(i,icell) = j
1062 exit
1063 end if
1064 end do
1065 if (j > vertexdegree) then
1066 call mpas_log_write('Unexpected error while identifying kiteForCell', messagetype=mpas_log_err)
1067 end if
1068 else
1069 kiteforcell(i,icell) = 1
1070 end if
1071 end do
1072 end do
1073
1074 end subroutine atm_compute_signs
