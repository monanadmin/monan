32def garplot(ncfile, label, color, marker):
33 """
34 add a plot of grounded area against time to current axes
35 """
36 try:
37 ncid = Dataset(ncfile, 'r')
38 except:
39 print "Failed to open file: {}. Skipping.".format(ncfile)
40 return 0
41 gar = ncid.variables["groundedArea"][:]*1e-6*1e-3
42 time = ncid.variables["time"][:]
43 if label == 'nolabel':
44 plt.plot(tscale(time), gar, 'o-', mfc=color,
45 color='black', label="", marker=marker)
46 else:
47 plt.plot(tscale(time), gar, 'o-', mfc=color,
48 color='black', label=label, marker=marker)
