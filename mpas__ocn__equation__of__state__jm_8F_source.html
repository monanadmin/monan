83 subroutine ocn_equation_of_state_jm_density(meshPool, scratchPool, nCells, k_displaced, displacement_type, &
84 indexT, indexS, tracers, density, err, &
85 tracersSurfaceLayerValue, thermalExpansionCoeff, salineContractionCoeff)!{{{
86 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
87 ! This module contains routines necessary for computing the density
88 ! from model temperature and salinity using an equation of state.
89 !
90 ! The UNESCO equation of state computed using the
91 ! potential-temperature-based bulk modulus from Jackett and
92 ! McDougall, JTECH, Vol.12, pp 381-389, April, 1995.
93 !
94 ! Input: mesh - mesh metadata
95 ! s - state: tracers
96 ! k_displaced
97
98 ! If k_displaced=0, density is returned with no displacement
99 ! If k_displaced>0,the density returned is that for a parcel
100 ! adiabatically displaced from its original level to level
101 ! k_displaced.
102
103 !
104 ! Output: s - state: computed density
105 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
106
107 implicit none
108
109 type (mpas_pool_type), intent(in) :: meshpool
110 type (mpas_pool_type), intent(in) :: scratchpool
111 integer, intent(in) :: ncells
112 integer, intent(in) :: k_displaced, indext, indexs
113 character(len=*), intent(in) :: displacement_type
114 real (kind=rkind), dimension(:,:,:), intent(in) :: tracers
115 real (kind=rkind), dimension(:,:), intent(out) :: density
116 integer, intent(out) :: err
117 real (kind=rkind), dimension(:,:), intent(in), optional :: tracerssurfacelayervalue
118 real (kind=rkind), dimension(:,:), intent(out), optional :: &
119 thermalexpansioncoeff, &! Thermal expansion coefficient (alpha), defined as $-1/\rho d\rho/dT$ (note negative sign)
120 salinecontractioncoeff ! Saline contraction coefficient (beta), defined as $1/\rho d\rho/dS$
121
122 integer :: iedge, icell, ivertex, k, k_displaced_local
123 integer, pointer :: nvertices, nvertlevels
124 integer, dimension(:), pointer :: maxlevelcell
125 character(len=60) :: displacement_type_local
126
127 real (kind=rkind) :: &
128 depth, &
129 drdt0, &! d(density)/d(temperature), for surface
130 drds0, &! d(density)/d(salinity ), for surface
131 dkdt, &! d(bulk modulus)/d(pot. temp.)
132 dkds, &! d(bulk modulus)/d(salinity )
133 drhodt, &! derivative of density with respect to temperature
134 drhods, &! derivative of density with respect to salinity
135 tmin, tmax, &! valid temperature range for level k
136 smin, smax ! valid salinity range for level k
137 real (kind=rkind), dimension(:), pointer :: &
138 refbottomdepth, prefeos
139 real (kind=rkind), dimension(:), allocatable :: &
140 p, p2 ! temporary pressure scalars
141 real (kind=rkind), dimension(:), pointer :: &
142 tq,sq, &! adjusted T,S
143 bulk_mod, &! Bulk modulus
144 sqr,denomk, &! work arrays
145 rho_s, &! density at the surface
146 work1, work2, work3, work4, t2
147 real (kind=rkind), dimension(:), allocatable :: &
148 tracertemp, tracersalt
149
150!-----------------------------------------------------------------------
151!
152! UNESCO EOS constants and JMcD bulk modulus constants
153!
154!-----------------------------------------------------------------------
155
156 !*** for density of fresh water (standard UNESCO)
157
158 real (kind=rkind), parameter :: &
159 unt0 = 999.842594_rkind, &
160 unt1 = 6.793952e-2_rkind, &
161 unt2 = -9.095290e-3_rkind, &
162 unt3 = 1.001685e-4_rkind, &
163 unt4 = -1.120083e-6_rkind, &
164 unt5 = 6.536332e-9_rkind
165
166 !*** for dependence of surface density on salinity (UNESCO)
167
168 real (kind=rkind), parameter :: &
169 uns1t0 = 0.824493_rkind , &
170 uns1t1 = -4.0899e-3_rkind, &
171 uns1t2 = 7.6438e-5_rkind, &
172 uns1t3 = -8.2467e-7_rkind, &
173 uns1t4 = 5.3875e-9_rkind, &
174 unsqt0 = -5.72466e-3_rkind, &
175 unsqt1 = 1.0227e-4_rkind, &
176 unsqt2 = -1.6546e-6_rkind, &
177 uns2t0 = 4.8314e-4_rkind
178
179 !*** from Table A1 of Jackett and McDougall
180
181 real (kind=rkind), parameter :: &
182 bup0s0t0 = 1.965933e+4_rkind, &
183 bup0s0t1 = 1.444304e+2_rkind, &
184 bup0s0t2 = -1.706103_rkind , &
185 bup0s0t3 = 9.648704e-3_rkind, &
186 bup0s0t4 = -4.190253e-5_rkind
187
188 real (kind=rkind), parameter :: &
189 bup0s1t0 = 5.284855e+1_rkind, &
190 bup0s1t1 = -3.101089e-1_rkind, &
191 bup0s1t2 = 6.283263e-3_rkind, &
192 bup0s1t3 = -5.084188e-5_rkind
193
194 real (kind=rkind), parameter :: &
195 bup0sqt0 = 3.886640e-1_rkind, &
196 bup0sqt1 = 9.085835e-3_rkind, &
197 bup0sqt2 = -4.619924e-4_rkind
198
199 real (kind=rkind), parameter :: &
200 bup1s0t0 = 3.186519_rkind , &
201 bup1s0t1 = 2.212276e-2_rkind, &
202 bup1s0t2 = -2.984642e-4_rkind, &
203 bup1s0t3 = 1.956415e-6_rkind
204
205 real (kind=rkind), parameter :: &
206 bup1s1t0 = 6.704388e-3_rkind, &
207 bup1s1t1 = -1.847318e-4_rkind, &
208 bup1s1t2 = 2.059331e-7_rkind, &
209 bup1sqt0 = 1.480266e-4_rkind
210
211 real (kind=rkind), parameter :: &
212 bup2s0t0 = 2.102898e-4_rkind, &
213 bup2s0t1 = -1.202016e-5_rkind, &
214 bup2s0t2 = 1.394680e-7_rkind, &
215 bup2s1t0 = -2.040237e-6_rkind, &
216 bup2s1t1 = 6.128773e-8_rkind, &
217 bup2s1t2 = 6.207323e-10_rkind
218
219 integer :: k_test, k_ref
220
221 err = 0
222
223 call mpas_pool_get_array(meshpool, 'maxLevelCell', maxlevelcell)
224 call mpas_pool_get_array(meshpool, 'refBottomDepth', refbottomdepth)
225
226 call mpas_pool_get_dimension(meshpool, 'nVertLevels', nvertlevels)
227
228 allocate(tracertemp(nvertlevels))
229 allocate(tracersalt(nvertlevels))
230
231! Jackett and McDougall
232 tmin = -2.0_rkind ! valid pot. temp. range
233 tmax = 40.0_rkind
234 smin = 0.0_rkind ! valid salinity, in psu
235 smax = 42.0_rkind
236
237! This function computes pressure in bars from depth in meters
238! using a mean density derived from depth-dependent global
239! average temperatures and salinities from Levitus 1994, and
240! integrating using hydrostatic balance.
241
242 allocate(prefeos(nvertlevels),p(nvertlevels),p2(nvertlevels))
243
244 allocate(sq(nvertlevels), tq(nvertlevels), sqr(nvertlevels), t2(nvertlevels), work1(nvertlevels), &
245 work2(nvertlevels), rho_s(nvertlevels), work3(nvertlevels), work4(nvertlevels), &
246 bulk_mod(nvertlevels), denomk(nvertlevels))
247
248 ! This could be put in the init routine.
249 ! Note I am using refBottomDepth, so pressure on top level does
250 ! not include SSH contribution. I am not sure if that matters, but
251 ! POP does it the same way.
252 depth = 0.5_rkind*refbottomdepth(1)
253 prefeos(1) = 0.059808_rkind*(exp(-0.025_rkind*depth) - 1.0_rkind) &
254 + 0.100766_rkind*depth + 2.28405e-7_rkind*depth**2
255 do k = 2,nvertlevels
256 depth = 0.5_rkind*(refbottomdepth(k)+refbottomdepth(k-1))
257 prefeos(k) = 0.059808_rkind*(exp(-0.025_rkind*depth) - 1.0_rkind) &
258 + 0.100766_rkind*depth + 2.28405e-7_rkind*depth**2
259 enddo
260
261 ! If k_displaced=0, in-situ density is returned (no displacement)
262 ! If k_displaced/=0, potential density is returned
263
264 ! if displacement_type = 'relative', potential density is calculated
265 ! referenced to level k + k_displaced
266 ! if displacement_type = 'absolute', potential density is calculated
267 ! referenced to level k_displaced for all k
268 ! NOTE: k_displaced = 0 or > nVertLevels is incompatible with 'absolute'
269 ! so abort if necessary
270 if (displacement_type == 'surfaceDisplaced') then
271 if(present(tracerssurfacelayervalue)) then
272 displacement_type_local = 'relative'
273 k_displaced_local = 0
274 else
275 call mpas_log_write( &
276 'tracersSurfaceLayerValue must be present when displacement_type is ' &
277 // '''surfaceDisplaced'' in JM EOS', &
278 mpas_log_crit)
279 endif
280 else
281 displacement_type_local = trim(displacement_type)
282 k_displaced_local = k_displaced
283 endif
284
285 if (displacement_type_local == 'absolute' .and. &
286 (k_displaced_local <= 0 .or. k_displaced_local > nvertlevels) ) then
287
288 call mpas_log_write('Abort: In equation_of_state_jm' // &
289 ' k_displaced must be between 1 and nVertLevels for ' // &
290 'displacement_type = absolute', mpas_log_crit)
291 endif
292
293 if (k_displaced_local == 0) then
294 do k=1,nvertlevels
295 p(k) = prefeos(k)
296 p2(k) = p(k)*p(k)
297 enddo
298 else ! k_displaced_local /= 0
299 do k=1,nvertlevels
300 if (displacement_type_local == 'relative') then
301 k_test = min(k + k_displaced_local, nvertlevels)
302 k_ref = max(k_test, 1)
303 else
304 k_test = min(k_displaced_local, nvertlevels)
305 k_ref = max(k_test, 1)
306 endif
307 p(k) = prefeos(k_ref)
308 p2(k) = p(k)*p(k)
309 enddo
310 endif
311
312 !$omp do schedule(runtime) private(k, DRDT0, DKDT, DRHODT, DRDS0, DKDS, DRHODS)
313 do icell=1,ncells
314 if (displacement_type == 'surfaceDisplaced') then
315 if(present(tracerssurfacelayervalue)) then
316 do k=1,nvertlevels
317 tracertemp(k) = tracerssurfacelayervalue(indext,icell)
318 tracersalt(k) = tracerssurfacelayervalue(indexs,icell)
319 enddo
320 else
321 call mpas_log_write( &
322 'tracersSurfaceLayerValue must be present in JM EOS call', &
323 mpas_log_crit)
324 endif
325 else
326 do k = 1, nvertlevels
327 tracertemp(k) = tracers(indext, k, icell)
328 tracersalt(k) = tracers(indexs, k, icell)
329 end do
330 endif
331
332 do k=1,maxlevelcell(icell)
333 sq(k) = max(min(tracersalt(k),smax),smin)
334 tq(k) = max(min(tracertemp(k),tmax),tmin)
335
336 sqr(k) = sqrt(sq(k))
337 t2(k) = tq(k)*tq(k)
338
339 !***
340 !*** first calculate surface (p=0) values from UNESCO eqns.
341 !***
342
343 work1(k) = uns1t0 + uns1t1*tq(k) + &
344 (uns1t2 + uns1t3*tq(k) + uns1t4*t2(k))*t2(k)
345 work2(k) = sqr(k)*(unsqt0 + unsqt1*tq(k) + unsqt2*t2(k))
346
347 rho_s(k) = unt1*tq(k) + (unt2 + unt3*tq(k) + (unt4 + unt5*tq(k))*t2(k))*t2(k) &
348 + (uns2t0*sq(k) + work1(k) + work2(k))*sq(k)
349
350 !***
351 !*** now calculate bulk modulus at pressure p from
352 !*** Jackett and McDougall formula
353 !***
354
355 work3(k) = bup0s1t0 + bup0s1t1*tq(k) + &
356 (bup0s1t2 + bup0s1t3*tq(k))*t2(k) + &
357 p(k) *(bup1s1t0 + bup1s1t1*tq(k) + bup1s1t2*t2(k)) + &
358 p2(k)*(bup2s1t0 + bup2s1t1*tq(k) + bup2s1t2*t2(k))
359 work4(k) = sqr(k)*(bup0sqt0 + bup0sqt1*tq(k) + bup0sqt2*t2(k) + &
360 bup1sqt0*p(k))
361
362 bulk_mod(k) = bup0s0t0 + bup0s0t1*tq(k) + &
363 (bup0s0t2 + bup0s0t3*tq(k) + bup0s0t4*t2(k))*t2(k) + &
364 p(k) *(bup1s0t0 + bup1s0t1*tq(k) + &
365 (bup1s0t2 + bup1s0t3*tq(k))*t2(k)) + &
366 p2(k)*(bup2s0t0 + bup2s0t1*tq(k) + bup2s0t2*t2(k)) + &
367 sq(k)*(work3(k) + work4(k))
368
369 denomk(k) = 1.0/(bulk_mod(k) - p(k))
370
371 density(k, icell) = (unt0 + rho_s(k))*bulk_mod(k)*denomk(k)
372
373 end do
374
375 if (present(thermalexpansioncoeff)) then
376 do k=1,maxlevelcell(icell)
377 drdt0 = unt1 + 2.0_rkind*unt2*tq(k) + &
378 (3.0_rkind*unt3 + 4.0_rkind*unt4*tq(k) + 5.0_rkind*unt5*t2(k))*t2(k) + &
379 (uns1t1 + 2.0_rkind*uns1t2*tq(k) + &
380 (3.0_rkind*uns1t3 + 4.0_rkind*uns1t4*tq(k))*t2(k) + &
381 (unsqt1 + 2.0_rkind*unsqt2*tq(k))*sqr(k) )*sq(k)
382
383 dkdt = bup0s0t1 + 2.0_rkind*bup0s0t2*tq(k) + &
384 (3.0_rkind*bup0s0t3 + 4.0_rkind*bup0s0t4*tq(k))*t2(k) + &
385 p(k) *(bup1s0t1 + 2.0_rkind*bup1s0t2*tq(k) + 3.0_rkind*bup1s0t3*t2(k)) + &
386 p2(k)*(bup2s0t1 + 2.0_rkind*bup2s0t2*tq(k)) + &
387 sq(k)*(bup0s1t1 + 2.0_rkind*bup0s1t2*tq(k) + 3.0_rkind*bup0s1t3*t2(k) + &
388 p(k) *(bup1s1t1 + 2.0_rkind*bup1s1t2*tq(k)) + &
389 p2(k) *(bup2s1t1 + 2.0_rkind*bup2s1t2*tq(k)) + &
390 sqr(k)*(bup0sqt1 + 2.0_rkind*bup0sqt2*tq(k)))
391
392 drhodt = (denomk(k)*(drdt0*bulk_mod(k) - &
393 p(k)*(unt0+rho_s(k))*dkdt*denomk(k)))
394
395 thermalexpansioncoeff(k,icell) = -drhodt/density(k,icell)
396 end do
397 end if
398
399 if (present(salinecontractioncoeff)) then
400 do k=1,maxlevelcell(icell)
401 drds0 = 2.0_rkind*uns2t0*sq(k) + work1(k) + 1.5_rkind*work2(k)
402 dkds = work3(k) + 1.5_rkind*work4(k)
403
404 drhods = denomk(k)*(drds0*bulk_mod(k) - &
405 p(k)*(unt0+rho_s(k))*dkds*denomk(k))
406
407 salinecontractioncoeff(k,icell) = drhods/density(k,icell)
408
409 end do
410
411 end if
412 end do
413 !$omp end do
414
415 deallocate(prefeos,p,p2)
416 deallocate(tracertemp)
417 deallocate(tracersalt)
418
419 deallocate(sq)
420 deallocate(tq)
421 deallocate(sqr)
422 deallocate(t2)
423 deallocate(work1)
424 deallocate(work2)
425 deallocate(rho_s)
426 deallocate(work3)
427 deallocate(work4)
428 deallocate(bulk_mod)
429 deallocate(denomk)
430
431 end subroutine ocn_equation_of_state_jm_density!}}}
