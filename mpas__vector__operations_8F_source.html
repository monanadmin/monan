114 real (kind=rkind), intent(in) :: p_1(3)

116 real (kind=rkind), intent(out) :: p_out(3)
117

633!***********************************************************************
634!
635! routine mpas_initialize_vectors
636!
651!-----------------------------------------------------------------------
652 subroutine mpas_initialize_vectors(meshPool)!{{{
653
654 implicit none
655
656 type (mpas_pool_type), intent(in) :: meshpool
657
658 integer :: iedge, icell, cell1, cell2
659 integer, pointer :: ncells, nedges
660 integer, dimension(:,:), pointer :: cellsonedge, verticesonedge, edgesoncell
661 real(kind=rkind), dimension(:), pointer :: xcell, ycell, zcell, xedge, yedge, zedge
662 real(kind=rkind), dimension(:,:), pointer :: localverticalunitvectors, edgenormalvectors
663 real(kind=rkind), dimension(:,:,:), pointer :: celltangentplane
664 real(kind=rkind), dimension(3) :: xhatplane, yhatplane, rhat
665 real(kind=rkind) :: normaldotrhat
666 logical, pointer :: on_a_sphere, is_periodic
667 real(kind=rkind), pointer :: x_period, y_period
668
669 call mpas_pool_get_config(meshpool, 'is_periodic', is_periodic)
670 call mpas_pool_get_config(meshpool, 'x_period', x_period)
671 call mpas_pool_get_config(meshpool, 'y_period', y_period)
672
673 call mpas_pool_get_config(meshpool, 'on_a_sphere', on_a_sphere)
674
675 call mpas_pool_get_array(meshpool, 'xCell', xcell)
676 call mpas_pool_get_array(meshpool, 'yCell', ycell)
677 call mpas_pool_get_array(meshpool, 'zCell', zcell)
678 call mpas_pool_get_array(meshpool, 'xEdge', xedge)
679 call mpas_pool_get_array(meshpool, 'yEdge', yedge)
680 call mpas_pool_get_array(meshpool, 'zEdge', zedge)
681 call mpas_pool_get_array(meshpool, 'cellsOnEdge', cellsonedge)
682 call mpas_pool_get_array(meshpool, 'verticesOnEdge', verticesonedge)
683 call mpas_pool_get_array(meshpool, 'edgesOnCell', edgesoncell)
684
685 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
686 call mpas_pool_get_dimension(meshpool, 'nEdges', nedges)
687
688 call mpas_pool_get_array(meshpool, 'localVerticalUnitVectors', localverticalunitvectors)
689 call mpas_pool_get_array(meshpool, 'edgeNormalVectors', edgenormalvectors)
690 call mpas_pool_get_array(meshpool, 'cellTangentPlane', celltangentplane)
691
692 ! init arrays
693 edgenormalvectors = 0
694 localverticalunitvectors = 0
695
696 ! loop over all cells to be solved on this block
697 do icell = 1, ncells
698 if (on_a_sphere) then
699 localverticalunitvectors(1,icell) = xcell(icell)
700 localverticalunitvectors(2,icell) = ycell(icell)
701 localverticalunitvectors(3,icell) = zcell(icell)
702 call mpas_unit_vec_in_r3(localverticalunitvectors(:,icell))
703 else ! on a plane
704 localverticalunitvectors(:,icell) = (/ 0., 0., 1. /)
705 end if
706 end do
707
708 ! Initialize normal unit vectors at each edge
709 ! These vectors point from cell to cell.
710 ! At boundaries, one cell does not exist, so it points from cell to edge or from edge to cell.
711 do iedge = 1,nedges
712 cell1 = cellsonedge(1,iedge)
713 cell2 = cellsonedge(2,iedge)
714
715 if (cell1 == ncells+1) then ! this is a boundary edge
716 ! the normal points from the edge location to the cell location
717 if (is_periodic) then
718 edgenormalvectors(1,iedge) = xcell(cell2) - mpas_fix_periodicity(xedge(iedge), xcell(cell2), x_period)
719 edgenormalvectors(2,iedge) = ycell(cell2) - mpas_fix_periodicity(yedge(iedge), ycell(cell2), y_period)
720 edgenormalvectors(3,iedge) = zcell(cell2) - zedge(iedge)
721 else
722 edgenormalvectors(1,iedge) = xcell(cell2) - xedge(iedge)
723 edgenormalvectors(2,iedge) = ycell(cell2) - yedge(iedge)
724 edgenormalvectors(3,iedge) = zcell(cell2) - zedge(iedge)
725 end if
726
727 else if (cell2 == ncells+1) then ! this is a boundary edge
728 ! the normal points from the cell location to the edge location
729 if (is_periodic) then
730 edgenormalvectors(1,iedge) = mpas_fix_periodicity(xedge(iedge), xcell(cell1), x_period) - xcell(cell1)
731 edgenormalvectors(2,iedge) = mpas_fix_periodicity(yedge(iedge), ycell(cell1), y_period) - ycell(cell1)
732 edgenormalvectors(3,iedge) = zedge(iedge) - zcell(cell1)
733 else
734 edgenormalvectors(1,iedge) = xedge(iedge) - xcell(cell1)
735 edgenormalvectors(2,iedge) = yedge(iedge) - ycell(cell1)
736 edgenormalvectors(3,iedge) = zedge(iedge) - zcell(cell1)
737 end if
738
739 else ! this is not a boundary cell
740 ! the normal points from the cell 1 to cell2
741 ! mrp problem: on periodic domains, vectors on edges of domain point the wrong way.
742 if (is_periodic) then
743 edgenormalvectors(1,iedge) = mpas_fix_periodicity(xcell(cell2), xcell(cell1), x_period) - xcell(cell1)
744 edgenormalvectors(2,iedge) = mpas_fix_periodicity(ycell(cell2), ycell(cell1), y_period) - ycell(cell1)
745 edgenormalvectors(3,iedge) = zcell(cell2) - zcell(cell1)
746 else
747 edgenormalvectors(1,iedge) = xcell(cell2) - xcell(cell1)
748 edgenormalvectors(2,iedge) = ycell(cell2) - ycell(cell1)
749 edgenormalvectors(3,iedge) = zcell(cell2) - zcell(cell1)
750 end if
751
752 end if
753 call mpas_unit_vec_in_r3(edgenormalvectors(:,iedge))
754 end do
755
756 do icell=1,ncells
757 iedge = edgesoncell(1,icell)
758 ! xHat and yHat are a local basis in the plane of the horizontal cell
759 ! we arbitrarily choose xHat to point toward the first edge
760 rhat = localverticalunitvectors(:,icell)
761 normaldotrhat = sum(edgenormalvectors(:,iedge)*rhat)
762 xhatplane = edgenormalvectors(:,iedge) - normaldotrhat*rhat
763 call mpas_unit_vec_in_r3(xhatplane)
764
765 call mpas_cross_product_in_r3(rhat, xhatplane, yhatplane)
766 call mpas_unit_vec_in_r3(yhatplane) ! just to be sure...
767 celltangentplane(:,1,icell) = xhatplane
768 celltangentplane(:,2,icell) = yhatplane
769 end do
770
771 end subroutine mpas_initialize_vectors!}}}
