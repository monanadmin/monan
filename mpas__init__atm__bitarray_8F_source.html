153 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
154 ! Name: bitarray_merge
155 !
156 ! Purpose: The first bitarray argument, b1, is set to the union of the .TRUE.
157 ! bits in b1 and b2. That is, after returning, a bit x in b1 is set if
158 ! either x was set in b1 or x was set in b2. Thus, b1 AND b2 MUST BE BIT
159 ! ARRAYS OF THE SAME DIMENSIONS.
160 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
161 subroutine bitarray_merge(b1, b2)
162
163 use mpas_derived_types, only : mpas_log_crit
164 use mpas_log, only : mpas_log_write
165
166 implicit none
167
168 ! Arguments
169 type (bitarray_type), intent(inout) :: b1, b2
170
171 ! Local variables
172 integer :: i, j
173
174 if (b1%x_int_dim /= b2%x_int_dim .or. b1%y_int_dim /= b2%y_int_dim) then
175 call mpas_log_write('In bitarray_merge(), b1 and b2 have different dimensions.', messagetype=mpas_log_crit)
176 end if
177
178 do i=1,b1%x_int_dim
179 do j=1,b1%y_int_dim
180 b1%iarray(i,j) = ior(b1%iarray(i,j), b2%iarray(i,j))
181 end do
182 end do
183
184 end subroutine bitarray_merge
