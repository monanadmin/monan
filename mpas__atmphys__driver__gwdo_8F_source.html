126!=================================================================================================================
127 subroutine gwdo_from_mpas(configs,mesh,sfc_input,diag_physics,tend_physics,its,ite)
128!=================================================================================================================
129
130!input arguments:
131 type(mpas_pool_type),intent(in):: configs
132 type(mpas_pool_type),intent(in):: mesh
133 type(mpas_pool_type),intent(in):: sfc_input
134 type(mpas_pool_type),intent(in):: diag_physics
135 type(mpas_pool_type),intent(in):: tend_physics
136
137 integer,intent(in):: its,ite
138
139!local variables:
140 integer:: i,k,j
141
142!local pointers:
143 integer,dimension(:),pointer:: kpbl
144 real(kind=rkind),pointer:: len_disp
145 real(kind=rkind),dimension(:),pointer :: meshdensity
146 real(kind=rkind),dimension(:),pointer :: oa1,oa2,oa3,oa4,ol1,ol2,ol3,ol4,con,var2d
147 real(kind=rkind),dimension(:),pointer :: dusfcg,dvsfcg
148 real(kind=rkind),dimension(:,:),pointer:: dtaux3d,dtauy3d,rublten,rvblten
149
150!-----------------------------------------------------------------------------------------------------------------
151
152 call mpas_pool_get_config(configs,'config_len_disp',len_disp)
153 call mpas_pool_get_array(mesh,'meshDensity',meshdensity)
154
155 call mpas_pool_get_array(sfc_input,'oa1' ,oa1 )
156 call mpas_pool_get_array(sfc_input,'oa2' ,oa2 )
157 call mpas_pool_get_array(sfc_input,'oa3' ,oa3 )
158 call mpas_pool_get_array(sfc_input,'oa4' ,oa4 )
159 call mpas_pool_get_array(sfc_input,'ol1' ,ol1 )
160 call mpas_pool_get_array(sfc_input,'ol2' ,ol2 )
161 call mpas_pool_get_array(sfc_input,'ol3' ,ol3 )
162 call mpas_pool_get_array(sfc_input,'ol4' ,ol4 )
163 call mpas_pool_get_array(sfc_input,'con' ,con )
164 call mpas_pool_get_array(sfc_input,'var2d',var2d)
165
166 call mpas_pool_get_array(diag_physics,'kpbl' ,kpbl )
167 call mpas_pool_get_array(diag_physics,'dusfcg' ,dusfcg )
168 call mpas_pool_get_array(diag_physics,'dvsfcg' ,dvsfcg )
169 call mpas_pool_get_array(diag_physics,'dtaux3d' ,dtaux3d )
170 call mpas_pool_get_array(diag_physics,'dtauy3d' ,dtauy3d )
171 call mpas_pool_get_array(tend_physics,'rublten' ,rublten )
172 call mpas_pool_get_array(tend_physics,'rvblten' ,rvblten )
173
174 do j = jts,jte
175 do i = its,ite
176
177 sina_p(i,j) = 0._rkind
178 cosa_p(i,j) = 1._rkind
179
180 var2d_p(i,j) = var2d(i)
181 con_p(i,j) = con(i)
182 oa1_p(i,j) = oa1(i)
183 oa2_p(i,j) = oa2(i)
184 oa3_p(i,j) = oa3(i)
185 oa4_p(i,j) = oa4(i)
186 ol1_p(i,j) = ol1(i)
187 ol2_p(i,j) = ol2(i)
188 ol3_p(i,j) = ol3(i)
189 ol4_p(i,j) = ol4(i)
190 enddo
191 enddo
192
193 do j = jts,jte
194 do i = its,ite
195 dx_p(i,j) = len_disp / meshdensity(i)**0.25
196 kpbl_p(i,j) = kpbl(i)
197 dusfcg_p(i,j) = dusfcg(i)
198 dvsfcg_p(i,j) = dvsfcg(i)
199 enddo
200 enddo
201
202 do j = jts,jte
203 do k = kts,kte
204 do i = its,ite
205 dtaux3d_p(i,k,j) = dtaux3d(k,i)
206 dtauy3d_p(i,k,j) = dtauy3d(k,i)
207 rublten_p(i,k,j) = rublten(k,i)
208 rvblten_p(i,k,j) = rvblten(k,i)
209 enddo
210 enddo
211 enddo
212
213 end subroutine gwdo_from_mpas

215!=================================================================================================================
216 subroutine gwdo_to_mpas(diag_physics,tend_physics,its,ite)
217!=================================================================================================================
218
219!input arguments:
220 integer,intent(in):: its,ite
221
222!inout arguments:
223 type(mpas_pool_type),intent(inout):: diag_physics
224 type(mpas_pool_type),intent(inout):: tend_physics
225
226!local variables:
227 integer:: i,k,j
228
229!local pointers:
230 real(kind=rkind),dimension(:),pointer :: dusfcg,dvsfcg
231 real(kind=rkind),dimension(:,:),pointer:: dtaux3d,dtauy3d,rubldiff,rvbldiff,rublten,rvblten
232
233!-----------------------------------------------------------------------------------------------------------------
234
235 call mpas_pool_get_array(diag_physics,'dusfcg' ,dusfcg )
236 call mpas_pool_get_array(diag_physics,'dvsfcg' ,dvsfcg )
237 call mpas_pool_get_array(diag_physics,'dtaux3d' ,dtaux3d )
238 call mpas_pool_get_array(diag_physics,'dtauy3d' ,dtauy3d )
239 call mpas_pool_get_array(diag_physics,'rubldiff',rubldiff)
240 call mpas_pool_get_array(diag_physics,'rvbldiff',rvbldiff)
241 call mpas_pool_get_array(tend_physics,'rublten' ,rublten )
242 call mpas_pool_get_array(tend_physics,'rvblten' ,rvblten )
243
244 do j = jts,jte
245 do i = its,ite
246 dusfcg(i) = dusfcg_p(i,j)
247 dvsfcg(i) = dvsfcg_p(i,j)
248 enddo
249 enddo
250
251 do j = jts,jte
252 do k = kts,kte
253 do i = its,ite
254 dtaux3d(k,i) = dtaux3d_p(i,k,j)
255 dtauy3d(k,i) = dtauy3d_p(i,k,j)
256 rubldiff(k,i) = rublten_p(i,k,j)-rublten(k,i)
257 rvbldiff(k,i) = rvblten_p(i,k,j)-rvblten(k,i)
258 rublten(k,i) = rublten_p(i,k,j)
259 rvblten(k,i) = rvblten_p(i,k,j)
260 enddo
261 enddo
262 enddo
263
264 end subroutine gwdo_to_mpas

266!=================================================================================================================
267 subroutine driver_gwdo(itimestep,configs,mesh,sfc_input,diag_physics,tend_physics,its,ite)
268!=================================================================================================================
269
270!input arguments:
271 type(mpas_pool_type),intent(in):: configs
272 type(mpas_pool_type),intent(in):: mesh
273 type(mpas_pool_type),intent(in):: sfc_input
274
275 integer,intent(in):: its,ite
276 integer,intent(in):: itimestep
277
278!inout arguments:
279 type(mpas_pool_type),intent(inout):: diag_physics
280 type(mpas_pool_type),intent(inout):: tend_physics
281
282!local variables:
283 character(len=StrKIND),pointer:: gwdo_scheme
284
285 integer:: i,icell,iedge
286 real(kind=rkind),dimension(:),allocatable:: dx_max
287
288!-----------------------------------------------------------------------------------------------------------------
289!call mpas_log_write('')
290!call mpas_log_write('--- enter subroutine driver_gwdo:')
291
292 call mpas_pool_get_config(configs,'config_gwdo_scheme',gwdo_scheme)
293
294!copy MPAS arrays to local arrays:
295 call gwdo_from_mpas(configs,mesh,sfc_input,diag_physics,tend_physics,its,ite)
296
297 gwdo_select: select case (trim(gwdo_scheme))
298
299 case("bl_ysu_gwdo")
300 call mpas_timer_start('bl_ysu_gwdo')
301 call gwdo ( &
302 p3d = pres_hydd_p , p3di = pres2_hydd_p , pi3d = pi_p , &
303 u3d = u_p , v3d = v_p , t3d = t_p , &
304 qv3d = qv_p , z = z_p , rublten = rublten_p , &
305 rvblten = rvblten_p , dtaux3d = dtaux3d_p , dtauy3d = dtauy3d_p , &
306 dusfcg = dusfcg_p , dvsfcg = dvsfcg_p , kpbl2d = kpbl_p , &
307 itimestep = itimestep , dt = dt_pbl , dx = dx_p , &
308 cp = cp , g = gravity , rd = r_d , &
309 rv = r_v , ep1 = ep_1 , pi = pii , &
310 var2d = var2d_p , oc12d = con_p , oa2d1 = oa1_p , &
311 oa2d2 = oa2_p , oa2d3 = oa3_p , oa2d4 = oa4_p , &
312 ol2d1 = ol1_p , ol2d2 = ol2_p , ol2d3 = ol3_p , &
313 ol2d4 = ol4_p , sina = sina_p , cosa = cosa_p , &
314 ids = ids , ide = ide , jds = jds , jde = jde , kds = kds , kde = kde , &
315 ims = ims , ime = ime , jms = jms , jme = jme , kms = kms , kme = kme , &
316 its = its , ite = ite , jts = jts , jte = jte , kts = kts , kte = kte &
317 )
318 call mpas_timer_stop('bl_ysu_gwdo')
319
320 case default
321
322 end select gwdo_select
323
324!copy local arrays to MPAS grid:
325 call gwdo_to_mpas(diag_physics,tend_physics,its,ite)
326
327!call mpas_log_write('--- end subroutine driver_gwdo.')
328!call mpas_log_write('')
329
330 end subroutine driver_gwdo
