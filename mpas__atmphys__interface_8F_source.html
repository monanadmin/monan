24!Interface for conversion between variables used in the MPAS dynamical core and variables needed in the
25!physics parameterizations.
26!Laura D. Fowler (send comments to laura@ucar.edu).
27!2013-05-01.
28!
29! subroutines in mpas_atmphys_interface_nhyd:
30! -------------------------------------------
31! allocate_forall_physics : allocation of all meteorological variables needed in the physics.
32! deallocate_forall_physics: deallocation of all meteorological variables needed in the physics.
33! MPAS_to_physics : conversion of input "MPAS" variables to "physics" variables.
34! microphysics_from_MPAS : initialize local arrays needed in cloud microphysics schemes.
35! microphysics_to_MPAS : copy local arrays needed in cloud microphysics schemesto MPAS arrays.
36!
37! add-ons and modifications to sourcecode:
38! ----------------------------------------
39! * in subroutine MPAS_to_physics, moved the calculation of the local arrays qv_p,qc_p, and qr_p above the
40! calculation of th_p so that we do not need to use the pointer qv.
41! * in subroutine microphysics_from_MPAS, moved the calculation of the local arrays qv_p,qc_p, and qr_p above
42! the calculation of th_p so that we do not need to use the pointer qv.
43! * in subroutine microphysics_to_MPAS, since microphysics schemes update the temperature and water vapor
44! mixing ratio, we first update the total pressure and exner functions. Then, we update the modified
45! potential temperature and calculate the diabatic tendency due to cloud microphysics processes.
46! Laura D. Fowler (laura@ucar.edu) / 2013-11-07.
47! * replaced the variable g (that originally pointed to gravity) with gravity, for simplicity.
48! Laura D. Fowler (laura@ucar.edu) / 2014-03-21.
49! * throughout the sourcecode, replaced all "var_struct" defined arrays by local pointers.
50! Laura D. Fowler (laura@ucar.edu) / 2014-04-22.
51! * modified sourcecode to use pools.
52! Laura D. Fowler (laura@ucar.edu) / 2014-05-15.
53! * added calculation of the surface pressure tendency. Moved the calculation of znu_p below the calculation
54! of the surface pressure to avoid dividing by zero if the surface pressure is not output in the init file.
55! Laura D. Fowler (birch.mmm.ucar.ecu) / 2014-06-23.
56! * renamed module mpas_atmphys_interface_nhyd to mpas_atmphys_interface.
57! Laura D. Fowler (laura@ucar.edu) / 2014-09-19.
58! * in subroutine microphysics_to_MPAS, reverted the calculation of cloud microphysics tendency rt_diabatic_tend,
59! and update of the state variables to git hash identifier 7a66f273e182f4. This change reflects the fact that
60! we want to compute rt_diabatic_tend at constant volume.
61! Laura D. Fowler (laura@ucar.edu) / 2014-01-015.
62! * added the initialization of local variables needed in the parameterization of the MYNN surface layer scheme
63! and planetary boundary layer scheme from WRF 3.6.1.
64! Laura D. Fowler (laura@ucar.edu) / 2016-04-11.
65! * corrected the calculation of the surface pressure, mainly extrapolation of the air density to the surface.
66! Laura D. Fowler (laura@ucar.edu) / 2016-04-25.

199!=================================================================================================================
200 subroutine mpas_to_physics(configs,mesh,state,time_lev,diag,diag_physics,its,ite)
201!=================================================================================================================
202
203!input variables:
204 type(mpas_pool_type),intent(in):: configs
205 type(mpas_pool_type),intent(in):: mesh
206 type(mpas_pool_type),intent(in):: state
207 type(mpas_pool_type),intent(in):: diag
208
209 integer,intent(in):: its,ite
210 integer,intent(in):: time_lev
211
212!inout variables:
213 type(mpas_pool_type),intent(inout):: diag_physics
214
215!local pointers:
216 character(len=StrKIND),pointer:: pbl_scheme
217
218 integer,pointer:: index_qv,index_qc,index_qr,index_qi,index_qs,index_qg
219 integer,pointer:: index_ni
220
221 real(kind=rkind),dimension(:),pointer :: latcell,loncell
222 real(kind=rkind),dimension(:),pointer :: fzm,fzp,rdzw
223 real(kind=rkind),dimension(:),pointer :: surface_pressure,plrad
224 real(kind=rkind),dimension(:,:),pointer :: zgrid
225 real(kind=rkind),dimension(:,:),pointer :: zz,exner,pressure_b,rtheta_p,rtheta_b
226 real(kind=rkind),dimension(:,:),pointer :: rho_zz,theta_m,pressure_p,u,v,w
227 real(kind=rkind),dimension(:,:),pointer :: qv,qc,qr,qi,qs,qg
228 real(kind=rkind),dimension(:,:),pointer :: ni
229 real(kind=rkind),dimension(:,:,:),pointer:: scalars
230
231!local variables:
232 integer:: i,k,j
233
234 real(kind=rkind):: z0,z1,z2,w1,w2
235 real(kind=rkind):: rho_a,rho1,rho2,tem1,tem2
236
237!-----------------------------------------------------------------------------------------------------------------
238!call mpas_log_write('')
239!call mpas_log_write('--- enter subroutine MPAS_to_phys:')
240!call mpas_log_write('ims=$i ime=$i',intArgs=(/ims,ime/))
241!call mpas_log_write('jms=$i jme=$i',intArgs=(/jms,jme/))
242!call mpas_log_write('kms=$i kme=$i',intArgs=(/kms,kme/))
243!call mpas_log_write('')
244!call mpas_log_write('its=$i ite=$i',intArgs=(/its,ite/))
245!call mpas_log_write('jts=$i jte=$i',intArgs=(/jts,jte/))
246!call mpas_log_write('kts=$i kte=$i',intArgs=(/kts,kte/))
247
248!initialization:
249 call mpas_pool_get_config(configs,'config_pbl_scheme',pbl_scheme)
250
251 call mpas_pool_get_array(mesh,'latCell',latcell)
252 call mpas_pool_get_array(mesh,'lonCell',loncell)
253 call mpas_pool_get_array(mesh,'fzm' ,fzm )
254 call mpas_pool_get_array(mesh,'fzp' ,fzp )
255 call mpas_pool_get_array(mesh,'rdzw' ,rdzw )
256 call mpas_pool_get_array(mesh,'zgrid' ,zgrid )
257 call mpas_pool_get_array(mesh,'zz' ,zz )
258
259 call mpas_pool_get_array(diag,'surface_pressure' ,surface_pressure)
260 call mpas_pool_get_array(diag,'exner' ,exner )
261 call mpas_pool_get_array(diag,'pressure_base' ,pressure_b )
262 call mpas_pool_get_array(diag,'pressure_p' ,pressure_p )
263 call mpas_pool_get_array(diag,'rtheta_base' ,rtheta_b )
264 call mpas_pool_get_array(diag,'rtheta_p' ,rtheta_p )
265 call mpas_pool_get_array(diag,'uReconstructZonal' ,u )
266 call mpas_pool_get_array(diag,'uReconstructMeridional',v )
267
268 call mpas_pool_get_array(state,'rho_zz' ,rho_zz ,time_lev)
269 call mpas_pool_get_array(state,'theta_m',theta_m,time_lev)
270 call mpas_pool_get_array(state,'w' ,w ,time_lev)
271
272 call mpas_pool_get_dimension(state,'index_qv',index_qv)
273 call mpas_pool_get_dimension(state,'index_qc',index_qc)
274 call mpas_pool_get_dimension(state,'index_qr',index_qr)
275 call mpas_pool_get_dimension(state,'index_qi',index_qi)
276 call mpas_pool_get_dimension(state,'index_qs',index_qs)
277 call mpas_pool_get_dimension(state,'index_qg',index_qg)
278
279 call mpas_pool_get_array(state,'scalars',scalars,time_lev)
280 qv => scalars(index_qv,:,:)
281 qc => scalars(index_qc,:,:)
282 qr => scalars(index_qr,:,:)
283 qi => scalars(index_qi,:,:)
284 qs => scalars(index_qs,:,:)
285 qg => scalars(index_qg,:,:)
286
287 call mpas_pool_get_array(diag_physics,'plrad',plrad)
288
289 do j = jts, jte
290 do k = kts, kte
291 do i = its, ite
292
293 !water vapor and moist arrays:
294 qv_p(i,k,j) = max(0.,qv(k,i))
295 qc_p(i,k,j) = max(0.,qc(k,i))
296 qr_p(i,k,j) = max(0.,qr(k,i))
297 qi_p(i,k,j) = max(0.,qi(k,i))
298 qs_p(i,k,j) = max(0.,qs(k,i))
299 qg_p(i,k,j) = max(0.,qg(k,i))
300
301 !arrays located at theta points:
302 u_p(i,k,j) = u(k,i)
303 v_p(i,k,j) = v(k,i)
304
305 zz_p(i,k,j) = zz(k,i)
306 rho_p(i,k,j) = zz(k,i) * rho_zz(k,i)
307 rho_p(i,k,j) = rho_p(i,k,j)*(1._rkind + qv_p(i,k,j))
308 th_p(i,k,j) = theta_m(k,i) / (1._rkind + r_v/r_d * qv_p(i,k,j))
309 t_p(i,k,j) = th_p(i,k,j)*exner(k,i)
310
311 pi_p(i,k,j) = exner(k,i)
312 pres_p(i,k,j) = pressure_p(k,i) + pressure_b(k,i)
313
314 zmid_p(i,k,j) = 0.5*(zgrid(k+1,i)+zgrid(k,i))
315 dz_p(i,k,j) = zgrid(k+1,i)-zgrid(k,i)
316
317 enddo
318 enddo
319 enddo
320
321 pbl_select: select case (trim(pbl_scheme))
322 case("bl_mynn")
323 call mpas_pool_get_dimension(state,'index_ni',index_ni)
324 ni => scalars(index_ni,:,:)
325
326 do j = jts,jte
327 do k = kts,kte
328 do i = its,ite
329 ni_p(i,k,j) = max(0.,ni(k,i))
330 enddo
331 enddo
332 enddo
333
334 case default
335
336 end select pbl_select
337
338!calculation of the surface pressure using hydrostatic assumption down to the surface::
339 do j = jts,jte
340 do i = its,ite
341 tem1 = zgrid(2,i)-zgrid(1,i)
342 tem2 = zgrid(3,i)-zgrid(2,i)
343 rho1 = rho_zz(1,i) * zz(1,i) * (1. + qv_p(i,1,j))
344 rho2 = rho_zz(2,i) * zz(2,i) * (1. + qv_p(i,2,j))
345! surface_pressure(i) = 0.5*gravity*(zgrid(2,i)-zgrid(1,i)) &
346! * (rho1 + 0.5*(rho2-rho1)*tem1/(tem1+tem2))
347 surface_pressure(i) = 0.5*gravity*(zgrid(2,i)-zgrid(1,i)) &
348 * (rho1 - 0.5*(rho2-rho1)*tem1/(tem1+tem2))
349 surface_pressure(i) = surface_pressure(i) + pressure_p(1,i) + pressure_b(1,i)
350 enddo
351
352 do k = kts,kte
353 do i = its,ite
354 znu_p(i,k,j) = pres_p(i,k,j) / surface_pressure(i)
355 enddo
356 enddo
357 enddo
358
359!arrays located at w points:
360 do j = jts, jte
361 do k = kts,kte+1
362 do i = its,ite
363 w_p(i,k,j) = w(k,i)
364 z_p(i,k,j) = zgrid(k,i)
365 enddo
366 enddo
367 enddo
368
369!check that the pressure in the layer above the surface is greater than that in the layer
370!above it:
371 do j = jts,jte
372 do i = its,ite
373 if(pres_p(i,1,j) .le. pres_p(i,2,j)) then
374 call mpas_log_write('')
375 call mpas_log_write('--- subroutine MPAS_to_phys - pressure(1) < pressure(2):')
376 call mpas_log_write('i =$i', intargs=(/i/))
377 call mpas_log_write('latCell=$r', realargs=(/latcell(i)/degrad/))
378 call mpas_log_write('lonCell=$r', realargs=(/loncell(i)/degrad/))
379 do k = kts,kte
380 call mpas_log_write('$i $i $i $r $r $r $r $r $r $r $r', intargs=(/j,i,k/),&
381 realargs=(/dz_p(i,k,j),pressure_b(k,i),pressure_p(k,i),pres_p(i,k,j), &
382 rho_p(i,k,j),th_p(i,k,j),t_p(i,k,j),qv_p(i,k,j)/))
383 enddo
384! call mpas_log_write('pressure increasing with height', messageType=MPAS_LOG_CRIT)
385 endif
386 enddo
387 enddo
388
389!interpolation of pressure and temperature from theta points to w points:
390 do j = jts,jte
391 do k = kts+1,kte
392 do i = its,ite
393 tem1 = 1./(zgrid(k+1,i)-zgrid(k-1,i))
394 fzm_p(i,k,j) = (zgrid(k,i)-zgrid(k-1,i)) * tem1
395 fzp_p(i,k,j) = (zgrid(k+1,i)-zgrid(k,i)) * tem1
396 t2_p(i,k,j) = fzm_p(i,k,j)*t_p(i,k,j) + fzp_p(i,k,j)*t_p(i,k-1,j)
397 pres2_p(i,k,j) = fzm_p(i,k,j)*pres_p(i,k,j) + fzp_p(i,k,j)*pres_p(i,k-1,j)
398 enddo
399 enddo
400 enddo
401
402!interpolation of pressure and temperature from theta points to the top-of-the-model: follows
403!the calculation of the top-of-the-model pressure and temperature in WRF (subroutine phy_prep
404!in ./dyn_em/module_big_step_utilities.F):
405 k = kte+1
406 do j = jts,jte
407 do i = its,ite
408 z0 = zgrid(k,i)
409 z1 = 0.5*(zgrid(k,i)+zgrid(k-1,i))
410 z2 = 0.5*(zgrid(k-1,i)+zgrid(k-2,i))
411 w1 = (z0-z2)/(z1-z2)
412 w2 = 1.-w1
413 t2_p(i,k,j) = w1*t_p(i,k-1,j) + w2*t_p(i,k-2,j)
414 !use log of pressure to avoid occurrences of negative top-of-the-model pressure.
415 pres2_p(i,k,j) = exp(w1*log(pres_p(i,k-1,j))+w2*log(pres_p(i,k-2,j)))
416 enddo
417 enddo
418
419!ldf (2012-06-22): recalculates the pressure at the surface as an extrapolation of the
420!pressures in the 2 layers above the surface, as was originally done:
421 k = kts
422 do j = jts,jte
423 do i = its,ite
424 z0 = zgrid(k,i)
425 z1 = 0.5*(zgrid(k,i)+zgrid(k+1,i))
426 z2 = 0.5*(zgrid(k+1,i)+zgrid(k+2,i))
427 w1 = (z0-z2)/(z1-z2)
428 w2 = 1.-w1
429 t2_p(i,k,j) = w1*t_p(i,k,j)+w2*t_p(i,k+1,j)
430 pres2_p(i,k,j) = w1*pres_p(i,k,j)+w2*pres_p(i,k+1,j)
431! psfc_p(i,j) = pres2_p(i,k,j)
432 psfc_p(i,j) = surface_pressure(i)
433 enddo
434 enddo
435
436!calculation of the hydrostatic pressure:
437 do j = jts,jte
438 do i = its,ite
439 !pressure at w-points:
440 k = kte+1
441 pres2_hyd_p(i,k,j) = pres2_p(i,k,j)
442 pres2_hydd_p(i,k,j) = pres2_p(i,k,j)
443 do k = kte,1,-1
444 rho_a = rho_p(i,k,j) / (1.+qv_p(i,k,j))
445 pres2_hyd_p(i,k,j) = pres2_hyd_p(i,k+1,j) + gravity*rho_p(i,k,j)*dz_p(i,k,j)
446 pres2_hydd_p(i,k,j) = pres2_hydd_p(i,k+1,j) + gravity*rho_a*dz_p(i,k,j)
447 enddo
448 !pressure at theta-points:
449 do k = kte,1,-1
450 pres_hyd_p(i,k,j) = 0.5*(pres2_hyd_p(i,k+1,j)+pres2_hyd_p(i,k,j))
451 pres_hydd_p(i,k,j) = 0.5*(pres2_hydd_p(i,k+1,j)+pres2_hydd_p(i,k,j))
452 enddo
453 !surface pressure:
454 psfc_hyd_p(i,j) = pres2_hyd_p(i,1,j)
455 psfc_hydd_p(i,j) = pres2_hydd_p(i,1,j)
456 !znu:
457 do k = kte,1,-1
458 znu_hyd_p(i,k,j) = pres_hyd_p(i,k,j) / psfc_hyd_p(i,j)
459 enddo
460 enddo
461 enddo
462
463!save the model-top pressure:
464 do j = jts,jte
465 do i = its,ite
466 plrad(i) = pres2_p(i,kte+1,j)
467 enddo
468 enddo
469
470 end subroutine mpas_to_physics

588 do j = jts,jte

593 enddo

615!=================================================================================================================
616 subroutine microphysics_to_mpas(configs,mesh,state,time_lev,diag,diag_physics,tend,itimestep,its,ite)
617!=================================================================================================================
618
619!input variables:
620 type(mpas_pool_type),intent(in):: configs
621 type(mpas_pool_type),intent(in):: mesh
622
623 integer,intent(in):: itimestep,time_lev
624 integer,intent(in):: its,ite
625
626!output variables:
627 type(mpas_pool_type),intent(inout):: state
628 type(mpas_pool_type),intent(inout):: diag
629 type(mpas_pool_type),intent(inout):: tend
630 type(mpas_pool_type),intent(inout):: diag_physics
631
632!local pointers:
633 character(len=StrKIND),pointer:: microp_scheme
634 integer,pointer:: index_qv,index_qc,index_qr,index_qi,index_qs,index_qg
635 integer,pointer:: index_ni,index_nr
636 real(kind=rkind),dimension(:),pointer :: surface_pressure,tend_sfc_pressure
637 real(kind=rkind),dimension(:,:),pointer :: zgrid
638 real(kind=rkind),dimension(:,:),pointer :: zz,exner,exner_b,pressure_b,rtheta_p,rtheta_b
639 real(kind=rkind),dimension(:,:),pointer :: rho_zz,theta_m,pressure_p
640 real(kind=rkind),dimension(:,:),pointer :: rt_diabatic_tend
641 real(kind=rkind),dimension(:,:),pointer :: qv,qc,qr,qi,qs,qg
642 real(kind=rkind),dimension(:,:),pointer :: ni,nr
643 real(kind=rkind),dimension(:,:),pointer :: rainprod,evapprod
644 real(kind=rkind),dimension(:,:),pointer :: re_cloud,re_ice,re_snow
645 real(kind=rkind),dimension(:,:,:),pointer:: scalars
646
647!local variables:
648 integer:: icount
649 integer:: i,k,j
650 real(kind=rkind):: rho1,rho2,tem1,tem2
651
652!-----------------------------------------------------------------------------------------------------------------
653
654 call mpas_pool_get_config(configs,'config_microp_scheme',microp_scheme)
655
656 call mpas_pool_get_array(mesh,'zz' ,zz )
657 call mpas_pool_get_array(mesh,'zgrid',zgrid)
658
659 call mpas_pool_get_array(diag,'exner' ,exner )
660 call mpas_pool_get_array(diag,'exner_base' ,exner_b )
661 call mpas_pool_get_array(diag,'pressure_base' ,pressure_b )
662 call mpas_pool_get_array(diag,'pressure_p' ,pressure_p )
663 call mpas_pool_get_array(diag,'rtheta_base' ,rtheta_b )
664 call mpas_pool_get_array(diag,'rtheta_p' ,rtheta_p )
665 call mpas_pool_get_array(diag,'surface_pressure',surface_pressure)
666
667 call mpas_pool_get_array(diag_physics,'rainprod',rainprod)
668 call mpas_pool_get_array(diag_physics,'evapprod',evapprod)
669 call mpas_pool_get_array(diag_physics,'re_cloud',re_cloud)
670 call mpas_pool_get_array(diag_physics,'re_ice' ,re_ice )
671 call mpas_pool_get_array(diag_physics,'re_snow' ,re_snow )
672
673 call mpas_pool_get_array(tend,'tend_sfc_pressure',tend_sfc_pressure)
674
675 call mpas_pool_get_array(state,'rho_zz' ,rho_zz ,time_lev)
676 call mpas_pool_get_array(state,'theta_m',theta_m,time_lev)
677
678 call mpas_pool_get_dimension(state,'index_qv' ,index_qv )
679 call mpas_pool_get_dimension(state,'index_qc' ,index_qc )
680 call mpas_pool_get_dimension(state,'index_qr' ,index_qr )
681 call mpas_pool_get_dimension(state,'index_qi' ,index_qi )
682 call mpas_pool_get_dimension(state,'index_qs' ,index_qs )
683 call mpas_pool_get_dimension(state,'index_qg' ,index_qg )
684 call mpas_pool_get_dimension(state,'index_ni' ,index_ni )
685 call mpas_pool_get_dimension(state,'index_nr' ,index_nr )
686
687 call mpas_pool_get_array(state,'scalars',scalars,time_lev)
688 qv => scalars(index_qv,:,:)
689 qc => scalars(index_qc,:,:)
690 qr => scalars(index_qr,:,:)
691
692 call mpas_pool_get_array(tend,'rt_diabatic_tend',rt_diabatic_tend)
693
694!update variables needed in the dynamical core:
695 do j = jts,jte
696 do k = kts,kte
697 do i = its,ite
698 qv(k,i) = qv_p(i,k,j)
699 qc(k,i) = qc_p(i,k,j)
700 qr(k,i) = qr_p(i,k,j)
701
702 !potential temperature and diabatic forcing:
703 rt_diabatic_tend(k,i) = theta_m(k,i)
704 theta_m(k,i) = th_p(i,k,j) * (1. + r_v/r_d * qv_p(i,k,j))
705 rt_diabatic_tend(k,i) = (theta_m(k,i) - rt_diabatic_tend(k,i)) / dt_dyn
706
707 !density-weighted perturbation potential temperature:
708 rtheta_p(k,i) = rho_zz(k,i) * theta_m(k,i) - rtheta_b(k,i)
709
710 !exner function:
711 exner(k,i) = (zz(k,i)*(r_d/p0)*(rtheta_p(k,i)+rtheta_b(k,i)))**rcv
712
713 !pertubation pressure:
714 pressure_p(k,i) = zz(k,i)*r_d*(exner(k,i)*rtheta_p(k,i) &
715 + (exner(k,i)-exner_b(k,i))*rtheta_b(k,i))
716
717 enddo
718 enddo
719 enddo
720
721!updates the surface pressure and calculates the surface pressure tendency:
722 do j = jts,jte
723 do i = its,ite
724 tem1 = zgrid(2,i)-zgrid(1,i)
725 tem2 = zgrid(3,i)-zgrid(2,i)
726 rho1 = rho_zz(1,i) * zz(1,i) * (1. + qv_p(i,1,j))
727 rho2 = rho_zz(2,i) * zz(2,i) * (1. + qv_p(i,2,j))
728
729 tend_sfc_pressure(i) = surface_pressure(i)
730! surface_pressure(i) = 0.5*gravity*(zgrid(2,i)-zgrid(1,i)) &
731! * (rho1 + 0.5*(rho2-rho1)*tem1/(tem1+tem2))
732 surface_pressure(i) = 0.5*gravity*(zgrid(2,i)-zgrid(1,i)) &
733 * (rho1 - 0.5*(rho2-rho1)*tem1/(tem1+tem2))
734 surface_pressure(i) = surface_pressure(i) + pressure_p(1,i) + pressure_b(1,i)
735 tend_sfc_pressure(i) = (surface_pressure(i)-tend_sfc_pressure(i)) / dt_dyn
736 enddo
737 enddo
738
739!variables specific to different cloud microphysics schemes:
740 microp_select_init: select case(microp_scheme)
741
742 case ("mp_thompson","mp_wsm6")
743 qi => scalars(index_qi,:,:)
744 qs => scalars(index_qs,:,:)
745 qg => scalars(index_qg,:,:)
746
747 do j = jts, jte
748 do k = kts, kte
749 do i = its, ite
750 qi(k,i) = qi_p(i,k,j)
751 qs(k,i) = qs_p(i,k,j)
752 qg(k,i) = qg_p(i,k,j)
753 re_cloud(k,i) = recloud_p(i,k,j)
754 re_ice(k,i) = reice_p(i,k,j)
755 re_snow(k,i) = resnow_p(i,k,j)
756 enddo
757 enddo
758 enddo
759
760 microp2_select: select case(microp_scheme)
761
762 case("mp_thompson")
763 ni => scalars(index_ni,:,:)
764 nr => scalars(index_nr,:,:)
765
766 do j = jts, jte
767 do k = kts, kte
768 do i = its, ite
769 ni(k,i) = ni_p(i,k,j)
770 nr(k,i) = nr_p(i,k,j)
771 rainprod(k,i) = rainprod_p(i,k,j)
772 evapprod(k,i) = evapprod_p(i,k,j)
773 enddo
774 enddo
775 enddo
776
777 case default
778
779 end select microp2_select
780
781 case default
782
783 end select microp_select_init
784
785 end subroutine microphysics_to_mpas
