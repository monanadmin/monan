456 subroutine compute_mask(nCells, nCellsSolve, iRegion, lonCell, latCell, workMask)!{{{
457 ! this subroutines produces a 0/1 mask that is multiplied with workArray to
458 ! allow for min/max/avg to represent specific regions of the ocean domain
459 !
460 ! NOTE: computes_mask is temporary. workMask should be intent(in) to this entire module !
461 !
462 integer, intent(in) :: ncells, ncellssolve, iregion
463 real(kind=rkind), dimension(:), intent(in) :: loncell, latcell
464 real(kind=rkind), dimension(:), intent(out) :: workmask
465 integer :: icell
466 real(kind=rkind) :: dtr
467
468 dtr = 4.0_rkind*atan(1.0_rkind) / 180.0_rkind
469 workmask(:) = 0.0_rkind
470 do icell=1,ncellssolve
471 workmask(icell) = 1.0_rkind
472 enddo
473
474 if (iregion.eq.1) then
475 ! Arctic
476 do icell=1,ncellssolve
477 if(latcell(icell).lt. 60.0_rkind*dtr) workmask(icell) = 0.0_rkind
478 enddo
479 elseif (iregion.eq.2) then
480 ! Equatorial
481 do icell=1,ncellssolve
482 if(latcell(icell).gt. 15.0_rkind*dtr) workmask(icell) = 0.0_rkind
483 if(latcell(icell).lt.-15.0_rkind*dtr) workmask(icell) = 0.0_rkind
484 enddo
485 elseif (iregion.eq.3) then
486 ! Southern Ocean
487 do icell=1,ncellssolve
488 if(latcell(icell).gt.-50.0_rkind*dtr) workmask(icell) = 0.0_rkind
489 enddo
490 elseif (iregion.eq.4) then
491 ! Nino 3
492 do icell=1,ncellssolve
493 if(latcell(icell).gt. 5.0_rkind*dtr) workmask(icell) = 0.0_rkind
494 if(latcell(icell).lt. -5.0_rkind*dtr) workmask(icell) = 0.0_rkind
495 if(loncell(icell).lt.210.0_rkind*dtr) workmask(icell) = 0.0_rkind
496 if(loncell(icell).gt.270.0_rkind*dtr) workmask(icell) = 0.0_rkind
497 enddo
498 elseif (iregion.eq.5) then
499 ! Nino 4
500 do icell=1,ncellssolve
501 if(latcell(icell).gt. 5.0_rkind*dtr) workmask(icell) = 0.0_rkind
502 if(latcell(icell).lt. -5.0_rkind*dtr) workmask(icell) = 0.0_rkind
503 if(loncell(icell).lt.160.0_rkind*dtr) workmask(icell) = 0.0_rkind
504 if(loncell(icell).gt.210.0_rkind*dtr) workmask(icell) = 0.0_rkind
505 enddo
506 elseif (iregion.eq.6) then
507 ! Nino 3.4
508 do icell=1,ncellssolve
509 if(latcell(icell).gt. 5.0_rkind*dtr) workmask(icell) = 0.0_rkind
510 if(latcell(icell).lt. -5.0_rkind*dtr) workmask(icell) = 0.0_rkind
511 if(loncell(icell).lt.190.0_rkind*dtr) workmask(icell) = 0.0_rkind
512 if(loncell(icell).gt.240.0_rkind*dtr) workmask(icell) = 0.0_rkind
513 enddo
514 else
515 ! global (do nothing!)
516 endif
517
518 end subroutine compute_mask!}}}

521 subroutine compute_statistics(nDefinedDataFields, nCellsSolve, workArray, workMask, workMin, workMax, workSum)!{{{
522 ! this subroutines does the actual summing, min, max, masking ect
523 ! this hides the messy code from the high-level subroutine
524
525 integer, intent(in) :: ndefineddatafields, ncellssolve
526 real(kind=rkind), dimension(:,:), intent(in) :: workarray
527 real(kind=rkind), dimension(:), intent(in) :: workmask
528 real(kind=rkind), dimension(:), intent(out) :: workmin, workmax, worksum
529 integer :: icell, idata
530
531 worksum = 0.0_rkind
532 do icell=1,ncellssolve
533 worksum(1) = worksum(1) + workmask(icell)
534 worksum(2) = worksum(2) + workarray(2,icell)*workmask(icell)
535 do idata=3,ndefineddatafields
536 worksum(idata) = worksum(idata) + workarray(2,icell)*workarray(idata,icell)*workmask(icell)
537 enddo
538 enddo
539
540 do idata=1,ndefineddatafields
541 workmin(idata) = minval(workarray(idata,1:ncellssolve),workmask(1:ncellssolve)>0.5_rkind)
542 workmax(idata) = maxval(workarray(idata,1:ncellssolve),workmask(1:ncellssolve)>0.5_rkind)
543 enddo
544
545 end subroutine compute_statistics!}}}
