7def smoothGeometry(landFraction, floatingFraction, bed, draft, filterSigma):
8 import scipy.ndimage.filters as filters
9
10 # Smoothing is performed using only the topography in the portion of the grid that is ocean.
11 # This prevents the kink in the ice draft across the grounding line or regions of bare bedrock
12 # from influencing the smoothed topography. (Parts of the Ross ice shelf near the Trans-Antarctic
13 # Mountains are particularly troublesome if topogrpahy is smoothed across the grounding line.)
14
15 # Unlike in POP, the calving front is smoothed as well because MPAS-O does not support a
16 # sheer calving face
17
18 threshold = 0.01 # we won't normalize bed topography or ice draft where the mask is below this threshold
19
20 oceanFraction = 1. - landFraction
21 smoothedMask = filters.gaussian_filter(oceanFraction,filterSigma,mode='constant',cval=0.)
22 mask = smoothedMask > threshold
23
24 draft = filters.gaussian_filter(draft*oceanFraction,filterSigma,mode='constant',cval=0.)
25 draft[mask] /= smoothedMask[mask]
26 bed = filters.gaussian_filter(bed*oceanFraction,filterSigma,mode='constant',cval=0.)
27 bed[mask] /= smoothedMask[mask]
28
29 smoothedDraftMask = filters.gaussian_filter(floatingFraction,filterSigma,mode='constant',cval=0.)
30 smoothedDraftMask[mask] /= smoothedMask[mask]
31
32 return (bed,draft,smoothedDraftMask)
