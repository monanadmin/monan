71 subroutine ocn_init_expand_sphere(domain, stream_manager, newRadius, err)!{{{
72
73 !--------------------------------------------------------------------
74
75 type (domain_type), intent(inout) :: domain
76 type (mpas_streammanager_type), intent(inout) :: stream_manager
77 real (kind=rkind), intent(in) :: newradius
78 integer, intent(out) :: err
79
80 type (block_type), pointer :: block_ptr
81
82 type (mpas_pool_type), pointer :: meshpool
83
84 character (len=StrKIND) :: streamid
85 integer :: directionproperty
86
87 logical, pointer :: config_expand_sphere, config_realistic_coriolis_parameter
88 logical, pointer :: on_a_sphere
89 real (kind=rkind), pointer :: sphere_radius
90
91 integer, pointer :: ncells, ncellssolve, nedgessolve, nverticessolve, vertexdegree
92
93 integer, dimension(:, :), pointer :: cellsonvertex
94
95 real (kind=rkind), dimension(:), pointer :: areacell, areatriangle
96 real (kind=rkind), dimension(:), pointer :: dvedge, dcedge
97 real (kind=rkind), dimension(:), pointer :: fcell, fedge, fvertex
98 real (kind=rkind), dimension(:), pointer :: xcell, ycell, zcell, latcell, loncell
99 real (kind=rkind), dimension(:), pointer :: xedge, yedge, zedge, latedge, lonedge
100 real (kind=rkind), dimension(:), pointer :: xvertex, yvertex, zvertex, latvertex, lonvertex
101 real (kind=rkind), dimension(:, :), pointer :: kiteareasonvertex
102
103 real (kind=rkind) :: oldradius, ratio
104 real (kind=rkind) :: norm
105 real (kind=rkind) :: oldx, oldy, oldz
106 integer :: icell, iedge, ivertex, i
107
108 err = 0
109
110 call mpas_pool_get_config(domain % configs, 'config_expand_sphere', config_expand_sphere)
111
112 if ( .not. config_expand_sphere ) return
113
114 call mpas_pool_get_config(domain % configs, 'config_realistic_coriolis_parameter', config_realistic_coriolis_parameter)
115
116 call mpas_pool_get_subpool(domain % blocklist % structs, 'mesh', meshpool)
117
118 call mpas_pool_get_config(meshpool, 'on_a_sphere', on_a_sphere)
119 call mpas_pool_get_config(meshpool, 'sphere_radius', sphere_radius)
120
121 if ( .not. on_a_sphere ) then
122 call mpas_log_write( 'Warning: Only spherical meshes can been expanded.')
123 call mpas_log_write( 'Skipping expansion')
124 return
125 end if
126
127 if ( sphere_radius == 0.0_rkind ) then
128 call mpas_log_write( 'Sphere radius is 0.0', mpas_log_crit)
129 err = 1
130 return
131 end if
132
133 call mpas_log_write( 'Expanding mesh to a radius of size: $f m.', realargs=(/ newradius /) )
134
135 call mpas_stream_mgr_begin_iteration(stream_manager)
136 do while (mpas_stream_mgr_get_next_stream(stream_manager, streamid, directionproperty))
137 if ( directionproperty == mpas_stream_output .or. directionproperty == mpas_stream_input_output ) then
138 call mpas_stream_mgr_add_att(stream_manager, 'sphere_radius', newradius, streamid)
139 end if
140 end do
141
142 oldradius = sphere_radius
143 ratio = newradius / oldradius
144
145 block_ptr => domain % blocklist
146 do while(associated(block_ptr))
147 call ocn_init_set_pools_sphere_radius(block_ptr % structs, newradius)
148
149 ! Expand cell quantities
150 call mpas_pool_get_subpool(block_ptr % structs, 'mesh', meshpool)
151
152 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
153 call mpas_pool_get_dimension(meshpool, 'nCellsSolve', ncellssolve)
154 call mpas_pool_get_dimension(meshpool, 'nEdgesSolve', nedgessolve)
155 call mpas_pool_get_dimension(meshpool, 'nVerticesSolve', nverticessolve)
156 call mpas_pool_get_dimension(meshpool, 'vertexDegree', vertexdegree)
157
158 call mpas_pool_get_array(meshpool, 'xCell', xcell)
159 call mpas_pool_get_array(meshpool, 'yCell', ycell)
160 call mpas_pool_get_array(meshpool, 'zCell', zcell)
161 call mpas_pool_get_array(meshpool, 'latCell', latcell)
162 call mpas_pool_get_array(meshpool, 'lonCell', loncell)
163 call mpas_pool_get_array(meshpool, 'areaCell', areacell)
164 call mpas_pool_get_array(meshpool, 'fCell', fcell)
165
166 call mpas_pool_get_array(meshpool, 'xEdge', xedge)
167 call mpas_pool_get_array(meshpool, 'yEdge', yedge)
168 call mpas_pool_get_array(meshpool, 'zEdge', zedge)
169 call mpas_pool_get_array(meshpool, 'latEdge', latedge)
170 call mpas_pool_get_array(meshpool, 'lonEdge', lonedge)
171 call mpas_pool_get_array(meshpool, 'dvEdge', dvedge)
172 call mpas_pool_get_array(meshpool, 'dcEdge', dcedge)
173 call mpas_pool_get_array(meshpool, 'fEdge', fedge)
174
175 call mpas_pool_get_array(meshpool, 'xVertex', xvertex)
176 call mpas_pool_get_array(meshpool, 'yVertex', yvertex)
177 call mpas_pool_get_array(meshpool, 'zVertex', zvertex)
178 call mpas_pool_get_array(meshpool, 'latVertex', latvertex)
179 call mpas_pool_get_array(meshpool, 'lonVertex', lonvertex)
180 call mpas_pool_get_array(meshpool, 'areaTriangle', areatriangle)
181 call mpas_pool_get_array(meshpool, 'kiteAreasOnVertex', kiteareasonvertex)
182 call mpas_pool_get_array(meshpool, 'cellsOnVertex', cellsonvertex)
183 call mpas_pool_get_array(meshpool, 'fVertex', fvertex)
184
185 do icell = 1, ncellssolve
186 oldx = xcell(icell)
187 oldy = ycell(icell)
188 oldz = zcell(icell)
189
190 norm = sqrt(oldx**2 + oldy**2 + oldz**2)
191
192 xcell(icell) = (oldx / norm) * newradius
193 ycell(icell) = (oldy / norm) * newradius
194 zcell(icell) = (oldz / norm) * newradius
195 areacell(icell) = (areacell(icell) / (oldradius**2) )* newradius**2
196
197 if(config_realistic_coriolis_parameter) then
198 fcell(icell) = 2.0_rkind * omega * sin(latcell(icell))
199 end if
200 end do
201
202 ! Expand vertex quantities
203 do ivertex = 1, nverticessolve
204 oldx = xvertex(ivertex)
205 oldy = yvertex(ivertex)
206 oldz = zvertex(ivertex)
207
208 norm = sqrt(oldx**2 + oldy**2 + oldz**2)
209
210 xvertex(ivertex) = (oldx / norm) * newradius
211 yvertex(ivertex) = (oldy / norm) * newradius
212 zvertex(ivertex) = (oldz / norm) * newradius
213 areatriangle(ivertex) = 0.0_rkind
214
215 do i = 1, vertexdegree
216 if (cellsonvertex(i, ivertex) < ncells+1) then
217 kiteareasonvertex(i, ivertex) = ( kiteareasonvertex(i, ivertex) / oldradius**2) * newradius**2
218 else
219 kiteareasonvertex(i, ivertex) = 0.0_rkind
220 end if
221 areatriangle(ivertex) = areatriangle(ivertex) + kiteareasonvertex(i, ivertex)
222 end do
223
224 if(config_realistic_coriolis_parameter) then
225 fvertex(ivertex) = 2.0_rkind * omega * sin( latvertex(ivertex) )
226 end if
227 end do
228
229 ! Expand edge quantities
230 do iedge = 1, nedgessolve
231 oldx = xedge(iedge)
232 oldy = yedge(iedge)
233 oldz = zedge(iedge)
234
235 norm = sqrt(oldx**2 + oldy**2 + oldz**2)
236
237 xedge(iedge) = (oldx / norm) * newradius
238 yedge(iedge) = (oldy / norm) * newradius
239 zedge(iedge) = (oldz / norm) * newradius
240 dvedge(iedge) = (dvedge(iedge) / oldradius) * newradius
241 dcedge(iedge) = (dcedge(iedge) / oldradius) * newradius
242
243 if(config_realistic_coriolis_parameter) then
244 fedge(iedge) = 2.0_rkind * omega * sin( latedge(iedge) )
245 end if
246 end do
247
248 block_ptr % domain % sphere_radius = newradius
249 sphere_radius = newradius
250 block_ptr => block_ptr % next
251
252 end do
253
254 !--------------------------------------------------------------------
255
256 end subroutine ocn_init_expand_sphere!}}}

291 end if
292 end do

318 subroutine ocn_transform_from_lonlat_to_xyz(xin, yin, zin, ulon, ulat, ux, uy, uz)!{{{
319 implicit none
320 real, intent(in) :: xin, yin, zin, ulon, ulat
321 real, intent(out) :: ux, uy, uz
322 real :: h(3,3), p(3), q(3), g(3), x1(3,3), x2(3,3), trans_x2_to_x1(3,3), r
323 integer :: i,j,k
324 logical :: l_pole
325 real, parameter :: epsvt = 1.0e-10_rkind
326
327 !-----------------------------------------------------------------------
328 ! define the e1, e2, and e3 directions
329 !-----------------------------------------------------------------------
330 x1(1,1) = 1.0_rkind; x1(1,2) = 0.0_rkind; x1(1,3) = 0.0_rkind
331 x1(2,1) = 0.0_rkind; x1(2,2) = 1.0_rkind; x1(2,3) = 0.0_rkind
332 x1(3,1) = 0.0_rkind; x1(3,2) = 0.0_rkind; x1(3,3) = 1.0_rkind
333
334 !-----------------------------------------------------------------------
335 ! find the vectors (measured in X1) that point in the local
336 ! east (h(1,:)), north (h(2,:)), and vertical (h(3,:)) direction
337 !-----------------------------------------------------------------------
338 h(3,1) = xin; h(3,2) = yin; h(3,3) = zin
339 call ocn_unit_vector_in_3space(h(3,:))
340
341 !-----------------------------------------------------------------------
342 ! g(:) is a work array and holds the vector pointing to the North Pole.
343 ! measured in X1
344 !-----------------------------------------------------------------------
345 g(:) = x1(3,:)
346
347 !-----------------------------------------------------------------------
348 ! determine if the local vertical hits a pole
349 !-----------------------------------------------------------------------
350 l_pole = .false.
351 r = g(1)*h(3,1) + g(2)*h(3,2) + g(3)*h(3,3)
352 r = abs(r) + epsvt
353 if(r.gt.1.0_rkind) then
354 l_pole = .true.
355 h(3,:) = h(3,:) + epsvt
356 call ocn_unit_vector_in_3space(h(3,:))
357 endif
358
359 !-----------------------------------------------------------------------
360 ! find the vector that is perpendicular to the local vertical vector
361 ! and points in the direction of of the North pole, this defines the local
362 ! north direction. measured in X1
363 !-----------------------------------------------------------------------
364 call ocn_vector_on_tangent_plane ( h(3,:), g(:), h(2,:) )
365
366 !-----------------------------------------------------------------------
367 ! take the cross product of the local North direction and the local vertical
368 ! to find the local east vector. still in X1
369 !-----------------------------------------------------------------------
370 call ocn_cross_product_in_3space ( h(2,:), h(3,:), h(1,:) )
371
372 !-----------------------------------------------------------------------
373 ! put these 3 vectors into a matrix X2
374 !-----------------------------------------------------------------------
375 x2(1,:) = h(1,:) ! local east (measured in X1)
376 x2(2,:) = h(2,:) ! local north (measured in X1)
377 x2(3,:) = h(3,:) ! local vertical (measured in X1)
378
379 !-----------------------------------------------------------------------
380 ! compute the transformation matrix
381 !-----------------------------------------------------------------------
382 trans_x2_to_x1(:,:) = matmul(x1,transpose(x2))
383
384 !-----------------------------------------------------------------------
385 ! transform (ulon, ulat) into (x,y,z)
386 !-----------------------------------------------------------------------
387 p(1) = ulon; p(2) = ulat; p(3) = 0
388 g(:) = matmul(trans_x2_to_x1(:, :), p(:))
389 ux = g(1); uy = g(2); uz = g(3)
390
391 end subroutine ocn_transform_from_lonlat_to_xyz!}}}

500!-----------------------------------------------------------------------
501 subroutine ocn_unit_vector_in_3space (p_1)!{{{
502
503 !-----------------------------------------------------------------------
504 ! PURPOSE : normalize p_1 to unit length and overwrite p_1
505 !-----------------------------------------------------------------------
506
507 !-----------------------------------------------------------------------
508 ! intent(inout)
509 !-----------------------------------------------------------------------
510 real , intent(inout) :: &
511 p_1 (:)
512
513 !-----------------------------------------------------------------------
514 ! local
515 !-----------------------------------------------------------------------
516 real :: length
517
518 length = sqrt(p_1(1)**2 + p_1(2)**2 + p_1(3)**2 )
519 length = 1.0_rkind/length
520 p_1(1) = p_1(1)*length
521 p_1(2) = p_1(2)*length
522 p_1(3) = p_1(3)*length
523

528! routine ocn_vector_on_tangent_plane
529!
541!
542!-----------------------------------------------------------------------
543 subroutine ocn_vector_on_tangent_plane(p_1, p_2, p_out)!{{{
544!-----------------------------------------------------------------------
545! intent(in)
546!-----------------------------------------------------------------------
547 real , intent(in) :: &
548 p_1 (:), &
549 p_2 (:)
550
551!-----------------------------------------------------------------------
552! intent(out)
553!-----------------------------------------------------------------------
554 real , intent(out) :: &
555 p_out (:)
556
557!-----------------------------------------------------------------------
558! local
559!-----------------------------------------------------------------------
560 real :: &
561 work (3), t1(3), t2(3)
562
563! work (1) = - p_1(2) * ( -p_1(2) * p_2(1) + p_1(1) * p_2(2) ) &
564! + p_1(3) * ( p_1(3) * p_2(1) - p_1(1) * p_2(3) )
565
566! work (2) = + p_1(1) * ( -p_1(2) * p_2(1) + p_1(1) * p_2(2) ) &
567! - p_1(3) * ( -p_1(3) * p_2(2) + p_1(2) * p_2(3) )
568
569! work (3) = - p_1(1) * ( p_1(3) * p_2(1) - p_1(1) * p_2(3) ) &
570! + p_1(2) * ( -p_1(3) * p_2(2) + p_1(2) * p_2(3) )
571
572
573 t1(:) = p_2(:) - p_1(:)
574 t2(:) = p_1
575
576 call ocn_unit_vector_in_3space (t1)
577 call ocn_unit_vector_in_3space (t2)
578
579 call ocn_cross_product_in_3space(t1(:), t2(:), work(:))
580 call ocn_unit_vector_in_3space (work)
581 call ocn_cross_product_in_3space(t2(:),work(:),p_out(:))
582 call ocn_unit_vector_in_3space (p_out)
583
584 end subroutine ocn_vector_on_tangent_plane!}}}

586!***********************************************************************
587!
588! routine ocn_cross_product_in_3space
589!
595!
596!-----------------------------------------------------------------------
597 subroutine ocn_cross_product_in_3space(p_1,p_2,p_out)!{{{
598!-----------------------------------------------------------------------
599! intent(in)
600!-----------------------------------------------------------------------
601 real , intent(in) :: &
602 p_1 (:), &
603 p_2 (:)
604
605!-----------------------------------------------------------------------
606! intent(out)
607!-----------------------------------------------------------------------
608 real , intent(out) :: &
609 p_out (:)
610
611 p_out(1) = p_1(2)*p_2(3)-p_1(3)*p_2(2)
612 p_out(2) = p_1(3)*p_2(1)-p_1(1)*p_2(3)
613 p_out(3) = p_1(1)*p_2(2)-p_1(2)*p_2(1)
614
615 end subroutine ocn_cross_product_in_3space!}}}
