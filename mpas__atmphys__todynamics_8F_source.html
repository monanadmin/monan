172!initialize the tendency for the potential temperature and all scalars due to PBL, convection,
173!and longwave and shortwave radiation:
174! allocate(theta(nVertLevels,nCellsSolve) )
175 allocate(tend_th(nvertlevels,ncellssolve))
176 tend_th = 0._rkind
177
178 tend_scalars(:,:,:) = 0._rkind

192 if (.not. associated(rqrcuten)) allocate(rqrcuten(0,0))
193 if (.not. associated(rqscuten)) allocate(rqscuten(0,0))
194 if (.not. associated(rniblten)) allocate(rniblten(0,0))
195 if (.not. associated(rthdynten)) allocate(rthdynten(0,0))
196 if (.not. associated(rublten)) allocate(rublten(0,0))
197 if (.not. associated(rvblten)) allocate(rvblten(0,0))
198 if (.not. associated(rthblten)) allocate(rthblten(0,0))
199 if (.not. associated(rqvblten)) allocate(rqvblten(0,0))
200 if (.not. associated(rqcblten)) allocate(rqcblten(0,0))
201 if (.not. associated(rqiblten)) allocate(rqiblten(0,0))
202 if (.not. associated(rthcuten)) allocate(rthcuten(0,0))
203 if (.not. associated(rqvcuten)) allocate(rqvcuten(0,0))
204 if (.not. associated(rqccuten)) allocate(rqccuten(0,0))
205 if (.not. associated(rqicuten)) allocate(rqicuten(0,0))

209 config_pbl_scheme, config_convection_scheme, config_radt_lw_scheme, config_radt_sw_scheme, &
210 index_qv, index_qc, index_qr, index_qi, index_qs, index_ni, &
211 rublten, rvblten, mass_edge, rublten_edge, &
212 tend_ru_physics, &
213 rucuten, rvcuten, rucuten_edge, &
214 tend_th, tend_scalars, mass, rthblten, rqvblten, rqcblten, rqiblten, rniblten, &
215 rthcuten, rqvcuten, rqccuten, rqrcuten, rqicuten, rqscuten, &
216 rthratenlw, rthratensw, rthdynten, &
217 tend_u_phys, tend_rtheta_adv, tend_diabatic, &
218 theta_m, scalars, &
219 tend_rtheta_physics, &
220 tend_theta_euler &
221 )
222

229 if (size(rqrcuten) == 0) deallocate(rqrcuten)

233 if (size(rublten) == 0) deallocate(rublten)
234 if (size(rvblten) == 0) deallocate(rvblten)
235 if (size(rthblten) == 0) deallocate(rthblten)
236 if (size(rqvblten) == 0) deallocate(rqvblten)
237 if (size(rqcblten) == 0) deallocate(rqcblten)
238 if (size(rqiblten) == 0) deallocate(rqiblten)
239 if (size(rthcuten) == 0) deallocate(rthcuten)
240 if (size(rqvcuten) == 0) deallocate(rqvcuten)
241 if (size(rqccuten) == 0) deallocate(rqccuten)
242 if (size(rqicuten) == 0) deallocate(rqicuten)
243
244! deallocate(theta)
245 deallocate(tend_th)
246
247! if(rk_step .eq. 3) then
248! call mpas_log_write('')
249! call mpas_log_write('--- enter subroutine physics_get_tend:')
250! call mpas_log_write('max rthblten = $r',realArgs=(/maxval(rthblten(:,1:nCellsSolve))/))
251! call mpas_log_write('min rthblten = $r',realArgs=(/minval(rthblten(:,1:nCellsSolve))/))
252! call mpas_log_write('max rthcuten = $r',realArgs=(/maxval(rthcuten(:,1:nCellsSolve))/))
253! call mpas_log_write('min rthcuten = $r',realArgs=(/minval(rthcuten(:,1:nCellsSolve))/))
254! call mpas_log_write('max rthratenlw = $r',realArgs=(/maxval(rthratenlw(:,1:nCellsSolve))/))
255! call mpas_log_write('min rthratenlw = $r',realArgs=(/minval(rthratenlw(:,1:nCellsSolve))/))
256! call mpas_log_write('max rthratensw = $r',realArgs=(/maxval(rthratensw(:,1:nCellsSolve))/))
257! call mpas_log_write('min rthratensw = $r',realArgs=(/minval(rthratensw(:,1:nCellsSolve))/))
258! call mpas_log_write('--- end subroutine physics_get_tend')
259! call mpas_log_write('')
260! endif
261
262 end subroutine physics_get_tend

264 !==================================================================================================
265 subroutine physics_get_tend_work(block, mesh, nCells, nEdges, nCellsSolve, nEdgesSolve, &
266 rk_step, dynamics_substep, &
267 config_pbl_scheme, config_convection_scheme, config_radt_lw_scheme, config_radt_sw_scheme, &
268 index_qv, index_qc, index_qr, index_qi, index_qs, index_ni, &
269 rublten, rvblten, mass_edge, rublten_Edge, tend_u, &
270 rucuten, rvcuten, rucuten_Edge, &
271 tend_th, tend_scalars, mass, rthblten, rqvblten, rqcblten, rqiblten, rniblten, &
272 rthcuten, rqvcuten, rqccuten, rqrcuten, rqicuten, rqscuten, &
273 rthratenlw, rthratensw, rthdynten, &
274 tend_u_phys, tend_rtheta_adv, tend_diabatic, &
275 theta_m, scalars, tend_theta, tend_theta_euler &
276 )
277!==================================================================================================
278
279 use mpas_atm_dimensions
280
281 implicit none
282
283 type(block_type), intent(in) :: block
284 type(mpas_pool_type), intent(in) :: mesh
285 integer, intent(in) :: nCells, nEdges, nCellsSolve, nEdgesSolve
286 integer, intent(in) :: rk_step, dynamics_substep
287 character(len=StrKIND), intent(in) :: config_pbl_scheme
288 character(len=StrKIND), intent(in) :: config_convection_scheme
289 character(len=StrKIND), intent(in) :: config_radt_lw_scheme
290 character(len=StrKIND), intent(in) :: config_radt_sw_scheme
291 integer, intent(in) :: index_qv, index_qc, index_qr, index_qi, index_qs, index_ni
292 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rublten
293 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rvblten
294 real (kind=rkind), dimension(nVertLevels,nEdges+1), intent(in) :: mass_edge
295 real (kind=rkind), dimension(nVertLevels,nEdges+1), intent(inout) :: rublten_edge
296 real (kind=rkind), dimension(nVertLevels,nEdges+1), intent(inout) :: tend_u
297 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rucuten
298 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rvcuten
299 real (kind=rkind), dimension(nVertLevels,nEdges+1), intent(inout) :: rucuten_edge
300 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(inout) :: tend_th
301 real (kind=rkind), dimension(num_scalars,nVertLevels,nCells+1), intent(inout) :: tend_scalars
302 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: mass
303 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rthblten
304 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqvblten
305 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqcblten
306 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqiblten
307 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rniblten
308 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rthcuten
309 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqvcuten
310 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqccuten
311 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqrcuten
312 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqicuten
313 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rqscuten
314 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rthratenlw
315 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: rthratensw
316 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(inout) :: rthdynten
317 real (kind=rkind), dimension(nVertLevels,nEdges+1), intent(inout) :: tend_u_phys
318 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: tend_rtheta_adv
319 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: tend_diabatic
320 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: theta_m
321 real (kind=rkind), dimension(num_scalars,nVertLevels,nCells+1), intent(in) :: scalars
322 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(inout) :: tend_theta
323 real (kind=rkind), dimension(nVertLevels,nCells+1), intent(in) :: tend_theta_euler
324
325 integer :: i, k
326 real (kind=rkind) :: coeff
327
328 !add coupled tendencies due to PBL processes:
329 if (config_pbl_scheme .ne. 'off') then
330 if (rk_step == 1 .and. dynamics_substep == 1) then
331 call tend_toedges(block,mesh,rublten,rvblten,rublten_edge)
332
333 !MGD for PV budget? should a similar line be in the cumulus section below?
334 tend_u_phys(1:nvertlevels,1:nedges) = rublten_edge(1:nvertlevels,1:nedges)
335 end if
336
337 do i = 1, nedgessolve
338 do k = 1, nvertlevels
339 tend_u(k,i)=tend_u(k,i)+rublten_edge(k,i)*mass_edge(k,i)
340 enddo
341 enddo
342
343 do i = 1, ncellssolve
344 do k = 1, nvertlevels
345 tend_th(k,i) = tend_th(k,i) + rthblten(k,i)*mass(k,i)
346 tend_scalars(index_qv,k,i) = tend_scalars(index_qv,k,i) + rqvblten(k,i)*mass(k,i)
347 tend_scalars(index_qc,k,i) = tend_scalars(index_qc,k,i) + rqcblten(k,i)*mass(k,i)
348 tend_scalars(index_qi,k,i) = tend_scalars(index_qi,k,i) + rqiblten(k,i)*mass(k,i)
349 enddo
350 enddo
351
352 pbl_select: select case (trim(config_pbl_scheme))
353
354 case("bl_mynn")
355
356 do i = 1, ncellssolve
357 do k = 1, nvertlevels
358 tend_scalars(index_ni,k,i) = tend_scalars(index_ni,k,i) + rniblten(k,i)*mass(k,i)
359 enddo
360 enddo
361
362 case default
363
364 end select pbl_select
365 endif
366
367 !add coupled tendencies due to convection:
368 if (config_convection_scheme .ne. 'off') then
369
370 do i = 1, ncellssolve
371 do k = 1, nvertlevels
372 tend_th(k,i) = tend_th(k,i) + rthcuten(k,i)*mass(k,i)
373 tend_scalars(index_qv,k,i) = tend_scalars(index_qv,k,i) + rqvcuten(k,i)*mass(k,i)
374 tend_scalars(index_qc,k,i) = tend_scalars(index_qc,k,i) + rqccuten(k,i)*mass(k,i)
375 tend_scalars(index_qi,k,i) = tend_scalars(index_qi,k,i) + rqicuten(k,i)*mass(k,i)
376 enddo
377 enddo
378
379 convection_select: select case(config_convection_scheme)
380
381 case('cu_kain_fritsch')
382 do i = 1, ncellssolve
383 do k = 1, nvertlevels
384 tend_scalars(index_qr,k,i) = tend_scalars(index_qr,k,i) + rqrcuten(k,i)*mass(k,i)
385 tend_scalars(index_qs,k,i) = tend_scalars(index_qs,k,i) + rqscuten(k,i)*mass(k,i)
386 enddo
387 enddo
388
389 case('cu_tiedtke','cu_ntiedtke')
390 if (rk_step == 1 .and. dynamics_substep == 1) then
391 call tend_toedges(block,mesh,rucuten,rvcuten,rucuten_edge)
392
393 tend_u_phys(1:nvertlevels,1:nedges) = tend_u_phys(1:nvertlevels,1:nedges) &
394 + rucuten_edge(1:nvertlevels,1:nedges)
395 end if
396 do i = 1, nedgessolve
397 do k = 1, nvertlevels
398 tend_u(k,i)=tend_u(k,i)+rucuten_edge(k,i)*mass_edge(k,i)
399 enddo
400 enddo
401
402 case default
403 end select convection_select
404 endif
405
406 !add coupled tendencies due to longwave radiation:
407 if (config_radt_lw_scheme .ne. 'off') then
408 do i = 1, ncellssolve
409 do k = 1, nvertlevels
410 tend_th(k,i) = tend_th(k,i) + rthratenlw(k,i)*mass(k,i)
411 enddo
412 enddo
413 endif
414
415 !add coupled tendencies due to shortwave radiation:
416 if (config_radt_sw_scheme .ne. 'off') then
417 do i = 1, ncellssolve
418 do k = 1, nvertlevels
419 tend_th(k,i) = tend_th(k,i) + rthratensw(k,i)*mass(k,i)
420 enddo
421 enddo
422 endif
423
424 !if non-hydrostatic core, convert the tendency for the potential temperature to a
425 !tendency for the modified potential temperature:
426 do i = 1, ncellssolve
427 do k = 1, nvertlevels
428 coeff = (1. + r_v/r_d * scalars(index_qv,k,i))
