123 config_temperature_init, & ! temperature initialization option ('linear' or 'file')
124 config_surface_air_temperature_source, & ! surface air temperature initialization option ('constant' or 'file' or 'lapse')
125 config_basal_heat_flux_source ! basal heat flux initialization option ('constant' or 'file')
126
127 real (kind=rkind), pointer :: &
128 config_ice_density, & ! ice density
129 config_ocean_density ! ocean density
130
131 real (kind=rkind), pointer :: &
132 config_surface_air_temperature_value, & ! constant value of surface air temperature

134 config_basal_heat_flux_value ! constant value of basal heat flux
135

137 config_stats_cell_id ! cell ID for diagnostic output
138
139 integer, pointer :: &
140 ncellssolve, & ! number of locally owned cells
141 nvertlevels ! number of vertical layers
142
143 integer, dimension(:), pointer :: &
144 indextocellid ! global ID for each local cell
145
146 real (kind=rkind), dimension(:), pointer :: &
147 layercentersigma, & ! sigma coordinate at midpoint of each layer
148 layerinterfacesigma ! sigma coordinate at layer interfaces (including top and bottom)
149
150 real (kind=rkind), dimension(:), pointer :: &
151 thickness, & ! ice thickness
152 uppersurface ! ice upper surface
153
154 real (kind=rkind), dimension(:,:), pointer :: &
155 temperature, & ! interior ice temperature (K)
156 waterfrac, & ! interior water fraction (unitless)
157 enthalpy ! interior ice enthalpy (J m^{-3})
158
159 real (kind=rkind), dimension(:), pointer :: &
160 surfaceairtemperature, & ! surface air temperature (K)
161 surfacetemperature, & ! surface ice temperature (K)
162 basaltemperature ! basal ice temperature (K)
163
164 real (kind=rkind), dimension(:), pointer :: &
165 basalheatflux ! basal heat flux into the ice (W m^{-2}, positive upward)
166
167 integer :: k, ilayer
168
169 integer :: icell
170
171 integer :: err_tmp
172
173 real(kind=rkind) :: temperaturevalue
174
175 real(kind=rkind), dimension(:), allocatable :: pmptemperaturecol
176

178 logical, parameter :: circular_shelf_test = .false.
179 logical, parameter :: dome_test = .false.

182 integer :: nrows
183 integer :: i, irow
184 real(kind=rkind), parameter :: surfaceairtemperaturecelsius = -15.0_rkind

186
187 call mpas_timer_start("thermal init")
188
189 if (circular_shelf_test) then
190 ncellsperrow = 40
191 nrows = 46
192 elseif (dome_test) then
193 ncellsperrow = 30
194 nrows = 34
195 endif
196
197 err = 0
198
199 ! get config options
200 call mpas_pool_get_config(liconfigs, 'config_thermal_solver', config_thermal_solver)
201 call mpas_pool_get_config(liconfigs, 'config_temperature_init', config_temperature_init)
202 call mpas_pool_get_config(liconfigs, 'config_surface_air_temperature_source', config_surface_air_temperature_source)
203 call mpas_pool_get_config(liconfigs, 'config_surface_air_temperature_value', config_surface_air_temperature_value)
204 call mpas_pool_get_config(liconfigs, 'config_surface_air_temperature_lapse_rate', config_surface_air_temperature_lapse_rate)
205 call mpas_pool_get_config(liconfigs, 'config_basal_heat_flux_source', config_basal_heat_flux_source)
206 call mpas_pool_get_config(liconfigs, 'config_basal_heat_flux_value', config_basal_heat_flux_value)
207 call mpas_pool_get_config(liconfigs, 'config_print_thermal_info', config_print_thermal_info)
208 call mpas_pool_get_config(liconfigs, 'config_do_restart', config_do_restart)
209 call mpas_pool_get_config(liconfigs, 'config_stats_cell_ID', config_stats_cell_id)
210
211 ! set some physical constants
212 ! (to avoid calling mpas_pool_get_config repeatedly in this module)
213 call mpas_pool_get_config(liconfigs, 'config_ice_density', config_ice_density)
214 call mpas_pool_get_config(liconfigs, 'config_ocean_density', config_ocean_density)
215 rhoi = config_ice_density
216 rhoo = config_ocean_density
217
218 ! block loop
219 block => domain % blocklist
220 do while (associated(block))
221
222 ! get pools
223 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
224 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
225 call mpas_pool_get_subpool(block % structs, 'thermal', thermalpool)
226
227 ! get dimensions
228 call mpas_pool_get_dimension(meshpool, 'nVertLevels', nvertlevels)
229 call mpas_pool_get_dimension(meshpool, 'nCellsSolve', ncellssolve)
230
231 ! get arrays from the mesh pool
232 call mpas_pool_get_array(meshpool, 'layerCenterSigma', layercentersigma)
233 call mpas_pool_get_array(meshpool, 'layerInterfaceSigma', layerinterfacesigma)
234 call mpas_pool_get_array(meshpool, 'indexToCellID', indextocellid)
235
236 ! get arrays from the geometry pool
237 call mpas_pool_get_array(geometrypool, 'thickness', thickness)
238 call mpas_pool_get_array(geometrypool, 'upperSurface', uppersurface)
239
240 ! get arrays from the thermal pool
241 call mpas_pool_get_array(thermalpool, 'temperature', temperature)
242 call mpas_pool_get_array(thermalpool, 'waterfrac', waterfrac)
243 call mpas_pool_get_array(thermalpool, 'enthalpy', enthalpy)
244 call mpas_pool_get_array(thermalpool, 'surfaceAirTemperature', surfaceairtemperature)
245 call mpas_pool_get_array(thermalpool, 'surfaceTemperature', surfacetemperature)
246 call mpas_pool_get_array(thermalpool, 'basalTemperature', basaltemperature)
247 call mpas_pool_get_array(thermalpool, 'basalHeatFlux', basalheatflux)
248
249 if (config_print_thermal_info) then
250 call mpas_log_write('Initialize thermal solver, config_temperature_init = ' // trim(config_temperature_init))
251 endif
252
253 ! init for surfaceAirTemperature
254 !TODO - Allow for reading surfaceAirTemperature from an external file
255 if (trim(config_surface_air_temperature_source) == 'constant') then
256 surfaceairtemperature(:) = config_surface_air_temperature_value
257 if (config_print_thermal_info) &
258 call mpas_log_write('Initialize sfc air temp: $r', realargs=(/config_surface_air_temperature_value/))
259 elseif (trim(config_surface_air_temperature_source) == 'lapse') then
260 surfaceairtemperature(:) = config_surface_air_temperature_value - config_surface_air_temperature_lapse_rate * uppersurface(:)
261 if (config_print_thermal_info) &
262 call mpas_log_write('Initialize sfc air temp to $r with lapse rate of $r', &
263 realargs=(/config_surface_air_temperature_value, config_surface_air_temperature_lapse_rate/))
264 endif
265
266 ! init for basalHeatFlux
267 !TODO - Allow for reading basalHeatFlux from an external file
268 if (trim(config_basal_heat_flux_source) == 'constant') then
269 basalheatflux(:) = config_basal_heat_flux_value
270 if (config_print_thermal_info) &
271 call mpas_log_write('Initialize basal heat flux: $r', realargs=(/config_basal_heat_flux_value/))
272 endif
273
274 ! Precompute some grid quantities used in the vertical temperature solve
275
276 allocate(dsigmaterm(nvertlevels,2))
277 dsigmaterm(:,:) = 0.0_rkind
278
279 k = 1
280 dsigmaterm(k,1) = 1.0_rkind/( (layerinterfacesigma(k+1) - layerinterfacesigma(k)) * &
281 (layercentersigma(k) - layerinterfacesigma(k)) )
282
283 do k = 2, nvertlevels
284 dsigmaterm(k,1) = 1.0_rkind/( (layerinterfacesigma(k+1) - layerinterfacesigma(k)) * &
285 (layercentersigma(k) - layercentersigma(k-1)) )
286 enddo
287
288 do k = 1, nvertlevels-1
289 dsigmaterm(k,2) = 1.0_rkind/( (layerinterfacesigma(k+1) - layerinterfacesigma(k)) * &
290 (layercentersigma(k+1) - layercentersigma(k)) )
291 end do
292
293 k = nvertlevels
294 dsigmaterm(k,2) = 1.0_rkind/( (layerinterfacesigma(k+1) - layerinterfacesigma(k)) * &
295 (layerinterfacesigma(k+1) - layercentersigma(k)) )
296
297 if (config_print_thermal_info) then
298 call mpas_log_write('dsigmaTerm coefficients:')
299 do k = 1, nvertlevels
300 call mpas_log_write("$i $r $r", intargs=(/k/), realargs=(/dsigmaterm(k,1), dsigmaterm(k,2)/))
301 enddo
302 endif
303
304 !WHL - debug - Temporary initialization for test cases
305
306 if (config_print_thermal_info .and. (circular_shelf_test .or. dome_test)) then
307
308 call mpas_log_write(' ')
309
310 if (circular_shelf_test) then
311
312 call mpas_log_write('Circular shelf test:')
313
314 elseif (dome_test) then
315
316 call mpas_log_write('Dome test:')
317! write(stdoutUnit,*) ' '
318! write(stdoutUnit,*) 'Cells with ice: iCell, thickness'
319! do iCell = 1, nCellsSolve
320! if (thickness(iCell) > 0.0_RKIND) then
321! write(stdoutUnit,*) iCell, thickness(iCell)
322! endif
323! enddo
324
325 endif
326
327 !WHL - debug dome - Adjust thickness slightly so the central cell has the same thickness as in CISM
328!! thickness(:) = thickness(:) * 707.1068115234375000_RKIND / 707.10678118654755_RKIND
329
330! call mpas_log_write(' ')
331! call mpas_log_write('Initial thickness')
332! do iRow = nRows, 1, -1
333! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
334! write(stdoutUnit,'(a3)',advance='no') ' '
335! endif
336! do i = nCellsPerRow/2 - 2, nCellsPerRow
337! iCell = (iRow-1)*nCellsPerRow + i
338! write(stdoutUnit,'(f8.2)',advance='no') thickness(iCell)
339! enddo
340! write(stdoutUnit,*) ' '
341! enddo
342!
343! write(stdoutUnit,*) ' '
344! write(stdoutUnit,*) 'Initial surfaceAirTemperature'
345! do iRow = nRows, 1, -1
346! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
347! write(stdoutUnit,'(a3)',advance='no') ' '
348! endif
349! do i = nCellsPerRow/2 - 2, nCellsPerRow
350! iCell = (iRow-1)*nCellsPerRow + i
351! write(stdoutUnit,'(f8.2)',advance='no') surfaceAirTemperature(iCell)
352! enddo
353! write(stdoutUnit,*) ' '
354! enddo
355
356 endif ! circ shelf or dome
357
358
359 ! Initialize vertical temperature profile.
360 ! Three possibilities:
361 ! (1) Set up a linear temperature profile, with T = artm at the surface and T <= Tpmp
362 ! at the bed (config_temperature_init = 'linear').
363 ! A parameter (pmpt_offset) controls how far below Tpmp the initial bed temp is set.
364 ! (2) Read ice temperature from an initial input file (config_temperature_init = 'file').
365 ! (3) Read ice temperature from a restart file.
366 !
367 ! The default is (1).
368 ! If restarting, we always do (3).
369 ! If (2) or (3), then the temperature should already have been read in, and there is
370 ! nothing to do here (except possibly to set waterfrac).
371
372 if (config_do_restart) then
373
374 ! nothing to do; temperature was read from the restart file
375 !TODO - Make sure waterfrac is also read, if needed
376 if (config_print_thermal_info) then
377 call mpas_log_write('Initialized ice temperature from the restart file')
378 endif
379
380 elseif (trim(config_temperature_init) == 'file') then
381
382 ! Temperature was read from the input file
383 if (config_print_thermal_info) then
384 call mpas_log_write('Initialized ice temperature from the input file')
385 endif
386
387 ! initialize waterfrac, in case we are using the enthalpy solver
388 !TODO - Allow waterfrac to be read from the input file?
389 waterfrac(:,:) = 0.0_rkind
390
391 elseif (trim(config_temperature_init) == 'sfc_air_temperature') then
392
393 allocate(pmptemperaturecol(nvertlevels))
394 do icell = 1, ncellssolve
395 ! set the surface temperature to the surface air temperature or 273.15, whichever is less
396 surfacetemperature(icell) = min(surfaceairtemperature(icell), kelvin_to_celsius)
397
398 ! set the column temperature to surface air temp or PMP, whichever is less
399 call pressure_melting_point_column(layercentersigma, thickness(icell), pmptemperaturecol)
400 do k = 1, nvertlevels
401 temperature(k,icell) = min(surfaceairtemperature(icell), pmptemperaturecol(k) + kelvin_to_celsius)
402 enddo
403
404 ! set the basal temperature to surface air temp or PMP, whichever is less
405 call pressure_melting_point(thickness(icell), temperaturevalue)
406 basaltemperature(icell) = min(surfaceairtemperature(icell), temperaturevalue + kelvin_to_celsius)
407 enddo
408 deallocate(pmptemperaturecol)
409
410 waterfrac(:,:) = 0.0_rkind
411
412 if (config_print_thermal_info) then
413 call mpas_log_write('Initialized ice column temperature to the surface air temperature')
414 endif
415
416 elseif (trim(config_temperature_init) == 'linear') then
417
418 ! set up a linear temperature profile in each column
419 ! T = surfaceAirTemperature at the ice surface, and T <= Tpmp at the bed
420
421 ! initialize T = 273.15 K = 0 C everywhere
422
423 temperature(:,:) = kelvin_to_celsius ! = 273.15
424
425 do icell = 1, ncellssolve
426
427 call li_init_linear_temperature_in_column(&
428 nvertlevels, &
429 layercentersigma, &
430 thickness(icell), &
431 surfaceairtemperature(icell), &
432 temperature(:,icell), &
433 waterfrac(:,icell), &
434 surfacetemperature(icell), &
435 basaltemperature(icell))
436
437 enddo ! iCell
438
439 if (config_print_thermal_info) then
440 call mpas_log_write('Initialized a linear temperature profile in each column')
441
442 !WHL - debug
443! if (circular_shelf_test .or. dome_test) then
444!
445! call mpas_log_write(' ')
446! call mpas_log_write('Initial layer 1 temperature')
447! iLayer = 1
448! do iRow = nRows, 1, -1
449! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
450! write(stdoutUnit,'(a3)',advance='no') ' '
451! endif
452! do i = nCellsPerRow/2 - 2, nCellsPerRow
453! iCell = (iRow-1)*nCellsPerRow + i
454! write(stdoutUnit,'(f8.2)',advance='no') temperature(iLayer,iCell)
455! enddo
456! write(stdoutUnit,*) ' '
457! enddo
458!
459! write(stdoutUnit,*) ' '
460! write(stdoutUnit,*) 'Initial basal temperature'
461! do iRow = nRows, 1, -1
462! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
463! call mpas_log_write(' '
464! endif
465! do i = nCellsPerRow/2 - 2, nCellsPerRow
466! iCell = (iRow-1)*nCellsPerRow + i
467! write(stdoutUnit,'(f8.2)',advance='no') basalTemperature(iCell)
468! enddo
469! write(stdoutUnit,*) ' '
470! enddo
471!
472! do iCell = 1, nCellsSolve
473! if (indexToCellID(iCell) == config_stats_cell_ID) then
474! write(stdoutUnit,*) ' '
475! write(stdoutUnit,*) 'Diagnostic cell =', iCell
476! write(stdoutUnit,*) 'Thickness =', thickness(iCell)
477! write(stdoutUnit,*) ' '
478! write(stdoutUnit,*) 'Temperature profile (C):'
479! write(stdoutUnit,*) 'Sfc:', surfaceTemperature(iCell) - kelvin_to_celsius
480! do iLayer = 1, nVertLevels
481! write(stdoutUnit,*) iLayer, layerCenterSigma(iLayer), temperature(iLayer,iCell) - kelvin_to_celsius
482! enddo
483! write(stdoutUnit,*) 'Bed:', basalTemperature(iCell) - kelvin_to_celsius
484! endif
485! enddo
486!
487! endif ! shelf or dome test
488
489 endif ! config_print_thermal_info
490
491 endif ! restart file, input file, or linear
492
493 block => block % next
494 enddo
495
496 !TODO - Add a debug check for bad values
497 ! E.g., make sure the temperature read from a file is in Kelvin and not Celsius
498
499 ! halo updates
500 call mpas_timer_start("halo updates")
501
502 call mpas_dmpar_field_halo_exch(domain, 'surfaceTemperature')
503 call mpas_dmpar_field_halo_exch(domain, 'basalTemperature')
504 call mpas_dmpar_field_halo_exch(domain, 'temperature')
505
506 if (trim(config_thermal_solver) == 'enthalpy') then
507 ! prognostic variables are temperature and waterfrac, so need to update waterfrac too
508 call mpas_dmpar_field_halo_exch(domain, 'waterfrac')
509 endif
510
511 call mpas_timer_stop("halo updates")
512
513 ! === error check
514 if (err > 0) then
515 call mpas_log_write("An error has occurred in li_thermal_init.", mpas_log_err)
516 endif
517
518 call mpas_timer_stop("thermal init")
519
520 !--------------------------------------------------------------------
521 end subroutine li_thermal_init

526!
535!-----------------------------------------------------------------------
536
537 subroutine li_init_linear_temperature_in_column(&
538 nVertLevels, &
539 layerCenterSigma, &
540 thickness, &
541 surfaceAirTemperature, &
542 temperature, &
543 waterfrac, &
544 surfaceTemperature, &
545 basalTemperature)
546
547 !-----------------------------------------------------------------
548 ! input variables
549 !-----------------------------------------------------------------
550
551 integer, intent(in) :: &
552 nvertlevels
553
554 real(kind=rkind), dimension(nVertLevels), intent(in) :: &
555 layercentersigma
556
557 real(kind=rkind), intent(in) :: &
558 thickness, &
559 surfaceairtemperature
560
561 !-----------------------------------------------------------------
562 ! output variables
563 !-----------------------------------------------------------------
564
565 real(kind=rkind), dimension(nVertLevels), intent(out) :: &
566 temperature, &
567 waterfrac
568
569 real(kind=rkind), intent(out) :: &
570 surfacetemperature, &
571 basaltemperature
572
573 !-----------------------------------------------------------------
574 ! local variables
575 !-----------------------------------------------------------------
576
577 real (kind=rkind), dimension(nVertLevels) :: &
578 pmptemp ! pressure melting point temp in ice interior
579
580 real (kind=rkind) :: &
581 pmptemp_bed ! pressure melting point temp at bed
582
583 integer :: nvert
584
585 !TODO - Make pmpt_offset a namelist parameter?
586 real (kind=rkind), parameter :: &
587 pmpt_offset = 2.0_rkind ! offset of initial Tbed from pressure melting point temperature (K)
588 ! Note: pmtp_offset is positive for T < Tpmp
589
590
591 ! set the surface ice temperature to the surface air temperature (or 273.15, whichever is less)
592 surfacetemperature = min(surfaceairtemperature, kelvin_to_celsius)
593
594 ! compute the pressure melting point temperature in the column and at the bed
595 ! (in degrees C)
596
597 call pressure_melting_point_column(&
598 layercentersigma, &
599 thickness, &
600 pmptemp)
601
602 call pressure_melting_point(&
603 thickness, &
604 pmptemp_bed)
605
606 ! convert to Kelvin
607 pmptemp(:) = pmptemp(:) + kelvin_to_celsius
608 pmptemp_bed = pmptemp_bed + kelvin_to_celsius
609
610 ! set the basal temperature to slightly below the pressure melting point temperature
611 basaltemperature = pmptemp_bed - pmpt_offset
612
613 ! set the interior temperatures
614 ! make sure T <= Tpmp - pmpt_offset in column interior
615
616 temperature(:) = surfacetemperature + &
617 (basaltemperature - surfacetemperature) * layercentersigma(:)
618
619 temperature(:) = min(temperature(:), pmptemp(:) - pmpt_offset)
620
621 ! set waterfrac = 0
622 waterfrac(:) = 0.0_rkind
623
624 end subroutine li_init_linear_temperature_in_column

652 domain
653
654 !-----------------------------------------------------------------
655 ! output variables
656 !-----------------------------------------------------------------
657 integer, intent(out) :: err
658
659 !-----------------------------------------------------------------
660 ! local variables
661 !-----------------------------------------------------------------
662
663 type (block_type), pointer :: block
664
665 type (mpas_pool_type), pointer :: meshpool
666 type (mpas_pool_type), pointer :: geometrypool
667 type (mpas_pool_type), pointer :: thermalpool
668 type (mpas_pool_type), pointer :: velocitypool ! needed for mask subroutine

691 config_surface_air_temperature_value, & ! constant value of surface air temperature

693

1599
1600 !-----------------------------------------------------------------
1601 ! input/output variables
1602 !-----------------------------------------------------------------
1603 type (domain_type), intent(inout) :: &
1604 domain
1605
1606 !-----------------------------------------------------------------
1607 ! output variables
1608 !-----------------------------------------------------------------

1613 !-----------------------------------------------------------------
1614
1615 type (block_type), pointer :: block
1616
1617 type (mpas_pool_type), pointer :: meshpool
1618 type (mpas_pool_type), pointer :: geometrypool
1619 type (mpas_pool_type), pointer :: velocitypool
1620 type (mpas_pool_type), pointer :: thermalpool
1621 type (mpas_pool_type), pointer :: scratchpool
1622
1623 logical, pointer :: &
1624 config_print_thermal_info ! if true, print debug info
1625
1626 integer, pointer :: &
1627 ncells, & ! number of cells
1628 nedges, & ! number of edges
1629 nvertlevels ! number of vertical layers
1630
1631 integer, dimension(:), pointer :: &
1632 nedgesoncell ! number of edges on each cell
1633
1634 integer, dimension(:,:), pointer :: &
1635 cellsonedge, & ! indices for 2 cells on each edge
1636 edgesoncell ! indices for edges on each cell
1637
1638 real(kind=rkind), dimension(:), pointer :: &
1639 areacell ! area of each cell
1640
1641 real(kind=rkind), dimension(:), pointer :: &
1642 dcedge, & ! distance between neighboring cells across edge
1643 dvedge ! distance between eighboring vertices along edge

1645 real(kind=rkind), dimension(:), pointer :: &
1646 layercentersigma ! vertical coordinate at center of each layer

1653

1663 real (kind=rkind), dimension(:,:), pointer :: &
1664 heatdissipationedge ! heat dissipation on edges
1665
1666 integer, dimension(:), pointer :: &
1667 indextocellid ! list of global cell IDs
1668
1669 integer, pointer :: &
1670 config_stats_cell_id ! global ID for diagnostic cell
1671
1672 real (kind=rkind), pointer :: &
1673 config_flowlawexponent ! flow law exponent
1674
1675 real (kind=rkind) :: &

1677 weightedge ! edge weight for averaging to cell center
1678

1680 flowparamaedge ! flow parameter averaged to edge
1681
1682 integer :: icell, icell1, icell2, iedge, iedgeoncell, ilayer
1683
1684 ! Here are notes from the Glimmer calculation of heat dissipation:
1685 !
1686 ! "Two methods of doing this calculation:
1687 ! 1. find dissipation at u-pts and then average
1688 ! 2. find dissipation at H-pts by averaging quantities from u-pts
1689 ! (2) works best for eismint divide (symmetry) but (1) may be better for full expts"
1690 !
1691 ! Glimmer uses (2).
1692 !
1693 ! For MPAS LI we use the C-grid variant of (1); we find the dissipation at edges and then average to cell centers.
1694 ! The heating rate phi, defined on an edge, is given by
1695 !
1696 ! phi = 2 * A(T) * (sigma * rhoi * g * H * |grad(s)|)^(n+1)
1697 !
1698 ! where A(T) is the flow factor, sigma is the vertical coordinate of the layer,
1699 ! H is the ice thickness averaged to the edge, and grad(s) is the surface elevation gradient.
1700 !
1701 ! phi has units of W m^{-3}.
1702 ! The heat dissipation in deg/s is given by phi / (rhoi * cp_ice).
1703
1704 err = 0
1705
1706 ! block loop
1707 block => domain % blocklist
1708 do while (associated(block))
1709
1710 ! get pools
1711 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
1712 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
1713 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
1714 call mpas_pool_get_subpool(block % structs, 'thermal', thermalpool)
1715 call mpas_pool_get_subpool(block % structs, 'scratch', scratchpool)
1716
1717 ! get fields from the mesh pool
1718 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
1719 call mpas_pool_get_dimension(meshpool, 'nEdges', nedges)
1720 call mpas_pool_get_dimension(meshpool, 'nVertLevels', nvertlevels)
1721
1722 call mpas_pool_get_array(meshpool, 'layerCenterSigma', layercentersigma)
1723 call mpas_pool_get_array(meshpool, 'cellsOnEdge', cellsonedge)
1724 call mpas_pool_get_array(meshpool, 'edgesOnCell', edgesoncell)
1725 call mpas_pool_get_array(meshpool, 'nEdgesOnCell', nedgesoncell)
1726 call mpas_pool_get_array(meshpool, 'dcEdge', dcedge)
1727 call mpas_pool_get_array(meshpool, 'dvEdge', dvedge)
1728 call mpas_pool_get_array(meshpool, 'areaCell', areacell)
1729 call mpas_pool_get_array(meshpool, 'indexToCellID', indextocellid) ! diagnostic only
1730
1731 ! get fields from the geometry pool
1732 call mpas_pool_get_array(geometrypool, 'thickness', thickness)
1733 call mpas_pool_get_array(geometrypool, 'slopeEdge', slopeedge)
1734
1735 ! get fields from the velocity pool
1736 call mpas_pool_get_array(velocitypool, 'flowParamA', flowparama)
1737
1738 ! get fields from the thermal pool
1739 call mpas_pool_get_array(thermalpool, 'heatDissipation', heatdissipation)
1740
1741 ! get config parameters
1742 call mpas_pool_get_config(liconfigs, 'config_flowLawExponent', config_flowlawexponent)
1743 call mpas_pool_get_config(liconfigs, 'config_stats_cell_ID', config_stats_cell_id)
1744 call mpas_pool_get_config(liconfigs, 'config_print_thermal_info', config_print_thermal_info)
1745
1746 ! get scratch fields
1747 call mpas_pool_get_field(scratchpool, 'workLevelEdge', heatdissipationedgefield)
1748 call mpas_allocate_scratch_field(heatdissipationedgefield, .true.)
1749 heatdissipationedge => heatdissipationedgefield % array
1750
1751 allocate(flowparamaedge(nvertlevels))
1752
1753 if (config_print_thermal_info) then
1754 call mpas_log_write('Compute SIA heat dissipation')
1755 if (maxval(thickness) == 0.0_rkind) then
1756 return ! no thicknessm so just return
1757 endif
1758 endif
1759
1760 ! compute the heat dissipation on edges
1761
1762 do iedge = 1, nedges
1763
1764 ! identify the cells on this edge
1765 icell1 = cellsonedge(1,iedge)
1766 icell2 = cellsonedge(2,iedge)
1767
1768 if (icell1 >= 1 .and. icell1 <= ncells .and. icell2 >= 1 .and. icell2 <= ncells) then ! both cells exist
1769
1770 ! average the thickness and flow parameter to the edge
1771 thicknessedge = 0.5_rkind * (thickness(icell1) + thickness(icell2))
1772 flowparamaedge(:) = 0.5_rkind * (flowparama(:,icell1) + flowparama(:,icell2))
1773
1774 ! compute the dissipation at each level
1775 !TODO - Verify that this equation gives the right answer
1776 heatdissipationedge(:,iedge) = 2.0_rkind * flowparamaedge(:) * &
1777 (layercentersigma(:) * rhoi * gravity * thicknessedge * &
1778 abs(slopeedge(iedge))) ** (config_flowlawexponent + 1.0_rkind)
1779
1780 if (config_print_thermal_info .and. indextocellid(icell1) == config_stats_cell_id) then
1781! write(stdoutUnit,*) ' '
1782! write(stdoutUnit,*) 'Heat dissipation, iEdge, iCell1 =', iEdge, iCell1
1783! write(stdoutUnit,*) 'thicknessEdge:', thicknessEdge
1784! write(stdoutUnit,*) 'flowParamAEdge:', flowParamAEdge(:)
1785! write(stdoutUnit,*) 'slopeEdge:', slopeEdge(iEdge)
1786! write(stdoutUnit,*) 'heatDissipationEdge:', heatDissipationEdge(:,iEdge)
1787 endif
1788
1789 else ! one neighbor cell does not exist
1790 !TODO = Confirm that the dissipation is not needed at such edges
1791
1792 heatdissipationedge(:,iedge) = 0.0_rkind
1793
1794 endif
1795
1796 enddo ! iEdge
1797
1798 ! average the heat dissipation to cell centers
1799
1800 do icell = 1, ncells
1801
1802 heatdissipation(:,icell) = 0.0_rkind
1803
1804 if (config_print_thermal_info .and. indextocellid(icell) == config_stats_cell_id) then

2488 ! (1) arithmetic average: diffusivity = f*diffusivityCold + (1-f)*diffusivityTemperate

2492 ! intermediate values.

2523 denthalpytemp = enthalpytemp(k) - enthalpytemp(k-1) ! = dEnthalpy in cold ice, < dEnthalpy in temperate ice
2524
2525 if (abs(denthalpy) > 1.e-20_rkind * rho_water * latent_heat_ice) then
2526 avgfactor = max(0.0_rkind, denthalpytemp/denthalpy)
2527 avgfactor = min(1.0_rkind, avgfactor)
2528 else
2529 avgfactor = 0.0_rkind
2530 endif
2531
2532 if (harmonic_avg) then ! take a harmonic average
2533 ! This gives slower cooling of temperate layers and allows
2534 ! large temperature gradients between cold and temperate layers
2535 diffusivity(k) = 1.0_rkind / ((avgfactor/diffusivitycold) + (1.0_rkind - avgfactor)/diffusivitytemperate)
2536 else ! take an arithmetic average
2537 ! This gives faster cooling of temperate layers and smaller gradients
2538 diffusivity(k) = avgfactor*diffusivitycold + (1.0_rkind - avgfactor)*diffusivitytemperate
2539 endif
2540
2541 end do
2542
2543 ! Compute subdiagonal, diagonal, and superdiagonal matrix elements
2544 ! Assume backward Euler time stepping
2545
2546 ! upper boundary: set to surfaceEnthalpy = surface temperature*rhoi*cp_ice
2547 subd(1) = 0.0_rkind
2548 diag(1) = 1.0_rkind
2549 supd(1) = 0.0_rkind
2550 rhs(1) = surfaceenthalpy
2551
2552 ! ice interior, layers 1:nVertLevels (matrix elements 2:nVertLevels+1)
2553
2554 factor = deltat / thickness**2
2555
2556 subd(2:nvertlevels+1) = -factor * diffusivity(1:nvertlevels) * dsigmaterm(1:nvertlevels,1)
2557 supd(2:nvertlevels+1) = -factor * diffusivity(2:nvertlevels+1) * dsigmaterm(1:nvertlevels,2)
2558 diag(2:nvertlevels+1) = 1.0_rkind - subd(2:nvertlevels+1) - supd(2:nvertlevels+1)
2559 rhs(2:nvertlevels+1) = enthalpy(1:nvertlevels) + heatdissipation(1:nvertlevels) * deltat * rhoi * cp_ice
2560
2561 ! Note: heatDissipation has units of phi/rhoi/cp_ice, where phi has units of W m^-3..
2562 ! For an enthalpy calculation, we want just phi, hence heatDissipation * rhoi * cp_ice
2563
2564 ! Basal boundary:
2565 ! For grounded ice, a heat flux is applied.
2566 ! For floating ice, the basal temperature is held constant.
2567 ! If basalTemperature < T_pmp, then require dT/dsigma = H/k * (G + taub*ubas)
2568 ! That is, net heat flux at lower boundary must equal zero.
2569 ! If basalTemperature >= Tpmp, then set basalTemperature = Tpmp
2570
2571 if (floatingmask == 1) then
2572
2573 subd(nvertlevels+2) = 0.0_rkind
2574 diag(nvertlevels+2) = 1.0_rkind
2575 supd(nvertlevels+2) = 0.0_rkind
2576 rhs(nvertlevels+2) = basalenthalpy
2577
2578 else ! grounded ice
2579
2580 if (abs(temperature(nvertlevels) - pmptemperature(nvertlevels)) < 0.001_rkind) then
2581
2582 ! Positive-thickness basal temperate boundary Layer
2583 !WHL - Not sure whether this condition is ideal. It implies that basalEnthalpy = enthalpy(nVertLevels).
2584
2585 subd(nvertlevels+2) = -1.0_rkind
2586 diag(nvertlevels+2) = 1.0_rkind
2587 supd(nvertlevels+2) = 0.0_rkind
2588 rhs(nvertlevels+2) = 0.0_rkind
2589
2590 elseif (abs(basaltemperature - pmptemperaturebed) < 0.001_rkind) then ! melting
2591
2592 ! Zero-thickness basal temperate boundary layer
2593 ! Hold basal temperature at pressure melting point
2594 subd(nvertlevels+2) = 0.0_rkind
2595 diag(nvertlevels+2) = 1.0_rkind
2596 supd(nvertlevels+2) = 0.0_rkind
2597 rhs(nvertlevels+2) = pmptemperaturebed * rhoi*cp_ice
2598
2599 else ! cold ice at bed
2600
2601 ! Frozen at bed: Maintain balance of heat sources and sinks
2602 ! (conductive flux, geothermal flux, and basal friction)
2603
2604 ! Note: basalHeatFlux is generally >= 0, since defined as positive up
2605
2606 ! calculate dsigma for the bottom layer between the basal boundary and the temperature point above
2607 dsigmabot = (1.0_rkind - layercentersigma(nvertlevels))
2608
2609 subd(nvertlevels+2) = -1.0_rkind
2610 diag(nvertlevels+2) = 1.0_rkind
2611 supd(nvertlevels+2) = 0.0_rkind
2612 rhs(nvertlevels+2) = (basalfrictionflux + basalheatflux) * dsigmabot * thickness *rhoi * cp_ice / iceconductivity
2613
2614 endif ! melting or frozen
2615
2616 end if ! floating or grounded
2617
2618 end subroutine enthalpy_matrix_elements

2631
2632 subroutine basal_melt_grounded_ice(&
2633 config_thermal_solver, &
2634 deltat, &
2635 nCellsSolve, &
2636 nVertLevels, &
2637 layerInterfaceSigma, &
2638 layerCenterSigma, &
2639 iceMask, &
2640 floatingMask, &
2641 thickness, &
2642 temperature, &
2643 basalTemperature, &
2644 waterfrac, &
2645 enthalpy, &
2646 basalFrictionFlux, &
2647 basalHeatFlux, &
2648 basalConductiveFlux, &
2649 basalWaterThickness, &
2650 groundedBasalMassBal)
2651
2652 !-----------------------------------------------------------------
2653 ! Note: For the temperature solver, any internal temperatures above
2654 ! the pressure melting point are reset to Tpmp. Excess energy
2655 ! is applied toward melting with immediate drainage to the bed.
2656 ! For the enthalpy solver, any meltwater in excess of the maximum allowed
2657 ! meltwater fraction (0.01 by default) is drained to the bed.
2658 !-----------------------------------------------------------------
2659
2660 !-----------------------------------------------------------------
2661 ! input variables
2662 !-----------------------------------------------------------------
2663
2664 character(len=StrKIND), intent(in) :: &
2665 config_thermal_solver
2666
2667 real(kind=rkind), intent(in) :: &
2668 deltat
2669
2670 integer, intent(in) :: &
2671 ncellssolve
2672
2673 integer, intent(in) :: &
2674 nvertlevels
2675
2676 real(kind=rkind), dimension(:), intent(in) :: &
2677 layerinterfacesigma
2678
2679 real(kind=rkind), dimension(:), intent(in) :: &
2680 layercentersigma
2681
2682 real(kind=rkind), dimension(:,:), intent(in) :: &
2683 enthalpy
2684
2685 real(kind=rkind), dimension(:), intent(in) :: &
2686 thickness, &
2687 basalfrictionflux, &
2688 basalheatflux, &
2689 basalconductiveflux, &
2690 basalwaterthickness
2691
2692 integer, dimension(:), intent(in) :: &
2693 icemask, & !< Input: = 1 where ice exists (thickness > config_thermal_thickness), else = 0
2694 floatingmask
2695
2696 !-----------------------------------------------------------------
2697 ! input/output variables
2698 !-----------------------------------------------------------------
2699
2700 real(kind=rkind), dimension(:,:), intent(inout) :: &
2701 temperature
2702
2703 real(kind=rkind), dimension(:), intent(inout) :: &
2704 basaltemperature
2705
2706 real(kind=rkind), dimension(:,:), intent(inout) :: &
2707 waterfrac
2708
2709 !-----------------------------------------------------------------
2710 ! output variables
2711 !-----------------------------------------------------------------
2712
2713 real(kind=rkind), dimension(:), intent(out):: &
2714 groundedbasalmassbal
2715
2716 !-----------------------------------------------------------------
2717 ! local variables
2718 !-----------------------------------------------------------------
2719
2720 logical, pointer :: config_thermal_calculate_bmb
2721 integer :: k, icell
2722
2723 real(kind=rkind), dimension(nVertLevels) :: &
2724 pmptemperature ! pressure melting point temperature in ice interior
2725
2726 real(kind=rkind) :: pmptemperaturebed ! pressure melting point temperature at bed
2727 real(kind=rkind) :: netbasalflux ! heat flux available for basal melting (W/m^2)
2728 real(kind=rkind) :: layerthickness ! layer thickness (m)
2729 real(kind=rkind) :: meltenergy ! energy available for internal melting (J/m^2)
2730 real(kind=rkind) :: internalmeltrate ! internal melt rate, transferred to bed (m/s)
2731 real(kind=rkind) :: excesswater ! thickness of excess meltwater (m)
2732
2733 real(kind=rkind), parameter :: &
2734 maxwaterfrac = 0.01_rkind ! maximum allowed water fraction; excess drains to bed
2735
2736 real(kind=rkind), parameter :: &
2737 eps11 = 1.0e-11_rkind ! small number
2738
2739
2740 ! Compute melt rate for grounded ice
2741
2742 groundedbasalmassbal(:) = 0.0_rkind
2743
2744 do icell = 1, ncellssolve
2745
2746 if (icemask(icell) == 1 .and. floatingmask(icell) == 0) then ! ice is present and grounded
2747
2748 ! Compute basal mass balance
2749 ! Note: basalMassBal < 0 for melting, > 0 for freeze-on
2750 ! basalFrictionFlux >= 0 by definition
2751 ! basalHeatFlux is positive up, so generally basalHeatFlux >= 0
2752 ! basalConductiveFlux is positive down, so basalConductiveFlux < 0 for heat flowing
2753 ! from the bed toward the surface
2754 !
2755 ! This equation allows for freeze-on (basalMassBal > 0) if the conductive term
2756 ! (basalConductiveFlux, positive down) is carrying enough heat away from the boundary.
2757 ! But freeze-on requires a local water supply, basalWaterThickness > 0.
2758 ! When basalWaterThickness = 0, we reset the bed temperature to a value slightly below the melting point.
2759 !
2760 !TODO - For the enthalpy scheme, deal with the rare case that the bottom layer melts completely,
2761 ! and overlying layers with a different enthalpy also melt.
2762
2763 netbasalflux = basalfrictionflux(icell) + basalheatflux(icell) + basalconductiveflux(icell) ! W/m^2
2764
2765 if (abs(netbasalflux) < eps11) then ! netBasalFlux might be slightly different from zero
2766 ! because of rounding errors; if so, then zero out
2767 netbasalflux = 0.0_rkind
2768 endif
2769
2770 if (trim(config_thermal_solver) == 'enthalpy') then
2771 groundedbasalmassbal(icell) = -netbasalflux / (latent_heat_ice * rhoi - enthalpy(nvertlevels,icell))
2772 else ! temperature solver
2773 groundedbasalmassbal(icell) = -netbasalflux / (latent_heat_ice * rhoi) ! m/s
2774 endif
2775
2776 endif ! ice is present and grounded
2777
2778 ! Add internal melting
2779 ! Note: It is possible to have internal melting for floating ice.
2780 ! If so, this melting will be switched later from groundedBasalMassBal to floatingBasalMassBal
2781
2782 if (icemask(icell) == 1) then ! ice is present
2783
2784 if (trim(config_thermal_solver) == 'enthalpy') then
2785
2786 ! Add internal melting associated with waterfrac > maxWaterfrac (1%)
2787 !TODO - Add correction for rhoi/rhow here? Or melting ice that is already partly melted?
2788
2789 do k = 1, nvertlevels
2790 if (waterfrac(k,icell) > maxwaterfrac) then
2791
2792 ! compute melt rate associated with excess water
2793 excesswater = (waterfrac(k,icell) - maxwaterfrac) * thickness(icell) * (layerinterfacesigma(k+1) &
2794 - layerinterfacesigma(k)) ! m
2795 internalmeltrate = excesswater / deltat
2796
2797 ! transfer meltwater to the bed
2798 ! Note: It is possible to have internal melting for floating ice.
2799 ! If so, then this melting will later be switched from groundedBasalMassBall to floatingBasalMassBal.
2800 groundedbasalmassbal(icell) = groundedbasalmassbal(icell) - internalmeltrate ! m/s
2801
2802 ! reset waterfrac to max value
2803 waterfrac(k,icell) = maxwaterfrac
2804
2805 endif ! waterfrac > maxWaterfrac
2806 enddo ! k
2807
2808 elseif (config_thermal_solver == 'temperature') then
2809
2810 ! Add internal melting associated with T > Tpmp
2811
2812 call pressure_melting_point_column(&
2813 layercentersigma, &
2814 thickness(icell), &
2815 pmptemperature)
2816
2817 do k = 1, nvertlevels
2818 if (temperature(k,icell) > pmptemperature(k)) then
2819
2820 ! compute excess energy available for melting
2821 layerthickness = thickness(icell) * (layerinterfacesigma(k+1) - layerinterfacesigma(k)) ! m
2822 meltenergy = rhoi*cp_ice * (temperature(k,icell) - pmptemperature(k)) * layerthickness ! J/m^2
2823 internalmeltrate = meltenergy / (rhoi * latent_heat_ice * deltat) ! m/s
2824
2825 ! transfer meltwater to the bed
2826 ! Note: It is possible to have internal melting for floating ice.
2827 ! If so, then this melting will later be switched from groundedBasalMassBall to floatingBasalMassBal.
2828 groundedbasalmassbal(icell) = groundedbasalmassbal(icell) - internalmeltrate ! m/s
2829
2830 ! reset T to Tpmp
2831 temperature(k,icell) = pmptemperature(k)
2832
2833 endif ! temperature > pmpTemperature
2834 enddo ! k
2835
2836 endif ! config_thermal_solver
2837
2838 endif ! ice is present
2839
2840 ! Cap basal temperature at pressure melting point
2841
2842 if (icemask(icell) == 1 .and. floatingmask(icell) == 0) then ! ice is present and grounded
2843
2844 call pressure_melting_point(&
2845 thickness(icell), &
2846 pmptemperaturebed)
2847
2848 basaltemperature(icell) = min(basaltemperature(icell), pmptemperaturebed)
2849
2850 ! If freeze-on was computed above (basalMassBal > 0) and Tbed = Tpmp but no basal water is present,
2851 ! then set basalTemperature < Tpmp.
2852 ! Note: In the matrix element subroutines, we solve for Tbed (instead of holding it at Tpmp) when Tbed < -0.001.
2853 ! With an offset here of 0.01, we will solve for T_bed at the next timestep.
2854 ! Note: I don't think energy conservation is violated here, because no energy is associated with
2855 ! the infinitesimally thin layer at the bed.
2856
2857 if (groundedbasalmassbal(icell) > 0.0_rkind .and. basalwaterthickness(icell) == 0.0_rkind .and. &
2858 basaltemperature(icell) >= pmptemperaturebed) then
2859 basaltemperature(icell) = pmptemperaturebed - 0.01_rkind
2860 endif
2861
2862 endif ! ice is present and grounded
2863
2864 ! change units from m/s to kg/m2/s
2865 groundedbasalmassbal(icell) = groundedbasalmassbal(icell) * rhoi
2866
2867 enddo ! iCell
2868
2869 ! Check if we want thermal calculation to apply to BMB
2870 ! In some idealized configurations, we want to evolve temperature (and possibly have it
2871 ! coupled to velocity), but we do not want the geometry to evolve based on the temeprature
2872 ! evolution. In this case, we zero groundedBasalMassBal. This is cleaner than avoiding
2873 ! its calculation in the lines above, because in this situation we still want the various
2874 ! temperature fields to be "set back" to appropriate values as if melt had occurred.
2875 ! This violates conservation of energy but is desired for some tests.
2876 call mpas_pool_get_config(liconfigs, 'config_thermal_calculate_bmb', config_thermal_calculate_bmb)
2877 if (.not. config_thermal_calculate_bmb) then
2878 groundedbasalmassbal(:) = 0.0_rkind
2879 endif
2880
2881 end subroutine basal_melt_grounded_ice

2923

2925 lowersurface, &

2932 icemask, & !< Input: = 1 where ice exists (thickness > config_thermal_thickness), else = 0
2933 floatingmask
2934

2939 config_bmlt_float_xlimit
2942

2948 floatingbasalmassbal
2949

2986 err = 0

2990 if (trim(config_basal_mass_bal_float) == 'none') then

2997

2999 ! allow basal melt in ice-free ocean cells, in case ice is advected to those cells by the transport scheme
3000

3003 floatingbasalmassbal(:) = 0.0_rkind

3008 ! ice is present and floating, or ice-free ocean

3010 ! Provided xCell > bmlt_float_xlimit, prescribe a uniform basal melt rate.
3011 ! The default is 0.0, but for MISMIP+ the prescribed value of xlimit is 480 km.
3012 if (abs(xcell(icell)) >= config_bmlt_float_xlimit) then ! basal melting is allowed
3013 floatingbasalmassbal(icell) = -bmlt_float_rate
3014 endif
3015
3016 endif ! ice is present and floating
3017
3018 enddo ! iCell
3019
3020 ! change units from m/s to kg/m2/s
3021 floatingbasalmassbal(:) = floatingbasalmassbal(:) * rhoi
3022
3023 elseif (trim(config_basal_mass_bal_float) == 'mismip') then
3024
3025 ! compute melt rate (m/s) based on bed depth and cavity thickness
3026 ! The MISMIP+ formula is as follows:
3027 !
3028 ! bmlt_float = omega * tanh(H_c/H_0) * max(z_0 - z_d, 0)
3029 !
3030 ! where H_c = lsrf - topg is the cavity thickness
3031 ! z_d = lsrf - eus is the ice draft
3032 ! omega = a time scale = 0.2 yr^{-1} by default
3033 ! H_0 = 75 m by default
3034 ! z_0 = -100 m by default
3035
3036 ! allow basal melt in ice-free ocean cells, in case ice is advected to those cells by the transport scheme
3037
3038 floatingbasalmassbal(:) = 0.0_rkind
3039
3040 do icell = 1, ncellssolve
3041
3042 if ( floatingmask(icell) == 1 .or. &
3043 (bedtopography(icell) < config_sea_level .and. icemask(icell) == 0) ) then
3044 ! ice is present and floating, or ice-free ocean
3045
3046 hcavity = lowersurface(icell) - bedtopography(icell)
3047 zdraft = lowersurface(icell) - config_sea_level
3048 floatingbasalmassbal(icell) = -bmlt_float_omega * tanh(hcavity/bmlt_float_h0) * max(bmlt_float_z0 - &
3049 zdraft, 0.0_rkind)
3050
3051 endif ! ice is present and floating
3052 enddo ! iCell
3053
3054 ! change units from m/s to kg/m2/s
3055 floatingbasalmassbal(:) = floatingbasalmassbal(:) * rhoi
3056
3057 elseif (trim(config_basal_mass_bal_float) == 'seroussi') then
3058
3059 ! Melt rate parameterization from:
3060 ! Seroussi, H., Y. Nakayama, E. Larour, D. Menemenlis, M. Morlighem, E. Rignot, and A. Khazendar (2017), Continued retreat of Thwaites Glacier, West Antarctica, controlled by bed topography and ocean circulation, Geophys. Res. Lett., 1-9, doi:10.1002/2017GL072910.
3061 ! for Thwaites Glacier.
3062 ! Specifically, this is a linear fit of melt with shelf draft from the Supplemental Information, Figure S1.
3063 ! The linear relation is modified by a:
3064 ! * depth above which there is no melt (Antarctic Surface Water saturation)
3065 ! * a maximum melt rate (Circumpolar Deep Water saturation)
3066 ! * a depth below which melt stops increasing (minimum sill height)
3067
3068 call mpas_pool_get_config(liconfigs, 'config_basal_mass_bal_seroussi_amplitude', config_basal_mass_bal_seroussi_amplitude) ! meters
3069 call mpas_pool_get_config(liconfigs, 'config_basal_mass_bal_seroussi_period', config_basal_mass_bal_seroussi_period) ! years
3070 call mpas_pool_get_config(liconfigs, 'config_basal_mass_bal_seroussi_phase', config_basal_mass_bal_seroussi_phase) ! cycles
3071
3072 slopeser = 0.088_rkind ! slope of relation between depth and melt rate (melt (m/yr) per depth (m))
3073 interceptser = -100.0_rkind ! depth (m) at which melting goes to 0 (negative meaning below sea level)
3074 maxmeltser = 50.0_rkind ! maximum allowable melt rate (m/yr) (positive meaning melting)
3075 silldepth = -650.0_rkind ! depth below which melt stops increasing (m) (negative meaning below sea level)
3076
3077 if (config_basal_mass_bal_seroussi_period <= 0.0_rkind) then
3078 call mpas_log_write("Value for config_basal_mass_bal_seroussi_period has to be a positive real value.", mpas_log_err)

3080 endif

3083 interceptser = interceptser + config_basal_mass_bal_seroussi_amplitude * &
3084 sin( (2.0_rkind * pii / config_basal_mass_bal_seroussi_period) * (dayssincestart/365.0_rkind) &
3085 + 2.0_rkind * pii * config_basal_mass_bal_seroussi_phase)

3088 floatingbasalmassbal(:) = 0.0_rkind
3089
3090 do icell = 1, ncellssolve
3091
3092 ! Shut off melt at an arbitrary shallow depth to discourage ice from disappearing.
3093 if ( (floatingmask(icell) == 1) .and. (lowersurface(icell) < -10.0_rkind) ) then
3094 ! ice is present and floating
