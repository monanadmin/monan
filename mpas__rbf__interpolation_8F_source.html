43 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
44 ! Routines for computing scalar interpolaiton coefficients in 3D (coplanar points
45 ! in 3D) with support for either constant or constant and linear basis
46 ! functions in addition to RBFs. In constrast to the two subroutines
47 ! above, these coefficients are valid for computing the value of the
48 ! interpolant at a fixe point in space using function values that may
49 ! vary (e.g., in time) at each of the interpolation "source" points.
50 ! The last 3 routines can be used to compute coefficients for imposing both Neumann
51 ! and Dirichlet boundary conditions.
52 ! Pseudocode for function reconstruction at the destinationPoint is as follows
53 ! Dirichlet: functionAtDestination = sum(functionAtSources*dirichletCoefficients)
54 ! Neumann: functionAtDestination = sum(functionOrDerivAtSources*neumannCoefficients)
55 ! where functionOrDerivAtSource = functionAtSources where .not.(isInterface)
56 ! = functionNormalDerivAtSources where isInterface
57 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
58 public :: mpas_rbf_interp_func_3d_sca_const_dir_comp_coeffs, &

63 mpas_rbf_interp_func_3d_sca_lin_dir_neu_comp_coeffs

65 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
66 ! Routines for computing vector interpolaiton coefficients in 3D (coplanar points
67 ! in 3D) with support for only constant basis functions in addition to RBFs.
68 ! (Linear basis functions can cause problems: a linear vortex flow u = y xHat - x yHat
69 ! cannot be reconstructed from vectors that are only normal to the cell edges in 2D.
70 ! Therefore, we don't support them). As with the scalar 3D subroutines
71 ! above, these coefficients are valid for computing the value of the
72 ! interpolant at a fixe point in space using function values that may
73 ! vary (e.g., in time) at each of the interpolation "source" points.
74 ! The user supplies to these routines a set of sourcePoints and unitVectors
75 ! as well as a destinationPoint and, for the last 2 routines, flags for
76 ! which points are tangent to the interface and which of the supplied unitVectors
77 ! is the normal at the corresponding point.
78 !
79 ! The first two routines compute coefficients that can be used to interpolate
80 ! a vector function to the destination point given "function dot unitVector" values
81 ! at each source point. These routines are useful, for example, for reconstructing
82 ! the full vector velocity at cell centers from the normal components of the velocity
83 ! at cell faces (or cell edges in 2D), or for computing the full velocity at an
84 ! immersed boundary image point based on the normal velocity at several faces and
85 ! the full velocity at boundary points (e.g., a no-slip boundary condition).
86 !
87 ! The last two routines compute coefficients that can be used to interpolate
88 ! a vector function to the destination point given "function dot unitVector" values
89 ! at non-tangent source point and "dFunction/dn dot unitVector" values at
90 ! tangent source point. These routines are useful, for example, for computing the
91 ! full velocity at an immersed boundary image point based on the normal velocity at
92 ! several faces, the normal velocity at boundary points (e.g., u dot n = 0 for a
93 ! no-penetration boundary condition on a fixed boundary), and the normal derivative
94 ! of the tangential components of velocity at the boundary points (e.g., a free-slip
95 ! boundary condition).
96 ! Pseudocode for function reconstruction at the destinationPoint is as follows
97 ! dirichlet: functionAtDestination_i = sum_j(functionDotUnitVectorAtSources_j*coefficients_j,i)
98 ! for i = x,y,z
99 ! tangentNeumann: functionAtDestination_i &
100 ! = sum_(j where .not isTangent) (functionDotUnitVectorAtSources_j*coefficients_j,i) &
101 ! + sum_(j where isTangent) ((dFunction/dn dot UnitVector)_j*coefficients_j,i)
102 ! for i = x,y,z
103 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
104 public :: mpas_rbf_interp_func_3d_vec_const_dir_comp_coeffs, &
105 mpas_rbf_interp_func_3d_plane_vec_const_dir_comp_coeffs!, &

304 subroutine mpas_rbf_interp_loc_2d_sca_const_eval_with_derivs(fieldCount, coeffCount, &!{{{
305 pointCount, coefficients, evaluationPoint, points, alpha, derivs)
306 integer, intent(in) :: fieldcount
307 integer, intent(in) :: coeffcount
308 integer, intent(in) :: pointcount
309 real(kind=rkind), dimension(coeffCount, fieldCount), intent(in) :: coefficients
310 real(kind=rkind), dimension(2), intent(in) :: evaluationpoint
311 real(kind=rkind), dimension(pointCount,2), intent(in) :: points
312 real(kind=rkind), intent(in) :: alpha
313
314 real(kind=rkind), dimension(6,fieldCount), intent(out) :: derivs
315
316 integer :: pointindex
317 real(kind=rkind) :: x, y, rsquared, rbfvalue, rbfderivoverr, rbfsecondderiv
318
319 derivs = 0.0
320 do pointindex = 1, pointcount
321 x = (evaluationpoint(1) - points(pointindex,1))
322 y = (evaluationpoint(2) - points(pointindex,2))
323 rsquared = x**2 + y**2
324 call mpas_evaluate_rbf_and_derivs(rsquared/alpha**2, rbfvalue, rbfderivoverr, rbfsecondderiv)
325 rbfderivoverr = rbfderivoverr/alpha**2
326 rbfsecondderiv = rbfsecondderiv/alpha**2
327 if(rsquared/alpha**2 < 1e-7) then
328 ! terms 2,3 and 5 are zero by radial symmetry of the RBF
329 derivs(1,:) = derivs(1,:) + coefficients(pointindex,:)*rbfvalue
330 derivs(4,:) = derivs(4,:) + coefficients(pointindex,:)*rbfsecondderiv
331 derivs(6,:) = derivs(6,:) + coefficients(pointindex,:)*rbfsecondderiv
332 else
333 call mpas_evaluate_rbf_and_derivs(rsquared, rbfvalue, rbfderivoverr, rbfsecondderiv)
334 derivs(1,:) = derivs(1,:) + coefficients(pointindex,:)*rbfvalue
335 derivs(2,:) = derivs(2,:) + coefficients(pointindex,:)*rbfderivoverr*x
336 derivs(3,:) = derivs(3,:) + coefficients(pointindex,:)*rbfderivoverr*y
337 derivs(4,:) = derivs(4,:) + coefficients(pointindex,:) &
338 * (rbfsecondderiv*x**2 + rbfderivoverr*y**2)/rsquared
339 derivs(5,:) = derivs(5,:) + coefficients(pointindex,:) &
340 * (rbfsecondderiv - rbfderivoverr)*x*y/rsquared
341 derivs(6,:) = derivs(6,:) + coefficients(pointindex,:) &
342 * (rbfsecondderiv*y**2 + rbfderivoverr*x**2)/rsquared
343 end if
344 end do
345 derivs(1,:) = derivs(1,:) + coefficients(pointcount+1,:)
346 end subroutine mpas_rbf_interp_loc_2d_sca_const_eval_with_derivs!}}}

1783!
1784! subroutine to perform the partial-pivoting Gaussian elimination.
1785! A(N,N) is the original matrix in the input and transformed matrix
1786! plus the pivoting element ratios below the diagonal in the output.
1787! INDX(N) records the pivoting order. Copyright (c) Tao Pang 2001.
1788!
1789 IMPLICIT NONE

1791 integer, INTENT (OUT), DIMENSION (N) :: indx

1794 real(kind=rkind) :: c1,pi,pi1,pj
1795 real(kind=rkind), DIMENSION (N) :: c
1796!
1797! Initialize the index
1798!
1799 DO i = 1, n
1800 indx(i) = i
1801 END DO
1802!
1803! Find the rescaling factors, one from each row
1804!
1805 DO i = 1, n
1806 c1= 0.0
1807 DO j = 1, n
1808 !C1 = AMAX1(C1,ABS(A(I,J)))
1809 c1 = max(c1,abs(a(i,j)))
1810 END DO
1811 c(i) = c1
1812 END DO
1813!
1814! Search the pivoting (largest) element from each column
1815!
1816 DO j = 1, n-1
1817 pi1 = 0.0
1818 DO i = j, n
1819 pi = abs(a(indx(i),j))/c(indx(i))
1820 IF (pi.GT.pi1) THEN
1821 pi1 = pi
1822 k = i
1823 ENDIF
1824 END DO

1827!

1829 indx(j) = indx(k)
1830 indx(k) = itmp

1832 pj = a(indx(i),j)/a(indx(j),j)

1835!

1839!
