80 function ocn_init_mode_init(domain, startTimeStamp) result(ierr)!{{{
81
82 type (domain_type), intent(inout) :: domain
83 character(len=*), intent(out) :: starttimestamp
84 integer :: ierr
85
86 real (kind=rkind) :: dt
87 type (block_type), pointer :: block
88
89 integer :: err_tmp
90 integer, pointer :: nvertlevels
91 real (kind=rkind) :: maxdensity, maxdensity_global
92 real (kind=rkind), dimension(:), pointer :: meshdensity
93 type (mpas_pool_type), pointer :: meshpool
94 type (mpas_pool_type), pointer :: diagnosticspool
95
96 character (len=StrKIND), pointer :: xtime
97 type (mpas_time_type) :: starttime
98 type (mpas_timeinterval_type) :: timestep
99
100 logical, pointer :: config_do_restart, config_filter_btr_mode, config_conduct_tests
101 logical, pointer :: config_write_stats_on_startup
102 character (len=StrKIND), pointer :: config_vert_coord_movement, config_pressure_gradient_type
103 real (kind=rkind), pointer :: config_maxmeshdensity
104
105 ierr = 0
106
107 !
108 ! Set startTimeStamp based on the start time of the simulation clock
109 !
110 starttime = mpas_get_clock_time(domain % clock, mpas_start_time, err_tmp)
111 call mpas_get_time(starttime, datetimestring=starttimestamp)
112 ierr = ior(ierr, err_tmp)
113
114 ! Setup ocean config pool
115 call ocn_constants_init(domain % configs, domain % packages)
116
117 !
118 ! Read input data for model
119 !
120 call mpas_timer_start('io_read', .false.)
121 call mpas_stream_mgr_read(domain % streamManager, streamid='input_init', ierr=err_tmp)
122 call mpas_timer_stop('io_read')
123
124 call mpas_timer_start('reset_io_alarms', .false.)
125 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='input_init', ierr=err_tmp)
126 ! call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=MPAS_STREAM_OUTPUT, ierr=err_tmp)
127 call mpas_timer_stop('reset_io_alarms')
128
129 ! Read the remaining input streams
130 call mpas_timer_start('io_read', .false.)
131 call mpas_stream_mgr_read(domain % streamManager, ierr=err_tmp)
132 ierr = ior(ierr, err_tmp)
133 call mpas_timer_stop('io_read')
134 call mpas_timer_start('reset_io_alarms', .false.)
135 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_input, ierr=err_tmp)
136 ierr = ior(ierr, err_tmp)
137 call mpas_timer_stop('reset_io_alarms')
138
139 ! Initialize submodules before initializing blocks.
140 call ocn_equation_of_state_init(err_tmp)
141 ierr = ior(ierr, err_tmp)
142 if(ierr.eq.1) then
143 call mpas_log_write('An error was encountered while initializing the MPAS-Ocean init mode', mpas_log_crit)
144 endif
145
146 !
147 ! Initialize core
148 !
149 timestep = mpas_get_clock_timestep(domain % clock, ierr=err_tmp)
150 call mpas_get_timeinterval(timestep, dt=dt)
151
152 block => domain % blocklist
153 do while (associated(block))
154 call mpas_pool_get_subpool(block % structs, 'diagnostics', diagnosticspool)
155 call mpas_pool_get_array(diagnosticspool, 'xtime', xtime)
156 xtime = starttimestamp
157 block => block % next
158 end do
159
160 ! Expand sphere if it needs to be expanded
161 call ocn_init_expand_sphere(domain, domain % streamManager, a, ierr)
162
163 end function ocn_init_mode_init!}}}

176 function ocn_init_mode_setup_clock(core_clock, configs) result(ierr)!{{{
177
178 implicit none
179
180 type (mpas_clock_type), intent(inout) :: core_clock
181 type (mpas_pool_type), intent(inout) :: configs
182 integer :: ierr
183
184 type (mpas_time_type) :: starttime, stoptime, alarmstarttime
185 type (mpas_timeinterval_type) :: runduration, timestep, alarmtimestep
186 character(len=StrKIND) :: restarttimestamp
187 character(len=StrKIND), pointer :: config_start_time, config_stop_time, config_run_duration
188 character(len=StrKIND), pointer :: config_dt, config_restart_timestamp_name
189 integer :: err_tmp
190
191 ierr = 0
192
193 call mpas_pool_get_config(configs, 'config_dt', config_dt)
194 call mpas_pool_get_config(configs, 'config_start_time', config_start_time)
195 call mpas_pool_get_config(configs, 'config_stop_time', config_stop_time)
196 call mpas_pool_get_config(configs, 'config_run_duration', config_run_duration)
197 call mpas_pool_get_config(configs, 'config_restart_timestamp_name', config_restart_timestamp_name)
198
199 call mpas_set_time(starttime, datetimestring=config_start_time, ierr=err_tmp)
200 call mpas_set_timeinterval(timestep, timestring=config_dt, ierr=err_tmp)
201 if (trim(config_run_duration) /= "none") then
202 call mpas_set_timeinterval(runduration, timestring=config_run_duration, ierr=err_tmp)
203 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, runduration=runduration, ierr=err_tmp)
204
205 if (trim(config_stop_time) /= "none") then
206 call mpas_set_time(curr_time=stoptime, datetimestring=config_stop_time, ierr=err_tmp)
207 if(starttime + runduration /= stoptime) then
208 call mpas_log_write( 'Warning: config_run_duration and config_stop_time are inconsitent: using config_run_duration.')
209 end if
210 end if
211 else if (trim(config_stop_time) /= "none") then
212 call mpas_set_time(curr_time=stoptime, datetimestring=config_stop_time, ierr=err_tmp)
213 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, stoptime=stoptime, ierr=err_tmp)
214 else
215 call mpas_log_write( ' Warning: config_run_duration and config_start_time were "none", setting run duration to 1 second.')
216 call mpas_set_timeinterval(runduration, timestring="0000_00:00:01", ierr=err_tmp)
217 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, runduration=runduration, ierr=err_tmp)
218 end if
219
220 end function ocn_init_mode_setup_clock!}}}
