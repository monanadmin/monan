31! subroutines in mpas_atmphys_driver_pbl:
32! ---------------------------------------

40! -----------------------------------

50! * throughout the sourcecode, replaced all "var_struct" defined arrays by local pointers.
51! Laura D. Fowler (laura@ucar.edu) / 2014-04-22.

53! Laura D. Fowler (laura@ucar.edu) / 2014-05-15.
54! * renamed "ysu" with "bl_ysu".
55! Laura D. Fowler (laura@ucar.edu) / 2016-03-25.

253!=================================================================================================================
254 subroutine pbl_from_mpas(configs,mesh,sfc_input,diag_physics,tend_physics,its,ite)
255!=================================================================================================================
256
257!input arguments:
258 type(mpas_pool_type),intent(in):: configs
259 type(mpas_pool_type),intent(in):: mesh
260 type(mpas_pool_type),intent(in):: diag_physics
261 type(mpas_pool_type),intent(in):: sfc_input
262 type(mpas_pool_type),intent(in):: tend_physics
263
264 integer,intent(in):: its,ite
265
266!local variables:
267 integer:: i,k,j
268
269!local pointers:
270 character(len=StrKIND),pointer:: pbl_scheme
271
272 real(kind=rkind),dimension(:),pointer:: hfx,hpbl,qfx,ust,wspd,xland,znt
273 real(kind=rkind),dimension(:),pointer:: delta,wstar
274
275!local pointers for YSU scheme:
276 logical,pointer:: config_ysu_pblmix
277 real(kind=rkind),dimension(:),pointer:: br,fh,fm,regime,u10,v10
278 real(kind=rkind),dimension(:,:),pointer:: rthratenlw,rthratensw
279
280!local pointers for MYNN scheme:
281 real(kind=rkind),pointer:: len_disp
282 real(kind=rkind),dimension(:),pointer :: meshdensity
283 real(kind=rkind),dimension(:),pointer :: ch,qsfc,qcg,rmol,skintemp
284 real(kind=rkind),dimension(:,:),pointer:: cov,qke,qsq,tsq,sh3d,tke_pbl,qke_adv,el_pbl
285
286!-----------------------------------------------------------------------------------------------------------------
287
288 call mpas_pool_get_config(configs,'config_pbl_scheme',pbl_scheme)
289
290 call mpas_pool_get_array(diag_physics,'hfx' ,hfx )
291 call mpas_pool_get_array(diag_physics,'hpbl' ,hpbl )
292 call mpas_pool_get_array(diag_physics,'qfx' ,qfx )
293 call mpas_pool_get_array(diag_physics,'ust' ,ust )
294 call mpas_pool_get_array(diag_physics,'wspd' ,wspd )
295 call mpas_pool_get_array(diag_physics,'znt' ,znt )
296 call mpas_pool_get_array(diag_physics,'delta' ,delta )
297 call mpas_pool_get_array(diag_physics,'wstar' ,wstar )
298
299 call mpas_pool_get_array(sfc_input ,'xland' ,xland )
300
301 do j = jts,jte
302 do i = its,ite
303 !from surface-layer model:
304 hfx_p(i,j) = hfx(i)
305 hpbl_p(i,j) = hpbl(i)
306 qfx_p(i,j) = qfx(i)
307 ust_p(i,j) = ust(i)
308 wspd_p(i,j) = wspd(i)
309 xland_p(i,j) = xland(i)
310 kpbl_p(i,j) = 1
311 znt_p(i,j) = znt(i)
312 delta_p(i,j) = delta(i)
313 wstar_p(i,j) = wstar(i)
314 !... ocean currents are set to zero:
315 uoce_p(i,j) = 0._rkind
316 voce_p(i,j) = 0._rkind
317 enddo
318 enddo
319
320 pbl_select: select case (trim(pbl_scheme))
321
322 case("bl_ysu")
323 call mpas_pool_get_config(configs,'config_ysu_pblmix',config_ysu_pblmix)
324
325 call mpas_pool_get_array(diag_physics,'br' ,br )
326 call mpas_pool_get_array(diag_physics,'fm' ,fm )
327 call mpas_pool_get_array(diag_physics,'fh' ,fh )
328 call mpas_pool_get_array(diag_physics,'regime',regime)
329 call mpas_pool_get_array(diag_physics,'u10' ,u10 )
330 call mpas_pool_get_array(diag_physics,'v10' ,v10 )
331
332 call mpas_pool_get_array(tend_physics,'rthratenlw',rthratenlw)
333 call mpas_pool_get_array(tend_physics,'rthratensw',rthratensw)
334
335 ysu_pblmix = 0
336 if(config_ysu_pblmix) ysu_pblmix = 1
337
338 do j = jts,jte
339 do i = its,ite
340 !from surface-layer model:
341 br_p(i,j) = br(i)
342 psim_p(i,j) = fm(i)
343 psih_p(i,j) = fh(i)
344 regime_p(i,j) = regime(i)
345 u10_p(i,j) = u10(i)
346 v10_p(i,j) = v10(i)
347 !initialization for YSU PBL scheme:
348 ctopo_p(i,j) = 1._rkind
349 ctopo2_p(i,j) = 1._rkind
350 enddo
351 enddo
352
353 do j = jts,jte
354 do k = kts,kte
355 do i = its,ite
356 exch_p(i,k,j) = 0._rkind
357 rthraten_p(i,k,j) = rthratenlw(k,i) + rthratensw(k,i)
358 enddo
359 enddo
360 enddo
361
362 case("bl_mynn")
363 call mpas_pool_get_config(configs,'config_len_disp',len_disp)
364 call mpas_pool_get_array(mesh,'meshDensity',meshdensity)
365
366 call mpas_pool_get_array(sfc_input ,'skintemp',skintemp)
367 call mpas_pool_get_array(diag_physics,'ch' ,ch )
368 call mpas_pool_get_array(diag_physics,'qcg' ,qcg )
369 call mpas_pool_get_array(diag_physics,'qsfc' ,qsfc )
370 call mpas_pool_get_array(diag_physics,'rmol' ,rmol )
371
372 call mpas_pool_get_array(diag_physics,'el_pbl' ,el_pbl )
373 call mpas_pool_get_array(diag_physics,'cov' ,cov )
374 call mpas_pool_get_array(diag_physics,'qke' ,qke )
375 call mpas_pool_get_array(diag_physics,'qke_adv',qke_adv )
376 call mpas_pool_get_array(diag_physics,'qsq' ,qsq )
377 call mpas_pool_get_array(diag_physics,'tsq' ,tsq )
378 call mpas_pool_get_array(diag_physics,'tke_pbl',tke_pbl )
379 call mpas_pool_get_array(diag_physics,'sh3d' ,sh3d )
380
381 do j = jts,jte
382 do i = its,ite
383 dx_p(i,j) = len_disp / meshdensity(i)**0.25
384 ch_p(i,j) = ch(i)
385 qcg_p(i,j) = qcg(i)
386 qsfc_p(i,j) = qsfc(i)
387 rmol_p(i,j) = rmol(i)
388 tsk_p(i,j) = skintemp(i)
389 !... no gravitational settling of fog/cloud droplets (grav_settling = 0):
390 vdfg_p(i,j) = 0._rkind
391 enddo
392 enddo
393
394 do j = jts,jte
395 do k = kts,kte
396 do i = its,ite
397 elpbl_p(i,k,j) = el_pbl(k,i)
398 cov_p(i,k,j) = cov(k,i)
399 qke_p(i,k,j) = qke(k,i)
400 qsq_p(i,k,j) = qsq(k,i)
401 tsq_p(i,k,j) = tsq(k,i)
402 tkepbl_p(i,k,j) = tke_pbl(k,i)
403 qkeadv_p(i,k,j) = qke_adv(k,i)
404 sh3d_p(i,k,j) = sh3d(k,i)
405 rniblten_p(i,k,j) = 0._rkind
406
407 !... outputs:
408 dqke_p(i,k,j) = 0._rkind
409 qbuoy_p(i,k,j) = 0._rkind
410 qdiss_p(i,k,j) = 0._rkind
411 qshear_p(i,k,j) = 0._rkind
412 qwt_p(i,k,j) = 0._rkind
413 enddo
414 enddo
415 enddo
416
417 case default
418
419 end select pbl_select
420
421 do j = jts,jte
422 do k = kts,kte
423 do i = its,ite
424 rublten_p(i,k,j) = 0._rkind
425 rvblten_p(i,k,j) = 0._rkind
426 rthblten_p(i,k,j) = 0._rkind
427 rqvblten_p(i,k,j) = 0._rkind
428 rqcblten_p(i,k,j) = 0._rkind
429 rqiblten_p(i,k,j) = 0._rkind
430
431 kzh_p(i,k,j) = 0._rkind
432 kzm_p(i,k,j) = 0._rkind
433 kzq_p(i,k,j) = 0._rkind
434 enddo
435 enddo
436 enddo
437
438 end subroutine pbl_from_mpas
