45def xtime2numtime(xtime):
46 """Define a function to convert xtime character array to numeric time values using datetime objects"""
47 # First parse the xtime character array into a string
48 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
49
50 dt = []
51 for stritem in xtimestr:
52 itemarray = stritem.strip().replace('_', '-').replace(':', '-').split('-') # Get an array of strings that are Y,M,D,h,m,s
53 results = [int(i) for i in itemarray]
54 if (results[0] < 1900): # datetime has a bug where years less than 1900 are invalid on some systems
55 results[0] += 1900
56 dt.append( datetime.datetime(*results) ) # * notation passes in the array as arguments
57
58 numtime = netCDF4.date2num(dt, units='seconds since '+str(dt[0])) # use the netCDF4 module's function for converting a datetime to a time number
59 return numtime / (3600.0*24.0*365.0) # in years

61def xtime2numtimeMy(xtime):
62 """Define a function to convert xtime character array to numeric time values using local arithmetic"""
63 # First parse the xtime character array into a string
64 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
65
66 numtime = np.zeros( (len(xtimestr),) )
67 ii = 0
68 dayOfMonthStart = [1, 32, 60, 91, 121, 152, 182, 213, 244, 274, 305, 335]
69 for stritem in xtimestr:
70 itemarray = stritem.strip().replace('_', '-').replace(':', '-').split('-') # Get an array of strings that are Y,M,D,h,m,s
71 results = [int(i) for i in itemarray]
72 numtime[ii] = results[0] + (dayOfMonthStart[results[1]-1]-1 + results[2]) / 365.0 # decimal year
73 ii += 1
74 return numtime

79class modelRun:
80 def __init__(self, run):
81 '''
82 This reads results from a model run and saves and analyzes the needed results.
83
84 run = name of subdir in which the run was performed
85 '''
86
87 # some metadata about run
88 if 'amp' in run:
89 self.amp = float(run[3:6])
90 self.per = float(run[10:12])
91 self.pha = float(run[16:])
92 else:
93 self.amp = 0.0
94 self.per = 0.0
95 self.pha = 0.0
96
97 #print self.amp, self.per, self.pha
98
99 f = netCDF4.Dataset(run + '/' + outfname, 'r')
100 self.nt = len(f.dimensions['Time'])
101 self.yrs = np.zeros((self.nt,))
102 #yrs = f.variables['daysSinceStart'][:] / 365.0
103 self.xtimes = f.variables['xtime'][:]
104
105
106 self.yrs = xtime2numtimeMy(self.xtimes)
107 self.dyrs = self.yrs[1:] - self.yrs[0:-1]
108
109
110 # resampled version of time array - needed for filtering.
111 # (Filtering needed b/c the occasional tiny time step that the model is exhibiting leads to inaccurate (noisy) derivatives)
112 resampEndtime = self.yrs.max()
113 resampEndtime = 300.0
114 self.resampYrs = np.linspace(0.0, resampEndtime, num=resampEndtime*12*2)
115 self.dresampYrs = self.resampYrs[1:] - self.resampYrs[0:-1]
116
117 self.melt = f.variables['avgSubshelfMelt'][:]
118 self.resampMelt = np.interp(self.resampYrs, self.yrs, self.melt) # generate y values for each x
119 self.totalmelt = f.variables['totalFloatingBasalMassBal'][:] / 1.0e12
120 self.cumumelt = self.totalmelt*0.0
121 for t in np.arange(self.nt-1)+1:
122 self.cumumelt[t] = self.cumumelt[t-1] + self.totalmelt[t-1] * self.dyrs[t-1]
123
124 self.VAF = f.variables['volumeAboveFloatation'][:] / 1.0e12 * rhoi
125 self.resampVAF = np.interp(self.resampYrs, self.yrs, self.VAF) # generate y values for each x
126 self.VAFsmooth = scipy.signal.savgol_filter(self.resampVAF, window_length=windowLength, polyorder=1)
127 #VAFrate = (VAF[1:]-VAF[0:-1] ) / dyrs
128 self.VAFsmoothrate = (self.VAFsmooth[1:] - self.VAFsmooth[0:-1] ) / self.dresampYrs
129
130 self.GA = f.variables['groundedIceArea'][:] / 1.0e3**2
131 self.resampGA = np.interp(self.resampYrs, self.yrs, self.GA) # generate y values for each x
132 self.GAsmooth = scipy.signal.savgol_filter(self.resampGA, window_length=windowLength, polyorder=1)
133 self.GArate = (self.GAsmooth[1:] - self.GAsmooth[0:-1]) / self.dresampYrs
134
135
136 f.close()
