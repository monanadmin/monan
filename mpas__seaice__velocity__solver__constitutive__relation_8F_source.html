189
190 ! convert from stress11 to stress1 etc
191 straindivergence = strain11 + strain22
192 straintension = strain11 - strain22
193 strainshearing = strain12 * 2.0_rkind
194
195 stress1 = stress11 + stress22
196 stress2 = stress11 - stress22
197
198 ! perform the constituitive relation
199 delta = sqrt(straindivergence**2 + (straintension**2 + strainshearing**2) / eccentricitysquared)
200
201 pressurecoefficient = icepressure / max(delta,puny)
202 replacementpressure = pressurecoefficient * delta
203
204 pressurecoefficient = (pressurecoefficient * dtelastic) / (2.0_rkind * dampingtimescale)
205
206 denominator = 1.0_rkind + (0.5_rkind * dtelastic) / dampingtimescale
207
208 stress1 = (stress1 + pressurecoefficient * (straindivergence - delta)) / denominator
209 stress2 = (stress2 + (pressurecoefficient / eccentricitysquared) * straintension ) / denominator
210 stress12 = (stress12 + (pressurecoefficient / eccentricitysquared) * strainshearing * 0.5_rkind) / denominator
211
212 ! convert back
213 stress11 = 0.5_rkind * (stress1 + stress2)
214 stress22 = 0.5_rkind * (stress1 - stress2)
215
216 end subroutine seaice_evp_constitutive_relation
