142 call mpas_pool_get_array(meshpool, 'refBottomDepth', refbottomdepth)
143 call mpas_pool_get_config(ocnconfigs, 'config_surface_buoyancy_depth', config_surface_buoyancy_depth)
144
145 allocate(weights(nvertlevels+1))
146 weights = 0.0_rkind
147 weights(1) = 1.0_rkind
148 avals(:)=0.0_rkind
149 kvals(:)=0.0_rkind
150
151 call mpas_pool_get_config(ocnconfigs, 'config_sw_absorption_type', config_sw_absorption_type)

153
154 call mpas_pool_get_array(swforcingpool,'zenithAngle',zenithangle)

156
157 ncells = ncellsarray( 3 )
158
159 !$omp do schedule(runtime) private(depth, k, cloudRatio, depLev)
160 do icell = 1, ncells
161 depth = 0.0_rkind
162 cloudratio = min(1.0_rkind, 1.0_rkind - penetrativetemperatureflux(icell)/(hflux_factor*(1.0e-15_rkind + &
163 clearskyradiation(icell))))
164 cloudratio = max(0.0_rkind, cloudratio)
165
166 call ocn_get_os00_coeffs(chlorophylla(icell),zenithangle(icell),cloudratio,avals, kvals)

169 depth = depth + layerthickness(k, icell)
170
171 call ocn_get_variable_sw_fraction(depth, weights(k+1), avals, kvals)
172 tend(index_temperature, k, icell) = tend(index_temperature, k, icell) + penetrativetemperatureflux(icell) &
173 * (weights(k) - weights(k+1) )
174 end do
175
176 depth = 0.0_rkind

178 depth = depth + layerthickness(k,icell)

180 enddo
181
182 if(k == maxlevelcell(icell) .or. k == 1) then
183 deplev=2
184 else
185 deplev=k

189 end do
190 !$omp end do

192 deallocate(weights)

384 subroutine ocn_get_variable_sw_fraction(depth, weight, Avals, Kvals)!{{{
385! Note: below 200m the solar penetration gets set to zero,
386! otherwise the limit for the exponent ($+/- 5678$) needs to be
387! taken care of.
388
389 real (kind=rkind), intent(in) :: depth
390 real (kind=rkind), intent(in),dimension(4) :: avals
391 real (kind=rkind), intent(in),dimension(4) :: kvals
392 real (kind=rkind), intent(out) :: weight
393
394!-----------------------------------------------------------------------
395!
396! local variables
397!
398!-----------------------------------------------------------------------
399
400!
401 integer :: k
402
403 real (kind=rkind), parameter :: depth_cutoff = -200.0_rkind
404
405!-----------------------------------------------------------------------
406!
407! compute absorption fraction
408!
409!-----------------------------------------------------------------------
410
411 if (-depth < depth_cutoff) then
412 weight = 0.0_rkind
413 else
414 weight=0.0_rkind
415 do k=1,4
416 weight = weight + avals(k)*exp(-depth*kvals(k))
417 enddo
418 endif

446 subroutine ocn_get_os00_coeffs(chlorophyllA,zenithAngle,cloudFraction,Avals,Kvals)!{{{
447
448 real(kind=rkind), intent(in) :: chlorophylla, zenithangle, cloudfraction
449 real(kind=rkind), intent(out), dimension(4) :: avals, kvals
450
451 if(cloudfraction > 0.1_rkind) then ! cloudy skies
452 avals(1) = 0.026_rkind*chlorophylla + 0.112_rkind*cloudfraction + 0.366_rkind
453 avals(2) = -0.009_rkind*chlorophylla + 0.034_rkind*cloudfraction + 0.207_rkind
454 avals(3) = -0.015_rkind*chlorophylla -0.006_rkind*cloudfraction + 0.188_rkind
455 avals(4) = -0.003_rkind*chlorophylla -0.131_rkind*cloudfraction + 0.169_rkind
456 kvals(1) = 0.063_rkind*chlorophylla -0.015_rkind*cloudfraction + 0.082_rkind
457 kvals(2) = 0.278_rkind*chlorophylla -0.562_rkind*cloudfraction + 1.02_rkind
458 kvals(3) = 3.91_rkind*chlorophylla -12.91_rkind*cloudfraction + 16.62_rkind
459 kvals(4) = 16.64_rkind*chlorophylla -478.28_rkind*cloudfraction + 736.56_rkind
460 else ! clear skies
461 avals(1) = 0.033_rkind*chlorophylla -0.025_rkind*zenithangle + 0.419_rkind
462 avals(2) = -0.010_rkind*chlorophylla -0.007_rkind*zenithangle + 0.231_rkind
463 avals(3) = -0.019_rkind*chlorophylla -0.003_rkind*zenithangle + 0.195_rkind
464 avals(4) = -0.006_rkind*chlorophylla -0.004_rkind*zenithangle + 0.154_rkind
465 kvals(1) = 0.066_rkind*chlorophylla + 0.006_rkind*zenithangle + 0.066_rkind
466 kvals(2) = 0.396_rkind*chlorophylla -0.027_rkind*zenithangle + 0.866_rkind
467 kvals(3) = 7.68_rkind*chlorophylla -2.49_rkind*zenithangle + 17.81_rkind
