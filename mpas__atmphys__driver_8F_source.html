33!MPAS top physics driver.
34!Laura D. Fowler (send comments to laura@ucar.edu).
35!2013-05-01.
36!
37! subroutine physics_driver is the top physics driver from which separate drivers for all physics
38! parameterizations, except cloud microphysics parameterizations are called.
39!
40! subroutines called in mpas_atmphys_driver:
41! ------------------------------------------
42! allocate_forall_physics : allocate local arrays defining atmospheric soundings (pressure,..)
43! allocate_cloudiness : allocate all local arrays used in driver_cloudiness.
44! allocate_convection : allocate all local arrays used in driver_convection.
45! allocate_gwdo : allocate all local arrays used in driver_gwdo.
46! allocate_lsm : allocate all local arrays used in driver_lsm.
47! allocate_pbl : allocate all local arrays used in driver_pbl.
48! allocate_radiation_lw : allocate all local arrays used in driver_radiation_lw.
49! allocate_radiation_sw : allocate all local arrays used in driver_radiation_sw.
50! allocate_sfclayer : allocate all local arrays used in driver_sfclayer.
51!
52! deallocate_forall_physics : deallocate local arrays defining atmospheric soundings.
53! deallocate_cloudiness : dedeallocate all local arrays used in driver_cloudiness.
54! deallocate_convection : deallocate all local arrays used in driver_convection.
55! deallocate_gwdo : deallocate all local arrays used in driver_gwdo.
56! deallocate_lsm : deallocate all local arrays used in driver_lsm.
57! deallocate_pbl : deallocate all local arrays used in driver_pbl.
58! deallocate_radiation_lw : deallocate all local arrays used in driver_radiation_lw.
59! deallocate_radiation_sw : deallocate all local arrays used in driver_radiation_sw.
60! deallocate_sfclayer : deallocate all local arrays used in driver_sfclayer.
61!
62! MPAS_to_physics :
63! driver_cloudiness : driver for parameterization of fractional cloudiness.
64! driver_convection : driver for parameterization of convection.
65! driver_gwdo : driver for parameterization of gravity wave drag over orography.
66! driver_lsm : driver for land-surface scheme.
67! driver_pbl : driver for planetary boundary layer scheme.
68! driver_radiation_sw : driver for short wave radiation schemes.
69! driver_radiation_lw : driver for long wave radiation schemes.
70! driver_sfclayer : driver for surface layer scheme.
71! update_convection_step1 : updates lifetime of deep convective clouds in Kain-Fritsch scheme.
72! update_convection_step2 : updates accumulated precipitation output from convection schemes.
73! update_radiation_diagnostics: updates accumualted radiation diagnostics from radiation schemes.
74!
75! add-ons and modifications to sourcecode:
76! ----------------------------------------
77! * removed call to calculate atmospheric soundings for the hydrostatic dynamical core.
78! Laura D. Fowler (2013-05-06).
79! * removed the namelist option config_eddy_scheme and associated sourcecode.
80! * removed the namelist option config_conv_shallow_scheme and associated sourcecode.
81! Laura D. Fowler (birch.ucar.edu) / 2013-05-29.
82! * added block%atm_input in calls to subroutines driver_radiation_lw amd driver_radiation_lw.
83! Laura D. Fowler (laura@ucar.edu) / 2013-07-03.
84! * modified sourcecode to use pools.
85! Laura D. Fowler (laura@ucar.edu) / 2014-05-15.
86! * renamed config_conv_deep_scheme to config_convection_scheme.
87! Laura D. Fowler (laura@ucar.edu) / 2014-09-18.
88! * in the call to driver_convection, added block%configs needed for the implementation of the
89! Grell-Freitas convection scheme.
90! Laura D. Fowler (laura@ucar.edu) / 2016-03-30.
91! * modified the call to the subroutines driver_sfclayer and driver_pbl for the implementation
92! of the MYNN surface layer scheme and PBL schemes. itimestep and block%configs are added to
93! the argument list.
94! Laura D. Fowler (laura@ucar.edu) / 2015-01-06.
95! * now only call subroutine update_convection_step2 when config_convection_scheme is not off.
96! Laura D. Fowler (laura@ucar.edu) / 2016-04-13.
97! * modified call to driver_cloudiness to accomodate the calculation of the cloud fraction with the Thompson
98! cloud microphysics scheme.
99! Laura D. Fowler (laura@ucar.edu) / 2016-06-04.

105!=================================================================================================================
106 subroutine physics_driver(domain,itimestep,xtime_s)
107!=================================================================================================================
108
109!input arguments:
110 integer,intent(in):: itimestep
111 real(kind=rkind),intent(in):: xtime_s
112
113!inout arguments:
114 type(domain_type),intent(inout):: domain
115
116!local pointers:
117 type(mpas_pool_type),pointer:: configs, &
118 mesh, &
119 state, &
120 diag, &
121 diag_physics, &
122 tend_physics, &
123 atm_input, &
124 sfc_input
125
126 logical,pointer:: config_frac_seaice
127
128 character(len=StrKIND),pointer:: config_bucket_update, &
129 config_convection_scheme, &
130 config_gwdo_scheme, &
131 config_lsm_scheme, &
132 config_pbl_scheme, &
133 config_radt_lw_scheme, &
134 config_radt_sw_scheme, &
135 config_sfclayer_scheme
136
137 logical, pointer:: config_oml1d
138 real(kind=rkind),pointer:: config_bucket_radt
139
140!local variables:
141 type(block_type),pointer:: block
142
143 integer:: time_lev
144 integer:: thread
145
146 integer,pointer:: nthreads
147 integer,dimension(:),pointer:: cellsolvethreadstart, cellsolvethreadend
148
149!=================================================================================================================
150!call mpas_log_write('')
151!call mpas_log_write('--- enter subroutine mpas_atmphys_driver:')
152
153 call mpas_timer_start('physics driver')
154
155 call mpas_pool_get_config(domain%configs,'config_convection_scheme',config_convection_scheme)
156 call mpas_pool_get_config(domain%configs,'config_gwdo_scheme' ,config_gwdo_scheme )
157 call mpas_pool_get_config(domain%configs,'config_lsm_scheme' ,config_lsm_scheme )
158 call mpas_pool_get_config(domain%configs,'config_pbl_scheme' ,config_pbl_scheme )
159 call mpas_pool_get_config(domain%configs,'config_radt_lw_scheme' ,config_radt_lw_scheme )
160 call mpas_pool_get_config(domain%configs,'config_radt_sw_scheme' ,config_radt_sw_scheme )
161 call mpas_pool_get_config(domain%configs,'config_sfclayer_scheme' ,config_sfclayer_scheme )
162 call mpas_pool_get_config(domain%configs,'config_bucket_radt' ,config_bucket_radt )
163 call mpas_pool_get_config(domain%configs,'config_bucket_update' ,config_bucket_update )
164 call mpas_pool_get_config(domain%configs,'config_frac_seaice' ,config_frac_seaice )
165 call mpas_pool_get_config(domain%configs,'config_oml1d' ,config_oml1d )
166
167 if(config_convection_scheme .ne. 'off' .or. &
168 config_lsm_scheme .ne. 'off' .or. &
169 config_pbl_scheme .ne. 'off' .or. &
170 config_radt_lw_scheme .ne. 'off' .or. &
171 config_radt_sw_scheme .ne. 'off' .or. &
172 config_sfclayer_scheme .ne. 'off') then
173
174 block => domain % blocklist
175 do while(associated(block))
176
177 call mpas_pool_get_subpool(block%structs,'mesh' ,mesh )
178 call mpas_pool_get_subpool(block%structs,'state' ,state )
179 call mpas_pool_get_subpool(block%structs,'diag' ,diag )
180 call mpas_pool_get_subpool(block%structs,'diag_physics',diag_physics)
181 call mpas_pool_get_subpool(block%structs,'atm_input' ,atm_input )
182 call mpas_pool_get_subpool(block%structs,'sfc_input' ,sfc_input )
183 call mpas_pool_get_subpool(block%structs,'tend_physics',tend_physics)
184
185 call mpas_pool_get_dimension(block % dimensions, 'nThreads', nthreads)
186
187 call mpas_pool_get_dimension(block % dimensions, 'cellSolveThreadStart', cellsolvethreadstart)
188 call mpas_pool_get_dimension(block % dimensions, 'cellSolveThreadEnd', cellsolvethreadend)
189
190 !allocate arrays shared by all physics parameterizations:
191 call allocate_forall_physics(block%configs)
192
193 !physics prep step:
194 time_lev = 1
195
196!$OMP PARALLEL DO
197 do thread=1,nthreads
198 call mpas_to_physics(block%configs,mesh,state,time_lev,diag,diag_physics, &
199 cellsolvethreadstart(thread), cellsolvethreadend(thread))
200 end do
201!$OMP END PARALLEL DO
202
203 !call to cloud scheme:
204 if(l_radtlw .or. l_radtsw) then
205 call allocate_cloudiness
206!$OMP PARALLEL DO
207 do thread=1,nthreads
208 call driver_cloudiness(block%configs,mesh,diag_physics,sfc_input, &
209 cellsolvethreadstart(thread),cellsolvethreadend(thread))
210 end do
211!$OMP END PARALLEL DO
212 endif
213
214 !call to short wave radiation scheme:
215 if(l_radtsw) then
216 time_lev = 1
217 call allocate_radiation_sw(block%configs,xtime_s)
218!$OMP PARALLEL DO
219 do thread=1,nthreads
220 call driver_radiation_sw(itimestep,block%configs,mesh,state,time_lev,diag_physics, &
221 atm_input,sfc_input,tend_physics,xtime_s, &
222 cellsolvethreadstart(thread), cellsolvethreadend(thread))
223 end do
224!$OMP END PARALLEL DO
225 endif
226
227 !call to long wave radiation scheme:
228 if(l_radtlw) then
229 time_lev = 1
230 call allocate_radiation_lw(block%configs,xtime_s)
231!$OMP PARALLEL DO
232 do thread=1,nthreads
233 call driver_radiation_lw(xtime_s,block%configs,mesh,state,time_lev,diag_physics, &
234 atm_input,sfc_input,tend_physics, &
235 cellsolvethreadstart(thread), cellsolvethreadend(thread))
236 end do
237!$OMP END PARALLEL DO
238 endif
239
240 !call to accumulate long- and short-wave diagnostics if needed:
241 if(config_bucket_update /= 'none' .and. config_bucket_radt .gt. 0._rkind) then
242!$OMP PARALLEL DO
243 do thread=1,nthreads
244 call update_radiation_diagnostics(block%configs,mesh,diag_physics, &
245 cellsolvethreadstart(thread), cellsolvethreadend(thread))
246 end do
247!$OMP END PARALLEL DO
248 endif
249
250 !deallocate all radiation arrays:
251 if(config_radt_sw_scheme.ne.'off' .or. config_radt_lw_scheme.ne.'off') &
252 call deallocate_cloudiness
253 if(config_radt_sw_scheme.ne.'off') call deallocate_radiation_sw(block%configs)
254 if(config_radt_lw_scheme.ne.'off') call deallocate_radiation_lw(block%configs)
255
256 !call to surface-layer scheme:
257 if(config_sfclayer_scheme .ne. 'off') then
258 call allocate_sfclayer(block%configs)
259!$OMP PARALLEL DO
260 do thread=1,nthreads
261 call driver_sfclayer(itimestep,block%configs,mesh,diag_physics,sfc_input, &
262 cellsolvethreadstart(thread), cellsolvethreadend(thread))
263 end do
264!$OMP END PARALLEL DO
265 call deallocate_sfclayer(block%configs)
266 endif
267
268 !call to 1d ocean mixed-layer model
269 if(config_oml1d) call driver_oml1d(block%configs, mesh, diag, diag_physics, sfc_input)
270
271 !call to land-surface scheme:
272 if(config_lsm_scheme .ne. 'off') then
273 call allocate_lsm(config_frac_seaice)
274!$OMP PARALLEL DO
275 do thread=1,nthreads
276 call driver_lsm(itimestep,block%configs,mesh,diag_physics,sfc_input, &
277 cellsolvethreadstart(thread), cellsolvethreadend(thread))
278 end do
279!$OMP END PARALLEL DO
280 call deallocate_lsm(config_frac_seaice)
281 endif
282
283 !call to pbl schemes:
284 if(config_pbl_scheme .ne. 'off' .and. config_sfclayer_scheme .ne. 'off') then
285 call allocate_pbl(block%configs)
286!$OMP PARALLEL DO
287 do thread=1,nthreads
288 call driver_pbl(itimestep,block%configs,mesh,sfc_input,diag_physics,tend_physics, &
289 cellsolvethreadstart(thread),cellsolvethreadend(thread))
290 end do
291!$OMP END PARALLEL DO
292 call deallocate_pbl(block%configs)
293
294 endif
295
296 !call to gravity wave drag over orography scheme:
297 if(config_gwdo_scheme .ne. 'off') then
298 call allocate_gwdo
299!$OMP PARALLEL DO
300 do thread=1,nthreads
301 call driver_gwdo(itimestep,block%configs,mesh,sfc_input,diag_physics,tend_physics, &
302 cellsolvethreadstart(thread),cellsolvethreadend(thread))
303 end do
304!$OMP END PARALLEL DO
305 call deallocate_gwdo
306 endif
307
308 !call to convection scheme:
309!$OMP PARALLEL DO
310 do thread=1,nthreads
311 call update_convection_step1(block%configs,diag_physics,tend_physics, &
312 cellsolvethreadstart(thread),cellsolvethreadend(thread))
313 end do
314!$OMP END PARALLEL DO
315 if(l_conv) then
316 call allocate_convection(block%configs)
317!$OMP PARALLEL DO
318 do thread=1,nthreads
319 call driver_convection(itimestep,block%configs,mesh,sfc_input,diag_physics,tend_physics, &
320 cellsolvethreadstart(thread), cellsolvethreadend(thread))
321 end do
322!$OMP END PARALLEL DO
323 call deallocate_convection(block%configs)
324 endif
325 !update diagnostics:
326 if(config_convection_scheme .ne. 'off') then
327!$OMP PARALLEL DO
328 do thread=1,nthreads
329 call update_convection_step2(block%configs,diag_physics, &
330 cellsolvethreadstart(thread),cellsolvethreadend(thread))
331 end do
332!$OMP END PARALLEL DO
333 end if
334
335 !deallocate arrays shared by all physics parameterizations:
336 call deallocate_forall_physics(block%configs)
337
338 block => block % next
339 end do
340
341 endif
342
343 call mpas_timer_stop('physics driver')
344
345!call mpas_log_write('--- enter subroutine mpas_atmphys_driver:')
346!call mpas_log_write('')
347
348 end subroutine physics_driver
