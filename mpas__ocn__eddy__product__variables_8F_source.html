175 type (mpas_pool_type), pointer :: meshpool
176 type (mpas_pool_type), pointer :: statepool
177 type (mpas_pool_type), pointer :: tracerspool

194 block => domain % blocklist
195 do while (associated(block))
196 call mpas_pool_get_subpool(block % structs, 'state', statepool)

215 call mpas_pool_get_array(eddyproductvariablesampool, 'velocityMeridionalSquared', velocitymeridionalsquared)
216 call mpas_pool_get_array(eddyproductvariablesampool, 'velocityZonalTimesTemperature', velocityzonaltimestemperature)
217 call mpas_pool_get_array(eddyproductvariablesampool, 'velocityMeridionalTimesTemperature', &
218 velocitymeridionaltimestemperature)
219
220 do icell = 1,ncellssolve
221 sshsquared(icell) = ssh(icell)**2
222 do k = 1, maxlevelcell(icell)
223 velocityzonalsquared(k,icell) = velocityzonal(k,icell)**2
224 velocitymeridionalsquared(k,icell) = velocitymeridional(k,icell)**2
225 velocityzonaltimestemperature(k,icell) = velocityzonal(k,icell)*activetracers(index_temperature,k,icell)
226 velocitymeridionaltimestemperature(k,icell) = velocitymeridional(k,icell)*activetracers(index_temperature,k,icell)
227 end do
228 end do

230 block => block % next
231 end do
