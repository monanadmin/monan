32 if (! (argc >= 5) ) {

54 fprintf(stderr, "\nError: Option %s is not valid for stream order.\n", argv[4]);
55 fprintf(stderr, "\tOrder values are:\n");
56 fprintf(stderr, "\t\tlisted -- Defines streams in the listed order in registry.\n");
57 fprintf(stderr, "\t\tmutable -- Defines immutable streams first, followed by mutable streams.\n");
58 return 1;

60
61
62 pairs = argc-5;
63 if ( pairs > 0 ) {
64 keys = malloc(sizeof(char*) * pairs);
65 values = malloc(sizeof(char*) * pairs);
66 } else {
67 pairs = 0;
68 }
69
70 for(i = 0; i < pairs; i++){
71 keys[i] = malloc(sizeof(char)*1024);
72 values[i] = malloc(sizeof(char)*1024);
73
74 string = strdup(argv[5+i]);
75 tofree = string;
76
77 token = strsep(&string, "=");
78 sprintf(keys[i], "%s", token);
79
80 token = strsep(&string, "=");
81 sprintf(values[i], "%s", token);
82
83 free(tofree);
84 }
85
86 ezxml_t registry = ezxml_parse_fp(regfile);
87
88 err = generate_streams(registry, streamsfile, stream_file_prefix, order, pairs, keys, values);
89
90 if ( pairs > 0 ) {
91 free(keys);
92 free(values);
93 }
