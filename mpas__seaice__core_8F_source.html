39
40 function seaice_core_init(domain, startTimeStamp) result(iErr)
41
42 use mpas_derived_types
43 use mpas_stream_manager
44
45 use seaice_diagnostics, only: &
46 seaice_initialize_time_diagnostics
47
48 implicit none
49
50 type (domain_type), intent(inout) :: domain
51 character(len=*), intent(out) :: starttimestamp
52
53 real (kind=rkind) :: dt
54
55 character(len=strKIND), pointer :: config_initial_condition_type
56 logical, pointer :: config_do_restart
57 real (kind=rkind), pointer :: config_dt
58 type (mpas_time_type) :: starttime
59 integer :: ierr
60 integer :: numthreads
61
62 call mpas_pool_set_error_level(mpas_pool_warn)
63
64 ierr = 0
65
66 numthreads = mpas_threading_get_max_threads()
67
68 call mpas_log_write(' ')
69 call mpas_log_write(' *****************************************************************************')
70 call mpas_log_write(' MPI Task $i has access to $i threads', intargs=(/domain % dminfo % my_proc_id, numthreads/))
71 call mpas_log_write(' *****************************************************************************')
72 call mpas_log_write(' ')
73
74 ! write out namelist options
75 call mpas_log_write(' ') ; call mpas_log_write(" ----- namelist options used -----") ; call mpas_log_write(' ')
76 call mpas_pool_print_summary(domain % configs, mpas_pool_config)
77 call mpas_log_write(' ') ; call mpas_log_write(" ----- done listing namelist options used -----") ; call mpas_log_write(' ')
78
79 call mpas_pool_get_config(domain % configs, 'config_initial_condition_type', config_initial_condition_type)
80 call mpas_pool_get_config(domain % configs, 'config_do_restart', config_do_restart)
81 call mpas_pool_get_config(domain % configs, 'config_dt', config_dt)
82
83 !
84 ! Set "local" clock to point to the clock contained in the domain type
85 !
86 clock => domain % clock
87
88
89 !
90 ! Set startTimeStamp based on the start time of the simulation clock
91 !
92 starttime = mpas_get_clock_time(clock, mpas_start_time, ierr)
93 call mpas_get_time(starttime, datetimestring=starttimestamp)
94
95 !
96 ! If this is a restart run, read the restart stream, else read the input stream.
97 ! Regardless of which stream we read for initial conditions, reset the
98 ! input alarms for both input and restart before reading any remaining input streams.
99 !
100 if (config_do_restart) then
101 call mpas_stream_mgr_read(domain % streamManager, streamid='restart', ierr=ierr)
102 else
103 if (trim(config_initial_condition_type) == "restart") then
104 call mpas_stream_mgr_read(domain % streamManager, streamid='restart_ic', ierr=ierr)
105 else
106 call mpas_stream_mgr_read(domain % streamManager, streamid='input', ierr=ierr)
107 end if
108 endif
109 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='input', direction=mpas_stream_input, ierr=ierr)
110 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='restart', direction=mpas_stream_input, ierr=ierr)
111 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='restart_ic', direction=mpas_stream_input, ierr=ierr)
112
113 ! bootstrap analysis members
114 call seaice_analysis_bootstrap(domain, ierr)
115
116 ! Read all other inputs
117 call mpas_stream_mgr_read(domain % streamManager, ierr=ierr)
118 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_input, ierr=ierr)
119
120 !
121 ! Initialize core
122 !
123 dt = config_dt
124
125 ! initialize time diagnostics
126 call seaice_initialize_time_diagnostics(domain)
127
128 ! initialize analysis members
129 call seaice_analysis_init(domain, ierr)
130
131 ! bootstrap analysis has used config_do_restart: now modify for rest of code
132 if (trim(config_initial_condition_type) == "restart") config_do_restart = .true.
133
134 call mpas_init_block(domain, dt)
135
136 end function seaice_core_init

150 subroutine seaice_simulation_clock_init(core_clock, configs, ierr)
151
152 implicit none
153
154 type (mpas_clock_type), intent(inout) :: core_clock
155 type (mpas_pool_type), intent(inout) :: configs
156 integer, intent(out) :: ierr
157
158 type (mpas_time_type) :: starttime, stoptime, alarmstarttime
159 type (mpas_timeinterval_type) :: runduration, timestep, alarmtimestep
160 integer :: local_err
161
162 character (len=StrKIND), pointer :: config_start_time, config_run_duration, config_stop_time, config_restart_timestamp_name
163 character (len=StrKIND) :: restarttimestamp
164 real (kind=rkind), pointer :: config_dt
165
166
167 ierr = 0
168
169 call mpas_pool_get_config(configs, 'config_dt', config_dt)
170 call mpas_pool_get_config(configs, 'config_start_time', config_start_time)
171 call mpas_pool_get_config(configs, 'config_run_duration', config_run_duration)
172 call mpas_pool_get_config(configs, 'config_stop_time', config_stop_time)
173 call mpas_pool_get_config(configs, 'config_restart_timestamp_name', config_restart_timestamp_name)
174
175
176 ! Set time to the user-specified start time OR use a restart time from file
177 if ( trim(config_start_time) == "file" ) then
178 open(22, file=config_restart_timestamp_name, form='formatted', status='old', iostat=local_err)
179 if (local_err /= 0) then
180 call mpas_log_write("seaice_simulation_clock_init: problem opening restart timestamp file for reading: $i", &
181 mpas_log_crit, intargs=(/local_err/))
182 endif
183 read(22,*,iostat=local_err) restarttimestamp
184 if (local_err /= 0) then
185 call mpas_log_write("seaice_simulation_clock_init: problem reading restart timestamp file $i: ", &
186 mpas_log_crit, intargs=(/local_err/))
187 endif
188 close(22)
189 call mpas_set_time(curr_time=starttime, datetimestring=restarttimestamp, ierr=local_err)
190 else
191 call mpas_set_time(curr_time=starttime, datetimestring=config_start_time, ierr=local_err)
192 end if
193
194 call mpas_set_timeinterval(timestep, dt=config_dt, ierr=local_err)
195
196 if (trim(config_run_duration) /= "none") then
197 call mpas_set_timeinterval(runduration, timestring=config_run_duration, ierr=local_err)
198 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, runduration=runduration, ierr=local_err)
199
200 if (trim(config_stop_time) /= "none") then
201 call mpas_set_time(curr_time=stoptime, datetimestring=config_stop_time, ierr=local_err)
202 if(starttime + runduration /= stoptime) then
203 call mpas_log_write(&
204 'config_run_duration and config_stop_time are inconsitent: using config_run_duration.', &
205 mpas_log_warn)
206 end if
207 end if
208 else if (trim(config_stop_time) /= "none") then
209 call mpas_set_time(curr_time=stoptime, datetimestring=config_stop_time, ierr=local_err)
210 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, stoptime=stoptime, ierr=local_err)
211 else
212 call mpas_log_write(&
213 'Error: Neither config_run_duration nor config_stop_time were specified.', &
214 mpas_log_warn)!! unclear what kind of error this shoudl be
215 ierr = 1
216 end if
217
218 end subroutine seaice_simulation_clock_init

260 function seaice_core_run(domain) result(iErr)
261
262 use mpas_derived_types
263 use mpas_kind_types
264 use mpas_stream_manager
265 use mpas_timer
266 use seaice_time_integration, only: seaice_timestep_finalize
267 use seaice_forcing, only: &
268 seaice_forcing_get, &
269 seaice_forcing_write_restart_times
270 use seaice_initialize, only: seaice_init_post_clock_advance
271
272 implicit none
273
274 type (domain_type), intent(inout) :: domain
275
276 integer :: itimestep
277 type (block_type), pointer :: block_ptr
278
279 type (mpas_time_type) :: currtime
280 character(len=StrKIND) :: timestamp
281 integer :: ierr
282
283 character(len=StrKIND), pointer :: config_restart_timestamp_name
284
285 logical, pointer :: config_write_output_on_startup
286
287 ierr = 0
288
289 ! configurations
290 call mpas_pool_get_config(domain % configs, 'config_restart_timestamp_name', config_restart_timestamp_name)
291
292 currtime = mpas_get_clock_time(clock, mpas_now, ierr)
293 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=ierr)
294 call mpas_log_write('Initial timestep '//trim(timestamp))
295
296 ! compute analysis at startup if required
297 ! (note: might require some reordering of call here if seaice_forcing_get data is used in analysis members)
298 call seaice_analysis_compute_startup(domain, ierr)
299
300 ! Avoid writing a restart file at the initial time
301 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='restart', direction=mpas_stream_output, ierr=ierr)
302
303 call mpas_pool_get_config(domain % configs, 'config_write_output_on_startup', config_write_output_on_startup)
304 if (config_write_output_on_startup) then
305 call mpas_stream_mgr_write(domain % streamManager, ierr=ierr)
306 endif
307 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)
308
309 itimestep = 1
310 call mpas_advance_clock(clock)
311
312 ! initial forcing
313 call seaice_forcing_get(domain % streamManager, domain, clock, .true.)
314
315 ! final initialization after clock advance
316 call seaice_init_post_clock_advance(domain, clock)
317
318 timeloop: do
319
320 currtime = mpas_get_clock_time(clock, mpas_now, ierr)
321 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=ierr)
322 call mpas_log_write('Doing timestep '//trim(timestamp))
323
324 ! pre-timestep analysis computation
325 call seaice_analysis_precompute(domain, ierr)
326
327 ! do the time integration
328 call mpas_timer_start("time integration")
329 call mpas_timestep(domain, itimestep, timestamp)
330 call mpas_timer_stop("time integration")
331
332 ! update analysis members
333 call seaice_analysis_compute(domain, ierr)
334 call seaice_analysis_restart(domain, ierr)
335 call seaice_analysis_write(domain, ierr)
336
337 if ( mpas_stream_mgr_ringing_alarms(domain % streamManager, &
338 streamid='restart', direction=mpas_stream_output, ierr=ierr) ) then
339 call seaice_forcing_write_restart_times(domain)
340 endif
341
342 call mpas_stream_mgr_write(domain % streamManager, ierr=ierr)
343
344 ! update the restart_timestamp file with the new time if needed
345 if ( mpas_stream_mgr_ringing_alarms(domain % streamManager, &
346 streamid='restart', direction=mpas_stream_output, ierr=ierr) ) then
347 open(22, file=config_restart_timestamp_name, form='formatted', status='replace', iostat=ierr)
348 if (ierr /= 0) then
349 call mpas_log_write("seaice_core_run: problem opening restart timestamp file for writing: $i", &
350 mpas_log_crit, intargs=(/ierr/))
351 endif
352 write(22, *, iostat=ierr) timestamp
353 if (ierr /= 0) then
354 call mpas_log_write("seaice_core_run: problem writing restart timestamp file: $i", &
355 mpas_log_crit, intargs=(/ierr/))
356 endif
357 close(22)
358 end if
359
360 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_output, ierr=ierr)
361
362 ! finalize the timstep
363 call seaice_timestep_finalize(domain)
364
365 ! check if the simulation is finished
366 if (mpas_is_clock_stop_time(clock)) exit timeloop
367
368 ! advance the clock
369 itimestep = itimestep + 1
370 call mpas_advance_clock(clock)
371
372 ! forcing
373 call seaice_forcing_get(domain % streamManager, domain, clock, .false.)
374
375 end do timeloop
376
377 end function seaice_core_run
