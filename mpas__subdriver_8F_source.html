399 subroutine add_stream_attributes(domain)
400
401 use mpas_stream_manager, only : mpas_stream_mgr_add_att

403 implicit none
404

408 integer, pointer :: intAtt

416 if (domain % dminfo % nProcs < 10) then
417 write(histatt, '(A,I1,A,A,A)') 'mpirun -n ', domain % dminfo % nProcs, ' ./', trim(domain % core % coreName), '_model'
418 else if (domain % dminfo % nProcs < 100) then
419 write(histatt, '(A,I2,A,A,A)') 'mpirun -n ', domain % dminfo % nProcs, ' ./', trim(domain % core % coreName), '_model'
420 else if (domain % dminfo % nProcs < 1000) then
421 write(histatt, '(A,I3,A,A,A)') 'mpirun -n ', domain % dminfo % nProcs, ' ./', trim(domain % core % coreName), '_model'
422 else if (domain % dminfo % nProcs < 10000) then
423 write(histatt, '(A,I4,A,A,A)') 'mpirun -n ', domain % dminfo % nProcs, ' ./', trim(domain % core % coreName), '_model'
424 else if (domain % dminfo % nProcs < 100000) then
425 write(histatt, '(A,I5,A,A,A)') 'mpirun -n ', domain % dminfo % nProcs, ' ./', trim(domain % core % coreName), '_model'
426 else
427 write(histatt, '(A,I6,A,A,A)') 'mpirun -n ', domain % dminfo % nProcs, ' ./', trim(domain % core % coreName), '_model'
428 end if
429
430 call mpas_stream_mgr_add_att(domain % streamManager, 'model_name', domain % core % modelName)
431 call mpas_stream_mgr_add_att(domain % streamManager, 'core_name', domain % core % coreName)
432 call mpas_stream_mgr_add_att(domain % streamManager, 'source', domain % core % source)
433 call mpas_stream_mgr_add_att(domain % streamManager, 'Conventions', domain % core % Conventions)
434 call mpas_stream_mgr_add_att(domain % streamManager, 'git_version', domain % core % git_version)
435
436 call mpas_stream_mgr_add_att(domain % streamManager, 'on_a_sphere', domain % on_a_sphere)
437 call mpas_stream_mgr_add_att(domain % streamManager, 'sphere_radius', domain % sphere_radius)
438 call mpas_stream_mgr_add_att(domain % streamManager, 'is_periodic', domain % is_periodic)
439 call mpas_stream_mgr_add_att(domain % streamManager, 'x_period', domain % x_period)
440 call mpas_stream_mgr_add_att(domain % streamManager, 'y_period', domain % y_period)
441 ! DWJ 10/01/2014: Eventually add the real history attribute, for now (due to length restrictions)
442 ! add a shortened version.
443! call MPAS_stream_mgr_add_att(domain % streamManager, 'history', domain % history)
444 call mpas_stream_mgr_add_att(domain % streamManager, 'history', histatt)
445 call mpas_stream_mgr_add_att(domain % streamManager, 'parent_id', domain % parent_id)
446 call mpas_stream_mgr_add_att(domain % streamManager, 'mesh_spec', domain % mesh_spec)
447
448 call mpas_pool_begin_iteration(domain % configs)
449
450 do while (mpas_pool_get_next_member(domain % configs, itr))
451
452 if ( itr % memberType == mpas_pool_config) then
453
454 if ( itr % dataType == mpas_pool_real ) then
455 call mpas_pool_get_config(domain % configs, itr % memberName, realatt)
456 call mpas_stream_mgr_add_att(domain % streamManager, itr % memberName, realatt, ierr=local_ierr)
457 else if ( itr % dataType == mpas_pool_integer ) then
458 call mpas_pool_get_config(domain % configs, itr % memberName, intatt)
459 call mpas_stream_mgr_add_att(domain % streamManager, itr % memberName, intatt, ierr=local_ierr)
460 else if ( itr % dataType == mpas_pool_character ) then
461 call mpas_pool_get_config(domain % configs, itr % memberName, charatt)
462 call mpas_stream_mgr_add_att(domain % streamManager, itr % memberName, charatt, ierr=local_ierr)
463 else if ( itr % dataType == mpas_pool_logical ) then
464 call mpas_pool_get_config(domain % configs, itr % memberName, logatt)
465 if (logatt) then
466 call mpas_stream_mgr_add_att(domain % streamManager, itr % memberName, 'YES', ierr=local_ierr)
467 else
468 call mpas_stream_mgr_add_att(domain % streamManager, itr % memberName, 'NO', ierr=local_ierr)
469 end if
470 end if
471
472 end if
473 end do

475 end subroutine add_stream_attributes
