83!-----------------------------------------------------------------------
84 subroutine ocn_rbfvertex(meshPool, filterNum, layerThickness, u, uvReconstructX, uvReconstructY, uvReconstructZ, & !{{{
85 initialize_weights, scratchPool, particleCellPool)
86
87 implicit none
88
89 ! inputs
90 type (mpas_pool_type), pointer, intent(in) :: meshPool
91 type (mpas_pool_type), pointer, intent(in) :: scratchPool
92 type (mpas_pool_type), pointer, intent(in) :: particleCellPool
93 real (kind=rkind), dimension(:,:), pointer, intent(in) :: u
94 real (kind=rkind), dimension(:,:), pointer, intent(in) :: layerthickness
95 integer, intent(in) :: filterNum
96 logical, intent(in) :: initialize_weights
97
98 ! outputs
99 type (field2DReal), pointer, intent(inout) :: uvReconstructX
100 type (field2DReal), pointer, intent(inout) :: uvReconstructY
101 type (field2DReal), pointer, intent(inout) :: uvReconstructZ
102
103 ! local / temporary arrays needed in the compute procedure
104 type (field2DReal), pointer :: &
105 ucReconstructX, ucReconstructY, ucReconstructZ, ucReconstructZonal, ucReconstructMeridional ! cell center values
106 type (field2DReal), pointer :: ucStore, vcStore, wcStore
107 type (field2DInteger), pointer :: boundaryVertex, boundaryCell, boundaryCellGlobal, boundaryVertexGlobal
108
109 ! get pointers
110 call mpas_pool_get_field(scratchpool, 'ucReconstructX', ucreconstructx)
111 call mpas_pool_get_field(scratchpool, 'ucReconstructY', ucreconstructy)
112 call mpas_pool_get_field(scratchpool, 'ucReconstructZ', ucreconstructz)
113 call mpas_pool_get_field(scratchpool, 'ucReconstructZonal', ucreconstructzonal)
114 call mpas_pool_get_field(scratchpool, 'ucReconstructMeridional', ucreconstructmeridional)
115 call mpas_pool_get_field(scratchpool, 'boundaryVertexGlobal', boundaryvertexglobal)
116
117 ! allocate memory
118 call mpas_allocate_scratch_field(ucreconstructx, .true.)
119 call mpas_allocate_scratch_field(ucreconstructy, .true.)
120 call mpas_allocate_scratch_field(ucreconstructz, .true.)
121 call mpas_allocate_scratch_field(ucreconstructzonal, .true.)
122 call mpas_allocate_scratch_field(ucreconstructmeridional, .true.)
123 call mpas_allocate_scratch_field(boundaryvertexglobal, .true.)
124
125 ucreconstructx % array = 0.0_rkind
126 ucreconstructy % array = 0.0_rkind
127 ucreconstructz % array = 0.0_rkind
128
129 ! initialize weights (should be pre-initialized)
130 if (initialize_weights) then
131 call mpas_init_reconstruct(meshpool)
132 end if
133
134 ! get cell center reconstructed RBF values
135 call mpas_reconstruct(meshpool, u, ucreconstructx % array, ucreconstructy % array, ucreconstructz % array, &
136 ucreconstructzonal % array, ucreconstructmeridional % array)
137
138 ! need to do exchange for uc components (we don't use Zonal / Meridional for this calculation)
139 call mpas_dmpar_exch_halo_field(ucreconstructx)
140 call mpas_dmpar_exch_halo_field(ucreconstructy)
141 call mpas_dmpar_exch_halo_field(ucreconstructz)
142
143 ! get boundaries
144 call mpas_pool_get_field(meshpool,'boundaryVertex', boundaryvertex)
145 call mpas_pool_get_field(meshpool,'boundaryCell', boundarycell)
146
147
148 if (filternum > 0) then
149 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
150 ! filter the cell velocity field
151 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
152 call ocn_second_order_shapiro_filter_ops(filternum, meshpool, scratchpool, boundaryvertex, boundarycell, &
153 layerthickness, ucreconstructx, ucreconstructy, ucreconstructz)
154
155 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
156 ! store filter data &
157 ! write data to file for output
158 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
159 call mpas_pool_get_field(particlecellpool, 'filteredVelocityU', ucstore)
160 call mpas_pool_get_field(particlecellpool, 'filteredVelocityV', vcstore)
161 call mpas_pool_get_field(particlecellpool, 'filteredVelocityW', wcstore)
162 ucstore % array = ucreconstructx % array
163 vcstore % array = ucreconstructy % array
164 wcstore % array = ucreconstructz % array
165
166 end if
167
168 ! interpolate to vertex locations for use in Wachspress
169 call ocn_vector_cell_center_to_vertex(meshpool, boundaryvertex % array, boundarycell % array, &
170 ucreconstructx % array, ucreconstructy % array, ucreconstructz % array, &
171 uvreconstructx % array, uvreconstructy % array, uvreconstructz % array)
172
173 ! handle boundary vertices (should be zero). Can potentially remove if mpas_init_block initializes to 0 vs -1e34
174 boundaryvertexglobal % array = boundaryvertex % array
175 call mpas_dmpar_exch_halo_field(boundaryvertexglobal)
176 ! definite change between these fields!
177 uvreconstructx % array = uvreconstructx % array * (1.0_rkind - boundaryvertexglobal % array)
178 uvreconstructy % array = uvreconstructy % array * (1.0_rkind - boundaryvertexglobal % array)
179 uvreconstructz % array = uvreconstructz % array * (1.0_rkind - boundaryvertexglobal % array)
180
181 ! do halo exchanges
182 call mpas_dmpar_exch_halo_field(uvreconstructx)
183 call mpas_dmpar_exch_halo_field(uvreconstructy)
184 call mpas_dmpar_exch_halo_field(uvreconstructz)
185
186 ! deallocate memory
187 call mpas_deallocate_scratch_field(ucreconstructx, .true.)
188 call mpas_deallocate_scratch_field(ucreconstructy, .true.)
189 call mpas_deallocate_scratch_field(ucreconstructz, .true.)
190 call mpas_deallocate_scratch_field(ucreconstructzonal, .true.)
191 call mpas_deallocate_scratch_field(ucreconstructmeridional, .true.)
192 call mpas_deallocate_scratch_field(boundaryvertexglobal, .true.)
193
194 end subroutine ocn_rbfvertex!}}}

209 subroutine ocn_vector_cell_center_to_vertex(meshPool, boundaryVertex, boundaryCell, & !{{{
210 ucReconstructX, ucReconstructY, ucReconstructZ, &
211 uvReconstructX, uvReconstructY, uvReconstructZ)
212
213 implicit none
214
215 ! input variables
216 type (mpas_pool_type), pointer, intent(in) :: meshPool
217 real (kind=rkind), dimension(:,:), pointer, intent(in) :: ucreconstructx, &
218 ucreconstructy, &
219 ucreconstructz
220 integer, dimension(:,:), pointer, intent(in) :: boundaryVertex, boundaryCell
221
222 ! output variables
223 real (kind=rkind), dimension(:,:), pointer, intent(inout) :: uvreconstructx
224 real (kind=rkind), dimension(:,:), pointer, intent(inout) :: uvreconstructy
225 real (kind=rkind), dimension(:,:), pointer, intent(inout) :: uvreconstructz
226
227 ! local variables
228 integer, pointer :: nVerticesSolve, nCells, vertexDegree, nVertLevels
229 integer :: aVertex, aCell, aLevel
230 integer, dimension(:,:), pointer :: cellsOnVertex
231 real (kind=rkind), dimension(:), pointer :: xcell, ycell, zcell, xvertex, yvertex, zvertex
232 real (kind=rkind), dimension(:,:), pointer :: kiteareasonvertex
233 ! temporary arrays needed in the (to be constructed) init procedure
234 ! note that lambda is going to be constant for this and could be cached
235 real (kind=rkind), dimension(:), allocatable :: lambda
236 real (kind=rkind), dimension(:,:), allocatable :: pointvertex
237 real (kind=rkind), dimension(3) :: pointinterp
238 real (kind=rkind) :: xp,yp,zp , sumarea, kitearea
239 logical, pointer :: is_periodic
240 real(kind=rkind), pointer :: x_period, y_period
241
242 uvreconstructx = 0.0_rkind
243 uvreconstructy = 0.0_rkind
244 uvreconstructz = 0.0_rkind
245
246 call mpas_pool_get_dimension(meshpool, 'vertexDegree', vertexdegree)
247
248 allocate(lambda(vertexdegree), pointvertex(3,vertexdegree))
249
250 ! setup pointers
251 call mpas_pool_get_dimension(meshpool, 'nVerticesSolve', nverticessolve)
252 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
253 call mpas_pool_get_dimension(meshpool, 'nVertLevels', nvertlevels)
254 call mpas_pool_get_array(meshpool, 'cellsOnVertex', cellsonvertex)
255
256 call mpas_pool_get_array(meshpool, 'xCell', xcell)
257 call mpas_pool_get_array(meshpool, 'yCell', ycell)
258 call mpas_pool_get_array(meshpool, 'zCell', zcell)
259
260 call mpas_pool_get_array(meshpool, 'xVertex', xvertex)
261 call mpas_pool_get_array(meshpool, 'yVertex', yvertex)
262 call mpas_pool_get_array(meshpool, 'zVertex', zvertex)
263
264 call mpas_pool_get_array(meshpool, 'kiteAreasOnVertex', kiteareasonvertex)
265
266 call mpas_pool_get_config(meshpool, 'is_periodic', is_periodic)
267 call mpas_pool_get_config(meshpool, 'x_period', x_period)
268 call mpas_pool_get_config(meshpool, 'y_period', y_period)
269
270 ! loop over all vertices
271 do avertex = 1, nverticessolve
272 ! could precompute the list as an optimization
273 ! really, condition is any boundaryVertex in column greater than 0
274 if(any(boundaryvertex(:,avertex) < 1)) then
275 ! get vertex location and cell center locations
276 do acell = 1, vertexdegree
277 ! logical could be moved outside of code block as an optimization
278 ! (then essentially would have two nearly identical code blocks...)
279 if (is_periodic) then
280 ! fix periodicity with respect to pointInterp (xVertex)
281 pointvertex(1,acell) = mpas_fix_periodicity(xcell(cellsonvertex(acell, avertex)), xvertex(avertex), x_period)
282 pointvertex(2,acell) = mpas_fix_periodicity(ycell(cellsonvertex(acell, avertex)), yvertex(avertex), y_period)
283 pointvertex(3,acell) = zcell(cellsonvertex(acell, avertex))
284 else
285 pointvertex(1,acell) = xcell(cellsonvertex(acell, avertex))
286 pointvertex(2,acell) = ycell(cellsonvertex(acell, avertex))
287 pointvertex(3,acell) = zcell(cellsonvertex(acell, avertex))
288 end if
289 end do
290 ! vertex point for reconstruction
291 pointinterp(1) = xvertex(avertex)
292 pointinterp(2) = yvertex(avertex)
293 pointinterp(3) = zvertex(avertex)
294 ! get interpolation constants (could be cached / optimized with areaBin)
295 lambda = mpas_wachspress_coordinates(vertexdegree, pointvertex , pointinterp, meshpool)
296 else
297 lambda = 0.0_rkind
298 end if
299
300 do alevel = 1, nvertlevels
301 if(boundaryvertex(alevel,avertex) < 1) then
302 ! perform interpolation
303 uvreconstructx(alevel,avertex) = sum(ucreconstructx(alevel,cellsonvertex(:,avertex)) * lambda)
304 uvreconstructy(alevel,avertex) = sum(ucreconstructy(alevel,cellsonvertex(:,avertex)) * lambda)
305 uvreconstructz(alevel,avertex) = sum(ucreconstructz(alevel,cellsonvertex(:,avertex)) * lambda)
306 end if
307 end do
308
309 ! need to specify boundary conditions for the vertexes (outside this subroutine)
310
311 end do
312
313 deallocate(lambda, pointvertex)
314
315 end subroutine ocn_vector_cell_center_to_vertex!}}}

510!***********************************************************************
511!
512! routine ocn_simple_vector_laplacian_filter
513!
521!-----------------------------------------------------------------------
522 subroutine ocn_simple_vector_laplacian_filter(meshPool, scratchPool, boundaryCell, layerThickness, ucReconstruct) !{{{
523 implicit none
524
525 type (mpas_pool_type), pointer, intent(in) :: meshPool, scratchPool
526 integer, dimension(:,:), pointer, intent(in) :: boundaryCell
527 real (kind=rkind), dimension(:,:), pointer, intent(inout) :: ucreconstruct
528 real (kind=rkind), dimension(:,:), pointer, intent(in) :: layerthickness
529
530 ! local variables
531 type (field2DReal), pointer :: ucTemp
532 integer :: aCell, aNeigh, aLevel
533 integer, pointer :: nCellsSolve, nVertLevels
534 integer, dimension(:), pointer :: nEdgesOnCell
535 integer, dimension(:,:), pointer :: cellsOnCell
536 real (kind=rkind), dimension(:), pointer :: areacell
537 real (kind=rkind) :: volsum, cellvol
538
539 ! allocate scratch memory
540 call mpas_pool_get_field(scratchpool, 'ucTemp', uctemp)
541 call mpas_allocate_scratch_field(uctemp,.true.)
542
543 ! get values from pools
544 call mpas_pool_get_dimension(meshpool,'nCellsSolve',ncellssolve)
545 call mpas_pool_get_dimension(meshpool,'nVertLevels',nvertlevels)
546 call mpas_pool_get_array(meshpool,'nEdgesOnCell',nedgesoncell)
547 call mpas_pool_get_array(meshpool,'cellsOnCell', cellsoncell)
548 call mpas_pool_get_array(meshpool,'areaCell',areacell)
549
550 uctemp % array = 0.0_rkind
551
552 ! perform laplacian filtering
553 do acell = 1, ncellssolve
554 do alevel = 1, nvertlevels
555 volsum = nedgesoncell(acell) * layerthickness(alevel,acell) * areacell(acell) * (1-boundarycell(alevel,acell))
556 uctemp % array(alevel, acell) = ucreconstruct(alevel, acell) * volsum
557 if (volsum /= 0 ) then
558 ! loop over all neighbors
559 do aneigh = 1, nedgesoncell(acell)
560 cellvol = layerthickness(alevel,cellsoncell(aneigh,acell)) * areacell(cellsoncell(aneigh,acell)) &
561 * (1-boundarycell(alevel, cellsoncell(aneigh,acell)))
562 volsum = volsum + cellvol
563 uctemp % array(alevel, acell) = uctemp % array(alevel, acell) + ucreconstruct(alevel,cellsoncell(aneigh,acell)) &
564 * cellvol
565 end do
566 uctemp % array(alevel, acell) = uctemp % array(alevel, acell) / volsum
567 end if
568 end do
569 end do
570
571 ! exchange halo values
572 call mpas_dmpar_exch_halo_field(uctemp)
573
574 ! replace input values with filtered values
575 ucreconstruct = uctemp % array
576
577 ! deallocate scratch memory
578 call mpas_deallocate_scratch_field(uctemp , .true.)
579
580 end subroutine ocn_simple_vector_laplacian_filter !}}}

594 subroutine ocn_simple_vector_shapiro_filter(meshPool, scratchPool, boundaryVertex, boundaryCell, &
595 ucReconstructX, ucReconstructY, ucReconstructZ) !{{{
596 implicit none
597
598 type (mpas_pool_type), pointer, intent(in) :: meshPool, scratchPool
