73!-----------------------------------------------------------------------
74 subroutine li_timestep(domain, err)
75
76 !-----------------------------------------------------------------
77 !
78 ! input variables
79 !
80 !-----------------------------------------------------------------
81
82 !-----------------------------------------------------------------
83 !
84 ! input/output variables
85 !
86 !-----------------------------------------------------------------
87 type (domain_type), intent(inout) :: domain
88
89 !-----------------------------------------------------------------
90 !
91 ! output variables
92 !
93 !-----------------------------------------------------------------
94 integer, intent(out) :: err
95
96 !-----------------------------------------------------------------
97 !
98 ! local variables
99 !
100 !-----------------------------------------------------------------
101 ! Pools pointers
102 type (block_type), pointer :: block
103 type (mpas_pool_type), pointer :: meshpool
104 character (len=StrKIND), pointer :: xtime
105 real (kind=rkind), pointer :: dayssincestart
106 character (len=StrKIND), pointer :: simulationstarttime
107 character (len=StrKIND), pointer :: config_time_integration
108 logical, pointer :: config_adaptive_timestep
109 real (kind=rkind), pointer :: deltat
110
111 ! Other local variables
112 type (mpas_timeinterval_type) :: timestepinterval
113 real (kind=rkind) :: dtseconds
114 type (mpas_time_type) :: currtime
115 character(len=StrKIND) :: timestamp
116 type (mpas_time_type) :: simulationstarttime_timetype
117
118 integer :: err_tmp
119
120 err = 0
121 err_tmp = 0
122
123 call mpas_pool_get_config(liconfigs, 'config_time_integration', config_time_integration)
124 call mpas_pool_get_config(liconfigs, 'config_adaptive_timestep', config_adaptive_timestep)
125
126 currtime = mpas_get_clock_time(domain % clock, mpas_now, err_tmp)
127 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=err_tmp)
128
129 ! ===
130 ! === Non-adaptive timestep: Get dt in seconds
131 ! ===
132 if (.not. config_adaptive_timestep) then
133 ! Get the interval at this point in time - will be fixed for nonadaptive timestep, but need to get it out of the clock
134 timestepinterval = mpas_get_clock_timestep(domain % clock, ierr=err_tmp)
135 err = ior(err,err_tmp)
136 ! Convert the clock's time interval into a dt in seconds to be used by the time stepper,
137 ! using the currTime as the start time for this interval.
138 ! (We want to do this conversion before doint the timestep and advancing the clock because the dt
139 ! in seconds may change as the base time changes, and we want the old time as the base time.
140 ! For example, the number of seconds in a year will be longer in a leap year.
141 ! That is why nonadaptive timesteps have to be handled before the timestep, while
142 ! adaptive timesteps are handled after the timestep.
143 ! It may be possible to have them handled in the same place within li_tendency.F if we want to embed it that deeply.)
144 call mpas_get_timeinterval(timestepinterval, starttimein=currtime, dt=dtseconds, ierr=err_tmp)
145 err = ior(err,err_tmp)
146 else
147 ! initialize the dt to 0 when using the adaptive timestepper
148 ! it will be set by the time stepper but needs to avoid triggering an error in tend_layerThickness_fo_upwind
149 dtseconds = 0.0_rkind
150 endif
151
152 ! Assign current value to variabels in all blocks
153 block => domain % blocklist
154 do while (associated(block))
155 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
156 call mpas_pool_get_array(meshpool, 'deltat', deltat)
157 deltat = dtseconds
158
159 block => block % next
160 end do
161
162
163 ! ===
164 ! === Perform timestep
165 ! ===
166 !call mpas_log_write('Using ' // trim(config_time_integration) // ' time integration.')
167 select case (config_time_integration)
168 case ('forward_euler')
169 call li_time_integrator_forwardeuler(domain, err_tmp)
170 case ('rk4')
171 call mpas_log_write(trim(config_time_integration) // ' is not currently supported.', mpas_log_err)
172 err_tmp = 1
173 case default
174 call mpas_log_write(trim(config_time_integration) // ' is not a valid land ice time integration option.', mpas_log_err)
175 err_tmp = 1
176 end select
177 err = ior(err,err_tmp)
178
179
180 ! ===
181 ! === Adaptive timestep: update clock information
182 ! ===
183 ! Set time step in clock object since the time step could have changed
184 ! Need to get value out of a block, but all blocks should have the same value, so just use first block
185 if (config_adaptive_timestep) then
186 block => domain % blocklist
187 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
188 call mpas_pool_get_array(meshpool, 'deltat', deltat)
189 ! convert dt in seconds to timeInterval type
190 call mpas_set_timeinterval(timestepinterval, dt=deltat, ierr=err_tmp)
191 err = ior(err,err_tmp)
192 ! update the clock with the timeInterval
193 call mpas_set_clock_timestep(domain % clock, timestepinterval, err_tmp)
194 err = ior(err,err_tmp)
195 endif
196
197 ! ===
198 ! === Update clock information
199 ! ===
200 ! Advance clock - needed to wait until after time step is completed in case the dt has changed!
201 call mpas_advance_clock(domain % clock)
202 currtime = mpas_get_clock_time(domain % clock, mpas_now, err_tmp)
203 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=err_tmp)
204 err = ior(err, err_tmp)
205 call mpas_log_write(' Completed timestep. New time is: ' // trim(timestamp), flushnow=.true.)
206
207
208 block => domain % blocklist
209 do while (associated(block))
210 ! Assign the time stamp for this time step
211 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
212 call mpas_pool_get_array(meshpool, 'xtime', xtime)
213 xtime = timestamp
214
215 ! compute time since start of simulation, in days
216 call mpas_pool_get_array(meshpool, 'simulationStartTime', simulationstarttime)
217 call mpas_pool_get_array(meshpool, 'daysSinceStart',dayssincestart)
218 call mpas_set_time(simulationstarttime_timetype, datetimestring=simulationstarttime)
219 call mpas_get_timeinterval(currtime - simulationstarttime_timetype, dt=dayssincestart)
220 dayssincestart = dayssincestart / seconds_per_day
221
222 block => block % next
223 end do
224
225 ! === error check
226 if (err > 0) then
227 call mpas_log_write("An error has occurred in li_timestep.", mpas_log_err)
228 endif
229
230 !--------------------------------------------------------------------
231 end subroutine li_timestep
