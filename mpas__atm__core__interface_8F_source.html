102 function atm_setup_packages(configs, packages, iocontext) result(ierr)
103
104 use mpas_dmpar
105 use mpas_derived_types, only : mpas_pool_type, mpas_io_context_type
106 use mpas_pool_routines, only : mpas_pool_get_config, mpas_pool_get_package
107
108#ifdef DO_PHYSICS
109 use mpas_atmphys_control
110 use mpas_atmphys_packages
111#endif
112
113 implicit none
114
115 type (mpas_pool_type), intent(inout) :: configs
116 type (mpas_pool_type), intent(inout) :: packages
117 type (mpas_io_context_type), intent(inout) :: iocontext
118 integer :: ierr
119
120 logical, pointer :: iauactive
121 character(len=StrKIND), pointer :: config_iau_option
122 logical, pointer :: limited_areaactive
123 logical, pointer :: config_apply_lbcs
124 logical, pointer :: config_jedi_da, jedi_daactive
125 integer :: local_ierr
126
127 ierr = 0
128
129 !
130 ! Incremental analysis update
131 !
132 nullify(config_iau_option)
133 call mpas_pool_get_config(configs, 'config_IAU_option', config_iau_option)
134
135 nullify(iauactive)
136 call mpas_pool_get_package(packages, 'iauActive', iauactive)
137
138 if (trim(config_iau_option) /= 'off') then
139 iauactive = .true.
140 else
141 iauactive = .false.
142 end if
143
144 !
145 ! Limited-area
146 !
147 nullify(config_apply_lbcs)
148 call mpas_pool_get_config(configs, 'config_apply_lbcs', config_apply_lbcs)
149
150 nullify(limited_areaactive)
151 call mpas_pool_get_package(packages, 'limited_areaActive', limited_areaactive)
152
153 if (config_apply_lbcs) then
154 limited_areaactive = .true.
155 else
156 limited_areaactive = .false.
157 end if
158
159 !
160 ! JEDI data assimilation
161 !
162 nullify(config_jedi_da)
163 call mpas_pool_get_config(configs, 'config_jedi_da', config_jedi_da)
164
165 nullify(jedi_daactive)
166 call mpas_pool_get_package(packages, 'jedi_daActive', jedi_daactive)
167
168 if (associated(config_jedi_da) .and. associated(jedi_daactive)) then
169 jedi_daactive = config_jedi_da
170 else
171 ierr = ierr + 1
172 call mpas_log_write('Package setup failed for ''jedi_da''. '// &
173 'Either ''jedi_da'' is not a package, or ''config_jedi_da'' is not a namelist option.', &
174 messagetype=mpas_log_err)
175 end if
176
177#ifdef DO_PHYSICS
178 !check that all the physics options are correctly defined and that at
179 !least one physics parameterization is called (using the logical moist_physics):
180 call physics_namelist_check(configs)
181
182 local_ierr = atmphys_setup_packages(configs, packages, iocontext)
183 if (local_ierr /= 0) then
184 ierr = ierr + 1
185 call mpas_log_write('Package setup failed for atmphys in core_atmosphere', messagetype=mpas_log_err)
186 end if
187#endif
188
189 end function atm_setup_packages
