89 ! Get dimensions
90 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
91 call mpas_pool_get_dimension(meshpool, 'nCellsSolve', ncellssolve)
92 call mpas_pool_get_dimension(meshpool, 'nEdges', nedges)
93 call mpas_pool_get_dimension(meshpool, 'maxEdges', maxedges)

97 ! Initialize pointers

99 call mpas_pool_get_array(meshpool, 'cellsOnEdge', cellsonedge)
100 call mpas_pool_get_array(meshpool, 'edgesOnCell', edgesoncell)
101 call mpas_pool_get_array(meshpool, 'cellsOnCell', cellsoncell)
102 call mpas_pool_get_array(meshpool, 'areaCell', areacell)
103 call mpas_pool_get_array(meshpool, 'nEdgesOnCell', nedgesoncell)

106 if(present(maxlevelcell_in)) then
107 maxlevelcell => maxlevelcell_in

109 allocate(maxlevelcell(ncells+1))
110 maxlevelcell(:) = nvertlevels
111 end if
112
113 if(present(maxleveledgetop_in)) then
114 maxleveledgetop => maxleveledgetop_in

116 allocate(maxleveledgetop(nedges+1))
117 maxleveledgetop(:) = nvertlevels
118 end if

120 if(present(highorderadvectionmask_in)) then
121 highorderadvectionmask => highorderadvectionmask_in

123 allocate(highorderadvectionmask(nvertlevels, nedges+1))

126 else

128 end if
129 end if

143 do i = 1, nedgesoncell(icell)
144 iedge = edgesoncell(i, icell)
145 if(icell == cellsonedge(1, iedge)) then
146 edgesignoncell(i, icell) = -1
147 else
148 edgesignoncell(i, icell) = 1

150 end do
151 end do
152 end if

154 ! Setup high order horizontal flux

157 ! allocate nCells arrays

159

163 ! Loop over tracers. One tracer is advected at a time. It is copied into a temporary array in order to improve locality
164 do itracer = 1, num_tracers
165 ! Initialize variables for use in this iTracer iteration
166 tracer_cur(:, :) = tracers(itracer, :, :)

168 high_order_vert_flux = 0.0_rkind
169 high_order_horiz_flux = 0.0_rkind

171 ! Compute the high order vertical flux. Also determine bounds on tracer_cur.

173 k = max(1, min(maxlevelcell(icell), 2))
174 verticalweightk = verticalcellsize(k-1, icell) / (verticalcellsize(k, icell) + verticalcellsize(k-1, icell))
175 verticalweightkm1 = verticalcellsize(k, icell) / (verticalcellsize(k, icell) + verticalcellsize(k-1, icell))
176 high_order_vert_flux(k,icell) = w(k,icell)*(verticalweightk*tracer_cur(k,icell)+verticalweightkm1*tracer_cur(k-1,icell))
177
178 do k=3,maxlevelcell(icell)-1
179 if(vert4thorder) then

181 tracer_cur(k ,icell),tracer_cur(k+1,icell), w(k,icell))
182 else if(vert3rdorder) then
183 high_order_vert_flux(k, icell) = mpas_tracer_advection_vflux3( tracer_cur(k-2,icell),tracer_cur(k-1,icell), &
184 tracer_cur(k ,icell),tracer_cur(k+1,icell), w(k,icell), coef_3rd_order )
185 else if (vert2ndorder) then
186 verticalweightk = verticalcellsize(k-1, icell) / (verticalcellsize(k, icell) + verticalcellsize(k-1, icell))
187 verticalweightkm1 = verticalcellsize(k, icell) / (verticalcellsize(k, icell) + verticalcellsize(k-1, icell))
188 high_order_vert_flux(k,icell) = w(k,icell)*(verticalweightk*tracer_cur(k,icell)+verticalweightkm1*tracer_cur(k-1,icell))
189 end if
190 end do
191
192 k = max(1, maxlevelcell(icell))
193 verticalweightk = verticalcellsize(k-1, icell) / (verticalcellsize(k, icell) + verticalcellsize(k-1, icell))
194 verticalweightkm1 = verticalcellsize(k, icell) / (verticalcellsize(k, icell) + verticalcellsize(k-1, icell))
195 high_order_vert_flux(k,icell) = w(k,icell)*(verticalweightk*tracer_cur(k,icell)+verticalweightkm1*tracer_cur(k-1,icell))
196 end do ! iCell Loop
197
198 ! Compute the high order horizontal flux
199 do iedge = 1, nedges
200 cell1 = cellsonedge(1, iedge)
201 cell2 = cellsonedge(2, iedge)
202
203 ! Compute 2nd order fluxes where needed.
204 do k = 1, maxleveledgetop(iedge)
205 tracer_weight = iand(highorderadvectionmask(k, iedge)+1, 1) * (dvedge(iedge) * 0.5_rkind) * normalthicknessflux(k, iedge)
206
207 high_order_horiz_flux(k, iedge) = high_order_horiz_flux(k, iedge) + tracer_weight * (tracer_cur(k, cell1) + tracer_cur(k, cell2))
208 end do ! k loop
209
210 ! Compute 3rd or 4th fluxes where requested.
211 do i = 1, nadvcellsforedge(iedge)
212 icell = advcellsforedge(i,iedge)
213 do k = 1, maxlevelcell(icell)
214 tracer_weight = highorderadvectionmask(k, iedge) * (adv_coefs(i,iedge) + coef_3rd_order*sign(1.0_rkind,normalthicknessflux(k,iedge))*adv_coefs_3rd(i,iedge))
215
216 tracer_weight = normalthicknessflux(k,iedge)*tracer_weight
217 high_order_horiz_flux(k,iedge) = high_order_horiz_flux(k,iedge) + tracer_weight * tracer_cur(k,icell)
218 end do ! k loop
219 end do ! i loop over nAdvCellsForEdge
220 end do ! iEdge loop
221
222 ! Accumulate the scaled high order horizontal tendencies
223 do icell = 1, ncells
224 invareacell1 = 1.0 / areacell(icell)
225 do i = 1, nedgesoncell(icell)
226 iedge = edgesoncell(i, icell)
227 do k = 1, maxleveledgetop(iedge)
228 tend(itracer, k, icell) = tend(itracer, k, icell) + edgesignoncell(i, icell) * high_order_horiz_flux(k, iedge) * invareacell1
229 end do
230 end do
231 end do
232
233 ! Accumulate the scaled high order vertical tendencies.
234 do icell = 1, ncellssolve
235 do k = 1,maxlevelcell(icell)
236 tend(itracer, k, icell) = tend(itracer, k, icell) + verticaldivergencefactor(k) * (high_order_vert_flux(k+1, icell) - high_order_vert_flux(k, icell))
237 end do ! k loop
238 end do ! iCell loop
239 end do ! iTracer loop
240
241 deallocate(tracer_cur)
242 deallocate(high_order_horiz_flux)
243 deallocate(high_order_vert_flux)
244
245 if(.not.present(maxlevelcell_in)) then
246 deallocate(maxlevelcell)
247 end if
248
249 if(.not.present(maxleveledgetop_in)) then
250 deallocate(maxleveledgetop)
251 end if
252
253 if(.not.present(highorderadvectionmask_in)) then
254 deallocate(highorderadvectionmask)
255 end if
256
257 if(.not.present(verticaldivergencefactor_in)) then
258 deallocate(verticaldivergencefactor)
259 end if
260 end subroutine mpas_tracer_advection_std_tend!}}}
