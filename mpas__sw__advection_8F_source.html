430 ! FUNCTION SPHERE_ANGLE
431 !
432 ! Computes the angle between arcs AB and AC, given points A, B, and C
433 ! Equation numbers w.r.t. http://mathworld.wolfram.com/SphericalTrigonometry.html
434 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
435 real (kind=rkind) function sphere_angle(ax, ay, az, bx, by, bz, cx, cy, cz)
436
437 implicit none
438
439 real (kind=rkind), intent(in) :: ax, ay, az, bx, by, bz, cx, cy, cz
440
441 real (kind=rkind) :: a, b, c ! Side lengths of spherical triangle ABC
442
443 real (kind=rkind) :: abx, aby, abz ! The components of the vector AB
444 real (kind=rkind) :: mab ! The magnitude of AB
445 real (kind=rkind) :: acx, acy, acz ! The components of the vector AC
446 real (kind=rkind) :: mac ! The magnitude of AC
447
448 real (kind=rkind) :: dx ! The i-components of the cross product AB x AC
449 real (kind=rkind) :: dy ! The j-components of the cross product AB x AC
450 real (kind=rkind) :: dz ! The k-components of the cross product AB x AC
451
452 real (kind=rkind) :: s ! Semiperimeter of the triangle
453 real (kind=rkind) :: sin_angle
454
455 a = acos(max(min(bx*cx + by*cy + bz*cz,1.0_rkind),-1.0_rkind)) ! Eqn. (3)
456 b = acos(max(min(ax*cx + ay*cy + az*cz,1.0_rkind),-1.0_rkind)) ! Eqn. (2)
457 c = acos(max(min(ax*bx + ay*by + az*bz,1.0_rkind),-1.0_rkind)) ! Eqn. (1)
458
459 abx = bx - ax
460 aby = by - ay
461 abz = bz - az
462
463 acx = cx - ax
464 acy = cy - ay
465 acz = cz - az
466
467 dx = (aby * acz) - (abz * acy)
468 dy = -((abx * acz) - (abz * acx))
469 dz = (abx * acy) - (aby * acx)
470
471 s = 0.5*(a + b + c)
472! sin_angle = sqrt((sin(s-b)*sin(s-c))/(sin(b)*sin(c))) ! Eqn. (28)
473 sin_angle = sqrt(min(1.0_rkind,max(0.0_rkind,(sin(s-b)*sin(s-c))/(sin(b)*sin(c))))) ! Eqn. (28)
474
475 if ((dx*ax + dy*ay + dz*az) >= 0.0) then
476 sphere_angle = 2.0 * asin(max(min(sin_angle,1.0_rkind),-1.0_rkind))
477 else
478 sphere_angle = -2.0 * asin(max(min(sin_angle,1.0_rkind),-1.0_rkind))
479 end if
480
481 end function sphere_angle

484 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
485 ! FUNCTION PLANE_ANGLE
486 !
487 ! Computes the angle between vectors AB and AC, given points A, B, and C, and
488 ! a vector (u,v,w) normal to the plane.
489 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
490 real (kind=rkind) function plane_angle(ax, ay, az, bx, by, bz, cx, cy, cz, u, v, w)
491
492 implicit none
493
494 real (kind=rkind), intent(in) :: ax, ay, az, bx, by, bz, cx, cy, cz, u, v, w
495
496 real (kind=rkind) :: abx, aby, abz ! The components of the vector AB
497 real (kind=rkind) :: mab ! The magnitude of AB
498 real (kind=rkind) :: acx, acy, acz ! The components of the vector AC
499 real (kind=rkind) :: mac ! The magnitude of AC
500
501 real (kind=rkind) :: dx ! The i-components of the cross product AB x AC
502 real (kind=rkind) :: dy ! The j-components of the cross product AB x AC
503 real (kind=rkind) :: dz ! The k-components of the cross product AB x AC
504
505 real (kind=rkind) :: cos_angle
506
507 abx = bx - ax
508 aby = by - ay
509 abz = bz - az
510 mab = sqrt(abx**2.0 + aby**2.0 + abz**2.0)
511
512 acx = cx - ax
513 acy = cy - ay
514 acz = cz - az
515 mac = sqrt(acx**2.0 + acy**2.0 + acz**2.0)
516
517
518 dx = (aby * acz) - (abz * acy)
519 dy = -((abx * acz) - (abz * acx))
520 dz = (abx * acy) - (aby * acx)
521
522 cos_angle = (abx*acx + aby*acy + abz*acz) / (mab * mac)
523
524 if ((dx*u + dy*v + dz*w) >= 0.0) then
525 plane_angle = acos(max(min(cos_angle,1.0_rkind),-1.0_rkind))
526 else
527 plane_angle = -acos(max(min(cos_angle,1.0_rkind),-1.0_rkind))
528 end if
529
530 end function plane_angle

533 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
534 ! FUNCTION ARC_LENGTH
535 !
536 ! Returns the length of the great circle arc from A=(ax, ay, az) to
537 ! B=(bx, by, bz). It is assumed that both A and B lie on the surface of the
538 ! same sphere centered at the origin.
539 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
540 real (kind=rkind) function arc_length(ax, ay, az, bx, by, bz)
541
542 implicit none
543
544 real (kind=rkind), intent(in) :: ax, ay, az, bx, by, bz
545
546 real (kind=rkind) :: r, c
547 real (kind=rkind) :: cx, cy, cz
548
549 cx = bx - ax
550 cy = by - ay
551 cz = bz - az
552
553! r = ax*ax + ay*ay + az*az
554! c = cx*cx + cy*cy + cz*cz
555!
556! arc_length = sqrt(r) * acos(1.0 - c/(2.0*r))
557
558 r = sqrt(ax*ax + ay*ay + az*az)
559 c = sqrt(cx*cx + cy*cy + cz*cz)
560! arc_length = sqrt(r) * 2.0 * asin(c/(2.0*r))
561 arc_length = r * 2.0 * asin(c/(2.0*r))
562
563 end function arc_length

566 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
567 ! subroutine sw_arc_bisect
568 !
569 ! Returns the point C=(cx, cy, cz) that bisects the great circle arc from
570 ! A=(ax, ay, az) to B=(bx, by, bz). It is assumed that A and B lie on the
571 ! surface of a sphere centered at the origin.
572 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
573 subroutine sw_arc_bisect(ax, ay, az, bx, by, bz, cx, cy, cz)
574
575 implicit none
576
577 real (kind=rkind), intent(in) :: ax, ay, az, bx, by, bz
578 real (kind=rkind), intent(out) :: cx, cy, cz
579
580 real (kind=rkind) :: r ! Radius of the sphere
581 real (kind=rkind) :: d
582
583 r = sqrt(ax*ax + ay*ay + az*az)
584
585 cx = 0.5*(ax + bx)
586 cy = 0.5*(ay + by)
587 cz = 0.5*(az + bz)
588
589 if (cx == 0. .and. cy == 0. .and. cz == 0.) then
590 call mpas_log_write('arc_bisect: A and B are diametrically opposite', mpas_log_err)
591 else
592 d = sqrt(cx*cx + cy*cy + cz*cz)
593 cx = r * cx / d
594 cy = r * cy / d
595 cz = r * cz / d
596 end if
597
598 end subroutine sw_arc_bisect

601 subroutine sw_poly_fit_2(a_in,b_out,weights_in,m,n,ne)
602
603 implicit none

675subroutine sw_migs (A,N,X,INDX)
676!
677! subroutine to invert matrix A(N,N) with the inverse stored
678! in X(N,N) in the output. Copyright (c) Tao Pang 2001.
679!
680 IMPLICIT NONE
681 INTEGER, INTENT (IN) :: N
682 INTEGER :: I,J,K
683 INTEGER, INTENT (OUT), DIMENSION (N) :: INDX
684 REAL (kind=rkind), INTENT (INOUT), DIMENSION (N,N):: a
685 REAL (kind=rkind), INTENT (OUT), DIMENSION (N,N):: x
686 REAL (kind=rkind), DIMENSION (N,N) :: b
687!
688 DO i = 1, n
689 DO j = 1, n
690 b(i,j) = 0.0
691 END DO
692 END DO
693 DO i = 1, n
694 b(i,i) = 1.0
695 END DO
696!
697 call sw_elgs (a,n,indx)
698!
699 DO i = 1, n-1
700 DO j = i+1, n
701 DO k = 1, n
702 b(indx(j),k) = b(indx(j),k)-a(indx(j),i)*b(indx(i),k)
703 END DO
704 END DO
705 END DO
706!
707 DO i = 1, n
708 x(n,i) = b(indx(n),i)/a(indx(n),n)
709 DO j = n-1, 1, -1
710 x(j,i) = b(indx(j),i)
711 DO k = j+1, n
712 x(j,i) = x(j,i)-a(indx(j),k)*x(k,i)
713 END DO
714 x(j,i) = x(j,i)/a(indx(j),j)
715 END DO
716 END DO
717end subroutine sw_migs

720subroutine sw_elgs (A,N,INDX)
721!
722! subroutine to perform the partial-pivoting Gaussian elimination.
723! A(N,N) is the original matrix in the input and transformed matrix
724! plus the pivoting element ratios below the diagonal in the output.
725! INDX(N) records the pivoting order. Copyright (c) Tao Pang 2001.
726!
727 IMPLICIT NONE
728 INTEGER, INTENT (IN) :: N
729 INTEGER :: I,J,K,ITMP
730 INTEGER, INTENT (OUT), DIMENSION (N) :: INDX
731 REAL (kind=rkind) :: c1,pi,pi1,pj
732 REAL (kind=rkind), INTENT (INOUT), DIMENSION (N,N) :: a
733 REAL (kind=rkind), DIMENSION (N) :: c
734!
735! Initialize the index
736!
737 DO i = 1, n
738 indx(i) = i
739 END DO
740!
741! Find the rescaling factors, one from each row
742!
743 DO i = 1, n
744 c1= 0.0
745 DO j = 1, n
746 c1 = max(c1,abs(a(i,j)))
747 END DO
748 c(i) = c1
749 END DO
750!
751! Search the pivoting (largest) element from each column
752!
753 DO j = 1, n-1
754 pi1 = 0.0
755 DO i = j, n
756 pi = abs(a(indx(i),j))/c(indx(i))
757 IF (pi.GT.pi1) THEN
758 pi1 = pi
759 k = i
760 ENDIF
761 END DO
762!
763! Interchange the rows via INDX(N) to record pivoting order
764!
765 itmp = indx(j)
766 indx(j) = indx(k)
767 indx(k) = itmp
768 DO i = j+1, n
769 pj = a(indx(i),j)/a(indx(j),j)
770!
771! Record pivoting ratios below the diagonal
772!
773 a(indx(i),j) = pj
774!
775! Modify other elements accordingly
776!
777 DO k = j+1, n
778 a(indx(i),k) = a(indx(i),k)-pj*a(indx(j),k)
779 END DO
780 END DO
781 END DO
782!
783end subroutine sw_elgs
