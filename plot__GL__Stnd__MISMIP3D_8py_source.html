50def xtime2numtime(xtime):
51 """Define a function to convert xtime character array to numeric time values using datetime objects"""
52 # First parse the xtime character array into a string
53 xtimestr = netCDF4.chartostring(xtime) # convert from the character array to an array of strings using the netCDF4 module's function
54
55 dt = []
56 for stritem in xtimestr:
57 itemarray = stritem.strip().replace('_', '-').replace(':', '-').split('-') # Get an array of strings that are Y,M,D,h,m,s
58 results = [int(i) for i in itemarray]
59 if (results[0] < 1900): # datetime has a bug where years less than 1900 are invalid on some systems
60 results[0] += 1900
61 dt.append( datetime.datetime(*results) ) # * notation passes in the array as arguments
62
63 numtime = netCDF4.date2num(dt, units='seconds since '+str(dt[0])) # use the netCDF4 module's function for converting a datetime to a time number
64 numtime /= (3600.0 * 24.0 * 365.0)
65 numtime -= numtime[0] # return years from start
66 return numtime
