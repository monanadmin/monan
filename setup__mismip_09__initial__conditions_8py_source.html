93#The following function computes the MISMIP+ bed according to Asay-Davis et al. (2016)
94
95def computeBed(x,y):
96 x = x/1.e3 # m to km
97 y = y/1.e3 # m to km
98 B0 = -150. # m
99 B2 = -728.8 # m
100 B4 = 343.91 # m
101 B6 = -50.57 # m
102 x_bar = 300. # km
103 x_tilde = x/x_bar
104 dc = 500. # m
105 fc = 4. # km
106 wc = 24. # km
107 Ly = 80. # km
108 Bmax = -720. # m
109 B_x = B0 + B2*x_tilde**2 + B4*x_tilde**4 + B6*x_tilde**6
110 B_y = dc / (1 + np.exp(-2*(y-Ly/2-wc)/fc)) + dc / (1 + np.exp(2*(y-Ly/2+wc)/fc))
111 Bsum = B_x + B_y
112 B = np.maximum(Bsum, Bmax) # B >= Bmax
113 return B
