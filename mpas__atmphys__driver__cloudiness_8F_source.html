26!2013-05-01.

28! subroutines in mpas_atmphys_driver_cloudiness:
29! ----------------------------------------------
30! allocate_cloudiness : allocate local arrays for parameterization of diagnostic cloudiness.
31! deallocate_cloudiness: deallocate local arrays for parameterization of diagnostic cloudiness.
32! cloudiness_from_MPAS : initialize local arrays.
33! cloudiness_to_MPAS : copy local arrays to MPAS arrays.
34! driver_cloudiness : main driver (called from subroutine physics_driver).
35! calc_cldincidence : calculates the cloud fraction as 0 or 1, depending on cloud condensates.
36! calc_cldfraction : calculates the cloud fraction as a function of the relative humidity.
37!
38! add-ons and modifications to sourcecode:
39! ----------------------------------------
40! * throughout the sourcecode, replaced all "var_struct" defined arrays by local pointers.
41! Laura D. Fowler (laura@ucar.edu) / 2014-04-22.
42! * modified sourcecode to use pools.
43! Laura D. Fowler (laura@ucar.edu) / 2014-05-15.
44! * added subroutine cal_cldfra3 which contains the calculation of the radiative cloud fraction in conjunction
45! with the Thompson cloud microphysics.
46! Laura D. Fowler (laura@ucar.edu) / 2016-06-04.
47! * initialize the local "radiation" water vapor, cloud water, cloud ice, and snow mixing ratios. When using the
48! option "cld_fraction_thompson", the "radiative" cloud water and ice paths are updated in conjunction with
49! cloud formation, but changes to the cloud water and cloud ice mixing ratios only affect the long wave and
50! short wave radiation codes.
51! Laura D. Fowler (laura@ucar.edu) / 2016-07-05.
52! * since we removed the local variable radt_cld_scheme from mpas_atmphys_vars.F, now defines radt_cld_scheme
53! as a pointer to config_radt_cld_scheme.
54! Laura D. Fowler (laura@ucar.edu) / 2917-02-16.

92!input arguments:
93 integer,intent(in):: its,ite
94
95!input and inout arguments:
96 type(mpas_pool_type),intent(in) :: configs
97 type(mpas_pool_type),intent(in) :: mesh
98 type(mpas_pool_type),intent(in) :: sfc_input
99
100!inout arguments:
101 type(mpas_pool_type),intent(inout):: diag_physics
102
103!local variables and pointers:
104 integer:: i,j,k
105
106 real(kind=rkind),pointer:: len_disp
107 real(kind=rkind),dimension(:),pointer:: areacell,meshdensity
108 real(kind=rkind),dimension(:),pointer:: xland
109
110!-----------------------------------------------------------------------------------------------------------------
111
112 call mpas_pool_get_config(configs,'config_len_disp',len_disp)
113
114 call mpas_pool_get_array(mesh,'areaCell' ,areacell )

118
119 do j = jts,jte
120 do i = its,ite
121 dx_p(i,j) = len_disp / meshdensity(i)**0.25
122 xland_p(i,j) = xland(i)
123 enddo

126 ! and snow mixing ratios:

130 qcrad_p(i,k,j) = qc_p(i,k,j)
131 qirad_p(i,k,j) = qi_p(i,k,j)
132 qsrad_p(i,k,j) = qs_p(i,k,j)

134 enddo
135 enddo
136 enddo

170 subroutine driver_cloudiness(configs,mesh,diag_physics,sfc_input,its,ite)
171!=================================================================================================================
172
173!input arguments:
174 type(mpas_pool_type),intent(in) :: configs
175 type(mpas_pool_type),intent(in) :: mesh
176 type(mpas_pool_type),intent(in) :: sfc_input
177
178 integer,intent(in):: its,ite
179
180!inout arguments:
181 type(mpas_pool_type),intent(inout):: diag_physics
182
183!local variables and pointers:
184 character(len=StrKIND),pointer:: radt_cld_scheme
185
186 integer:: i,j,k
187
188!-----------------------------------------------------------------------------------------------------------------
189!call mpas_log_write('')
190!call mpas_log_write('--- enter subroutine driver_cloudiness:')
191
192 call mpas_pool_get_config(configs,'config_radt_cld_scheme',radt_cld_scheme)
193
194!copy MPAS arrays to local arrays:
195 call cloudiness_from_mpas(configs,mesh,diag_physics,sfc_input,its,ite)
196
197 cld_fraction_select: select case (trim(radt_cld_scheme))
198 case("cld_incidence")
199 call mpas_timer_start('calc_cldincidence')
200 call calc_cldincidence(cldfrac_p,qcrad_p,qirad_p,f_qc,f_qi,its,ite)
201 call mpas_timer_stop('calc_cldincidence')
202
203 case("cld_fraction")
204 call mpas_timer_start('calc_cldfraction')
205 call calc_cldfraction(cldfrac_p,t_p,pres_p,qvrad_p,qcrad_p,qirad_p,qs_p,its,ite)
206 call mpas_timer_stop('calc_cldfraction')
207
208 case("cld_fraction_thompson")
209 call mpas_timer_start('cal_cldfra3')
210 call cal_cldfra3( &
211 cldfra = cldfrac_p , qv = qvrad_p , qc = qcrad_p , qi = qirad_p , &
212 qs = qsrad_p , p = pres_hyd_p , t = t_p , rho = rho_p , &
213 xland = xland_p , gridkm = dx_p , &
214 ids = ids , ide = ide , jds = jds , jde = jde , kds = kds , kde = kde , &
215 ims = ims , ime = ime , jms = jms , jme = jme , kms = kds , kme = kme , &
216 its = its , ite = ite , jts = jts , jte = jte , kts = kts , kte = kte &
217 )
218 call mpas_timer_stop('cal_cldfra3')
219
220 case default
221
222 end select cld_fraction_select
223
224!copy local arrays to MPAS grid:
225 call cloudiness_to_mpas(diag_physics,its,ite)
226
227!call mpas_log_write('--- end subroutine driver_cloudiness.')
228
229 end subroutine driver_cloudiness

232 subroutine calc_cldincidence(cldfrac,qc,qi,f_qc,f_qi,its,ite)
233!=================================================================================================================
234
235!input arguments:
236 logical,intent(in):: f_qc,f_qi
237 integer,intent(in):: its,ite
238 real(kind=rkind),intent(in),dimension(ims:ime,kms:kme,jms:jme):: qc,qi
239
240!output arguments:
241 real(kind=rkind),intent(out),dimension(ims:ime,kms:kme,jms:jme):: cldfrac
242
243!local variables:
244 integer:: i,j,k
245
246 real(kind=rkind),parameter:: thresh = 1.e-06
247
248!-----------------------------------------------------------------------------------------------------------------
249
250 do j = jts,jte
251 do k = kts,kte
252 do i = its,ite
253 cldfrac(i,k,j) = 0._rkind
254 enddo
255 enddo
256 enddo
257
258 if(f_qc .and. f_qi) then
259 do j = jts,jte
260 do k = kts,kte
261 do i = its,ite
262 if(qc(i,k,j)+qi(i,k,j) .gt. thresh) cldfrac(i,k,j) = 1.0_rkind
263 enddo
264 enddo
265 enddo
266 elseif(f_qc) then
267 do j = jts,jte
268 do k = kts,kte
269 do i = its,ite
270 if(qc(i,k,j) .gt. thresh) cldfrac(i,k,j) = 1.0_rkind
271 enddo
272 enddo
273 enddo
274 endif

276 end subroutine calc_cldincidence

278!=================================================================================================================
279 subroutine calc_cldfraction(cldfrac,t_p,pres_p,qv,qc,qi,qs,its,ite)
280!=================================================================================================================
281
282!input arguments:
283 integer,intent(in):: its,ite
284 real(kind=rkind),intent(in),dimension(ims:ime,kms:kme,jms:jme):: qv,qc,qi,qs
285 real(kind=rkind),intent(in),dimension(ims:ime,kms:kme,jms:jme):: t_p,pres_p
286
287!output arguments:
288 real(kind=rkind),intent(out),dimension(ims:ime,kms:kme,jms:jme):: cldfrac
289
290!local variables:
291 integer:: i,j,k
292
293 real(kind=rkind),parameter:: alpha0 = 100.
294 real(kind=rkind),parameter:: gamma = 0.49
295 real(kind=rkind),parameter:: qcldmin = 1.e-12
296 real(kind=rkind),parameter:: pexp = 0.25
297 real(kind=rkind),parameter:: rhgrid = 1.0
298
299 real(kind=rkind),parameter:: svp1 = 0.61078
300 real(kind=rkind),parameter:: svp2 = 17.2693882
301 real(kind=rkind),parameter:: svpi2 = 21.8745584
302 real(kind=rkind),parameter:: svp3 = 35.86
303 real(kind=rkind),parameter:: svpi3 = 7.66
304 real(kind=rkind),parameter:: svpt0 = 273.15
305
306 real(kind=rkind):: esi,esw,qvsi,qvsw
307 real(kind=rkind):: arg,denom,qcld,qvs,rhum,subsat,weight
308
309!-----------------------------------------------------------------------------------------------------------------
310
311 do j = jts,jte
312 do k = kts,kte
313 do i = its,ite
314 cldfrac(i,k,j) = 0._rkind
315 enddo
316 enddo
317 enddo
318
319 do j = jts,jte
320 do k = kts,kte
321 do i = its,ite
322
323!... calculation of the saturation mixing ratios over water and over ice (Murray, 1966):
324 esw = 1000. * svp1 * exp(svp2 * (t_p(i,k,j) - svpt0) / (t_p(i,k,j) - svp3))
325 esi = 1000. * svp1 * exp(svpi2 * (t_p(i,k,j) - svpt0) / (t_p(i,k,j) - svpi3))
326
327 qvsw = ep_2 * esw / (pres_p(i,k,j) - esw)
328 qvsi = ep_2 * esi / (pres_p(i,k,j) - esi)
329
330 qcld = qc(i,k,j) + qi(i,k,j) + qs(i,k,j)
331 if(qcld .lt. qcldmin) then
332 weight = 0.
333 else
334 weight = (qi(i,k,j) + qs(i,k,j)) / qcld
335 endif
336
337 qvs = (1-weight) * qvsw + weight * qvsi
338 rhum = qv(i,k,j) / qvs
339
340 if(qcld .lt. qcldmin) then
341
342 !assume that the cloud fraction is equal to 0. when the cloudy mixing ratio equals 0.
343 cldfrac(i,k,j) = 0.
344
345 elseif(rhum .ge. rhgrid) then
346 !assume that the cloud fraction is equal to 1. when the relative humidity equal 100%.
347 cldfrac(i,k,j) = 1.
348
349 else
350 !computation of the cloud fraction:
351 subsat = max(1.e-10,rhgrid*qvs-qv(i,k,j))
352 denom = subsat**gamma
353 arg = max(-6.9,-alpha0*qcld/denom) ! exp(-6.9) = 0.001
354
355 rhum = max(1.e-10,rhum)
356 cldfrac(i,k,j) = (rhum/rhgrid)**pexp*(1.-exp(arg))
357 if(cldfrac(i,k,j) .lt. 0.01) cldfrac(i,k,j) = 0.
358
359 endif
360
361 enddo
362 enddo
363 enddo
364
365 end subroutine calc_cldfraction
