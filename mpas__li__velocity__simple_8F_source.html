129
130 use mpas_vector_reconstruction
131
132 !-----------------------------------------------------------------
133 !
134 ! input variables
135 !
136 !-----------------------------------------------------------------
137
138 !-----------------------------------------------------------------
139 !
140 ! input/output variables
141 !
142 !-----------------------------------------------------------------
143 type (block_type), intent(inout) :: &
144 block
145
146 !-----------------------------------------------------------------
147 !
148 ! output variables
149 !
150 !-----------------------------------------------------------------
151
152 integer, intent(out) :: err
153
154 !-----------------------------------------------------------------
155 !
156 ! local variables
157 !
158 !-----------------------------------------------------------------
159
160 type (mpas_pool_type), pointer :: meshpool
161 type (mpas_pool_type), pointer :: velocitypool
162 type (mpas_pool_type), pointer :: scratchpool
163
164 integer, pointer :: ncells, nedges
165 integer, pointer :: ncellssolve, nedgessolve
166 integer, pointer :: nvertinterfaces
167 integer, pointer :: config_stats_cell_id
168
169 character(len=StrKind), pointer :: config_simple_velocity_type
170
171 !NOTE: Assume a planar mesh, so z coordinates are not needed
172 real (kind=rkind), dimension(:), pointer :: &
173 xcell, ycell, & ! cell center coordinates
174 xedge, yedge, & ! edge midpoint coordinates
175 dcedge ! distance between the 2 cell centers on each side of an edge
176
177 ! prescribed velocity at cell centers
178 type (field1dreal), pointer :: uvelocityxfield
179 type (field1dreal), pointer :: uvelocityyfield
180 real (kind=rkind), dimension(:), pointer :: uvelocityx
181 real (kind=rkind), dimension(:), pointer :: uvelocityy
182
183 real (kind=rkind), dimension(:,:), pointer :: &
184 normalvelocityinitial, & ! normal component of velocity on edges
185 ureconstructx, ureconstructy, ureconstructz, & ! x/y/z velocity components at cell center
186 ureconstructzonal, ureconstructmeridional ! zonal and meridional velocity components at cell center
187
188 integer, dimension(:,:), pointer :: cellsonedge ! indices for the 2 cells on each edge
189
190 real (kind=rkind), dimension(2) :: unitnormalvector ! x/y components of normal vector on an edge
191
192 real (kind=rkind) :: magnitude, radius, speed, xdiff, ydiff
193
194 real (kind=rkind) :: uedgex, uedgey ! x/y components of velocity at edge midpoints
195
196 integer :: err_tmp
197
198 integer :: ilevel, iedge, icell, icell1, icell2
199
200 real (kind=rkind), parameter :: flowspeed = 1000._rkind/scyr ! flow speed (m/s)
201 ! applies to uniform flow
202 real (kind=rkind), parameter :: flowtheta = 0.0_rkind ! direction of flow (0 < theta < 2*pi)
203 ! applied to uniform flow (not radial)
204
205 !Note: For radial flow, the user may want to reset these parameters
206 real (kind=rkind), parameter :: flowgradient = 1.2e-3_rkind/scyr ! du/dr for radial flow
207
208 real (kind=rkind), parameter :: xcenter = 0.0_rkind ! x coordinate of center of radial flow
209 real (kind=rkind), parameter :: ycenter = 0.0_rkind ! y coordinate of center of radial flow
210
211 !WHL - debug diagnostics only
212 integer, dimension(:), pointer :: nedgesoncell
213 integer, dimension(:,:), pointer :: edgesoncell ! index for each edge on a cell
214 real(kind=rkind), dimension(:,:,:), pointer :: &
215 coeffsreconstruct ! coefficients for reconstructing edge-based fields at cell centers
216 integer :: iedgeoncell
217
218 logical, parameter :: velocity_simple_bug_check = .false.
219
220 !--------------------------------
221 !WHL - optional thickness, SMB and topography tweaking for the radial velocity field and circular-shelf test case
222 !TODO - Remove these options after testing
223
224 character(len=StrKIND), pointer :: config_calving
225
226 logical, parameter :: radialmelting = .false.
227!! logical, parameter :: radialMelting = .true.
228 type (mpas_pool_type), pointer :: geometrypool
229 real (kind=rkind), dimension(:), pointer :: thickness
230 real (kind=rkind), dimension(:), pointer :: sfcmassbal
231 real (kind=rkind), dimension(:), pointer :: bedtopography
232
233 real (kind=rkind), parameter :: &
234 maxradius = 21000.0_rkind ! ice radius (m) for circular shelf problem
235
236 real (kind=rkind), parameter :: &
237 maxmelt = 100.0_rkind * 910.0_rkind / scyr ! max melt rate, kg/m2/s (converted from 100 m/yr)
238 ! for radial melting option
239 real (kind=rkind), parameter :: &
240 spiketopography = -880.0_rkind ! elevation of spike that grounds the ice
241 ! for config_calving = 'floating'
242
243 integer, parameter :: ncellsperrow = 40
244 integer, parameter :: nrows = 46
245 integer :: i, irow
246
247 character (len=strKind) :: msg
248
249 !--------------------------------
250
251 ! No block init needed.
252 err = 0
253 err_tmp = 0
254
255 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
256 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
257 call mpas_pool_get_subpool(block % structs, 'scratch', scratchpool)
258
259 ! Set needed variables and pointers
260
261 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
262 call mpas_pool_get_dimension(meshpool, 'nEdges', nedges)
263 call mpas_pool_get_dimension(meshpool, 'nCellsSolve', ncellssolve)
264 call mpas_pool_get_dimension(meshpool, 'nEdgesSolve', nedgessolve)
265 call mpas_pool_get_dimension(meshpool, 'nVertInterfaces', nvertinterfaces)
266
267 call mpas_pool_get_array(meshpool, 'cellsOnEdge', cellsonedge)
268 call mpas_pool_get_array(meshpool, 'xCell', xcell)
269 call mpas_pool_get_array(meshpool, 'yCell', ycell)
270 call mpas_pool_get_array(meshpool, 'xEdge', xedge)
271 call mpas_pool_get_array(meshpool, 'yEdge', yedge)
272 call mpas_pool_get_array(meshpool, 'dcEdge', dcedge)
273
274 call mpas_pool_get_array(velocitypool, 'normalVelocityInitial', normalvelocityinitial)
275 call mpas_pool_get_array(velocitypool, 'uReconstructX', ureconstructx)
276 call mpas_pool_get_array(velocitypool, 'uReconstructY', ureconstructy)
277 call mpas_pool_get_array(velocitypool, 'uReconstructZ', ureconstructz)
278 call mpas_pool_get_array(velocitypool, 'uReconstructZonal', ureconstructzonal)
279 call mpas_pool_get_array(velocitypool, 'uReconstructMeridional', ureconstructmeridional)
280
281 call mpas_pool_get_field(scratchpool, 'workCell', uvelocityxfield)
282 call mpas_allocate_scratch_field(uvelocityxfield, .true.)
283 uvelocityx => uvelocityxfield % array
284
285 call mpas_pool_get_field(scratchpool, 'workCell2', uvelocityyfield)
286 call mpas_allocate_scratch_field(uvelocityyfield, .true.)
287 uvelocityy => uvelocityyfield % array
288
289 call mpas_pool_get_config(liconfigs, 'config_stats_cell_ID', config_stats_cell_id)
290 call mpas_pool_get_config(liconfigs, 'config_simple_velocity_type', config_simple_velocity_type)
291
292 !WHL - debug diagnostics only
293 call mpas_pool_get_array(meshpool, 'edgesOnCell', edgesoncell)
294 call mpas_pool_get_array(meshpool, 'nEdgesOnCell', nedgesoncell)
295 call mpas_pool_get_array(meshpool, 'coeffs_reconstruct', coeffsreconstruct)
296
297 uvelocityx(:) = 0.0_rkind
298 uvelocityy(:) = 0.0_rkind
299
300 ! prescribe the x and y velocity components at cell centers (with no vertical variation)
301
302 if (trim(config_simple_velocity_type) == 'uniform') then
303
304 uvelocityx(:) = flowspeed * cos(flowtheta)
305 uvelocityy(:) = flowspeed * sin(flowtheta)
306
307 elseif (trim(config_simple_velocity_type) == 'radial') then
308
309 do icell = 1, ncells
310 xdiff = xcell(icell) - xcenter
311 ydiff = ycell(icell) - ycenter
312 radius = sqrt(xdiff**2 + ydiff**2)
313 if (radius > 0.0_rkind) then
314 speed = flowgradient * radius
315 uvelocityx(icell) = speed * xdiff/radius
316 uvelocityy(icell) = speed * ydiff/radius
317 else
318 uvelocityx(icell) = 0.0_rkind
319 uvelocityy(icell) = 0.0_rkind
320 endif
321 enddo
322
323 endif
324
325 ! given the velocity components at cell centers, compute the normal velocity component on edges
326
327 normalvelocityinitial(:,:) = 0.0_rkind
328
329 do iedge = 1, nedgessolve
330
331 ilevel = 1
332 icell1 = cellsonedge(1,iedge)
333 icell2 = cellsonedge(2,iedge)
334
335 ! average the velocity from the neighboring cells to the edge
336 uedgex = 0.5_rkind * (uvelocityx(icell1) + uvelocityx(icell2))
337 uedgey = 0.5_rkind * (uvelocityy(icell1) + uvelocityy(icell2))
338
339 ! Compute the components of the normal vector on the edge
340
341 unitnormalvector(1) = xedge(iedge) - xcell(icell1)
342 unitnormalvector(2) = yedge(iedge) - ycell(icell1)
343 magnitude = sqrt(unitnormalvector(1)**2 + unitnormalvector(2)**2)
344
345 ! Note: The magnitude should be dcEdge/2.
346 ! But this may not be the case for edges at the border of a periodic domain;
347 ! for these cells, the magnitude may be comparable to the domain size.
348 ! For such edges, create the normal vector from iCell2 instead.
349 if (magnitude > dcedge(iedge)) then
350! write(stderrUnit,*) 'Use iCell2 instead: iEdge, iCell1, iCell2, magnitude, dcEdge =', &
351! iEdge, iCell1, iCell2, magnitude, dcEdge(iEdge)
352 unitnormalvector(1) = -(xedge(iedge) - xcell(icell2))
353 unitnormalvector(2) = -(yedge(iedge) - ycell(icell2))
354 magnitude = sqrt(unitnormalvector(1)**2 + unitnormalvector(2)**2)
355 endif
356
357 unitnormalvector(:) = unitnormalvector(:)/magnitude
358
359 ! Compute the dot product of the velocity with the normal vector
360 ! Set to the same value everywhere in the column
361 normalvelocityinitial(:,iedge) = uedgex*unitnormalvector(1) + uedgey*unitnormalvector(2)
362
363 enddo ! iEdge
364
365 !WHL - debug
366 icell = config_stats_cell_id
367 ilevel = 1
368 call mpas_log_write(' ')
369 write(msg,*) 'Prescribed velocity, iCell, uvel, vvel (m/yr):', icell, uvelocityx(icell)*scyr, uvelocityy(icell)*scyr
370 call mpas_log_write(msg)
371 write(msg,*) 'xCell, yCell:', xcell(icell), ycell(icell)
372 call mpas_log_write(msg)
373 call mpas_log_write('iEdgeOnCell, cellsOnEdge, normalVelocity:')
374 do iedgeoncell = 1, nedgesoncell(icell)
375 iedge = edgesoncell(iedgeoncell,icell)
376 write(msg,*) iedgeoncell, cellsonedge(:,iedge), normalvelocityinitial(ilevel,iedge)*scyr
377 call mpas_log_write(msg)
378 enddo
379
380 if (velocity_simple_bug_check) then
381
382 ! Make sure we can recover the cell-center velocity to a good approximation
383 call mpas_reconstruct(meshpool, normalvelocityinitial, &
384 ureconstructx, ureconstructy, ureconstructz, &
385 ureconstructzonal, ureconstructmeridional )
386
387 ! Loop over cells, comparing the reconstructed velocity to the prescribed velocity
388 ! Note: Currently, the reconstruction coefficients are not correct for cells at the edge of a periodic domain,
389 ! so errors will be generated even though the normal velocities are correct.
390 ! For this reason I have commented out the warning messages.
391
392 do icell = 1, ncellssolve
393
394 speed = sqrt(uvelocityx(icell)**2 + uvelocityy(icell)**2)
395
396 if (icell == config_stats_cell_id) then
397 ilevel = 1
398 call mpas_log_write(' ')
399 write(msg,*) 'Velocity reconstruction, iCell =', icell
400 call mpas_log_write(msg)
401 write(msg,*) 'Initial velocity:', uvelocityx(icell), uvelocityy(icell)
402 call mpas_log_write(msg)
403 write(msg,*) 'Reconstructed velocity:', ureconstructx(ilevel,icell), ureconstructy(ilevel,icell)
404 call mpas_log_write('Reconstruction coefficients:')
405 call mpas_log_write(' ')
406 do iedgeoncell = 1, nedgesoncell(icell)
407 write(msg,*) iedgeoncell, coeffsreconstruct(:,iedgeoncell,icell)
408 call mpas_log_write(msg)
409 enddo
410 call mpas_log_write(' ')
411 call mpas_log_write('iEdgeOnCell, cellsOnEdge, normalVelocity:')
412 do iedgeoncell = 1, nedgesoncell(icell)
413 iedge = edgesoncell(iedgeoncell,icell)
414 write(msg,*) iedgeoncell, cellsonedge(:,iedge), normalvelocityinitial(ilevel,iedge)
415 call mpas_log_write(msg)
416 enddo
417 endif
418
419 ilevel = 1 ! Check at one level only, since the velocity is vertically uniform
420
421 if (abs(ureconstructx(ilevel,icell) - uvelocityx(icell)) > 1.e-8_rkind*speed .or. &
422 abs(ureconstructy(ilevel,icell) - uvelocityy(icell)) > 1.e-8_rkind*speed) then
423
424 xdiff = abs(ureconstructx(ilevel,icell) - uvelocityx(icell)) / speed
425 ydiff = abs(ureconstructy(ilevel,icell) - uvelocityy(icell)) / speed
426
427 call mpas_log_write(' ')
428 write(msg,*) 'WARNING: Reconstructed velocity not equal to uniform velocity, iCell, xDiff, yDiff=', &
429 icell, xdiff, ydiff
430 call mpas_log_write(msg)
431 write(msg,*) 'Prescribed velocity: ', uvelocityx(icell), uvelocityy(icell)
432 call mpas_log_write(msg)
433 write(msg,*) 'Reconstructed velocity:', ureconstructx(ilevel,icell), ureconstructy(ilevel,icell)
434 call mpas_log_write(msg)
435 call mpas_log_write('iEdgeOnCell, cellsOnEdge, normal velocity:')
436 do iedgeoncell = 1, nedgesoncell(icell)
437 iedge = edgesoncell(iedgeoncell,icell)
438 write(msg,*) iedgeoncell, cellsonedge(:,iedge), normalvelocityinitial(ilevel,iedge)
439 call mpas_log_write(msg)
440 enddo
441 err = 1
442 endif
443
444 enddo ! iCell
445
446 endif ! bug check
447
448
449 !--------------------------------
450 !TODO - Remove these options after testing the calving scheme
451
452 call mpas_pool_get_config(liconfigs, 'config_calving', config_calving)
453
454 if (radialmelting) then ! force the calving front to retreat
455
456 call mpas_log_write('Setting up radially symmetric melting')
457 write(msg,*) 'Melt rate at periphery (m/yr) =', maxmelt * scyr / 910.0_rkind
458 call mpas_log_write(msg)
459
460 ! Zero out the normal velocities, since we are testing ice retreat
461 normalvelocityinitial(:,:) = 0.0_rkind
462
463 ! Set the thickness and melt rate
464 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
465 call mpas_pool_get_array(geometrypool, 'thickness', thickness)
466 call mpas_pool_get_array(geometrypool, 'sfcMassBal', sfcmassbal)
467
468 do icell = 1, ncells
469 xdiff = xcell(icell) - xcenter
470 ydiff = ycell(icell) - ycenter
471 radius = sqrt(xdiff**2 + ydiff**2)
472 ! set thickness to taper away from the center
473 thickness(icell) = thickness(icell) * (1.0_rkind - radius/maxradius)
474 ! set melting to increase away from the center
475 sfcmassbal(icell) = maxmelt * (-radius/maxradius)
476 enddo
477
478! call mpas_log_write(' ')
479! call mpas_log_write('thickness (m):')
480! do iRow = nRows, 1, -1
481! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
482! write(stderrUnit,'(a3)',advance='no') ' '
483! endif
484! do i = nCellsPerRow/2 - 2, nCellsPerRow
485! iCell = (iRow-1)*nCellsPerRow + i
486! write(stderrUnit,'(f8.2)',advance='no') thickness(iCell)
487! enddo
488! write(stderrUnit,*) ' '
489! enddo
490!
491! write(stderrUnit,*) ' '
492! write(stderrUnit,*) 'melt rate (m/yr):'
493! do iRow = nRows, 1, -1
494! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
495! write(stderrUnit,'(a3)',advance='no') ' '
496! endif
497! do i = nCellsPerRow/2 - 2, nCellsPerRow
498! iCell = (iRow-1)*nCellsPerRow + i
499! write(stderrUnit,'(f8.2)',advance='no') -sfcMassBal(iCell)*scyr/910.0_RKIND
500! enddo
501! write(stderrUnit,*) ' '
502! enddo
503
504 endif ! radialMelting
505
506 if (trim(config_calving) == 'topographic_threshold') then
507
508 call mpas_log_write('Setting topography to drop off at periphery')
509
510 ! Set the bed topography to drop off near the periphery of the ice
511 ! so as to check that the calving topographic threshold option is working.
512 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
513 call mpas_pool_get_array(geometrypool, 'bedTopography', bedtopography)
514
515 do icell = 1, ncells
516 xdiff = xcell(icell) - xcenter
517 ydiff = ycell(icell) - ycenter
518 radius = sqrt(xdiff**2 + ydiff**2)
519 if (radius > 0.9_rkind*maxradius) then ! close to the edge
520 bedtopography(icell) = bedtopography(icell) * radius/(0.9_rkind*maxradius)
521 endif
522 enddo
523
524! write(stderrUnit,*) ' '
525! write(stderrUnit,*) 'bedTopography (m):'
526! do iRow = nRows, 1, -1
527! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
528! write(stderrUnit,'(a3)',advance='no') ' '
529! endif
530! do i = nCellsPerRow/2 - 2, nCellsPerRow
531! iCell = (iRow-1)*nCellsPerRow + i
532! write(stderrUnit,'(f8.2)',advance='no') -bedTopography(iCell)
533! enddo
534! write(stderrUnit,*) ' '
535! enddo
536
537 elseif (trim(config_calving) == 'floating') then
538
539 call mpas_log_write('Setting topography to be mostly grounded')
540 write(msg,*) 'Spike depth =', spiketopography
541 call mpas_log_write(msg)
542
543 ! Put in a large spike that grounds most of the ice, but leaves the peripheral ice floating,
544 ! so as to check that the calving no-float option is working.
545 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
546 call mpas_pool_get_array(geometrypool, 'bedTopography', bedtopography)
547
548 do icell = 1, ncells
549 xdiff = xcell(icell) - xcenter
550 ydiff = ycell(icell) - ycenter
551 radius = sqrt(xdiff**2 + ydiff**2)
552 if (radius < 0.9_rkind*maxradius) then ! inner part of ice shelf
553 bedtopography(icell) = spiketopography
554 endif
555 enddo
556
557! write(stderrUnit,*) ' '
558! write(stderrUnit,*) 'bed depth (m):'
559! do iRow = nRows, 1, -1
560! if (mod(iRow,2) == 0) then ! indent for even-numbered rows
561! write(stderrUnit,'(a3)',advance='no') ' '
562! endif
563! do i = nCellsPerRow/2 - 2, nCellsPerRow
564! iCell = (iRow-1)*nCellsPerRow + i
565! write(stderrUnit,'(f8.2)',advance='no') -bedTopography(iCell)
566! enddo
567! write(stderrUnit,*) ' '
568! enddo
569
570 endif ! config_calving
571 !--------------------------------
572
573 ! clean up
574 call mpas_allocate_scratch_field(uvelocityxfield, .true.)
575 call mpas_allocate_scratch_field(uvelocityyfield, .true.)
576
577 ! === error check
578 if (err > 0) then
579 call mpas_log_write("An error has occurred in li_velocity_uniform_init.", mpas_log_err)
580 endif
581
582 !--------------------------------------------------------------------
583 end subroutine li_velocity_simple_block_init
