51 subroutine get_julgmt(date_str,julyr,julday,gmt)
52!=================================================================================================================
53
54!input arguments:
55 character(len=*),intent(in):: date_str
56
57!output arguments:
58 integer,intent(out):: julyr
59 integer,intent(out):: julday
60
61 real(kind=rkind),intent(out) :: gmt
62
63!local variables:
64 integer:: ny , nm , nd , nh , ni , ns , nt
65 integer:: my1, my2, my3, monss
66 integer,dimension(12):: mmd
67 data mmd /31,28,31,30,31,30,31,31,30,31,30,31/
68
69!-----------------------------------------------------------------------------------------------------------------
70
71 call split_date_char(date_str,ny,nm,nd,nh,ni,ns,nt)
72
73 gmt = nh + float(ni)/60. + float(ns)/3600.
74 my1 = mod(ny,4)
75 my2 = mod(ny,100)
76 my3 = mod(ny,400)
77 if(my1.eq.0.and.my2.ne.0.or.my3.eq.0)mmd(2)=29
78 julday=nd
79 julyr=ny
80 do monss=1,nm-1
81 julday=julday+mmd(monss)
82 enddo
83
84 end subroutine get_julgmt

87 subroutine cal_mon_day(julday,julyr,jmonth,jday)
88!=================================================================================================================
89
90!input arguments:
91 integer,intent(in):: julday,julyr
92
93!output arguments:
94 integer,intent(out):: jmonth,jday
95
96!local variables:
97 logical:: leap,not_find_date
98 integer:: month (12),itmpday,itmpmon,i
99 data month/31,28,31,30,31,30,31,31,30,31,30,31/
100
101 not_find_date = .true.
102
103 itmpday = julday
104 itmpmon = 1
105 leap=.false.
106 if(mod(julyr,4).eq.0)then
107 month(2)=29
108 leap=.true.
109 endif
110
111 i = 1
112 do while (not_find_date)
113 if(itmpday.gt.month(i))then
114 itmpday=itmpday-month(i)
115 else
116 jday=itmpday
117 jmonth=i
118 not_find_date = .false.
119 endif
120 i = i+1
121 enddo
122
123 end subroutine cal_mon_day

147!=================================================================================================================
148 subroutine monthly_interp_to_date(npoints,date_str,field_in,field_out)
149!=================================================================================================================
150
151!input arguments:
152 character(len=StrKIND),intent(in):: date_str
153 integer,intent(in):: npoints
154 real(kind=rkind),intent(in) ,dimension(12,npoints):: field_in
155
156!output arguments:
157 real(kind=rkind),intent(out),dimension(npoints):: field_out
158
159!local variables:
160 character(len=2):: day15,mon
161
162 integer:: l,n
163 integer:: julyr,julday,int_month,month1,month2
164 integer:: target_julyr,target_julday,target_date
165 integer,dimension(0:13):: middle
166
167 real(kind=rkind):: gmt
168
169!-----------------------------------------------------------------------------------------------------------------
170!call mpas_log_write('')
171!call mpas_log_write('--- enter subroutine monthly_interp_to_date:')
172!call mpas_log_write('--- current_date = '//trim(date_str))
173
174 write(day15,fmt='(I2.2)') 15
175 do l = 1 , 12
176 write(mon,fmt='(I2.2)') l
177 call get_julgmt(date_str(1:4)//'-'//mon//'-'//day15//'_'//'00:00:00.0000' , &
178 julyr,julday,gmt)
179 middle(l) = julyr*1000 + julday
180 enddo
181
182 l = 0
183 middle(l) = middle( 1) - 31
184
185 l = 13
186 middle(l) = middle(12) + 31
187
188 call get_julgmt(date_str,target_julyr,target_julday,gmt)
189 target_date = target_julyr * 1000 + target_julday
190!call mpas_log_write('--- target_julday = $i',intArgs=(/target_julday/))
191!call mpas_log_write('--- target_date = $i',intArgs=(/target_date/))
192
193 find_month : do l = 0 , 12
194 if((middle(l) .lt. target_date) .and. (middle(l+1) .ge. target_date)) then
195 do n = 1, npoints
196 int_month = l
197 if((int_month .eq. 0) .or. (int_month .eq. 12)) then
198 month1 = 12
199 month2 = 1
200 else
201 month1 = int_month
202 month2 = month1 + 1
203 endif
204 if(n == 1) then
205! write(0,*) '--- month1 =',month1
206! write(0,*) '--- month2 =',month2
207 endif
208 field_out(n) = ( field_in(month2,n) * (target_date - middle(l)) &
209 + field_in(month1,n) * (middle(l+1) - target_date)) &
210 / (middle(l+1) - middle(l))
211! if(field_out(n) .ne. 8.) write(0,201) n,field_in(month2,n),field_in(month2,n), &
212! field_out(n)
213 enddo
214 exit find_month
215 endif
216 enddo find_month
217
218
219 end subroutine monthly_interp_to_date

222 subroutine monthly_min_max(npoints,field_in,field_min,field_max)
223!=================================================================================================================
224
225!input arguments:
226 integer,intent(in):: npoints
227 real(kind=rkind),intent(in) ,dimension(12,npoints):: field_in
228
229!output arguments:
230 real(kind=rkind),intent(out),dimension(npoints):: field_min,field_max
231
232!local variables:
233 integer:: n,nn
234 real(kind=rkind):: minner,maxxer
235
236!-----------------------------------------------------------------------------------------------------------------
237
238 do n = 1, npoints
239 minner = field_in(1,n)
240 maxxer = field_in(1,n)
241
242 do nn = 2, 12
243 if(field_in(nn,n) .lt. minner) then
244 minner = field_in(nn,n)
245 endif
246 if(field_in(nn,n) .gt. maxxer) then
247 maxxer = field_in(nn,n)
248 endif
249 field_min(n) = minner
250 field_max(n) = maxxer
251 enddo
252
253 enddo
254
255 end subroutine monthly_min_max
