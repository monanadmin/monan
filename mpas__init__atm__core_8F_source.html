14 function init_atm_core_init(domain, startTimeStamp) result(ierr)
15
16 use mpas_derived_types
17 use mpas_stream_manager
18 use mpas_io_streams, only : mpas_stream_nearest
19 use init_atm_cases

21 implicit none

23 type (domain_type), intent(inout) :: domain
24 character(len=*), intent(out) :: starttimestamp
25
26 type (block_type), pointer :: block
27 type (mpas_pool_type), pointer :: state, mesh
28 character (len=StrKIND), pointer :: xtime
29 character (len=StrKIND), pointer :: initial_time
30 character (len=StrKIND), pointer :: config_start_time
31 real (kind=rkind), pointer :: sphere_radius
32 integer :: ierr
33
34
35 ierr = 0
36
37 block => domain % blocklist
38 do while (associated(block))
39 call mpas_pool_get_subpool(block % structs, 'state', state)
40 call mpas_pool_get_subpool(block % structs, 'mesh', mesh)
41 call mpas_pool_get_array(state, 'xtime', xtime)
42 call mpas_pool_get_array(state, 'initial_time', initial_time)
43 call mpas_pool_get_config(mesh, 'sphere_radius', sphere_radius)
44 call mpas_pool_get_config(block % configs, 'config_start_time', config_start_time)
45
46 starttimestamp = config_start_time
47 xtime = config_start_time
48 initial_time = config_start_time
49 domain % sphere_radius = a ! Appears in output files
50 sphere_radius = a ! Used in setting up test cases
51
52 block => block % next
53 end do
54
55 call mpas_stream_mgr_add_att(domain % streamManager, 'sphere_radius', domain % sphere_radius, streamid='output', ierr=ierr)
56 call mpas_stream_mgr_add_att(domain % streamManager, 'sphere_radius', domain % sphere_radius, streamid='surface', ierr=ierr)
57
58 !
59 ! We don't actually expect the time in the (most likely 'static') file to
60 ! match the time in the namelist, so just read whatever time we find in
61 ! the input file.
62 !
63 call mpas_stream_mgr_read(domain % streamManager, whence=mpas_stream_nearest, ierr=ierr)
64 call mpas_stream_mgr_reset_alarms(domain % streamManager, direction=mpas_stream_input, ierr=ierr)
65
66 end function init_atm_core_init
