61 function ocn_analysis_mode_init(domain, startTimeStamp) result(ierr)!{{{
62
63 type (domain_type), intent(inout) :: domain
64 character(len=*), intent(out) :: starttimestamp
65 integer :: ierr
66
67 type (block_type), pointer :: block
68 type (mpas_pool_type), pointer :: diagnosticspool
69
70 integer :: err_tmp
71
72 type (mpas_timeinterval_type) :: timestep
73
74 ! remove dt later
75 real (kind=rkind) :: dt
76 character (len=StrKIND), pointer :: xtime, simulationstarttime
77 real (kind=rkind), pointer :: dayssincestartofsim
78 type (mpas_time_type) :: xtime_timetype, simulationstarttime_timetype
79 type (mpas_time_type) :: starttime
80
81 ierr = 0
82
83 !
84 ! Set startTimeStamp based on the start time of the simulation clock
85 !
86 starttime = mpas_get_clock_time(domain % clock, mpas_start_time, err_tmp)
87 call mpas_get_time(starttime, datetimestring=starttimestamp)
88 ierr = ior(ierr, err_tmp)
89
90 ! Setup ocean config pool
91 call ocn_constants_init(domain % configs, domain % packages)
92
93 !
94 ! Read input data for model
95 !
96 call mpas_timer_start('io_read')
97 call mpas_stream_mgr_read(domain % streamManager, streamid='mesh', whence=mpas_stream_nearest, ierr=err_tmp)
98 call mpas_stream_mgr_read(domain % streamManager, streamid='input', ierr=err_tmp)
99 call mpas_timer_stop('io_read')
100 call mpas_timer_start('io_reset_alarms')
101 call mpas_stream_mgr_reset_alarms(domain % streamManager, streamid='input', ierr=err_tmp)
102 call mpas_timer_stop('io_reset_alarms')
103
104 ! Initialize submodules before initializing blocks.
105 call ocn_equation_of_state_init(err_tmp)
106 ierr = ior(ierr, err_tmp)
107
108 call ocn_analysis_init(domain, err_tmp)
109 ierr = ior(ierr, err_tmp)
110
111 call ocn_init_routines_vert_coord(domain)
112
113 call ocn_init_routines_compute_max_level(domain)
114
115 timestep = mpas_get_clock_timestep(domain % clock, ierr=err_tmp)
116 call mpas_get_timeinterval(timestep, dt=dt)
117
118 block => domain % blocklist
119 do while (associated(block))
120 call mpas_pool_get_subpool(block % structs, 'diagnostics', diagnosticspool)
121 call mpas_pool_get_array(diagnosticspool, 'xtime', xtime)
122
123 call ocn_init_routines_block(block, dt, ierr)
124 if(ierr.eq.1) then
125 call mpas_log_write('An error was encountered in ocn_init_routines_block', mpas_log_crit)
126 endif
127
128 xtime = starttimestamp
129
130 ! Set simulationStartTime only if that variable is not read from the restart file.
131 call mpas_pool_get_array(diagnosticspool, 'simulationStartTime', simulationstarttime)
132 if (trim(simulationstarttime)=="no_date_available") then
133 simulationstarttime = starttimestamp
134 end if
135
136 ! compute time since start of simulation, in days
137 call mpas_pool_get_array(diagnosticspool, 'daysSinceStartOfSim',dayssincestartofsim)
138 call mpas_set_time(xtime_timetype, datetimestring=xtime)
139 call mpas_set_time(simulationstarttime_timetype, datetimestring=simulationstarttime)
140 call mpas_get_timeinterval(xtime_timetype - simulationstarttime_timetype,dt=dayssincestartofsim)
141 dayssincestartofsim = dayssincestartofsim*days_per_second
142
143 block => block % next
144 end do
145
146 end function ocn_analysis_mode_init!}}}

169 type (mpas_timeinterval_type) :: runduration, timestep, alarmtimestep
170 character(len=StrKIND) :: restarttimestamp
171 integer :: err_tmp
172 character (len=StrKIND), pointer :: config_start_time, config_stop_time
173 character (len=StrKIND), pointer :: config_run_duration
174 character (len=StrKIND), pointer :: config_dt, config_restart_timestamp_name
175
176 ierr = 0
177
178 call mpas_pool_get_config(configs, 'config_dt', config_dt)
179 call mpas_pool_get_config(configs, 'config_start_time', config_start_time)
180 call mpas_pool_get_config(configs, 'config_stop_time', config_stop_time)
181 call mpas_pool_get_config(configs, 'config_run_duration', config_run_duration)
182 call mpas_pool_get_config(configs, 'config_restart_timestamp_name', config_restart_timestamp_name)
183
184 if ( trim(config_start_time) == "file" ) then
185 open(22,file=config_restart_timestamp_name,form='formatted',status='old')
186 read(22,*) restarttimestamp
187 close(22)
188 call mpas_set_time(curr_time=starttime, datetimestring=restarttimestamp, ierr=err_tmp)
189 else
190 call mpas_set_time(curr_time=starttime, datetimestring=config_start_time, ierr=err_tmp)
191 end if
192
193 call mpas_set_timeinterval(timestep, timestring="0000_00:00:00", ierr=err_tmp)
194 call mpas_set_timeinterval(runduration, timestring="0000_00:00:00", ierr=err_tmp)
195 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, runduration=runduration, ierr=err_tmp)
196
197 end function ocn_analysis_mode_setup_clock!}}}

211 function ocn_analysis_mode_run(domain) result(ierr)!{{{
212
213 type (domain_type), intent(inout) :: domain
214
215 integer :: itimestep
216 real (kind=rkind) :: dt
217 type (block_type), pointer :: block_ptr
218
219 type (mpas_time_type) :: currtime
220 character(len=StrKIND) :: timestamp
221 integer :: err, ierr
222
223 type (mpas_pool_type), pointer :: statepool
224 type (mpas_pool_type), pointer :: tracerspool
225 type (mpas_pool_type), pointer :: forcingpool
226 type (mpas_pool_type), pointer :: meshpool
227 type (mpas_pool_type), pointer :: diagnosticspool
228 type (mpas_pool_type), pointer :: scratchpool
229
230 type (mpas_timeinterval_type) :: timestep
231 character (len=StrKIND), pointer :: config_dt
232 logical, pointer :: config_write_output_on_startup
233
234 ierr = 0
235
236 call mpas_pool_get_config(ocnconfigs, 'config_dt', config_dt)
237 call mpas_pool_get_config(ocnconfigs, 'config_write_output_on_startup', config_write_output_on_startup)
238
239 timestep = mpas_get_clock_timestep(domain % clock, ierr=ierr)
240 call mpas_get_timeinterval(timestep, dt=dt)
241
242 currtime = mpas_get_clock_time(domain % clock, mpas_now, ierr)
243 call mpas_get_time(curr_time=currtime, datetimestring=timestamp, ierr=ierr)
244 call mpas_log_write('Initial time ' // trim(timestamp))
245
246 ! fill in diagnostics variables
247 block_ptr => domain % blocklist
248 do while(associated(block_ptr))
249 call mpas_pool_get_subpool(block_ptr % structs, 'state', statepool)
250 call mpas_pool_get_subpool(statepool, 'tracers', tracerspool)
251 call mpas_pool_get_subpool(block_ptr % structs, 'forcing', forcingpool)
252 call mpas_pool_get_subpool(block_ptr % structs, 'mesh', meshpool)
253 call mpas_pool_get_subpool(block_ptr % structs, 'diagnostics', diagnosticspool)
254 call mpas_pool_get_subpool(block_ptr % structs, 'scratch', scratchpool)
255
256 call ocn_diagnostic_solve(dt, statepool, forcingpool, meshpool, diagnosticspool, scratchpool, tracerspool, 1)
257 block_ptr => block_ptr % next
258 end do
259
260 if (config_write_output_on_startup) then
261 call ocn_analysis_compute_startup(domain, err)
262 call mpas_timer_start('io_write')
263 call mpas_stream_mgr_write(domain % streamManager, ierr=ierr)
264 call mpas_timer_stop('io_write')
265 endif
266
267 end function ocn_analysis_mode_run!}}}
