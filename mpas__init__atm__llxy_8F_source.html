8MODULE init_atm_llxy
9
10! Module that defines constants, data structures, and
11! subroutines used to convert grid indices to lat/lon
12! and vice versa.
13!
14! SUPPORTED PROJECTIONS
15! ---------------------
16! Cylindrical Lat/Lon (code = PROJ_LATLON)
17! Mercator (code = PROJ_MERC)
18! Lambert Conformal (code = PROJ_LC)
19! Gaussian (code = PROJ_GAUSS)
20! Polar Stereographic (code = PROJ_PS)
21! Rotated Lat/Lon (code = PROJ_ROTLL)
22!
23! REMARKS
24! -------
25! The routines contained within were adapted from routines
26! obtained from NCEP's w3 library. The original NCEP routines were less
27! flexible (e.g., polar-stereo routines only supported truelat of 60N/60S)
28! than what we needed, so modifications based on equations in Hoke, Hayes, and
29! Renninger (AFGWC/TN/79-003) were added to improve the flexibility.
30! Additionally, coding was improved to F90 standards and the routines were
31! combined into this module.
32!
33! ASSUMPTIONS
34! -----------
35! Grid Definition:
36! For mercator, lambert conformal, and polar-stereographic projections,
37! the routines within assume the following:
38!
39! 1. Grid is dimensioned (i,j) where i is the East-West direction,
40! positive toward the east, and j is the north-south direction,
41! positive toward the north.
42! 2. Origin is at (1,1) and is located at the southwest corner,
43! regardless of hemispere.
44! 3. Grid spacing (dx) is always positive.
45! 4. Values of true latitudes must be positive for NH domains
46! and negative for SH domains.
47!
48! For the latlon and Gaussian projection, the grid origin may be at any
49! of the corners, and the deltalat and deltalon values can be signed to
50! account for this using the following convention:
51! Origin Location Deltalat Sign Deltalon Sign
52! --------------- ------------- -------------
53! SW Corner + +
54! NE Corner - -
55! NW Corner - +
56! SE Corner + -
57!
58! Data Definitions:
59! 1. Any arguments that are a latitude value are expressed in
60! degrees north with a valid range of -90 -> 90
61! 2. Any arguments that are a longitude value are expressed in
62! degrees east with a valid range of -180 -> 180.
63! 3. Distances are in meters and are always positive.
64! 4. The standard longitude (stdlon) is defined as the longitude
65! line which is parallel to the grid's y-axis (j-direction), along
66! which latitude increases (NOT the absolute value of latitude, but
67! the actual latitude, such that latitude increases continuously
68! from the south pole to the north pole) as j increases.
69! 5. One true latitude value is required for polar-stereographic and
70! mercator projections, and defines at which latitude the
71! grid spacing is true. For lambert conformal, two true latitude
72! values must be specified, but may be set equal to each other to
73! specify a tangent projection instead of a secant projection.
74!
75! USAGE
76! -----
77! To use the routines in this module, the calling routines must have the
78! following statement at the beginning of its declaration block:
79! USE map_utils
80!
81! The use of the module not only provides access to the necessary routines,
82! but also defines a structure of TYPE (proj_info) that can be used
83! to declare a variable of the same type to hold your map projection
84! information. It also defines some integer parameters that contain
85! the projection codes so one only has to use those variable names rather
86! than remembering the acutal code when using them. The basic steps are
87! as follows:
88!
89! 1. Ensure the "USE map_utils" is in your declarations.
90! 2. Declare the projection information structure as type(proj_info):
91! TYPE(proj_info) :: proj
92! 3. Populate your structure by calling the map_set routine:
93! CALL map_set(code,lat1,lon1,knowni,knownj,dx,stdlon,truelat1,truelat2,proj)
94! where:
95! code (input) = one of PROJ_LATLON, PROJ_MERC, PROJ_LC, PROJ_PS,
96! PROJ_GAUSS, or PROJ_ROTLL
97! lat1 (input) = Latitude of grid origin point (i,j)=(1,1)
98! (see assumptions!)
99! lon1 (input) = Longitude of grid origin
100! knowni (input) = origin point, x-location
101! knownj (input) = origin point, y-location
102! dx (input) = grid spacing in meters (ignored for LATLON projections)
103! stdlon (input) = Standard longitude for PROJ_PS and PROJ_LC,
104! deltalon (see assumptions) for PROJ_LATLON,
105! ignored for PROJ_MERC
106! truelat1 (input) = 1st true latitude for PROJ_PS, PROJ_LC, and
107! PROJ_MERC, deltalat (see assumptions) for PROJ_LATLON
108! truelat2 (input) = 2nd true latitude for PROJ_LC,
109! ignored for all others.
110! proj (output) = The structure of type (proj_info) that will be fully
111! populated after this call
112!
113! 4. Now that the proj structure is populated, you may call either
114! of the following routines:
115!
116! latlon_to_ij(proj, lat, lon, i, j)
117! ij_to_latlon(proj, i, j, lat, lon)
118!
119! It is incumbent upon the calling routine to determine whether or
120! not the values returned are within your domain's bounds. All values
121! of i, j, lat, and lon are REAL values.
122!
123!
124! REFERENCES
125! ----------
126! Hoke, Hayes, and Renninger, "Map Preojections and Grid Systems for
127! Meteorological Applications." AFGWC/TN-79/003(Rev), Air Weather
128! Service, 1985.
129!
130! NCAR MM5v3 Modeling System, REGRIDDER program, module_first_guess_map.F
131! NCEP routines w3fb06, w3fb07, w3fb08, w3fb09, w3fb11, w3fb12
132!

548 ENDIF

703 SUBROUTINE set_ps(proj)
704 ! Initializes a polar-stereographic map projection from the partially
705 ! filled proj structure. This routine computes the radius to the
706 ! southwest corner and computes the i/j location of the pole for use
707 ! in llij_ps and ijll_ps.
708 IMPLICIT NONE
709
710 ! Declare args
711 TYPE(proj_info), INTENT(INOUT) :: proj
712
713 ! Local vars
714 REAL (kind=rkind) :: ala1
715 REAL (kind=rkind) :: alo1
716 REAL (kind=rkind) :: reflon
717 REAL (kind=rkind) :: scale_top
718
719 ! Executable code
720 reflon = proj%stdlon + 90.
721
722 ! Compute numerator term of map scale factor
723 scale_top = 1. + proj%hemi * sin(proj%truelat1 * rad_per_deg)
724
725 ! Compute radius to lower-left (SW) corner
726 ala1 = proj%lat1 * rad_per_deg
727 proj%rsw = proj%rebydx*cos(ala1)*scale_top/(1.+proj%hemi*sin(ala1))
728
729 ! Find the pole point
730 alo1 = (proj%lon1 - reflon) * rad_per_deg
731 proj%polei = proj%knowni - proj%rsw * cos(alo1)
732 proj%polej = proj%knownj - proj%hemi * proj%rsw * sin(alo1)
733
734 RETURN
735
736 END SUBROUTINE set_ps

739 SUBROUTINE llij_ps(lat,lon,proj,i,j)
740 ! Given latitude (-90 to 90), longitude (-180 to 180), and the
741 ! standard polar-stereographic projection information via the
742 ! public proj structure, this routine returns the i/j indices which
743 ! if within the domain range from 1->nx and 1->ny, respectively.
744
745 IMPLICIT NONE
746
747 ! Delcare input arguments
748 REAL (kind=rkind), INTENT(IN) :: lat
749 REAL (kind=rkind), INTENT(IN) :: lon
750 TYPE(proj_info),INTENT(IN) :: proj
751
752 ! Declare output arguments
753 REAL (kind=rkind), INTENT(OUT) :: i !(x-index)
754 REAL (kind=rkind), INTENT(OUT) :: j !(y-index)
755
756 ! Declare local variables
757
758 REAL (kind=rkind) :: reflon
759 REAL (kind=rkind) :: scale_top
760 REAL (kind=rkind) :: ala
761 REAL (kind=rkind) :: alo
762 REAL (kind=rkind) :: rm
763
764 ! BEGIN CODE
765
766 reflon = proj%stdlon + 90.
767
768 ! Compute numerator term of map scale factor
769
770 scale_top = 1. + proj%hemi * sin(proj%truelat1 * rad_per_deg)
771
772 ! Find radius to desired point
773 ala = lat * rad_per_deg
774 rm = proj%rebydx * cos(ala) * scale_top/(1. + proj%hemi *sin(ala))
775 alo = (lon - reflon) * rad_per_deg
776 i = proj%polei + rm * cos(alo)
777 j = proj%polej + proj%hemi * rm * sin(alo)
778
779 RETURN
780
781 END SUBROUTINE llij_ps

784 SUBROUTINE ijll_ps(i, j, proj, lat, lon)
785
786 ! This is the inverse subroutine of llij_ps. It returns the
787 ! latitude and longitude of an i/j point given the projection info
788 ! structure.
789
790 IMPLICIT NONE
791
792 ! Declare input arguments
793 REAL (kind=rkind), INTENT(IN) :: i ! Column
794 REAL (kind=rkind), INTENT(IN) :: j ! Row
795 TYPE (proj_info), INTENT(IN) :: proj
796
797 ! Declare output arguments
798 REAL (kind=rkind), INTENT(OUT) :: lat ! -90 -> 90 north
799 REAL (kind=rkind), INTENT(OUT) :: lon ! -180 -> 180 East
800
801 ! Local variables
802 REAL (kind=rkind) :: reflon
803 REAL (kind=rkind) :: scale_top
804 REAL (kind=rkind) :: xx,yy
805 REAL (kind=rkind) :: gi2, r2
806 REAL (kind=rkind) :: arccos
807
808 ! Begin Code
809
810 ! Compute the reference longitude by rotating 90 degrees to the east
811 ! to find the longitude line parallel to the positive x-axis.
812 reflon = proj%stdlon + 90.
813
814 ! Compute numerator term of map scale factor
815 scale_top = 1. + proj%hemi * sin(proj%truelat1 * rad_per_deg)
816
817 ! Compute radius to point of interest
818 xx = i - proj%polei
819 yy = (j - proj%polej) * proj%hemi
820 r2 = xx**2 + yy**2
821
822 ! Now the magic code
823 IF (r2 .EQ. 0.) THEN
824 lat = proj%hemi * 90.
825 lon = reflon
826 ELSE
827 gi2 = (proj%rebydx * scale_top)**2.
828 lat = deg_per_rad * proj%hemi * asin((gi2-r2)/(gi2+r2))
829 arccos = acos(xx/sqrt(r2))
830 IF (yy .GT. 0) THEN
831 lon = reflon + deg_per_rad * arccos
832 ELSE
833 lon = reflon - deg_per_rad * arccos
834 ENDIF
835 ENDIF
836
837 ! Convert to a -180 -> 180 East convention
838 IF (lon .GT. 180.) lon = lon - 360.
839 IF (lon .LT. -180.) lon = lon + 360.
840
841 RETURN
842
843 END SUBROUTINE ijll_ps

846 SUBROUTINE set_ps_wgs84(proj)
847 ! Initializes a polar-stereographic map projection (WGS84 ellipsoid)
848 ! from the partially filled proj structure. This routine computes the
849 ! radius to the southwest corner and computes the i/j location of the
850 ! pole for use in llij_ps and ijll_ps.
851
852 IMPLICIT NONE
853
854 ! Arguments
855 TYPE(proj_info), INTENT(INOUT) :: proj
856
857 ! Local variables
858 REAL (kind=rkind) :: h, mc, tc, t, rho
859
860 h = proj%hemi
861
862 mc = cos(h*proj%truelat1*rad_per_deg)/sqrt(1.0-(e_wgs84*sin(h*proj%truelat1*rad_per_deg))**2.0)
863 tc = sqrt(((1.0-sin(h*proj%truelat1*rad_per_deg))/(1.0+sin(h*proj%truelat1*rad_per_deg)))* &
864 (((1.0+e_wgs84*sin(h*proj%truelat1*rad_per_deg))/(1.0-e_wgs84*sin(h*proj%truelat1*rad_per_deg)))**e_wgs84 ))
865

867 t = sqrt(((1.0-sin(h*proj%lat1*rad_per_deg))/(1.0+sin(h*proj%lat1*rad_per_deg)))* &
868 (((1.0+e_wgs84*sin(h*proj%lat1*rad_per_deg))/(1.0-e_wgs84*sin(h*proj%lat1*rad_per_deg)) )**e_wgs84 ) )
869 rho = h * (a_wgs84 / proj%dx) * mc * t / tc
870 proj%polei = rho * sin((h*proj%lon1 - h*proj%stdlon)*rad_per_deg)
871 proj%polej = -rho * cos((h*proj%lon1 - h*proj%stdlon)*rad_per_deg)

873 RETURN

878 SUBROUTINE llij_ps_wgs84(lat,lon,proj,i,j)
879 ! Given latitude (-90 to 90), longitude (-180 to 180), and the
880 ! standard polar-stereographic projection information via the
881 ! public proj structure, this routine returns the i/j indices which
882 ! if within the domain range from 1->nx and 1->ny, respectively.
883
884 IMPLICIT NONE
885
886 ! Arguments
887 REAL (kind=rkind), INTENT(IN) :: lat
888 REAL (kind=rkind), INTENT(IN) :: lon
889 REAL (kind=rkind), INTENT(OUT) :: i !(x-index)
890 REAL (kind=rkind), INTENT(OUT) :: j !(y-index)
891 TYPE(proj_info),INTENT(IN) :: proj
892
893 ! Local variables
894 REAL (kind=rkind) :: h, mc, tc, t, rho
895
896 h = proj%hemi
897
898 mc = cos(h*proj%truelat1*rad_per_deg)/sqrt(1.0-(e_wgs84*sin(h*proj%truelat1*rad_per_deg))**2.0)
899 tc = sqrt(((1.0-sin(h*proj%truelat1*rad_per_deg))/(1.0+sin(h*proj%truelat1*rad_per_deg)))* &
900 (((1.0+e_wgs84*sin(h*proj%truelat1*rad_per_deg))/(1.0-e_wgs84*sin(h*proj%truelat1*rad_per_deg)))**e_wgs84 ))
901
902 t = sqrt(((1.0-sin(h*lat*rad_per_deg))/(1.0+sin(h*lat*rad_per_deg))) * &
903 (((1.0+e_wgs84*sin(h*lat*rad_per_deg))/(1.0-e_wgs84*sin(h*lat*rad_per_deg)))**e_wgs84))
904
905 ! Find the x/y location of the requested lat/lon with respect to the pole of the projection
906 rho = (a_wgs84 / proj%dx) * mc * t / tc
907 i = h * rho * sin((h*lon - h*proj%stdlon)*rad_per_deg)
908 j = h *(-rho)* cos((h*lon - h*proj%stdlon)*rad_per_deg)
909
910 ! Get i/j relative to reference i/j
911 i = proj%knowni + (i - proj%polei)
912 j = proj%knownj + (j - proj%polej)
913
914 RETURN
915
916 END SUBROUTINE llij_ps_wgs84

919 SUBROUTINE ijll_ps_wgs84(i, j, proj, lat, lon)
920
921 ! This is the inverse subroutine of llij_ps. It returns the
922 ! latitude and longitude of an i/j point given the projection info
923 ! structure.
924
925 implicit none
926
927 ! Arguments
928 REAL (kind=rkind), INTENT(IN) :: i ! Column
929 REAL (kind=rkind), INTENT(IN) :: j ! Row
930 REAL (kind=rkind), INTENT(OUT) :: lat ! -90 -> 90 north
931 REAL (kind=rkind), INTENT(OUT) :: lon ! -180 -> 180 East
932 TYPE (proj_info), INTENT(IN) :: proj
933
934 ! Local variables
935 REAL (kind=rkind) :: h, mc, tc, t, rho, x, y
936 REAL (kind=rkind) :: chi, a, b, c, d
937
938 h = proj%hemi
939 x = (i - proj%knowni + proj%polei)
940 y = (j - proj%knownj + proj%polej)
941
942 mc = cos(h*proj%truelat1*rad_per_deg)/sqrt(1.0-(e_wgs84*sin(h*proj%truelat1*rad_per_deg))**2.0)
943 tc = sqrt(((1.0-sin(h*proj%truelat1*rad_per_deg))/(1.0+sin(h*proj%truelat1*rad_per_deg))) * &
944 (((1.0+e_wgs84*sin(h*proj%truelat1*rad_per_deg))/(1.0-e_wgs84*sin(h*proj%truelat1*rad_per_deg)))**e_wgs84 ))
945
946 rho = sqrt((x*proj%dx)**2.0 + (y*proj%dx)**2.0)
947 t = rho * tc / (a_wgs84 * mc)
948
949 lon = h*proj%stdlon + h*atan2(h*x,h*(-y))
950
951 chi = pi/2.0-2.0*atan(t)
952 a = 1./2.*e_wgs84**2. + 5./24.*e_wgs84**4. + 1./40.*e_wgs84**6. + 73./2016.*e_wgs84**8.
953 b = 7./24.*e_wgs84**4. + 29./120.*e_wgs84**6. + 54113./40320.*e_wgs84**8.
954 c = 7./30.*e_wgs84**6. + 81./280.*e_wgs84**8.
955 d = 4279./20160.*e_wgs84**8.
956
957 lat = chi + sin(2.*chi)*(a + cos(2.*chi)*(b + cos(2.*chi)*(c + d*cos(2.*chi))))
958 lat = h * lat
959
960 lat = lat*deg_per_rad
961 lon = lon*deg_per_rad
962
963 RETURN
964
965 END SUBROUTINE ijll_ps_wgs84

968 SUBROUTINE set_albers_nad83(proj)
969 ! Initializes an Albers equal area map projection (NAD83 ellipsoid)
970 ! from the partially filled proj structure. This routine computes the
971 ! radius to the southwest corner and computes the i/j location of the
972 ! pole for use in llij_albers_nad83 and ijll_albers_nad83.
973
974 IMPLICIT NONE
975
976 ! Arguments
977 TYPE(proj_info), INTENT(INOUT) :: proj
978
979 ! Local variables
980 REAL (kind=rkind) :: h, m1, m2, q1, q2, theta, q, sinphi
981
982 h = proj%hemi
983
984 m1 = cos(h*proj%truelat1*rad_per_deg)/sqrt(1.0-(e_nad83*sin(h*proj%truelat1*rad_per_deg))**2.0)
985 m2 = cos(h*proj%truelat2*rad_per_deg)/sqrt(1.0-(e_nad83*sin(h*proj%truelat2*rad_per_deg))**2.0)
986
987 sinphi = sin(proj%truelat1*rad_per_deg)
988 q1 = (1.0-e_nad83**2.0) * &
989 ((sinphi/(1.0-(e_nad83*sinphi)**2.0)) - 1.0/(2.0*e_nad83) * log((1.0-e_nad83*sinphi)/(1.0+e_nad83*sinphi)))
990
991 sinphi = sin(proj%truelat2*rad_per_deg)
992 q2 = (1.0-e_nad83**2.0) * &
993 ((sinphi/(1.0-(e_nad83*sinphi)**2.0)) - 1.0/(2.0*e_nad83) * log((1.0-e_nad83*sinphi)/(1.0+e_nad83*sinphi)))
994
995 if (proj%truelat1 == proj%truelat2) then
996 proj%nc = sin(proj%truelat1*rad_per_deg)
997 else
998 proj%nc = (m1**2.0 - m2**2.0) / (q2 - q1)
999 end if
1000
1001 proj%bigc = m1**2.0 + proj%nc*q1
1002
1003 ! Find the i/j location of reference lat/lon with respect to the pole of the projection
1004 sinphi = sin(proj%lat1*rad_per_deg)
1005 q = (1.0-e_nad83**2.0) * &
1006 ((sinphi/(1.0-(e_nad83*sinphi)**2.0)) - 1.0/(2.0*e_nad83) * log((1.0-e_nad83*sinphi)/(1.0+e_nad83*sinphi)))
1007
1008 proj%rho0 = h * (a_nad83 / proj%dx) * sqrt(proj%bigc - proj%nc * q) / proj%nc
1009 theta = proj%nc*(proj%lon1 - proj%stdlon)*rad_per_deg
1010
1011 proj%polei = proj%rho0 * sin(h*theta)
1012 proj%polej = proj%rho0 - proj%rho0 * cos(h*theta)
1013
1014 RETURN
1015
1016 END SUBROUTINE set_albers_nad83

1019 SUBROUTINE llij_albers_nad83(lat,lon,proj,i,j)

1021 ! standard projection information via the
1022 ! public proj structure, this routine returns the i/j indices which
1023 ! if within the domain range from 1->nx and 1->ny, respectively.
1024
1025 IMPLICIT NONE

1181 SUBROUTINE ijll_lc( i, j, proj, lat, lon)
1182

1184 ! geographical latitude and longitude for a Lambert Conformal projection.

1187 ! 25 Jul 01: Corrected by B. Shaw, NOAA/FSL
1188 !

1192 REAL (kind=rkind), INTENT(IN) :: i ! Cartesian X coordinate

1194 TYPE(proj_info),INTENT(IN) :: proj ! Projection info structure

1198 REAL (kind=rkind), INTENT(OUT) :: lon ! Longitude (-180->180 E)
1199

1202 REAL (kind=rkind) :: jnew

1206 REAL (kind=rkind) :: xx
1207 REAL (kind=rkind) :: yy

1210
1211 chi1 = (90. - proj%hemi*proj%truelat1)*rad_per_deg
1212 chi2 = (90. - proj%hemi*proj%truelat2)*rad_per_deg
1213
1214 ! See if we are in the southern hemispere and flip the indices
1215 ! if we are.
1216 inew = proj%hemi * i
1217 jnew = proj%hemi * j
1218
1219 ! Compute radius**2 to i/j location
1220 xx = inew - proj%polei
1221 yy = proj%polej - jnew
1222 r2 = (xx*xx + yy*yy)
1223 r = sqrt(r2)/proj%rebydx
1224

1226 IF (r2 .EQ. 0.) THEN
1227 lat = proj%hemi * 90.
1228 lon = proj%stdlon
1229 ELSE
1230

1232 lon = proj%stdlon + deg_per_rad * atan2(proj%hemi*xx,yy)/proj%cone
1233# if ( defined (G95) && ( DA_CORE == 1 ) )
1234 lon = dmod(lon+360., 360.0_rkind)
1235# else
1236 lon = mod(lon+360., 360.0_rkind)
1237# endif
1238
1239 ! Latitude. Latitude determined by solving an equation adapted
1240 ! from:
1241 ! Maling, D.H., 1973: Coordinate Systems and Map Projections
1242 ! Equations #20 in Appendix I.
1243
1244 IF (chi1 .EQ. chi2) THEN
1245 chi = 2.0*atan( ( r/tan(chi1) )**(1./proj%cone) * tan(chi1*0.5) )
1246 ELSE
1247 chi = 2.0*atan( (r*proj%cone/sin(chi1))**(1./proj%cone) * tan(chi1*0.5))
1248 ENDIF
1249 lat = (90.0-chi*deg_per_rad)*proj%hemi
1250
1251 ENDIF
1252
1253 IF (lon .GT. +180.) lon = lon - 360.
1254 IF (lon .LT. -180.) lon = lon + 360.
1255

1978 SUBROUTINE lggaus( nlat, cosc, gwt, sinc, colat, wos2 )
1979
1980 IMPLICIT NONE
1981
1982 ! LGGAUS finds the Gaussian latitudes by finding the roots of the
1983 ! ordinary Legendre polynomial of degree NLAT using Newton's
1984 ! iteration method.
1985
1986 ! On entry:
1987 integer nlat ! the number of latitudes (degree of the polynomial)
1988
1989 ! On exit: for each Gaussian latitude
1990 ! COSC - cos(colatitude) or sin(latitude)
1991 ! GWT - the Gaussian weights
1992 ! SINC - sin(colatitude) or cos(latitude)
1993 ! COLAT - the colatitudes in radians
1994 ! WOS2 - Gaussian weight over sin**2(colatitude)

1997 REAL (kind=high) , PARAMETER :: pi = 3.141592653589793

1999 ! Convergence criterion for iteration of cos latitude
2000

2016 ! Constants for determining the derivative of the polynomial
2017 fi = nlat
2018 fi1 = fi+1.0
2019 a = fi*fi1 / sqrt(4.0*fi1*fi1-1.0)
2020 b = fi1*fi / sqrt(4.0*fi*fi-1.0)
2021
2022 ! Loop over latitudes, iterating the search for each root
2023
2024 DO i=1,nzero
2025 j=0
2026

2029
