95 subroutine ocn_constants_init(configPool, packagePool)!{{{
96 type (mpas_pool_type), pointer :: configpool
97 type (mpas_pool_type), pointer :: packagepool
98 integer :: n
99
100 real (kind=rkind), pointer :: config_density0
101
102 ocnconfigs => configpool
103 ocnpackages => packagepool
104
105 call mpas_pool_get_config(configpool, 'config_density0', config_density0)
106
107 !-----------------------------------------------------------------------
108 !
109 ! physical constants
110 ! some of these constants may be over-ridden by CSM-defined
111 ! constants if the CSM shared constants are available
112 !
113 !-----------------------------------------------------------------------
114
115 t0_kelvin = 273.16_rkind ! zero point for Celsius
116 rho_air = 1.2_rkind ! ambient air density (kg/m^3)
117 rho_sw = config_density0 ! density of salt water (kg/m^3)
118 rho_fw = 1.0e3_rkind ! avg. water density (kg/m^3)
119 rho_ice = 0.917e3_rkind ! density of ice (kg/m^3)
120 cp_sw = 3.996e3_rkind ! specific heat salt water
121 cp_air = 1005.0_rkind ! heat capacity of air (J/kg/K)
122 sound = 1.5e2_rkind ! speed of sound (m/s)
123 vonkar = 0.4_rkind ! von Karman constant
124 emissivity = 1.0_rkind !
125 stefan_boltzmann = 567.0e-10_rkind ! W/m^2/K^4
126 latent_heat_vapor_mks = 2.5e6_rkind ! lat heat of vaporization (J/kg)
127 latent_heat_fusion = 3.34e6_rkind ! lat heat of fusion (erg/kg)
128 latent_heat_fusion_mks = 3.337e5_rkind ! lat heat of fusion (J/kg)
129 sea_ice_salinity = 4.0_rkind ! (psu)
130 ocn_ref_salinity = 34.7_rkind ! (psu)
131
132
133 !-----------------------------------------------------------------------
134 !
135 ! conversion factors
136 !
137 !-----------------------------------------------------------------------
138
139 days_per_second = 1._rkind/86400._rkind ! days per second
140 salt_to_ppt = 1000._rkind ! salt (kg/kg) to ppt
141 ppt_to_salt = 1.e-3_rkind ! salt ppt to kg/kg
142 mass_to_sv = 1.0e-12_rkind ! mass flux to Sverdrups
143 heat_to_pw = 4.186e-15_rkind ! heat flux to Petawatts
144 salt_to_svppt = 1.0e-9_rkind ! salt flux to Sv*ppt
145 salt_to_mmday = 3.1536e+5_rkind ! salt to water (mm/day)
146
147 !-----------------------------------------------------------------------
148 !
149 ! physical constants -- CCSM override
150 !
151 !-----------------------------------------------------------------------
152
153#ifdef MPAS_ESM_SHR_CONST
154 t0_kelvin = shr_const_tkfrz ! zero point for Celsius
155 cp_sw = shr_const_cpsw ! erg/g/K
156 cp_air = shr_const_cpdair ! J/kg/K
157 rho_air = shr_const_rhodair ! kg/m^3
158 rho_sw = shr_const_rhosw ! kg/m^3
159 rho_fw = shr_const_rhofw ! kg/m^3
160 rho_ice = shr_const_rhoice ! kg/m^3
161 vonkar = shr_const_karman
162 stefan_boltzmann = shr_const_stebol ! W/m^2/K^4
163 latent_heat_vapor_mks = shr_const_latvap ! J/kg
164 latent_heat_fusion_mks = shr_const_latice ! J/kg
165 ocn_ref_salinity = shr_const_ocn_ref_sal ! psu
166 sea_ice_salinity = shr_const_ice_ref_sal ! psu
167#endif
168
169!#ifdef ZERO_SEA_ICE_REF_SAL
170! sea_ice_salinity = c0
171!#endif
172
173 !-----------------------------------------------------------------------
174 !
175 ! convert heat, solar flux (W/m^2) to temperature flux (C*cm/s):
176 ! --------------------------------------------------------------
177 ! heat_flux in (W/m^2) = (J/s/m^2) = 1(kg/s^3)
178 ! density of seawater rho_sw in (kg/m^3)
179 ! specific heat of seawater cp_sw in (erg/kg/C) = (m^2/s^2/C)
180 !
181 ! temp_flux = heat_flux / (rho_sw*cp_sw)
182 ! temp_flux (C*cm/s) = heat_flux (W/m^2)
183 ! * 1 (kg/s^3)/(W/m^2)
184 ! / [(rho_sw*cp_sw) (kg/m/s^2/C)]
185 !
186 ! = heat_flux (W/m^2)
187 ! * hflux_factor (C*m/s)/(W/m^2)
188 !
189 ! ==> hflux_factor = 1/(rho_sw*cp_sw)
190 !
191 !-----------------------------------------------------------------------
192
193 hflux_factor = 1.0_rkind/(rho_sw*cp_sw)
194
195 !-----------------------------------------------------------------------
196 !
197 ! convert fresh water flux (kg/m^2/s) to virtual salt flux (msu*cm/s):
198 ! --------------------------------------------------------------------
199 ! ocean reference salinity in (o/oo=psu)
200 ! density of freshwater rho_fw = 1.0e3 (kg/m^3)
201 ! h2o_flux in (kg/m^2/s) = 1.0e2 (kg/m^2/s)
202 !
203 ! salt_flux = - h2o_flux * ocn_ref_salinity / rho_fw
204 ! salt_flux (msu*cm/s) = - h2o_flux (kg/m^2/s)
205 ! * ocn_ref_salinity (psu)
206 ! * 1.e-3 (msu/psu)
207 ! * 1.0e2 (kg/m^2/s)/(kg/m^2/s)
208 ! / 1.0e3 (kg/m^3)
209 ! = - h2o_flux (kg/m^2/s)
210 ! * ocn_ref_salinity (psu)
211 ! * fwflux_factor (m/s)(msu/psu)/(kg/m^2/s)
212 !
213 ! ==> fwflux_factor = 1.e-6
214 !
215 ! salt_flux(msu*cm/s) = h2oflux(kg/m^2/s) * salinity_factor
216 !
217 ! ==> salinity_factor = - ocn_ref_salinity(psu) * fwflux_factor
218 !
219 !-----------------------------------------------------------------------
220
221 fwflux_factor = 1.e-6_rkind
222 salinity_factor = -ocn_ref_salinity*fwflux_factor
223
224 !-----------------------------------------------------------------------
225 !
226 ! convert salt flux (kg/m^2/s) to salt flux (psu*m/s):
227 ! -----------------------------------------------------
228 ! density of freshwater rho_fw = 1.0e3 (kg/m^3)
229 ! salt_flux_kg in (kg/m^2/s) = 1.0e2 (kg/m^2/s)
230 !
231 ! salt_flux = - h2o_flux * ocn_ref_salinity / rho_fw
232 ! salt_flux (msu*cm/s) = salt_flux_kg (kg/m^2/s)
233 ! * 1.0e2 (kg/m^2/s)/(kg/m^2/s)
234 ! / 1.0e3 (kg/m^3)
235 ! = salt_flux_kg (kg/m^2/s)
236 ! * sflux_factor (psu*m/s)/(kg/m^2/s)
237 !
238 ! ==> sflux_factor = 1.0
239 !
240 !-----------------------------------------------------------------------
241
242 sflux_factor = 1.0_rkind
243
244 !-----------------------------------------------------------------------
245 !
246 ! convert fresh water mass flux (kg/m^2/s) to fresh water flux (m/s):
247 ! --------------------------------------------------------------------
248 ! density of freshwater rho_fw = 1.0e3 (kg/m^3)
249 ! h2o_flux in (kg/m^2/s) = 1.0e2 (kg/m^2/s)
250 !
251 ! fw_flux = h2o_flux / rho_fw
252 ! fw_flux (cm/s) = h2o_flux (kg/m^2/s)
253 ! * 1.0e2 (kg/m^2/s)
254 ! / 1.0e3 (kg/m^3)
255 ! = h2o_flux (kg/m^2/s)
256 ! * fwmass_to_fwflux (m/s)/(kg/m^2/s)
257 !
258 ! ==> fwmass_to_fwflux = 0.1
259 !
260 !-----------------------------------------------------------------------
261
262 fwmass_to_fwflux = 0.1_rkind
263
264 end subroutine ocn_constants_init!}}}
265
