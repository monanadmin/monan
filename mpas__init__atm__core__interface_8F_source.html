103 function init_atm_setup_packages(configs, packages, iocontext) result(ierr)
104
105 use mpas_derived_types, only : mpas_pool_type, mpas_io_context_type
106 use mpas_pool_routines, only : mpas_pool_get_config, mpas_pool_get_package
107
108 implicit none
109
110 type (mpas_pool_type), intent(inout) :: configs
111 type (mpas_pool_type), intent(inout) :: packages
112 type (mpas_io_context_type), intent(inout) :: iocontext
113 integer :: ierr
114
115 logical, pointer :: initial_conds, sfc_update, lbcs
116 logical, pointer :: gwd_stage_in, gwd_stage_out, vertical_stage_in, vertical_stage_out, met_stage_in, met_stage_out
117 logical, pointer :: config_native_gwd_static, config_static_interp, config_vertical_grid, config_met_interp
118 integer, pointer :: config_init_case
119
120
121 ierr = 0
122
123 call mpas_pool_get_config(configs, 'config_init_case', config_init_case)
124 call mpas_pool_get_config(configs, 'config_static_interp', config_static_interp)
125 call mpas_pool_get_config(configs, 'config_native_gwd_static', config_native_gwd_static)
126 call mpas_pool_get_config(configs, 'config_vertical_grid', config_vertical_grid)
127 call mpas_pool_get_config(configs, 'config_met_interp', config_met_interp)
128
129 nullify(initial_conds)
130 call mpas_pool_get_package(packages, 'initial_condsActive', initial_conds)
131
132 nullify(sfc_update)
133 call mpas_pool_get_package(packages, 'sfc_updateActive', sfc_update)
134
135 nullify(lbcs)
136 call mpas_pool_get_package(packages, 'lbcsActive', lbcs)
137
138 nullify(gwd_stage_in)
139 call mpas_pool_get_package(packages, 'gwd_stage_inActive', gwd_stage_in)
140
141 nullify(gwd_stage_out)
142 call mpas_pool_get_package(packages, 'gwd_stage_outActive', gwd_stage_out)
143
144 nullify(vertical_stage_in)
145 call mpas_pool_get_package(packages, 'vertical_stage_inActive', vertical_stage_in)
146
147 nullify(vertical_stage_out)
148 call mpas_pool_get_package(packages, 'vertical_stage_outActive', vertical_stage_out)
149
150 nullify(met_stage_in)
151 call mpas_pool_get_package(packages, 'met_stage_inActive', met_stage_in)
152
153 nullify(met_stage_out)
154 call mpas_pool_get_package(packages, 'met_stage_outActive', met_stage_out)
155
156 if (.not. associated(initial_conds) .or. &
157 .not. associated(sfc_update) .or. &
158 .not. associated(gwd_stage_in) .or. &
159 .not. associated(gwd_stage_out) .or. &
160 .not. associated(vertical_stage_in) .or. &
161 .not. associated(vertical_stage_out) .or. &
162 .not. associated(met_stage_in) .or. &
163 .not. associated(met_stage_out)) then
164 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
165 call mpas_log_write('* Error while setting up packages for init_atmosphere core.', messagetype=mpas_log_err)
166 call mpas_log_write('********************************************************************************', messagetype=mpas_log_err)
167 ierr = 1
168 return
169 end if
170
171 if (config_init_case == 8) then
172 initial_conds = .false.
173 sfc_update = .true.
174 else
175 initial_conds = .true.
176 sfc_update = .false.
177 end if
178
179 if (config_init_case == 9) then
180 lbcs = .true.
181 else
182 lbcs = .false.
183 end if
184
185 if (config_init_case == 7) then
186
187 !
188 ! The logic here is a little convoluted
189 ! For input, we want to read in fields from all earlier stages, except if those earlier stages are being run now
190 ! For output, we want to output the fields that were computed in a stage and all of those from earlier stages
191 !
192 gwd_stage_in = config_native_gwd_static .and. &
193 (.not. config_static_interp)
194 gwd_stage_out = config_native_gwd_static
195 vertical_stage_in = config_vertical_grid .and. &
196 (.not. config_native_gwd_static) .and. &
197 (.not. config_static_interp)
198 vertical_stage_out = config_vertical_grid
199 met_stage_in = config_met_interp .and. &
200 (.not. config_native_gwd_static) .and. &
201 (.not. config_static_interp) .and. &
202 (.not. config_vertical_grid)
203 met_stage_out = config_met_interp
204
205 else if (config_init_case == 8) then
206 gwd_stage_in = .false.
207 gwd_stage_out = .false.
208 vertical_stage_in = .true.
209 vertical_stage_out = .false.
210 met_stage_in = .false.
211 met_stage_out = .false.
212
213 !
214 ! When interpolating LBC fields, we need all inputs that would be needed for the interpolation
215 ! of ICs, so met_stage_in = .true.
216 !
217 else if (config_init_case == 9) then
218 gwd_stage_in = .false.
219 gwd_stage_out = .false.
220 vertical_stage_in = .false.
221 vertical_stage_out = .false.
222 met_stage_in = .true.
223 met_stage_out = .true.
224
225 initial_conds = .false. ! Also, turn off the initial_conds package to avoid writing the IC "output" stream
226
227 else
228 gwd_stage_in = .false.
229 gwd_stage_out = .false.
230 vertical_stage_in = .false.
231 vertical_stage_out = .false.
232 met_stage_in = .false.
233 met_stage_out = .true.
234 end if
235
236 end function init_atm_setup_packages
