254 subroutine ocn_init_gm_test_functions(diagnosticsPool, meshPool, scratchPool)!{{{
255
256 !-----------------------------------------------------------------
257 !
258 ! input variables
259 !
260 !-----------------------------------------------------------------
261
262 type (mpas_pool_type), intent(in) :: &
263 meshpool
264
265 !-----------------------------------------------------------------
266 !
267 ! input/output variables
268 !
269 !-----------------------------------------------------------------
270
271 type (mpas_pool_type), intent(inout) :: diagnosticspool
272 type (mpas_pool_type), intent(in) :: scratchpool
273
274 !-----------------------------------------------------------------
275 !
276 ! local variables
277 !
278 !-----------------------------------------------------------------
279
280 integer :: k, icell
281 integer, pointer :: ncells
282 integer, dimension(:), pointer :: maxlevelcell
283
284 real(kind=rkind) :: ztop, config_gm_analytic_temperature2, config_gm_analytic_temperature3, config_gm_analytic_ymax, &
285 config_gm_analytic_bottom_depth, l, r, c1, c2, zmax, zbot
286 real (kind=rkind), pointer :: config_gravwavespeed_trunc, config_standardgm_tracer_kappa, config_eos_linear_alpha
287
288 real(kind=rkind), dimension(:), pointer :: bottomdepth, refbottomdepthtopofcell, ycell, yedge
289 real(kind=rkind), dimension(:,:), pointer :: yrelativeslopesolution, ygmstreamfuncsolution, ygmbolusvelocitysolution, zmid
290
291 type (field2dreal), pointer :: yrelativeslopesolutionfield, ygmstreamfuncsolutionfield, ygmbolusvelocitysolutionfield
292
293 call mpas_pool_get_config(ocnconfigs, 'config_eos_linear_alpha', config_eos_linear_alpha)
294 call mpas_pool_get_config(ocnconfigs, 'config_gravWaveSpeed_trunc',config_gravwavespeed_trunc)
295 call mpas_pool_get_config(ocnconfigs, 'config_standardGM_tracer_kappa',config_standardgm_tracer_kappa)
296
297 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
298 call mpas_pool_get_array(meshpool, 'maxLevelCell', maxlevelcell)
299 call mpas_pool_get_array(meshpool, 'refBottomDepthTopOfCell',refbottomdepthtopofcell)
300 call mpas_pool_get_array(meshpool, 'yCell',ycell)
301 call mpas_pool_get_array(meshpool, 'yEdge',yedge)
302 call mpas_pool_get_array(meshpool, 'bottomDepth',bottomdepth)
303
304 call mpas_pool_get_array(diagnosticspool, 'zMid', zmid)
305
306 ! move solution variables to diagnostics pool to include them in output.
307 call mpas_pool_get_field(scratchpool, 'yRelativeSlopeSolution',yrelativeslopesolutionfield)
308 call mpas_pool_get_field(scratchpool, 'yGMStreamFuncSolution',ygmstreamfuncsolutionfield)
309 call mpas_pool_get_field(scratchpool, 'yGMBolusVelocitySolution',ygmbolusvelocitysolutionfield)
310
311 yrelativeslopesolution => yrelativeslopesolutionfield % array
312 ygmstreamfuncsolution => ygmstreamfuncsolutionfield % array
313 ygmbolusvelocitysolution => ygmbolusvelocitysolutionfield % array
314
315 ! These are flags that must match your initial conditions settings. See gm_analytic initial condition in mode_init.
316 config_gm_analytic_temperature2 = 10
317 config_gm_analytic_temperature3 = -10
318 config_gm_analytic_ymax = 500000
319 config_gm_analytic_bottom_depth = 1000
320
321 ! zMax is associated with linear temperature profile in z
322 zmax = -config_gm_analytic_bottom_depth
323 ! zBot is location we apply boundary conditions on the ODE for stream function.
324 zbot = zmax
325
326 l = config_gravwavespeed_trunc * sqrt(rho_sw * zmax / gravity / config_eos_linear_alpha / config_gm_analytic_temperature3)
327 r = - config_standardgm_tracer_kappa * config_gm_analytic_temperature2 * zmax / config_gm_analytic_temperature3 &
328 / config_gm_analytic_ymax
329 c1 = r*(1-exp(-zbot/l))/(exp(zbot/l) - exp(-zbot/l))
330 c2 = r-c1
331
332 !$omp do schedule(runtime) private(k, zTop)
333 do icell = 1, ncells
334
335 do k = 1, maxlevelcell(icell)
336 ! placed at mid-depth of cell center:
337 ygmbolusvelocitysolution(k,icell) = 1/l*(c1*exp(zmid(k,icell)/l) - c2*exp(-zmid(k,icell)/l) );
338 end do
339
340 do k = 1, maxlevelcell(icell)
341 ! placed at top interface, cell center.
342 ztop = - refbottomdepthtopofcell(k)
343 ygmstreamfuncsolution(k,icell) = c1*exp(ztop/l) + c2*exp(-ztop/l) - r;
344
345 end do
346
347 k = maxlevelcell(icell)+1
348 ! placed at top interface, cell center.
349 ztop = zbot
350 ygmstreamfuncsolution(k,icell) = c1*exp(ztop/l) + c2*exp(-ztop/l) - r;
351
352 end do
353 !$omp end do
354
355 end subroutine ocn_init_gm_test_functions!}}}
