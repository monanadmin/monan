51 subroutine soundings_setup(configs, all_pools, simulation_clock, dminfo)
52
53 use mpas_derived_types, only : mpas_pool_type, mpas_clock_type, dm_info
54 use mpas_pool_routines, only : mpas_pool_get_subpool, mpas_pool_get_dimension, mpas_pool_get_array, mpas_pool_get_config
55 use mpas_io_units, only : mpas_new_unit, mpas_release_unit
56 use mpas_timekeeping, only : mpas_timeinterval_type, mpas_time_type, mpas_set_timeinterval, &
57 mpas_get_clock_time, mpas_add_clock_alarm, mpas_now
58 use mpas_dmpar, only : io_node, mpas_dmpar_bcast_int, mpas_dmpar_bcast_logical, mpas_dmpar_bcast_char
59
60 implicit none
61
62 type (mpas_pool_type), pointer :: configs
63 type (mpas_pool_type), pointer :: all_pools
64 type (mpas_clock_type), pointer :: simulation_clock
65 type (dm_info), intent(in) :: dminfo
66
67 character(len=StrKIND), pointer :: soundinginterval
68
69 integer :: i, ierr
70 integer :: sndunit
71 real (kind=rkind) :: station_lat, station_lon
72 character (len=StrKIND) :: tempstr
73 character (len=StrKIND) :: station_name
74 logical :: exists
75 integer :: nearestcell
76 integer, pointer :: ncells, ncellssolve, maxedges
77 integer, dimension(:), pointer :: nedgesoncell
78 integer, dimension(:,:), pointer :: cellsoncell
79 real (kind=rkind), dimension(:), pointer :: latcell, loncell
80 type (mpas_timeinterval_type) :: intv
81 type (mpas_time_type) :: now
82
83
84 simulationclock => simulation_clock
85
86 call mpas_pool_get_subpool(all_pools, 'mesh', mesh)
87 call mpas_pool_get_subpool(all_pools, 'state', state)
88 call mpas_pool_get_subpool(all_pools, 'diag', diag)
89
90 call mpas_pool_get_config(configs, 'config_sounding_interval', soundinginterval)
91
92 if (trim(soundinginterval) == 'none') then
93 return
94 end if
95
96 if (dminfo % my_proc_id == io_node) then
97 inquire(file='sounding_locations.txt', exist=exists)
98 end if
99 call mpas_dmpar_bcast_logical(dminfo, exists)
100
101 if (.not. exists) then
102 call mpas_log_write('Sounding location file ''sounding_locations.txt'' not found.')
103 return
104 end if
105
106 call mpas_set_timeinterval(intv, timestring=trim(soundinginterval))
107 now = mpas_get_clock_time(simulationclock, mpas_now)
108 call mpas_add_clock_alarm(simulationclock, 'soundingAlarm', now, alarmtimeinterval=intv)
109
110 if (dminfo % my_proc_id == io_node) then
111 call mpas_new_unit(sndunit)
112
113 open(sndunit, file='sounding_locations.txt', form='formatted', status='old', iostat=ierr)
114 end if
115 call mpas_dmpar_bcast_int(dminfo, ierr)
116
117 if (ierr /= 0) then
118 call mpas_log_write('Error opening sounding location file ''sounding_locations.txt''')
119 if (dminfo % my_proc_id == io_node) then
120 call mpas_release_unit(sndunit)
121 end if
122 return
123 end if
124
125 if (dminfo % my_proc_id == io_node) then
126 do
127 read(sndunit, fmt=*, iostat=ierr) station_lat, station_lon, station_name
128 if (ierr == 0) then
129 nsoundings = nsoundings + 1
130 else
131 exit
132 end if
133 end do
134 end if
135 call mpas_dmpar_bcast_int(dminfo, nsoundings)
136
137 if (nsoundings == 0) then
138 if (dminfo % my_proc_id == io_node) then
139 close(sndunit)
140 call mpas_release_unit(sndunit)
141 end if
142 return
143 end if
144
145 call mpas_log_write('Read $i sounding locations from ''sounding_locations.txt''', intargs=(/nsoundings/))
146
147 if (dminfo % my_proc_id == io_node) then
148 rewind(sndunit)
149 end if
150
151 allocate(stationowned(nsoundings))
152 allocate(stationlats(nsoundings))
153 allocate(stationlons(nsoundings))
154 allocate(stationcells(nsoundings))
155 allocate(stationnames(nsoundings))
156
157 call mpas_pool_get_dimension(mesh, 'nCells', ncells)
158 call mpas_pool_get_dimension(mesh, 'nCellsSolve', ncellssolve)
159 call mpas_pool_get_dimension(mesh, 'maxEdges', maxedges)
160 call mpas_pool_get_array(mesh, 'nEdgesOnCell', nedgesoncell)
161 call mpas_pool_get_array(mesh, 'cellsOnCell', cellsoncell)
162 call mpas_pool_get_array(mesh, 'latCell', latcell)
163 call mpas_pool_get_array(mesh, 'lonCell', loncell)
164
165 nearestcell = ncells
166
167 do i=1,nsoundings
168 if (dminfo % my_proc_id == io_node) then
169 read(sndunit, fmt='(a)') tempstr
170 end if
171 call mpas_dmpar_bcast_char(dminfo, tempstr)
172 read(tempstr, fmt=*) stationlats(i), stationlons(i), stationnames(i)
173 nearestcell = nearest_cell((stationlats(i) * pi_const / 180.0_rkind), (stationlons(i) * pi_const / 180.0_rkind), &
174 nearestcell, ncells, maxedges, nedgesoncell, cellsoncell, latcell, loncell)
175 if (nearestcell <= ncellssolve) then
176 stationowned(i) = .true.
177 stationcells(i) = nearestcell
178 else
179 stationowned(i) = .false.
180 end if
181 end do
182
183 if (dminfo % my_proc_id == io_node) then
184 close(sndunit)
185
186 call mpas_release_unit(sndunit)
187 end if
188
189 end subroutine soundings_setup

193 ! routine soundings_compute
194 !
202 !
203 !-----------------------------------------------------------------------
204 subroutine soundings_compute()
205
206 use mpas_derived_types, only : mpas_pool_type
207 use mpas_pool_routines, only : mpas_pool_get_dimension, mpas_pool_get_array
208 use mpas_timekeeping, only : mpas_time_type, mpas_is_alarm_ringing, mpas_reset_clock_alarm, mpas_get_clock_time, &
209 mpas_get_time, mpas_now
210 use mpas_constants, only : rvord
211
212 implicit none
213
214 integer :: istn, k
215 integer, pointer :: nvertlevels, index_qv
216 real (kind=rkind), dimension(:,:), pointer :: pressure_base, pressure_p, ureconstructzonal, ureconstructmeridional, zgrid, &
217 theta_m, exner
218 real (kind=rkind), dimension(:,:,:), pointer :: scalars
219 real (kind=rkind) :: tmpc, tdpc, dir, spd, rh, log_rh, qvs, pres
220 type (mpas_time_type) :: now
221 character(len=StrKIND) :: nowstring
222 integer :: yyyy, mm, dd, h, m, s
223 character(len=StrKIND) :: fname
224 character(len=10) :: stid
225
226
227 if (mpas_is_alarm_ringing(simulationclock, 'soundingAlarm')) then
228
229 now = mpas_get_clock_time(simulationclock, mpas_now)
230 call mpas_get_time(now, yyyy=yyyy, mm=mm, dd=dd, h=h, m=m, s=s, datetimestring=nowstring)
231
232 call mpas_pool_get_dimension(mesh, 'nVertLevels', nvertlevels)
233 call mpas_pool_get_dimension(state, 'index_qv', index_qv)
234 call mpas_pool_get_array(mesh, 'zgrid', zgrid)
235 call mpas_pool_get_array(state, 'scalars', scalars, 1)
236 call mpas_pool_get_array(state, 'theta_m', theta_m, 1)
237 call mpas_pool_get_array(diag, 'pressure_base', pressure_base)
238 call mpas_pool_get_array(diag, 'pressure_p', pressure_p)
239 call mpas_pool_get_array(diag, 'exner', exner)
240 call mpas_pool_get_array(diag, 'uReconstructZonal', ureconstructzonal)
241 call mpas_pool_get_array(diag, 'uReconstructMeridional', ureconstructmeridional)
242
243! call mpas_log_write('--- Writing soundings at '//trim(nowString)//'---')
244
245 do istn=1,nsoundings
246 if (stationowned(istn)) then
247! call mpas_log_write('Writing sounding for station '//trim(stationNames(iStn)))
248
249 write(fname,'(a,i4.4,i2.2,i2.2,i2.2,i2.2,a)') trim(stationnames(istn))//'.', yyyy, mm, dd, h, m, '.snd'
250 open(97,file=trim(fname),form='formatted',status='replace')
251
252 write(stid,'(a)') trim(stationnames(istn))
253
254 write(97,'(a)') ' SNPARM = PRES;HGHT;TMPC;DWPC;DRCT;SPED;'
255 write(97,'(a)') ''
256 write(97,'(a,i2.2,i2.2,i2.2,a,i2.2,i2.2)') ' STID = '//stid//' STNM = 99999 TIME = ', mod(yyyy,100), mm, dd,'/', h, m
257 write(97,'(a,f6.2,a,f7.2,a)') ' SLAT = ', stationlats(istn), ' SLON = ', stationlons(istn), ' SELV = -999'
258 write(97,'(a)') ''
259 write(97,'(a)') ' PRES HGHT TMPC DWPC DRCT SPED'
260
261 do k=1,nvertlevels
262 tmpc = theta_m(k,stationcells(istn)) / (1.0_rkind + rvord * scalars(index_qv,k,stationcells(istn))) * exner(k,stationcells(istn))
263 pres = pressure_base(k,stationcells(istn)) + pressure_p(k,stationcells(istn))
264! if (tmpc >= 273.15_RKIND) then
265 qvs = rslf(pres, tmpc)
266! else
267! qvs = rsif(pres, tmpc)
268! end if
269 rh = max(1.0e-8,min(1.0,scalars(index_qv,k,stationcells(istn))/qvs))
270 log_rh = log(rh)
271 tmpc = tmpc - 273.15_rkind
272 pres = pres * 0.01
273 tdpc = 243.04*(log_rh+(17.625*tmpc/(243.04+tmpc))) / (17.625-log_rh-((17.625*tmpc)/(243.04+tmpc)))
274 spd = sqrt(ureconstructzonal(k,stationcells(istn))**2 + ureconstructmeridional(k,stationcells(istn))**2)
275 if (spd == 0.0) then
276 dir = 0.0
277 else
278 dir = acos(-ureconstructmeridional(k,stationcells(istn)) / spd)
279 if (ureconstructzonal(k,stationcells(istn)) > 0.0) then
280 dir = 2.0 * pi_const - dir
281 end if
282 dir = dir * 180.0_rkind / pi_const
283 end if
284 write(97,'(f10.2,f10.2,f9.2,f9.2,f9.2,f9.2)') &
285 pres, &
286 0.5 * (zgrid(k,stationcells(istn)) + zgrid(k+1,stationcells(istn))), & ! Avg to layer midpoint
287 tmpc, &
288 tdpc, &
289 dir, &
290 spd
291 end do
292
293 close(97)
294 end if
295 end do
296
297 call mpas_reset_clock_alarm(simulationclock, 'soundingAlarm')
298 else
299! call mpas_log_write('--- Not yet time to write soundings ---')
300 end if
301
302 end subroutine soundings_compute

333 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
334 ! Finds the MPAS grid cell nearest to (target_lat, target_lon)
335 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
336 integer function nearest_cell(target_lat, target_lon, start_cell, nCells, maxEdges, &
337 nEdgesOnCell, cellsOnCell, latCell, lonCell)
338
339 implicit none
340
341 real (kind=rkind), intent(in) :: target_lat, target_lon
342 integer, intent(in) :: start_cell
343 integer, intent(in) :: ncells, maxedges
344 integer, dimension(nCells), intent(in) :: nedgesoncell
345 integer, dimension(maxEdges,nCells), intent(in) :: cellsoncell
346 real (kind=rkind), dimension(nCells), intent(in) :: latcell, loncell
347
348 integer :: i
349 integer :: icell
350 integer :: current_cell
351 real (kind=rkind) :: current_distance, d
352 real (kind=rkind) :: nearest_distance
353
354 nearest_cell = start_cell
355 current_cell = -1
356
357 do while (nearest_cell /= current_cell)
358 current_cell = nearest_cell
359 current_distance = sphere_distance(latcell(current_cell), loncell(current_cell), target_lat, &
360 target_lon, 1.0_rkind)
361 nearest_cell = current_cell
362 nearest_distance = current_distance
363 do i = 1, nedgesoncell(current_cell)
364 icell = cellsoncell(i,current_cell)
365 if (icell <= ncells) then
366 d = sphere_distance(latcell(icell), loncell(icell), target_lat, target_lon, 1.0_rkind)
367 if (d < nearest_distance) then
368 nearest_cell = icell
369 nearest_distance = d
370 end if
371 end if
372 end do
373 end do
374
375 end function nearest_cell

378 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
379 ! Compute the great-circle distance between (lat1, lon1) and (lat2, lon2)
380 ! on a sphere with given radius.
381 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
382 real (kind=rkind) function sphere_distance(lat1, lon1, lat2, lon2, radius)
383
384 implicit none
385
386 real (kind=rkind), intent(in) :: lat1, lon1, lat2, lon2, radius
387 real (kind=rkind) :: arg1
388
389 arg1 = sqrt( sin(0.5*(lat2-lat1))**2 + cos(lat1)*cos(lat2)*sin(0.5*(lon2-lon1))**2 )
390 sphere_distance = 2.0 * radius * asin(arg1)
391
392 end function sphere_distance

401!+---+-----------------------------------------------------------------+
402! THIS FUNCTION CALCULATES THE LIQUID SATURATION VAPOR MIXING RATIO AS
403! A FUNCTION OF TEMPERATURE AND PRESSURE
404!
405 REAL(kind=rkind) FUNCTION rslf(P,T)
406
407 IMPLICIT NONE
408 REAL(kind=rkind), INTENT(IN):: p, t
409 REAL(kind=rkind):: esl,x
410 REAL(kind=rkind), PARAMETER:: c0= .611583699e03
411 REAL(kind=rkind), PARAMETER:: c1= .444606896e02
412 REAL(kind=rkind), PARAMETER:: c2= .143177157e01
413 REAL(kind=rkind), PARAMETER:: c3= .264224321e-1
414 REAL(kind=rkind), PARAMETER:: c4= .299291081e-3
415 REAL(kind=rkind), PARAMETER:: c5= .203154182e-5
416 REAL(kind=rkind), PARAMETER:: c6= .702620698e-8
417 REAL(kind=rkind), PARAMETER:: c7= .379534310e-11
418 REAL(kind=rkind), PARAMETER:: c8=-.321582393e-13
419
420 x=max(-80.,t-273.16)
421
422! ESL=612.2*EXP(17.67*X/(T-29.65))
423 esl=c0+x*(c1+x*(c2+x*(c3+x*(c4+x*(c5+x*(c6+x*(c7+x*c8)))))))
424 rslf=.622*esl/(p-esl)
425
426! ALTERNATIVE
427! ; Source: Murphy and Koop, Review of the vapour pressure of ice and
428! supercooled water for atmospheric applications, Q. J. R.
429! Meteorol. Soc (2005), 131, pp. 1539-1565.
430! ESL = EXP(54.842763 - 6763.22 / T - 4.210 * ALOG(T) + 0.000367 * T
431! + TANH(0.0415 * (T - 218.8)) * (53.878 - 1331.22
432! / T - 9.44523 * ALOG(T) + 0.014025 * T))
433
434 END FUNCTION rslf
