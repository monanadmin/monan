28!mpas_atmphys_o3climatology contains the subroutines needed to initialize,interpolate,and update
29!the climatological monthly-mean ozone volume mixing ratios o3clim to the MPAS grid. Input data
30!files are the same as the ones used in the CAM long- and short-wave radiation codes.
31!when namelist parameter config_o3climatology is set to true, o3clim is used in the RRTMG long-
32!wave and short-wave radiation codes,and replaces the annual-mean ozone sounding used by default.
33!Laura D. Fowler (send comments to laura@ucar.edu).
34!2013-07-03.
35!
36! subroutines in mpas_atmphys_o3climatology:
37! ------------------------------------------
38! init_o3climatology : read the CAM ozone data files.
39! update_o3climatology : interpolates the ozone volume mixing ratio to the current Julian day
40! as done for the greeness fraction in the MPAS time manager.
41! o3climatology_from_MPAS: interpolates the ozone volume mixing ratio to the current Julian day
42! as in the CAM radiation codes.
43!
44! add-ons and modifications to sourcecode:
45! ----------------------------------------
46! * throughout the sourcecode, replaced all "var_struct" defined arrays by local pointers.
47! Laura D. Fowler (laura@ucar.edu) / 2014-04-22.
48! * modified sourcecode to use pools.
49! Laura D. Fowler (laura@ucar.edu) / 2014-05-15.
50! * moved the subroutine vinterp_ozn to its own module module_ra_rrtmg_vinterp.F in physics_wrf.
51! Laura D. Fowler (laura@ucar.edu) / 2017-01-27.

57!=================================================================================================================
58 subroutine init_o3climatology(mesh,atm_input)
59!=================================================================================================================
60
61!This subroutine assumes a uniform distribution of ozone concentration. It should be replaced
62!with monthly climatology varying ozone distribution.
63
64!input arguments:
65 type(mpas_pool_type),intent(in):: mesh
66
67!inout arguments:
68 type(mpas_pool_type),intent(inout):: atm_input
69
70!local pointers:
71 integer, pointer:: ncells,num_months,levsiz
72 real(kind=rkind),dimension(:),pointer:: latcell,loncell
73 real(kind=rkind),dimension(:),pointer:: pin
74 real(kind=rkind),dimension(:,:,:),pointer:: ozmixm
75
76!local variables:
77 integer,parameter:: pin_unit = 27
78 integer,parameter:: lat_unit = 28
79 integer,parameter:: oz_unit = 29
80 integer,parameter:: open_ok = 0
81
82 integer:: i,i1,i2,istat,k,j,m
83 integer:: icell
84
85 real(kind=rkind):: lat,lon,dlat,dlatcell
86 real(kind=rkind),dimension(latsiz):: lat_ozone
87 real(kind=rkind),dimension(:,:,:,:),allocatable:: ozmixin
88
89!-----------------------------------------------------------------------------------------------------------------
90!call mpas_log_write('')
91!call mpas_log_write('--- enter subroutine physics_init_o3:')
92
93 call mpas_pool_get_dimension(mesh,'nCells',ncells)
94 call mpas_pool_get_dimension(mesh,'nMonths',num_months)
95 call mpas_pool_get_dimension(mesh,'nOznLevels',levsiz)
96
97 call mpas_pool_get_array(atm_input,'pin',pin)
98 call mpas_pool_get_array(atm_input,'ozmixm',ozmixm)
99 call mpas_pool_get_array(mesh,'latCell',latcell)
100 call mpas_pool_get_array(mesh,'lonCell',loncell)
101
102!-- read in ozone pressure data:
103 open(pin_unit,file='OZONE_PLEV.TBL',action='READ',status='OLD',iostat=istat)
104 if(istat /= open_ok) &
105 call physics_error_fatal('subroutine oznini: ' // &
106 'failure opening OZONE_PLEV.TBL')
107 do k = 1,levsiz
108 read(pin_unit,*) pin(k)
109 enddo
110 close(pin_unit)
111
112!-- read in ozone lat data:
113 open(lat_unit, file='OZONE_LAT.TBL',action='READ',status='OLD',iostat=istat)
114 if(istat /= open_ok) &
115 call physics_error_fatal('subroutine oznini: ' // &
116 'failure opening OZONE_LAT.TBL')
117 do j = 1, latsiz
118 read(lat_unit,*) lat_ozone(j)
119! call mpas_log_write('$i $r', intArgs=(/j/), realArgs=(/lat_ozone(j)/))
120 enddo
121 close(lat_unit)
122
123!-- read in ozone data:
124 open(oz_unit,file='OZONE_DAT.TBL',action='READ',status='OLD',iostat=istat)
125 if(istat /= open_ok) &
126 call physics_error_fatal('subroutine oznini: ' // &
127 'failure opening OZONE_DAT.TBL')
128
129 allocate(ozmixin(lonsiz,levsiz,latsiz,num_months))
130 do m=1,num_months
131 do j=1,latsiz ! latsiz=64
132 do k=1,levsiz ! levsiz=59
133 do i=1,lonsiz ! lonsiz=1
134 read(oz_unit,*) ozmixin(i,k,j,m)
135 enddo
136 enddo
137 enddo
138 enddo
139 close(oz_unit)
140
141!INTERPOLATION OF INPUT OZONE DATA TO MPAS GRID:
142!call mpas_log_write('max latCell=$r', realArgs=(/maxval(latCell)/degrad/))
143!call mpas_log_write('min latCell=$r', realArgs=(/minval(latCell)/degrad/))
144!call mpas_log_write('max lonCell=$r', realArgs=(/maxval(lonCell)/degrad/))
145!call mpas_log_write('min lonCell=$r', realArgs=(/minval(lonCell)/degrad/))
146!call mpas_log_write('')
147!call mpas_log_write('max lat_ozone=$r', realArgs=(/maxval(lat_ozone)/))
148!call mpas_log_write('min lat_ozone=$r', realArgs=(/minval(lat_ozone)/))
149 do icell = 1,ncells
150 lat = latcell(icell)/degrad
151 lon = loncell(icell)/degrad
152 if(lat .gt. lat_ozone(latsiz)) then
153 i1 = latsiz
154 i2 = latsiz
155 elseif(lat .lt. lat_ozone(1)) then
156 i1 = 1
157 i2 = 1
158 else
159 do i = 1, latsiz
160 if(lat.ge.lat_ozone(i) .and. lat.lt.lat_ozone(i+1)) exit
161 enddo
162 i1 = i
163 i2 = i+1
164 endif
165
166 do m = 1,num_months
167 do k = 1,levsiz
168 do j = 1,lonsiz
169 dlat = lat_ozone(i2)-lat_ozone(i1)
170 dlatcell = lat-lat_ozone(i1)
171 if(dlat == 0.) then
172 ozmixm(m,k,icell) = ozmixin(j,k,i1,m)
173 else
174 ozmixm(m,k,icell) = ozmixin(j,k,i1,m) &
175 + (ozmixin(j,k,i2,m)-ozmixin(j,k,i1,m))*dlatcell/dlat
176 endif
177 enddo
178 enddo
179 enddo
180! do k = 1, levsiz
181! call mpas_log_write('$i $i $i $r $r $r $r $r $r', intArgs=(/iCell,i1,i2/), &
182! realArgs=(/lat_ozone(i1),lat,lat_ozone(i2),ozmixin(1,k,i1,1),ozmixm(1,k,iCell),ozmixin(1,k,i2,1)/))
183! enddo
184 enddo
185 deallocate(ozmixin)
186
187! call mpas_log_write('--- end subroutine physics_init_o3.')
188
189 end subroutine init_o3climatology

228!=================================================================================================================
229 subroutine o3climatology_from_mpas(julian,mesh,atm_input,diag_physics)
230!=================================================================================================================
231
232!input arguments:
233 type(mpas_pool_type),intent(in):: mesh
234 real(kind=rkind),intent(in):: julian
235 type(mpas_pool_type),intent(in):: atm_input
236
237!inout arguments:
238 type(mpas_pool_type),intent(inout):: diag_physics
239
240!local pointers:
241 integer,pointer:: noznlevels,nvertlevels,ncellssolve,nmonths
242 real(kind=rkind),dimension(:,:),pointer :: o3clim
243 real(kind=rkind),dimension(:,:,:),pointer:: ozmixm
244
245!local variables:
246 logical:: finddate
247 logical:: ozncyc
248
249 integer:: icell,k
250 integer:: ijul,m,nm,np,np1
251 integer, dimension(12) :: date_oz
252 data date_oz/16, 45, 75, 105, 136, 166, 197, 228, 258, 289, 319, 350/
253
254 real(kind=r8):: intjulian_r8
255 real(kind=r8):: cdayozp_r8, cdayozm_r8
256 real(kind=r8):: fact1_r8, fact2_r8
257 real(kind=rkind):: fact1,fact2
258
259!-----------------------------------------------------------------------------------------------------------------
260!call mpas_log_write('')
261!call mpas_log_write('--- enter subroutine o3climatology_from_MPAS:')
262
263 call mpas_pool_get_dimension(mesh,'nOznLevels',noznlevels)
264 call mpas_pool_get_dimension(mesh,'nVertLevels',nvertlevels)
265 call mpas_pool_get_dimension(mesh,'nCellsSolve',ncellssolve)
266 call mpas_pool_get_dimension(mesh,'nMonths',nmonths)
267
268 call mpas_pool_get_array(atm_input,'ozmixm',ozmixm)
269 call mpas_pool_get_array(diag_physics,'o3clim',o3clim)
270
271 ozncyc = .true.
272
273!julian starts from 0.0 at 0Z on 1 Jan.
274 intjulian_r8 = real(julian + 1.0_rkind, r8) ! offset by one day
275
276!jan 1st 00z is julian=1.0 here
277 ijul=int(intjulian_r8)
278!note that following will drift. need to use actual month/day info to compute julian.
279 intjulian_r8 = intjulian_r8 - real(ijul, r8)
280 ijul = mod(ijul,365)
281 if(ijul .eq. 0) ijul=365
282 intjulian_r8 = intjulian_r8 + real(ijul, r8)
283 np1=1
284
285 finddate=.false.
286 do m = 1, nmonths
287 if(date_oz(m).gt.intjulian_r8 .and. .not.finddate) then
288 np1 = m
289 finddate = .true.
290 endif
291 enddo
292 cdayozp_r8=date_oz(np1)
293
294 if(np1 .gt. 1) then
295 cdayozm_r8 = date_oz(np1-1)
296 np = np1
297 nm = np-1
298 else
299 cdayozm_r8 = date_oz(12)
300 np = np1
301 nm = nmonths
302 endif
303 call getfactors(ozncyc,np1,cdayozm_r8,cdayozp_r8,intjulian_r8,fact1_r8,fact2_r8)
304 fact1 = real(fact1_r8, rkind)
305 fact2 = real(fact2_r8, rkind)
306
307!call mpas_log_write('--- end subroutine getfactors:')
308!call mpas_log_write('fact1 =$r', realArgs=(/fact1/))
309!call mpas_log_write('fact2 =$r', realArgs=(/fact2/))
310
311!Time interpolation.
312 do k = 1, noznlevels
313 do icell = 1, ncellssolve
314 o3clim(k,icell) = fact1 * ozmixm(nm,k,icell) + fact2 * ozmixm(np,k,icell)
315 end do
316!call mpas_log_write('$i $r $r $r $r $r $r', intArgs=(/k/), realArgs=(/o3clim(k,1),ozmixm(nm,k,1),ozmixm(np,k,1), &
317! o3clim(k,nCellsSolve),ozmixm(nm,k,nCellsSolve),ozmixm(np,k,nCellsSolve)/))
318 end do
319
320!call mpas_log_write('--- end subroutine o3climatology_from_MPAS')
321
322 end subroutine o3climatology_from_mpas
