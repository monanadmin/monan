69! routine li_time_integrator_forwardeuler
70!
76!
77!-----------------------------------------------------------------------
78 subroutine li_time_integrator_forwardeuler(domain, err)
79
80 use li_subglacial_hydro
81 use li_velocity
82
83 !-----------------------------------------------------------------
84 ! input variables
85 !-----------------------------------------------------------------
86
87 !-----------------------------------------------------------------
88 ! input/output variables
89 !-----------------------------------------------------------------
90
91 type (domain_type), intent(inout) :: &
92 domain
93
94 !-----------------------------------------------------------------
95 ! output variables
96 !-----------------------------------------------------------------
97 integer, intent(out) :: err
98
99 !-----------------------------------------------------------------
100 ! local variables
101 !-----------------------------------------------------------------
102 type (block_type), pointer :: block
103 integer :: err_tmp
104
105 logical, pointer :: config_restore_calving_front
106
107 err = 0
108
109 call mpas_pool_get_config(liconfigs, 'config_restore_calving_front', config_restore_calving_front)
110
111! === Prepare for advection (including CFL checks) ===========
112! This has to come first currently, because it sets the time step!
113 call mpas_timer_start("advection prep")
114 call prepare_advection(domain, err_tmp)
115 err = ior(err, err_tmp)
116 call mpas_timer_stop("advection prep")
117
118! === Basal melting for floating ice ===========
119 call mpas_timer_start("basal melting for floating ice")
120 call li_basal_melt_floating_ice(domain, err_tmp)
121 err = ior(err, err_tmp)
122 call mpas_timer_stop("basal melting for floating ice")
123
124! === Implicit column physics (vertical temperature diffusion) ===========
125 call mpas_timer_start("vertical therm")
126 call li_thermal_solver(domain, err_tmp)
127 err = ior(err, err_tmp)
128 call mpas_timer_stop("vertical therm")
129
130! === Compute new state for prognostic variables ==================================
131 call mpas_timer_start("advect thickness and tracers")
132 call advection_solver(domain, err_tmp)
133 err = ior(err, err_tmp)
134 call mpas_timer_stop("advect thickness and tracers")
135
136! === Update subglacial hydrology ===========
137! It's not clear where the best place to call this should be.
138! Seems sensible to put it after thermal evolution is complete to get updated basal melting source term.
139! Also seems (might be?) better to put it after geometry evolution.
140! We want it before the velocity solve since the hydro model can control the velo basal b.c.
141 call mpas_timer_start("subglacial hydro")
142 call li_sgh_solve(domain, err_tmp)
143 err = ior(err, err_tmp)
144 call mpas_timer_stop("subglacial hydro")
145
146! === Calve ice ========================
147 call mpas_timer_start("calve_ice")
148
149 if (config_restore_calving_front) then
150
151 ! restore the calving front to its initial position; calving options are ignored
152 call li_restore_calving_front(domain, err_tmp)
153 err = ior(err, err_tmp)
154
155 else
156
157 ! ice calving
158 call li_calve_ice(domain, err_tmp)
159 err = ior(err, err_tmp)
160
161 endif
162
163 call mpas_timer_stop("calve_ice")
164
165 call mpas_timer_start("halo updates")
166 call mpas_dmpar_field_halo_exch(domain, 'cellMask')
167 call mpas_dmpar_field_halo_exch(domain, 'edgeMask')
168 call mpas_dmpar_field_halo_exch(domain, 'vertexMask')
169 call mpas_timer_stop("halo updates")
170
171! === Solve Velocity =====================
172 ! During time-stepping, we always solveVelo
173 call li_velocity_solve(domain, solvevelo=.true., err=err_tmp)
174 err = ior(err, err_tmp)
175
176! === Calculate diagnostic variables for new state =====================
177
178 call li_calculate_diagnostic_vars(domain, err=err_tmp)
179 err = ior(err, err_tmp)
180
181
182! === Cleanup & Misc. =============================
183
184 ! === error check
185 if (err == 1) then
186 call mpas_log_write("An error has occurred in li_time_integrator_forwardeuler.", mpas_log_err)
187 endif
188
189 !--------------------------------------------------------------------
190 end subroutine li_time_integrator_forwardeuler

215!-----------------------------------------------------------------------
216
217 subroutine prepare_advection(domain, err)
218
219 use mpas_timekeeping
220
221 !-----------------------------------------------------------------
222 ! input variables
223 !-----------------------------------------------------------------
224
225 !-----------------------------------------------------------------
226 ! input/output variables
227 !-----------------------------------------------------------------
228
229 type (domain_type), intent(inout) :: domain
230
231 !-----------------------------------------------------------------
232 ! output variables
233 !-----------------------------------------------------------------
234 integer, intent(out) :: err
235
236 !-----------------------------------------------------------------
237 ! local variables
238 !-----------------------------------------------------------------
239 type (dm_info), pointer :: dminfo
240 type (block_type), pointer :: block
241
242 type (mpas_pool_type), pointer :: meshPool
243 type (mpas_pool_type), pointer :: velocityPool
244 type (mpas_pool_type), pointer :: geometryPool
245 type (mpas_pool_type), pointer :: scratchPool
246
247 real (kind=rkind), dimension(:,:), pointer :: normalvelocity
248 real (kind=rkind), dimension(:,:), pointer :: layernormalvelocity
249
250 logical, pointer :: config_print_thickness_advection_info
251 logical, pointer :: config_adaptive_timestep
252 logical, pointer :: config_adaptive_timestep_include_DCFL
253
254 integer :: &
255 allowableAdvecDtProcNumberHere, &
256 allowableAdvecDtProcNumber
257
258 real (kind=rkind) :: &
259 allowableadvecdt, &
260 allowableadvecdtonproc, &
261 allowableadvecdtallprocs
262
263 type (MPAS_TimeInterval_type) :: &
264 allowableAdvecDtOnProcInterval, &
265 allowableAdvecDtAllProcsInterval
266
267 character (len=StrKIND) :: &
268 allowableAdvecDtOnProcString, &
269 allowableAdvecDtAllProcsString
270
271 integer :: &
272 allowableDiffDtProcNumberHere, &
273 allowableDiffDtProcNumber
274
275 real (kind=rkind) :: &
276 allowablediffdt, &
277 allowablediffdtonproc, &
278 allowablediffdtallprocs
279
280 type (MPAS_TimeInterval_type) :: &
281 allowableDiffDtOnProcInterval, &
282 allowableDiffDtAllProcsInterval
283
284 character (len=StrKIND) :: &
285 allowableDiffDtOnProcString, &
286 allowableDiffDtAllProcsString
287
288 real (kind=rkind), pointer :: &
289 allowabledtacfl, &
290 allowabledtdcfl
291
292 real (kind=rkind), pointer :: deltat ! variable in blocks
293
294 real (kind=rkind) :: dtseconds ! local variable
295
296 integer :: err_tmp
297
298 err = 0
299
300 call mpas_pool_get_config(liconfigs, 'config_print_thickness_advection_info', config_print_thickness_advection_info)
301 call mpas_pool_get_config(liconfigs, 'config_adaptive_timestep', config_adaptive_timestep)
302 call mpas_pool_get_config(liconfigs, 'config_adaptive_timestep_include_DCFL', config_adaptive_timestep_include_dcfl)
303
304 allowableadvecdtallprocs = 0.0_rkind
305 allowablediffdtallprocs = 0.0_rkind
306
307 dminfo => domain % dminfo
308
309 ! Initialize
310
311 err = 0
312
313 allowableadvecdtonproc = 1.0e36_rkind ! set to large number
314 allowablediffdtonproc = 1.0e36_rkind ! set to large number
315
316 block => domain % blocklist
317 do while (associated(block))
318
319 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
320 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
321 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
322 call mpas_pool_get_subpool(block % structs, 'scratch', scratchpool)
323
324 call mpas_pool_get_array(velocitypool, 'normalVelocity', normalvelocity)
325 call mpas_pool_get_array(velocitypool, 'layerNormalVelocity', layernormalvelocity)
326
327 ! compute normal velocities and advective CFL limit for this block
328
329 call li_layer_normal_velocity( &
330 meshpool, &
331 normalvelocity, &
332 layernormalvelocity, &
333 allowableadvecdt, &
334 err_tmp)
335
336 err = ior(err, err_tmp)
337
338 allowableadvecdtonproc = min(allowableadvecdtonproc, allowableadvecdt)
339
340 ! Calculate diffusive CFL timestep, if needed
341 ! This used to be only calculated if (config_adaptive_timestep_include_DCFL) but for simplicity,
342 ! now it is always calculated. That allows assessment of the DCFL even when it is not being obeyed
343 ! by the adaptive time stepper. Timer is added here to monitor if this unnecessary calculation becomes significant.
344 call mpas_timer_start("calculate apparent diffusivity")
345 call li_calculate_apparent_diffusivity(meshpool, velocitypool, scratchpool, geometrypool, allowablediffdt)
346 allowablediffdtonproc = min(allowablediffdtonproc, allowablediffdt)
347 call mpas_timer_stop("calculate apparent diffusivity")
348
349 ! Note: The ACFL and DCFL timesteps could be calculated in
350 ! diagnostic_variables_solve_after_velocity. In that case, we could also add
351 ! variables to store their values, rather than just relying on the values
352 ! written to the log files. However, the current logic only calculates these
353 ! values if certain config options are set, so that would need to be dealt with.
354 ! If the ACFL and DCFL timesteps are moved to diagnostic_variables_solve_after_velocity,
355 ! Then the setting of the timestep value could happen at the beginning of the timestep,
356 ! probably in li_timestep rather than here. That might be cleaner, but I have
357 ! persisted with doing it here, because since the calculation of ACFL/DCFL have a lot
358 ! in common with the advection calculation, and it seems kind of silly to do those calculations
359 ! on the previous time step. That said, the calculations are pretty cheap.
360
361 block => block % next
362 end do
363
364
365 ! Local advective CFL info
366 call mpas_set_timeinterval(allowableadvecdtonprocinterval, dt=allowableadvecdtonproc, ierr=err_tmp)
367 err = ior(err,err_tmp)
368 call mpas_get_timeinterval(allowableadvecdtonprocinterval, timestring=allowableadvecdtonprocstring, ierr=err_tmp)
369 err = ior(err,err_tmp)
370
371 if (config_print_thickness_advection_info) then
372 call mpas_log_write(' Maximum allowable time step on THIS processor based on advective CFL is (Days_hhh:mmm:sss): ' &
373 // trim(allowableadvecdtonprocstring))
374 endif
375
376
377 ! Local diffusive CFL info
378 ! This used to be only calculated if (config_adaptive_timestep_include_DCFL) but for simplicity,
379 ! now it is always calculated. That allows assessment of the DCFL even when it is not being obeyed
380 ! by the adaptive time stepper.
381 call mpas_set_timeinterval(allowablediffdtonprocinterval, dt=allowablediffdtonproc, ierr=err_tmp)
382 err = ior(err,err_tmp)
383 call mpas_get_timeinterval(allowablediffdtonprocinterval, timestring=allowablediffdtonprocstring, ierr=err_tmp)
384 err = ior(err,err_tmp)
385
386 if (config_print_thickness_advection_info) then
387 call mpas_log_write(' Maximum allowable time step on THIS processor based on diffusive CFL is (Days_hhh:mmm:sss): ' &
388 // trim(allowablediffdtonprocstring))
389 endif
390
391
392 ! These calculation could be (and used to be) restricted to only if we are
393 ! printing advection debug information or adaptive timestepping, because
394 ! it requires 2 unnecessary MPI communications for each of the ACFL and DCFL
395 ! However, those communications are probably small relative to other costs in
396 ! the model, so now they always happen. The timer has been added to allow
397 ! assessment of that assumption.
398 call mpas_timer_start("calculate global CFL limits")
399
400 ! Determine ACFL limit on all procs
401 call mpas_dmpar_min_real(dminfo, allowableadvecdtonproc, allowableadvecdtallprocs)
402
403 ! Determine which processor has the limiting CFL
404 if (allowableadvecdtonproc == allowableadvecdtallprocs) then
405 allowableadvecdtprocnumberhere = dminfo % my_proc_id
406 else
407 allowableadvecdtprocnumberhere = -1
408 endif
409
410 call mpas_dmpar_max_int(dminfo, allowableadvecdtprocnumberhere, allowableadvecdtprocnumber)
411 call mpas_set_timeinterval(allowableadvecdtallprocsinterval, dt=allowableadvecdtallprocs, ierr=err_tmp)
412 err = ior(err,err_tmp)
413 call mpas_get_timeinterval(allowableadvecdtallprocsinterval, timestring=allowableadvecdtallprocsstring, ierr=err_tmp)
414 err = ior(err,err_tmp)
415
416 ! Repeat for diffusive CFL
417 ! This used to be only calculated if (config_adaptive_timestep_include_DCFL) but for simplicity,
418 ! now it is always calculated. That allows assessment of the DCFL even when it is not being obeyed
419 ! by the adaptive time stepper.
420
421 ! Determine DCFL limit on all procs
422 call mpas_dmpar_min_real(dminfo, allowablediffdtonproc, allowablediffdtallprocs)
423
424 ! Determine which processor has the limiting CFL
425 if (allowablediffdtonproc == allowablediffdtallprocs) then
426 allowablediffdtprocnumberhere = dminfo % my_proc_id
427 else
428 allowablediffdtprocnumberhere = -1
429 endif
430
431 call mpas_dmpar_max_int(dminfo, allowablediffdtprocnumberhere, allowablediffdtprocnumber)
432 call mpas_set_timeinterval(allowablediffdtallprocsinterval, dt=allowablediffdtallprocs, ierr=err_tmp)
433 err = ior(err,err_tmp)
434 call mpas_get_timeinterval(allowablediffdtallprocsinterval, timestring=allowablediffdtallprocsstring, ierr=err_tmp)
435 err = ior(err,err_tmp)
436
437 call mpas_timer_stop("calculate global CFL limits")
438
439
440 ! Write messages if they are turned on
441 if (config_print_thickness_advection_info) then
442 call mpas_log_write(' Maximum allowable time step for all processors based on advective CFL is (Days_hhh:mmm:sss): ' &
443 // trim(allowableadvecdtallprocsstring) // ' Time step is limited by processor number $i', &
444 intargs=(/allowableadvecdtprocnumber/))
445 if (config_adaptive_timestep_include_dcfl) then
446 call mpas_log_write(' Maximum allowable time step for all processors based on diffusive CFL is (Days_hhh:mmm:sss): '&
447 // trim(allowablediffdtallprocsstring) // ' Time step is limited by processor number $i', &
448 intargs=(/allowablediffdtprocnumber/))
449 endif
450 endif
451
452
453 ! Set adaptive timestep if needed
454 if (config_adaptive_timestep) then
455 call set_timestep(allowableadvecdtallprocs, allowablediffdtallprocs, domain % clock, dtseconds, err_tmp)
456 err = ior(err,err_tmp)
457 ! Set new value on all blocks
458 block => domain % blocklist
459 do while (associated(block))
460 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
461 call mpas_pool_get_array(meshpool, 'deltat', deltat)
462
463 deltat = dtseconds
464
465 block => block % next
466 end do
467 else
468 ! If not adaptive, Get dt from any block to check for CFL violation below
469 block => domain % blocklist
470 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
471 call mpas_pool_get_array(meshpool, 'deltat', deltat)
472 end if
473
474
475 ! Check for CFL error before finishing
476 if (deltat > allowableadvecdtonproc) then
477 call mpas_log_write('Advective CFL violation on this processor. ' // &
478 'Maximum allowable time step for this processor is (Days_hhh:mmm:sss): ' // trim(allowableadvecdtonprocstring), mpas_log_err)
479 err = ior(err,1)
480 endif
481
482 ! Local diffusive CFL info
483 if ( (config_adaptive_timestep_include_dcfl) .and. (deltat > allowablediffdtonproc) ) then
484 call mpas_log_write('Diffusive CFL violation on this processor. ' // &
485 'Maximum allowable time step for this processor is (Days_hhh:mmm:sss): ' // trim(allowablediffdtonprocstring), mpas_log_warn)
486 endif
487
488 if (err > 0) then
489 call mpas_log_write('Error in calculating thickness advection (possibly CFL violation)', mpas_log_err)
490 endif
491
492
493 ! set CFL variables if they have been calculated - every block should be set to the same value!
494 block => domain % blocklist
495 do while (associated(block))
496 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
497 call mpas_pool_get_array(meshpool, 'allowableDtACFL', allowabledtacfl)
498 allowabledtacfl = allowableadvecdtallprocs
499 call mpas_pool_get_array(meshpool, 'allowableDtDCFL', allowabledtdcfl)
500 allowabledtdcfl = allowablediffdtallprocs
501
502 block => block % next
503 end do
504
505
506 ! Halo updates
507 call mpas_timer_start("halo updates")
508 call mpas_dmpar_field_halo_exch(domain, 'layerNormalVelocity')
509 call mpas_timer_stop("halo updates")
510
511 ! === error check
512 if (err > 0) then
513 call mpas_log_write("An error has occurred in prepare_advection.", mpas_log_err)
514 endif
515
516 !--------------------------------------------------------------------
517 end subroutine prepare_advection

522!
532!-----------------------------------------------------------------------
533
534 subroutine advection_solver(domain, err)
535
536 use mpas_timekeeping
537 use li_mask
538
539 !-----------------------------------------------------------------
540 ! input variables
541 !-----------------------------------------------------------------
542
543 !-----------------------------------------------------------------
544 ! input/output variables
545 !-----------------------------------------------------------------
546 type (domain_type), intent(inout) :: domain
547
548 !-----------------------------------------------------------------
549 ! output variables
550 !-----------------------------------------------------------------
551 integer, intent(out) :: err
552
553 !-----------------------------------------------------------------
554 ! local variables
555 !-----------------------------------------------------------------
556 type (dm_info), pointer :: dminfo
557 type (block_type), pointer :: block
558
559 type (mpas_pool_type), pointer :: meshPool
560 type (mpas_pool_type), pointer :: velocityPool
561 type (mpas_pool_type), pointer :: geometryPool
562 type (mpas_pool_type), pointer :: thermalPool
563 type (mpas_pool_type), pointer :: scratchPool
564
565 real (kind=rkind), pointer :: deltat ! variable in blocks
566 real (kind=rkind), dimension(:), pointer :: thicknessold
567 real (kind=rkind), dimension(:), pointer :: thicknessnew
568 real (kind=rkind), dimension(:), pointer :: thickness
569
570 real (kind=rkind), dimension(:,:), pointer :: temperature
571 real (kind=rkind), dimension(:,:), pointer :: waterfrac
572 real (kind=rkind), dimension(:,:), pointer :: enthalpy
573
574 integer, pointer :: nCells
575
576 character (len=StrKIND), pointer :: config_thickness_advection
577 character (len=StrKIND), pointer :: config_tracer_advection
578
579 logical, pointer :: config_print_thickness_advection_info
580
581 !TODO - Replace masktmp with a scratch field?
582 integer, dimension(:), allocatable :: masktmp ! Temporary mask for assessing new thickness field
583
584 integer :: err_tmp
585
586 err = 0
587 err_tmp = 0
588
589 call mpas_pool_get_config(liconfigs, 'config_thickness_advection', config_thickness_advection)
590 call mpas_pool_get_config(liconfigs, 'config_tracer_advection', config_tracer_advection)
591 call mpas_pool_get_config(liconfigs, 'config_print_thickness_advection_info', config_print_thickness_advection_info)
592
593 dminfo => domain % dminfo
594
595 ! Halo updates
596 ! Note: The layer thickness and tracers must be up to date in halos before calling the advection subroutines.
597 ! The thermal tracers (temperature, waterfrac, enthalpy) are updated at the end of li_thermal_solver.
598 ! But thickness (which is used by subroutine li_advection_thickness_tracers) needs an update here. TODO: confirm this
599
600 call mpas_timer_start("halo updates")
601 call mpas_dmpar_field_halo_exch(domain, 'thickness')
602 call mpas_timer_stop("halo updates")
603
604
605 ! ===
606 ! === Calculate layerThicknessEdge, which is needed for advection
607 ! ===
608 if (trim(config_thickness_advection) == 'fo' .or. trim(config_tracer_advection) == 'fo') then
609 block => domain % blocklist
610 do while (associated(block))
611
612 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
613 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
614 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
615
616 call calculate_layerthicknessedge(meshpool, geometrypool, velocitypool, err_tmp)
617 err = ior(err,err_tmp)
618
619 block => block % next
620 end do
621
622 ! Halo update layerThicknessEdge - outer halo may be wrong due to requiring velocity
623 call mpas_timer_start("halo updates")
624 call mpas_dmpar_field_halo_exch(domain, 'layerThicknessEdge')
625 call mpas_timer_stop("halo updates")
626 endif ! fo advection option
627
628
629
630 ! ===
631 ! === Advect thickness and tracers
632 ! ===
633
634 block => domain % blocklist
635 do while (associated(block))
636
637 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
638 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
639 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
640 call mpas_pool_get_subpool(block % structs, 'thermal', thermalpool)
641 call mpas_pool_get_subpool(block % structs, 'scratch', scratchpool)
642 call mpas_pool_get_array(meshpool, 'deltat', deltat)
643
644 if (trim(config_thickness_advection) == 'fo' .and. trim(config_tracer_advection) == 'fo') then
645
646 ! Note: This subroutine requires that thickness and tracers are correct in halos
647
648 if (config_print_thickness_advection_info) then
649 call mpas_log_write('Advect thickness and tracers, dt=$r', realargs=(/deltat/))
650 endif
651
652 call li_advection_thickness_tracers(&
653 deltat, &
654 meshpool, &
655 velocitypool, &
656 geometrypool, &
657 thermalpool, &
658 scratchpool, &
659 err_tmp, &
660 advecttracersin = .true.)
661
662 err = ior(err,err_tmp)
663
664 elseif (trim(config_thickness_advection) == 'fo' .and. trim(config_tracer_advection) == 'none') then
665
666 if (config_print_thickness_advection_info) then
667 call mpas_log_write('Advect thickness (but not tracers), dt=$r', realargs=(/deltat/))
668 endif
669
670 call li_advection_thickness_tracers(&
671 deltat, &
672 meshpool, &
673 velocitypool, &
674 geometrypool, &
675 thermalpool, &
676 scratchpool, &
677 err_tmp, &
678 advecttracersin = .false.)
679
680 !call mpas_log_write("errtmp=$i",intArgs=(/err_tmp/))
681 err = ior(err,err_tmp)
682
683 endif
684
685 block => block % next
686 end do
687
688 ! Reset negative thicknesses to zero if needed
689
690 block => domain % blocklist
691 do while (associated(block))
692
693 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
694 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
695
696 call mpas_pool_get_dimension(meshpool, 'nCells', ncells)
697 call mpas_pool_get_array(geometrypool, 'thickness', thickness, timelevel=1)
698
699
700 allocate( masktmp(ncells + 1) )
701 masktmp = 0
702
703 where (thickness < 0.0_rkind)
704 masktmp = 1
705 thickness = 0.0_rkind
706 end where
707
708 if (config_print_thickness_advection_info) then
709
710 if (sum(masktmp) > 0) then
711 call mpas_log_write(' Cells with negative thickness (set to 0): $i', intargs=(/sum(masktmp)/))
712 endif
713
714 ! Count how many cells have ice.
715 masktmp = 0
716 where (thickness > 0.0_rkind)
717 masktmp = 1
718 end where
719 call mpas_log_write(' Cells with nonzero thickness: $i', intargs=(/sum(masktmp)/))
720
721 endif
722
723 deallocate(masktmp)
724
725 block => block % next
726 end do
727
728 ! Halo updates
729 call mpas_timer_start("halo updates")
730
731 call mpas_dmpar_field_halo_exch(domain, 'thickness')
732 call mpas_dmpar_field_halo_exch(domain, 'temperature')
733 call mpas_dmpar_field_halo_exch(domain, 'waterfrac')
734 call mpas_dmpar_field_halo_exch(domain, 'enthalpy')
735
736 call mpas_timer_stop("halo updates")
737
738 ! Update mask and geometry
739 block => domain % blocklist
740 do while (associated(block))
741 call mpas_pool_get_subpool(block % structs, 'mesh', meshpool)
742 call mpas_pool_get_subpool(block % structs, 'velocity', velocitypool)
743 call mpas_pool_get_subpool(block % structs, 'geometry', geometrypool)
744
745 call li_calculate_mask(meshpool, velocitypool, geometrypool, err_tmp)
746 call li_update_geometry(geometrypool)
747
748 block => block % next
749 end do
750
751
752 ! === error check
753 if (err > 0) then
754 call mpas_log_write("An error has occurred in advection_solver.", mpas_log_err)
755 endif
756
757 !--------------------------------------------------------------------
758 end subroutine advection_solver

771 subroutine set_timestep(allowableAdvecDt, allowableDiffDt, clock, dtSeconds, err)
772 use mpas_timekeeping
773
774 !-----------------------------------------------------------------
775 ! input variables
776 !-----------------------------------------------------------------
777 real (kind=rkind), intent(in) :: allowableadvecdt
778 real (kind=rkind), intent(in) :: allowablediffdt
779 type (MPAS_Clock_type), intent(in) :: clock

782 ! output variables
783 !-----------------------------------------------------------------

785 integer, intent(out) :: err

787 !-----------------------------------------------------------------
788 ! local variables
789 !-----------------------------------------------------------------
790 logical, pointer :: config_adaptive_timestep_include_DCFL
791 real (kind=rkind), pointer :: config_adaptive_timestep_cfl_fraction
792 real (kind=rkind), pointer :: config_max_adaptive_timestep
793 real (kind=rkind), pointer :: config_min_adaptive_timestep
794 type (MPAS_Time_type) :: nextForceTime, currTime
795 type (MPAS_TimeInterval_type) :: intervalToNextForceTime
796 real (kind=rkind) :: secondstonextforcetime
797 real (kind=rkind) :: allowabledt
798 real (kind=rkind) :: proposeddt
799 integer :: err_tmp
800
801 err = 0
802 err_tmp = 0
803
804
805 call mpas_pool_get_config(liconfigs, 'config_adaptive_timestep_CFL_fraction', config_adaptive_timestep_cfl_fraction)
806 call mpas_pool_get_config(liconfigs, 'config_max_adaptive_timestep', config_max_adaptive_timestep)
807 call mpas_pool_get_config(liconfigs, 'config_min_adaptive_timestep', config_min_adaptive_timestep)
808 call mpas_pool_get_config(liconfigs, 'config_adaptive_timestep_include_DCFL', config_adaptive_timestep_include_dcfl)
809
810 if (config_adaptive_timestep_include_dcfl) then
811 allowabledt = min(allowableadvecdt, allowablediffdt)
812 else
813 allowabledt = allowableadvecdt
814 endif
815
816 ! Take minimum of the max adaptive timestep setting and the allowable dt from the CFL condition
817 proposeddt = min(allowabledt * config_adaptive_timestep_cfl_fraction, config_max_adaptive_timestep)
818 ! Round down the proposed dt to avoid complications with fractional seconds
819 ! (some timekeeping-related functionality, like restarts, don't support them)
820 ! (need to perform floor with an 8-bit integer to allow up 293 billion years in seconds)
821 proposeddt = real(floor(proposeddt, kind=8), rkind)
822
823 ! Check if we need to force a timestep length to hit the target interval
824 currtime = mpas_get_clock_time(clock, mpas_now, err_tmp)
825 !print *, 'curr', currTime % t % YR, currTime % t % basetime % S, currTime % t % basetime % Sn, currTime % t % basetime % Sd
826 err = ior(err,err_tmp)
827 nextforcetime = mpas_alarm_get_next_ring_time(clock, 'adaptiveTimestepForceInterval')
828 !print *, 'ring', nextForceTime % t % YR, nextForceTime % t % basetime % S, nextForceTime % t % basetime % Sn, &
829 ! nextForceTime % t % basetime % Sd
830 intervaltonextforcetime = nextforcetime - currtime
831 !print *, 'int', intervalToNextForceTime % ti % YR, intervalToNextForceTime % ti % MM, &
832 ! intervalToNextForceTime % ti % basetime % S, intervalToNextForceTime % ti % basetime % Sn, &
833 ! intervalToNextForceTime % ti % basetime % Sd
834 ! Due to roundoff errors, we might be just shy of the desired time.
835 ! To avoid this, add one to the numerator of the fractional seconds to
836 ! make sure we get pushed over the edge. The way ESMF does fractional
837 ! seconds, this means we get the desired interval to better than 1 part per 100 million seconds
838 ! Note that even though this is a *very* tiny fudge factor, it does not
839 ! affect conservation within MPAS-LI, but it could have a very, very tiny
840 ! effect on a climate model that thinks we ran for, say, 10 years, but we
841 ! actually ran for 10 years +/- 1e-8 seconds.
842 !intervalToNextForceTime % ti % basetime % Sn = intervalToNextForceTime % ti % basetime % Sn + 1
843 ! Note: commenting above line because it should no longer be relevant after rounding down to full second on line 815,
844 ! but we still include a check for fractional seconds.
845 call mpas_get_timeinterval(intervaltonextforcetime, dt=secondstonextforcetime, ierr=err_tmp)
846 err = ior(err,err_tmp)
847 if (secondstonextforcetime - real(floor(secondstonextforcetime, kind=8), rkind) /= 0.0_rkind) then
848 call mpas_log_write("set_timestep found secondsToNextForceTime not equal to 0.0: $r, decimal part=$r", mpas_log_err, realargs=(/secondstonextforcetime, secondstonextforcetime - real(floor(secondstonextforcetime, kind=8), rkind)/))
849 err = ior(err, 1)
850 endif
851 !print *, proposedDt, secondsToNextForceTime
852
853 ! --- Actually set the dt here ---
854 dtseconds = min(proposeddt, secondstonextforcetime)
855
856 call mpas_log_write(' Setting time step (days) to: $r', realargs=(/dtseconds / (86400.0_rkind)/))
857 if (dtseconds < config_min_adaptive_timestep) then
858 call mpas_log_write('New deltat is less than config_min_adaptive_timestep.', mpas_log_err)
859 err = ior(err, 1)
860 endif
861
862 !--------------------------------------------------------------------
863 end subroutine set_timestep
