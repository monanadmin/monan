164 function test_setup_clock(core_clock, configs) result(ierr)!{{{
165
166 use mpas_derived_types
167 use mpas_io_units
168
169 implicit none
170
171 type (mpas_clock_type), intent(inout) :: core_clock
172 type (mpas_pool_type), intent(inout) :: configs
173 integer :: ierr
174
175 type (mpas_time_type) :: starttime, stoptime
176 type (mpas_timeinterval_type) :: runduration, timestep
177 integer :: local_err
178
179 character (len=StrKIND), pointer :: config_start_time, config_run_duration, config_stop_time
180
181 ierr = 0
182
183 call mpas_pool_get_config(configs, 'config_start_time', config_start_time)
184 call mpas_pool_get_config(configs, 'config_run_duration', config_run_duration)
185 call mpas_pool_get_config(configs, 'config_stop_time', config_stop_time)
186
187 call mpas_set_time(curr_time=starttime, datetimestring=config_start_time, ierr=local_err)
188
189 if (trim(config_run_duration) /= "none") then
190 call mpas_set_timeinterval(runduration, timestring=config_run_duration, ierr=local_err)
191 call mpas_set_timeinterval(timestep, timestring=config_run_duration, ierr=local_err)
192 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, runduration=runduration, ierr=local_err)
193
194 if (trim(config_stop_time) /= "none") then
195 call mpas_set_time(curr_time=stoptime, datetimestring=config_stop_time, ierr=local_err)
196 if(starttime + runduration /= stoptime) then
197 call mpas_log_write('config_run_duration and config_stop_time are inconsitent: using config_run_duration.', mpas_log_warn)
198 end if
199 end if
200 else if (trim(config_stop_time) /= "none") then
201 call mpas_set_time(curr_time=stoptime, datetimestring=config_stop_time, ierr=local_err)
202 timestep = stoptime - starttime
203 call mpas_create_clock(core_clock, starttime=starttime, timestep=timestep, stoptime=stoptime, ierr=local_err)
204 else
205 call mpas_log_write('Neither config_run_duration nor config_stop_time were specified.', mpas_log_err)
206 ierr = 1
207 end if
208
209
210 end function test_setup_clock!}}}
