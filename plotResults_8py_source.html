133def computeSectionCellIndices():
134 y = options.sectionY
135 xMin = numpy.amin(xCell)
136 xMax = numpy.amax(xCell)
137 xs = numpy.linspace(xMin,xMax,10000)
138 cellIndices = []
139 for x in xs:
140 distanceSquared = (x - xCell)**2 + (y-yCell)**2
141 index = numpy.argmin(distanceSquared)
142 if(len(cellIndices) == 0 or cellIndices[-1] != index):
143 cellIndices.append(index)

147def cellToSectionEdges(field):
148 nx = len(sectionCellIndices)
149 fieldMid = field[sectionCellIndices]
150 fieldEdge = numpy.ma.masked_all(nx+1)
151 fieldEdge[1:-1] = 0.5*(fieldMid[0:-1]+fieldMid[1:])
152 # extrapolate ends
153 fieldEdge[0] = 2*fieldMid[0]-fieldEdge[1]
154 fieldEdge[-1] = 2*fieldMid[-1]-fieldEdge[-2]
155 return fieldEdge
